# 4.1.1 统一文档抽象接口

## 概念讲解

### 文档处理的演进：从碎片化到标准化

在现代AI应用开发中，文档处理是一个基础且关键的环节。LangChain v1.2.22通过**统一文档抽象接口**彻底改变了传统的碎片化处理方式。传统AI应用开发中，不同文件类型（TXT、PDF、CSV、DOCX等）需要独立的处理逻辑，导致代码重复、维护困难，且难以实现统一的文档抽象。

#### 传统文档处理的核心问题：

1. **碎片化处理逻辑**：每种文件类型独立实现，代码重复严重
2. **不一致的元数据结构**：不同格式的元数据字段不同，难以统一查询
3. **扩展性差**：硬编码的处理器映射，难以添加自定义处理器
4. **错误处理复杂**：每种格式的错误处理独立实现，缺乏标准化

#### LangChain的解决方案：

LangChain通过`Document`类和`DocumentLoader`抽象提供了统一的文档处理接口。`Document`类封装了文档内容和元数据，`DocumentLoader`提供了标准化的加载接口，支持多种文件格式的透明处理。

## 核心要点

### 1. Document类的核心设计

`Document`类是LangChain文档处理的基础单元，包含两个核心属性：
- **page_content**: 文档的核心文本内容
- **metadata**: 文档的元数据字典，包含来源、作者、时间等附加信息

### 2. DocumentLoader的统一接口

所有文档加载器都遵循统一的接口设计：
- `load()`: 同步加载所有文档
- `lazy_load()`: 流式懒加载文档，适用于大文件
- `load_and_split()`: 加载并自动分割文档

### 3. 元数据标准化系统

LangChain定义了标准化的元数据字段，包括：
- 基础元数据：source、type、language
- 时间元数据：created_date、modified_date、processed_date
- 结构元数据：page、section、chunk_index
- 内容元数据：title、author、keywords、summary

### 4. 插件化架构设计

LangChain采用插件化架构，支持：
- 动态注册新的文档加载器
- 自定义元数据提取器
- 灵活的文档处理管道

```python
    for doc in documents:
        doc.metadata.setdefault("file_type", ext)
        doc.metadata.setdefault("processed_at", datetime.now().isoformat())
    
    return documents
```

```python
from typing import List
from abc import ABC, abstractmethod

# 定义文档加载器基类接口
class BaseDocumentLoader(ABC):
    """文档加载器基类"""
    
    @abstractmethod
    def load(self) -> List[Document]:
        """加载所有文档"""
        pass
    
    @abstractmethod
    def lazy_load(self):
        """流式加载文档"""
        pass

# 具体加载器实现示例
class SimpleTextLoader(BaseDocumentLoader):
    """简单文本加载器实现"""
    
    def __init__(self, file_path: str, encoding: str = 'utf-8'):
        self.file_path = file_path
        self.encoding = encoding
    
    def load(self) -> List[Document]:
        """加载文本文件"""
        try:
            with open(self.file_path, 'r', encoding=self.encoding) as f:
                content = f.read()
            
            return [Document(
                page_content=content,
                metadata={
                    "source": self.file_path,
                    "type": "text",
                    "encoding": self.encoding,
                    "file_size": len(content.encode(self.encoding))
                }
            )]
        except Exception as e:
            raise IOError(f"加载文件失败: {e}")
    
    def lazy_load(self):
        """流式加载（按行）"""
        try:
            with open(self.file_path, 'r', encoding=self.encoding) as f:
                for line_num, line in enumerate(f, 1):
                    yield Document(
                        page_content=line.strip(),
                        metadata={
                            "source": self.file_path,
                            "type": "text",
                            "line": line_num,
                            "encoding": self.encoding
                        }
                    )
        except Exception as e:
            raise IOError(f"流式加载失败: {e}")
```

```python
from langchain_core.documents import BaseDocumentLoader
from typing import List, Iterator
import json

class CustomJSONLoader(BaseDocumentLoader):
    """自定义JSON文档加载器"""
    
    def __init__(self, file_path: str, content_field: str = "content"):
        self.file_path = file_path
        self.content_field = content_field
    
    def load(self) -> List[Document]:
        """加载JSON文档"""
        try:
            with open(self.file_path, 'r', encoding='utf-8') as f:
                data = json.load(f)
            
            documents = []
            if isinstance(data, list):
                # 处理JSON数组
                for item in data:
                    if isinstance(item, dict) and self.content_field in item:
                        doc = Document(
                            page_content=str(item[self.content_field]),
                            metadata={k: v for k, v in item.items() if k != self.content_field}
                        )
                        documents.append(doc)
            elif isinstance(data, dict):
                # 处理单个JSON对象
                if self.content_field in data:
                    doc = Document(
                        page_content=str(data[self.content_field]),
                        metadata={k: v for k, v in data.items() if k != self.content_field}
                    )
                    documents.append(doc)
            
            return documents
        except Exception as e:
            raise IOError(f"加载JSON文件失败: {e}")
    
    def lazy_load(self) -> Iterator[Document]:
        """流式加载JSON文档"""
        try:
            with open(self.file_path, 'r', encoding='utf-8') as f:
                data = json.load(f)
            
            if isinstance(data, list):
                for item in data:
                    if isinstance(item, dict) and self.content_field in item:
                        yield Document(
                            page_content=str(item[self.content_field]),
                            metadata={k: v for k, v in item.items() if k != self.content_field}
                        )
        except Exception as e:
            raise IOError(f"流式加载JSON失败: {e}")
```

### 应用2：文档处理管道设计

```python
from typing import List, Callable, Optional

class DocumentProcessingPipeline:
    """文档处理管道"""
    
    def __init__(self):
        self.loaders = {}
        self.preprocessors = []
        self.validators = []
    
    def register_loader(self, file_type: str, loader_class):
        """注册文档加载器"""
        self.loaders[file_type] = loader_class
    
    def add_preprocessor(self, preprocessor: Callable[[Document], Document]):
        """添加预处理器"""
        self.preprocessors.append(preprocessor)
    
    def add_validator(self, validator: Callable[[Document], bool]):
        """添加验证器"""
        self.validators.append(validator)
    
    def process_document(self, file_path: str, **kwargs) -> List[Document]:
        """处理文档"""
        import os
        
        # 1. 选择加载器
        _, ext = os.path.splitext(file_path)
        ext = ext.lower().lstrip('.')
        
        loader_class = self.loaders.get(ext)
        if not loader_class:
            raise ValueError(f"不支持的文件类型: {ext}")
        
        # 2. 加载文档
        loader = loader_class(file_path, **kwargs)
        documents = loader.load()
        
        processed_docs = []
        for doc in documents:
            # 3. 应用预处理器
            processed_doc = doc
            for preprocessor in self.preprocessors:
                processed_doc = preprocessor(processed_doc)
            
            # 4. 验证文档
            is_valid = True
            for validator in self.validators:
                if not validator(processed_doc):
                    is_valid = False
                    break
            
            if is_valid:
                processed_docs.append(processed_doc)
        
        return processed_docs

# 使用示例
def create_sample_pipeline():
    """创建示例处理管道"""
    pipeline = DocumentProcessingPipeline()
    
    # 注册加载器
    pipeline.register_loader("txt", SimpleTextLoader)
    pipeline.register_loader("json", CustomJSONLoader)
    
    # 添加预处理器
    def clean_text(doc: Document) -> Document:
        """清理文本"""
        cleaned = doc.page_content.strip()
        return Document(page_content=cleaned, metadata=doc.metadata)
    
    def add_timestamp(doc: Document) -> Document:
        """添加时间戳"""
        from datetime import datetime
        doc.metadata["processed_at"] = datetime.now().isoformat()
        return doc
    
    pipeline.add_preprocessor(clean_text)
    pipeline.add_preprocessor(add_timestamp)
    
    # 添加验证器
    def validate_content(doc: Document) -> bool:
        """验证内容"""
        return len(doc.page_content) > 10
    
    pipeline.add_validator(validate_content)
    
    return pipeline
```

### 应用3：元数据管理系统

```python
from typing import Dict, Any, List
from dataclasses import dataclass
from datetime import datetime

@dataclass
class DocumentMetadata:
    """文档元数据模型"""
    source: str
    type: str = "document"
    language: str = "unknown"
    title: Optional[str] = None
    author: Optional[str] = None
    created_date: Optional[str] = None
    keywords: List[str] = None
    custom_fields: Dict[str, Any] = None
    
    def __post_init__(self):
        if self.keywords is None:
            self.keywords = []
        if self.custom_fields is None:
            self.custom_fields = {}
    
    def to_dict(self) -> Dict[str, Any]:
        """转换为字典"""
        result = {
            "source": self.source,
            "type": self.type,
            "language": self.language,
            "title": self.title,
            "author": self.author,
            "created_date": self.created_date,
            "keywords": self.keywords.copy()
        }
        
        # 添加自定义字段
        if self.custom_fields:
            result.update({f"custom_{k}": v for k, v in self.custom_fields.items()})
        
        return result
    
    def validate(self) -> List[str]:
        """验证元数据"""
        errors = []
        
        if not self.source:
            errors.append("缺少必需字段: source")
        
        if self.language not in ["zh-CN", "en-US", "ja-JP", "ko-KR", "unknown"]:
            errors.append(f"不支持的语言: {self.language}")
        
        return errors

class MetadataManager:
    """元数据管理器"""
    
    def __init__(self):
        self.metadata_store = {}
    
    def register_document(self, doc_id: str, metadata: DocumentMetadata):
        """注册文档元数据"""
        # 验证元数据
        errors = metadata.validate()
        if errors:
            raise ValueError(f"元数据验证失败: {errors}")
        
        self.metadata_store[doc_id] = metadata
    
    def search_by_metadata(self, **criteria) -> List[str]:
        """根据元数据搜索文档"""
        results = []
        
        for doc_id, metadata in self.metadata_store.items():
            match = True
            
            for field, value in criteria.items():
                if hasattr(metadata, field):
                    actual_value = getattr(metadata, field)
                    if actual_value != value:
                        match = False
                        break
                elif field.startswith("custom_"):
                    custom_field = field[7:]  # 去掉"custom_"前缀
                    if custom_field not in metadata.custom_fields or metadata.custom_fields[custom_field] != value:
                        match = False
                        break
                else:
                    match = False
                    break
            
            if match:
                results.append(doc_id)
        
        return results
    
    def get_statistics(self) -> Dict[str, Any]:
        """获取统计信息"""
        if not self.metadata_store:
            return {"total_documents": 0}
        
        stats = {
            "total_documents": len(self.metadata_store),
            "language_distribution": {},
            "type_distribution": {},
            "keywords_usage": {}
        }
        
        for metadata in self.metadata_store.values():
            # 语言分布
            lang = metadata.language
            stats["language_distribution"][lang] = stats["language_distribution"].get(lang, 0) + 1
            
            # 类型分布
            doc_type = metadata.type
            stats["type_distribution"][doc_type] = stats["type_distribution"].get(doc_type, 0) + 1
            
            # 关键词使用
            for keyword in metadata.keywords:
                stats["keywords_usage"][keyword] = stats["keywords_usage"].get(keyword, 0) + 1
        
        return stats
```

## 常见问题

### Q1: 如何处理不支持的文件格式？
**A**: LangChain提供了两种方式处理不支持的文件格式：
1. 使用`UnstructuredFileLoader`通用加载器，它使用非结构化库处理多种格式
2. 实现自定义的`BaseDocumentLoader`子类，注册到系统中

### Q2: 文档元数据应该包含哪些字段？
**A**: 建议包含以下标准字段：
- 必需字段：source（文档来源）
- 建议字段：type、language、title、author、created_date
- 可选字段：根据业务需求添加自定义字段

### Q3: 如何处理大文件的内存问题？
**A**: 使用`lazy_load()`方法进行流式处理，避免一次性加载整个文件到内存。对于超大文件，可以考虑分块加载和处理。

### Q4: 如何保证文档加载的性能？
**A**: 性能优化建议：
1. 使用缓存机制存储已加载的文档
2. 实现并行加载多个文档
3. 对于频繁访问的文档，考虑预加载
4. 使用合适的文本分割策略减少处理量

### Q5: 文档验证失败时如何处理？
**A**: 建议实现分级验证策略：
1. 基础验证：检查必需字段和基本格式
2. 内容验证：检查内容质量和相关性
3. 业务验证：根据业务规则进行验证
4. 对于验证失败的文档，记录错误信息并提供修复建议

```python
# 传统文档处理的碎片化问题
class TraditionalDocumentSystem:
    def __init__(self):
        # 不同文件类型的独立处理逻辑
        self.file_processors = {
            "txt": self._process_text_file,
            "pdf": self._process_pdf_file,
            "csv": self._process_csv_file,
            "docx": self._process_docx_file,
            "html": self._process_html_file,
        }
        
        # 分散的元数据管理
        self.metadata_schemas = {
            "txt": {"encoding": "utf-8", "line_ending": "\n"},
            "pdf": {"pages": True, "ocr": False},
            "csv": {"delimiter": ",", "header": True},
        }
    
    def _process_text_file(self, file_path: str) -> Dict[str, Any]:
        """处理文本文件"""
        import codecs
        
        try:
            with codecs.open(file_path, 'r', encoding='utf-8') as f:
                content = f.read()
            
            return {
                "content": content,
                "metadata": {
                    "type": "text",
                    "encoding": "utf-8",
                    "file_size": len(content.encode('utf-8')),
                    "lines": content.count('\n') + 1
                }
            }
        except Exception as e:
            print(f"Error processing text file: {e}")
            return {"content": "", "metadata": {"error": str(e)}}
    
    def _process_pdf_file(self, file_path: str) -> Dict[str, Any]:
        """处理PDF文件"""
        import PyPDF2
        
        try:
            content_parts = []
            metadata = {"type": "pdf", "pages": 0, "titles": []}
            
            with open(file_path, 'rb') as f:
                pdf_reader = PyPDF2.PdfReader(f)
                metadata["pages"] = len(pdf_reader.pages)
                
                for i, page in enumerate(pdf_reader.pages):
                    text = page.extract_text()
                    content_parts.append(f"--- Page {i+1} ---\n{text}")
                    
                    # 尝试提取标题
                    if i == 0 and hasattr(pdf_reader, 'metadata'):
                        if pdf_reader.metadata:
                            metadata["title"] = pdf_reader.metadata.get('/Title', '')
            
            content = "\n\n".join(content_parts)
            return {"content": content, "metadata": metadata}
        except Exception as e:
            print(f"Error processing PDF file: {e}")
            return {"content": "", "metadata": {"error": str(e)}}
    
    def _process_csv_file(self, file_path: str) -> Dict[str, Any]:
        """处理CSV文件"""
        import csv
        
        try:
            content_parts = []
            metadata = {"type": "csv", "rows": 0, "columns": []}
            
            with open(file_path, 'r', encoding='utf-8') as f:
                csv_reader = csv.reader(f)
                rows = list(csv_reader)
                
                if rows:
                    metadata["columns"] = rows[0]
                    metadata["rows"] = len(rows) - 1
                    
                    for i, row in enumerate(rows[1:], 1):
                        row_content = " | ".join(row)
                        content_parts.append(f"Row {i}: {row_content}")
            
            content = "\n".join(content_parts)
            return {"content": content, "metadata": metadata}
        except Exception as e:
            print(f"Error processing CSV file: {e}")
            return {"content": "", "metadata": {"error": str(e)}}
    
    def _process_docx_file(self, file_path: str) -> Dict[str, Any]:
        """处理DOCX文件"""
        from docx import Document as DocxDocument
        
        try:
            content_parts = []
            metadata = {"type": "docx", "paragraphs": 0, "tables": 0}
            
            doc = DocxDocument(file_path)
            metadata["paragraphs"] = len(doc.paragraphs)
            metadata["tables"] = len(doc.tables)
            
            for para in doc.paragraphs:
                if para.text.strip():
                    content_parts.append(para.text)
            
            content = "\n".join(content_parts)
            return {"content": content, "metadata": metadata}
        except Exception as e:
            print(f"Error processing DOCX file: {e}")
            return {"content": "", "metadata": {"error": str(e)}}
    
    def _process_html_file(self, file_path: str) -> Dict[str, Any]:
        """处理HTML文件"""
        from bs4 import BeautifulSoup
        
        try:
            with open(file_path, 'r', encoding='utf-8') as f:
                html_content = f.read()
            
            soup = BeautifulSoup(html_content, 'html.parser')
            
            # 提取文本内容
            for script in soup(["script", "style"]):
                script.decompose()
            
            text = soup.get_text()
            lines = (line.strip() for line in text.splitlines())
            chunks = (phrase.strip() for line in lines for phrase in line.split("  "))
            content = '\n'.join(chunk for chunk in chunks if chunk)
            
            metadata = {
                "type": "html",
                "title": soup.title.string if soup.title else "",
                "tags": [tag.name for tag in soup.find_all()],
                "links": len(soup.find_all('a'))
            }
            
            return {"content": content, "metadata": metadata}
        except Exception as e:
            print(f"Error processing HTML file: {e}")
            return {"content": "", "metadata": {"error": str(e)}}
    
    def process_document(self, file_path: str) -> Dict[str, Any]:
```

```python
# Document类的完整结构
from pydantic import BaseModel, Field
from typing import Any, Dict, Optional
from datetime import datetime

class DetailedDocument(BaseModel):
    """详细的Document结构定义"""
    
    # 核心内容
    page_content: str = Field(
        ...,
        description="文档的核心文本内容，通常是经过清理和标准化的文本"
    )
    
    # 元数据
    metadata: Dict[str, Any] = Field(
        default_factory=dict,
        description="文档的元数据，包含来源、作者、时间等附加信息"
    )
    
    # 可选的唯一标识
    id: Optional[str] = Field(
        default=None,
        description="文档的唯一标识符，用于去重和引用"
    )
    
    # 类型信息
    type: str = Field(
        default="document",
        description="文档类型，如'document'、'chunk'、'summary'等"
    )
    
    # 向量嵌入
    embedding: Optional[List[float]] = Field(
        default=None,
        description="文档的向量嵌入表示，用于相似性搜索"
    )
    
    # 评分和权重
    score: Optional[float] = Field(
        default=None,
        description="文档的相关性评分，用于检索排序"
    )
    
    # 创建时间
    created_at: datetime = Field(
        default_factory=datetime.now,
        description="文档创建时间"
    )
    
    # 更新时间
    updated_at: datetime = Field(
        default_factory=datetime.now,
        description="文档最后更新时间"
    )
    
    # 验证方法
    def validate_content(self) -> bool:
        """验证文档内容"""
        if not self.page_content or not self.page_content.strip():
            return False
        
        # 检查内容长度
        if len(self.page_content) < 10:
            print("警告：文档内容过短")
        
        # 检查元数据完整性
        required_metadata = ["source"]
        for field in required_metadata:
            if field not in self.metadata:
                print(f"警告：缺少必需元数据字段: {field}")
        
        return True
    
    def get_content_summary(self) -> Dict[str, Any]:
        """获取内容摘要"""
        content = self.page_content
        
        return {
            "char_count": len(content),
            "word_count": len(content.split()),
            "line_count": content.count('\n') + 1,
            "byte_size": len(content.encode('utf-8')),
            "language": self.metadata.get("language", "unknown"),
            "has_metadata": bool(self.metadata)
        }
    
    def to_dict(self, include_embedding: bool = False) -> Dict[str, Any]:
        """转换为字典"""
        result = {
            "page_content": self.page_content,
            "metadata": self.metadata,
            "id": self.id,
            "type": self.type,
            "created_at": self.created_at.isoformat(),
            "updated_at": self.updated_at.isoformat()
        }
        
        if include_embedding and self.embedding:
            result["embedding"] = self.embedding
        
        if self.score is not None:
            result["score"] = self.score
        
        return result
```

```python
from abc import ABC, abstractmethod
from typing import List, Iterator, Optional, Dict, Any
import inspect

class BaseDocumentLoaderDesign(ABC):
    """DocumentLoader抽象设计"""
    
    def __init__(self, file_path: str, **kwargs):
        self.file_path = file_path
        self.kwargs = kwargs
        self.loaded = False
        self.documents = []
    
    @abstractmethod
    def load(self) -> List[Document]:
        """加载所有文档"""
        pass
    
    @abstractmethod 
    def lazy_load(self) -> Iterator[Document]:
        """懒加载文档（流式处理）"""
        pass
    
    def load_and_split(self, text_splitter) -> List[Document]:
        """加载并分割文档"""
        documents = self.load()
        return text_splitter.split_documents(documents)
    
    def get_file_info(self) -> Dict[str, Any]:
        """获取文件信息"""
        import os
        import hashlib
        
        if not os.path.exists(self.file_path):
            return {"error": "File not found"}
        
        try:
            file_stats = os.stat(self.file_path)
            
            # 计算文件哈希
            file_hash = None
            try:
                with open(self.file_path, 'rb') as f:
                    file_hash = hashlib.md5(f.read()).hexdigest()
            except:
                pass
            
            return {
                "path": self.file_path,
                "size": file_stats.st_size,
                "created": file_stats.st_ctime,
                "modified": file_stats.st_mtime,
                "hash": file_hash,
                "extension": os.path.splitext(self.file_path)[1]
            }
        except Exception as e:
            return {"error": str(e)}
    
    def validate_file(self) -> bool:
        """验证文件"""
        import os
        
        if not os.path.exists(self.file_path):
            raise FileNotFoundError(f"File not found: {self.file_path}")
        
        if not os.path.isfile(self.file_path):
            raise ValueError(f"Not a file: {self.file_path}")
        
        return True
    
    def get_loader_info(self) -> Dict[str, Any]:
        """获取加载器信息"""
        return {
            "loader_class": self.__class__.__name__,
            "file_path": self.file_path,
            "parameters": self.kwargs,
            "abstract_methods": [
                method for method in dir(self) 
                if callable(getattr(self, method)) and method.startswith('_') is False
            ]
        }

class TextLoaderImplementation(BaseDocumentLoaderDesign):
    """文本加载器实现"""
    
    def load(self) -> List[Document]:
        """加载文本文件"""
        
        self.validate_file()
        
        try:
            with open(self.file_path, 'r', encoding=self.kwargs.get('encoding', 'utf-8')) as f:
                content = f.read()
            
            document = Document(
                page_content=content,
                metadata={
                    "source": self.file_path,
                    "type": "text",
                    "encoding": self.kwargs.get('encoding', 'utf-8'),
                    "loaded_by": self.__class__.__name__
                }
            )
            
            self.documents = [document]
            self.loaded = True
            
            return self.documents
            
        except Exception as e:
            raise IOError(f"Failed to load text file: {e}")
    
    def lazy_load(self) -> Iterator[Document]:
        """懒加载文本文件（按行）"""
        
        self.validate_file()
        
        try:
            with open(self.file_path, 'r', encoding=self.kwargs.get('encoding', 'utf-8')) as f:
                for line_num, line in enumerate(f, 1):
                    yield Document(
                        page_content=line.strip(),
                        metadata={
                            "source": self.file_path,
                            "type": "text",
                            "line": line_num,
                            "loaded_by": self.__class__.__name__
                        }
                    )
                    
        except Exception as e:
            raise IOError(f"Failed to lazy load text file: {e}")

    def create_loader(cls, file_path: str, **kwargs):
        """创建加载器"""
        import os
        
        _, ext = os.path.splitext(file_path)
        ext = ext.lower().lstrip('.')

        if not loader_class:
            # 使用通用加载器

            return UnstructuredFileLoader(file_path, **kwargs)
        
        return loader_class(file_path, **kwargs)
    
    @classmethod
    def get_registered_loaders(cls):
        """获取注册的加载器"""
        return cls._loader_registry.copy()
```

```python
from typing import Dict, Any, List, Optional, Set
from pydantic import BaseModel, Field, validator
import hashlib

class StandardizedMetadata(BaseModel):
    """标准化元数据模型"""

    custom: Dict[str, Any] = Field(default_factory=dict, description="自定义元数据")
    
    @validator('language')
    def validate_language(cls, v):
        """验证语言代码"""
        valid_languages = ["zh-CN", "zh-TW", "en-US", "en-GB", "ja-JP", "ko-KR"]
        if v != "unknown" and v not in valid_languages:
            print(f"警告：不支持的语言代码: {v}")
        return v

    def validate_quality_score(cls, v):
        """验证质量评分"""
        if v is not None and (v < 0 or v > 1):
            raise ValueError("质量评分必须在0到1之间")
        return v
    
    def to_flat_dict(self) -> Dict[str, Any]:
        """转换为扁平字典"""
        result = {}
        
        # 处理基本字段
        for field_name, field_value in self.dict().items():
            if field_value is not None:
                if field_name == "custom":
                    # 展开自定义字段
                    for custom_key, custom_value in field_value.items():
                        result[f"custom_{custom_key}"] = custom_value
                else:
                    result[field_name] = field_value
        
        return result

class MetadataManager:
    """元数据管理器"""
    
    def __init__(self):
        self.metadata_registry = {}
        self.fingerprint_index = {}
        
        # 定义元数据验证规则
        self.validation_rules = {
            "source": {"required": True, "type": str},
            "type": {"required": True, "type": str, "default": "document"},
            "language": {"required": False, "type": str, "default": "unknown"},
            "page": {"required": False, "type": int, "min": 1},
            "quality_score": {"required": False, "type": float, "min": 0, "max": 1}
        }
    
    def validate_metadata(self, metadata: Dict[str, Any]) -> Dict[str, List[str]]:
        """验证元数据"""
        errors = []
        warnings = []
        
        # 检查必需字段
        for field, rules in self.validation_rules.items():
            if rules.get("required", False) and field not in metadata:
                errors.append(f"缺少必需字段: {field}")
        
        # 检查字段类型
        for field, value in metadata.items():
            if field in self.validation_rules:
                expected_type = self.validation_rules[field].get("type")
                if expected_type and not isinstance(value, expected_type):
                    errors.append(f"字段类型错误: {field}应为{expected_type}，实际为{type(value)}")
            
            # 检查数值范围
            if field == "page" and value is not None and value < 1:
                errors.append(f"页码必须大于0: {value}")
            elif field == "quality_score" and value is not None:
                if value < 0 or value > 1:
                    errors.append(f"质量评分必须在0到1之间: {value}")
        
        # 检查未知字段
        known_fields = set(self.validation_rules.keys()) | {"custom"}
        unknown_fields = set(metadata.keys()) - known_fields
        if unknown_fields:
            warnings.append(f"未知字段: {unknown_fields}")
        
        return {"errors": errors, "warnings": warnings}
    
    def standardize_metadata(self, raw_metadata: Dict[str, Any]) -> StandardizedMetadata:
        """标准化元数据"""
        
        # 验证元数据
        validation_result = self.validate_metadata(raw_metadata)
        
        if validation_result["errors"]:
            print(f"元数据验证错误: {validation_result['errors']}")
        
        if validation_result["warnings"]:
            print(f"元数据验证警告: {validation_result['warnings']}")
        
        # 提取自定义字段
        custom_fields = {}
        standard_fields = {}
        
        known_fields = set(self.validation_rules.keys())
        
        for key, value in raw_metadata.items():
            if key in known_fields:
                standard_fields[key] = value
            else:
                custom_fields[key] = value
        
        # 设置默认值
        for field, rules in self.validation_rules.items():
            if field not in standard_fields and "default" in rules:
                standard_fields[field] = rules["default"]
        
        # 创建标准化元数据
        if custom_fields:
            standard_fields["custom"] = custom_fields
        
        return StandardizedMetadata(**standard_fields)
    
    def register_document_metadata(self, document_id: str, metadata: StandardizedMetadata):
        """注册文档元数据"""
        self.metadata_registry[document_id] = metadata
        
        # 建立指纹索引
        fingerprint = metadata.generate_fingerprint()
        if fingerprint not in self.fingerprint_index:
            self.fingerprint_index[fingerprint] = []
        self.fingerprint_index[fingerprint].append(document_id)
    
    def find_by_fingerprint(self, fingerprint: str) -> List[str]:
        """通过指纹查找文档"""
        return self.fingerprint_index.get(fingerprint, [])
    
    def search_by_metadata(self, query: Dict[str, Any]) -> List[str]:
        """通过元数据查询文档"""
        results = []
        
        for doc_id, metadata in self.metadata_registry.items():
            match = True
            
            for field, expected_value in query.items():
                actual_value = getattr(metadata, field, None)
                
                if actual_value != expected_value:
                    match = False
                    break
            
            if match:
                results.append(doc_id)
        
        return results
    
    def get_metadata_summary(self) -> Dict[str, Any]:
        """获取元数据摘要"""
        if not self.metadata_registry:
            return {"total_documents": 0}
        
        # 统计信息
        total_docs = len(self.metadata_registry)
        
        # 语言分布
        language_dist = {}
        type_dist = {}
        
        for metadata in self.metadata_registry.values():
            # 语言分布
            lang = metadata.language
            language_dist[lang] = language_dist.get(lang, 0) + 1
            
            # 类型分布
            doc_type = metadata.type
            type_dist[doc_type] = type_dist.get(doc_type, 0) + 1
        
        # 质量评分统计
        quality_scores = [
            m.quality_score for m in self.metadata_registry.values() 
            if m.quality_score is not None
        ]
        
        quality_stats = {}
        if quality_scores:
            quality_stats = {
                "min": min(quality_scores),
                "max": max(quality_scores),
                "avg": sum(quality_scores) / len(quality_scores),
                "count": len(quality_scores)
            }
        
        return {
            "total_documents": total_docs,
            "language_distribution": language_dist,
            "type_distribution": type_dist,
            "quality_statistics": quality_stats,
            "unique_fingerprints": len(self.fingerprint_index)
        }
```

```python
class DocumentProcessingPipeline:
    """文档处理管道"""
    
    def __init__(self):
        self.loaders = {}
        self.processors = []
        self.validators = []
        
    def register_loader(self, file_type: str, loader_class):
        """注册加载器"""
        self.loaders[file_type] = loader_class
    
    def add_processor(self, processor_func):
        """添加处理器"""
        self.processors.append(processor_func)
    
    def add_validator(self, validator_func):
        """添加验证器"""
        self.validators.append(validator_func)
    
    def process_document(self, file_path: str, **kwargs) -> List[Document]:
        """处理文档"""
        
        # 1. 验证文件
        if not os.path.exists(file_path):
            raise FileNotFoundError(f"File not found: {file_path}")
        
        # 2. 选择加载器
        _, ext = os.path.splitext(file_path)
        ext = ext.lower().lstrip('.')
        
        loader_class = self.loaders.get(ext)
        if not loader_class:
            raise ValueError(f"No loader registered for file type: {ext}")
        
        # 3. 加载文档
        loader = loader_class(file_path, **kwargs)
        documents = loader.load()
        
        # 4. 应用处理器
        processed_documents = []
        for doc in documents:
            processed_doc = doc
            
            for processor in self.processors:
                processed_doc = processor(processed_doc)
            
            processed_documents.append(processed_doc)
        
        # 5. 验证文档
        valid_documents = []
        validation_errors = []
        
        for doc in processed_documents:
            is_valid = True
            
            for validator in self.validators:
                validation_result = validator(doc)
                if not validation_result["valid"]:
                    is_valid = False
                    validation_errors.extend(validation_result["errors"])
                    break
            
            if is_valid:
                valid_documents.append(doc)
        
        # 6. 返回结果
        return {
            "documents": valid_documents,
            "total_loaded": len(documents),
            "total_valid": len(valid_documents),
            "validation_errors": validation_errors
        }
```
