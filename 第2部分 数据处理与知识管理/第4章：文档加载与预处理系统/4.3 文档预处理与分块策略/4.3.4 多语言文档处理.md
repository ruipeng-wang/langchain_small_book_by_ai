# 4.3.4 多语言文档处理

## 概念讲解

### 全球化时代的文档智能挑战

在全球化的业务环境中，多语言文档处理已成为AI应用开发的**核心挑战与关键机遇**。LangChain v1.2.22通过先进的多语言处理技术，帮助企业应对语言多样性带来的复杂性，构建真正全球化、智能化的文档处理系统。

#### 多语言处理的核心技术挑战

**语言多样性挑战**：
- **语言识别歧义**：短文本、混合文本、领域特定术语的语言识别困难
- **编码兼容性问题**：不同语言使用不同编码标准（UTF-8、GBK、Big5、Shift-JIS等）
- **语义边界差异**：不同语言的句子边界、词语分割规则存在显著差异
- **资源不均衡**：主流语言与低资源语言的工具质量差异巨大
- **文化语境理解**：语言表达中的文化特定性和语境依赖性

**技术实现复杂度**：
- **混合语言处理**：同一文档中多种语言的混合和交替使用
- **实时处理需求**：需要支持流式、实时的大规模多语言处理
- **质量保障困难**：不同语言的处理质量评估标准和方法不同
- **性能优化挑战**：多语言处理的计算资源和时间成本优化

#### LangChain多语言处理架构设计

LangChain采用**分层、模块化、可扩展**的多语言处理架构：

```
├── 处理层（Processing Layer）
```

```python
from langchain_core.documents import Document
from langchain_text_splitters import RecursiveCharacterTextSplitter

# 创建多语言感知的分割器
multilingual_splitter = RecursiveCharacterTextSplitter(
    chunk_size=500,
    chunk_overlap=100,
    separators=["\n\n", "\n", ".", "。", "!", "！", "?", "？", ";", "；", ",", "，", " ", ""]
)

# 处理混合语言文档
mixed_text = """
这是中文段落。人工智能技术正在快速发展。

This is an English paragraph. AI technology is evolving rapidly.

これは日本語の段落です。人工知能技術が進歩しています。

이것은 한국어 단락입니다. 인공지능 기술이 발전하고 있습니다.
"""

# 创建文档对象
mixed_doc = Document(
    page_content=mixed_text,
    metadata={"source": "multilingual_report.txt", "language": "mixed"}
)

# 分割文档
chunks = multilingual_splitter.split_documents([mixed_doc])

print(f"混合语言文档分割为 {len(chunks)} 个块")
for i, chunk in enumerate(chunks[:3]):  # 显示前3个块
    print(f"\n块 {i+1} (长度: {len(chunk.page_content)} 字符):")

```

### 语言检测与编码处理示例

```python
import chardet
from langdetect import detect, detect_langs
from typing import Tuple, Dict, Any

class BasicMultilingualProcessor:
    """基础多语言处理器"""
    
    def __init__(self):
        # 常见编码列表（按优先级排序）
        self.common_encodings = [
            'utf-8', 'utf-8-sig',  # UTF-8 及其带BOM版本
            'gbk', 'gb2312', 'gb18030',  # 中文编码
            'big5',  # 繁体中文编码
            'shift_jis', 'euc-jp',  # 日文编码
            'euc-kr', 'cp949',  # 韩文编码
            'iso-8859-1', 'windows-1252',  # 拉丁编码
            'ascii'  # ASCII编码
        ]
    
    def detect_encoding(self, raw_bytes: bytes) -> Tuple[str, float]:
        """检测文本编码"""
        try:
            # 使用chardet检测编码
            result = chardet.detect(raw_bytes)
            encoding = result['encoding']
            confidence = result['confidence']
            
            # 如果检测置信度低，尝试其他方法
            if confidence < 0.7:
                encoding = self._try_common_encodings(raw_bytes)
                confidence = 0.5  # 降低置信度
                
            return encoding, confidence
            
        except Exception as e:
            print(f"编码检测出错: {e}")
            return 'utf-8', 0.3  # 默认使用UTF-8
    
    def _try_common_encodings(self, raw_bytes: bytes) -> str:
        """尝试常见编码"""
        for encoding in self.common_encodings:
            try:
                # 尝试解码
                raw_bytes.decode(encoding, errors='strict')
                return encoding
            except UnicodeDecodeError:
                continue
        
        return 'utf-8'  # 默认编码
    
    def detect_language(self, text: str) -> Dict[str, Any]:
        """检测文档语言"""
        try:
            # 使用langdetect检测语言
            primary_lang = detect(text)
            lang_probabilities = detect_langs(text)
            
            # 判断是否为混合语言
            is_mixed = len([lp for lp in lang_probabilities if lp.prob > 0.2]) > 1
            
            return {
                'primary_language': primary_lang,
                'language_probabilities': [
                    {'lang': str(lang.lang), 'probability': lang.prob}
                    for lang in lang_probabilities
                ],
                'is_mixed_language': is_mixed,
                'confidence': max([lp.prob for lp in lang_probabilities]) if lang_probabilities else 0.0
            }
            
        except Exception as e:
            print(f"语言检测出错: {e}")
            return {
                'primary_language': 'unknown',
                'language_probabilities': [],
                'is_mixed_language': False,
                'confidence': 0.0
            }
    
    def process_document(self, raw_bytes: bytes) -> Dict[str, Any]:
        """处理多语言文档"""
        # 检测编码
        encoding, encoding_confidence = self.detect_encoding(raw_bytes)
        
        # 解码文本
        try:
            text = raw_bytes.decode(encoding, errors='replace')
        except Exception as e:
            print(f"解码失败: {e}")
            # 使用UTF-8作为回退
            text = raw_bytes.decode('utf-8', errors='replace')
            encoding = 'utf-8'
            encoding_confidence = 0.1
        
        # 检测语言
        language_info = self.detect_language(text)
        
        # 返回处理结果
        return {
            'text': text,
            'encoding': {
                'detected': encoding,
                'confidence': encoding_confidence,
                'used': encoding
            },
            'language': language_info,
            'processed_length': len(text),
            'has_encoding_errors': '' in text  # 检查替换字符
        }

# 使用示例
processor = BasicMultilingualProcessor()

# 示例文本（混合中英文）
sample_bytes = """
这是中文内容。This is English content.
Artificial intelligence is transforming the world.
人工智能正在改变世界。
""".encode('utf-8')

result = processor.process_document(sample_bytes)

print("多语言文档处理结果：")
print(f"检测编码: {result['encoding']['detected']} (置信度: {result['encoding']['confidence']:.2%})")
print(f"主要语言: {result['language']['primary_language']}")
print(f"混合语言: {result['language']['is_mixed_language']}")
print(f"处理长度: {result['processed_length']} 字符")
print(f"是否有编码错误: {result['has_encoding_errors']}")
```

### 语言特定处理配置示例

```python
from typing import Dict, Any

class LanguageSpecificConfig:
    """语言特定配置管理器"""
    
    def __init__(self):
        # 语言特定配置
        self.language_configs = {
            'zh': {  # 中文配置
                'name': 'Chinese',
                'splitter_separators': ["\n\n", "\n", "。", "！", "？", "；", "，", " ", ""],
                'chunk_size': 300,
                'chunk_overlap': 60,
                'special_handling': ['remove_extra_spaces', 'normalize_punctuation']
            },
            'en': {  # 英文配置
                'name': 'English',
                'splitter_separators': ["\n\n", "\n", ".", "!", "?", ";", ",", " ", ""],
                'chunk_size': 500,
                'chunk_overlap': 100,
                'special_handling': ['normalize_whitespace', 'fix_apostrophes']
            },
            'ja': {  # 日文配置
                'name': 'Japanese',
                'splitter_separators': ["\n\n", "\n", "。", "！", "？", "；", "、", " ", ""],
                'chunk_size': 400,
                'chunk_overlap': 80,
                'special_handling': ['normalize_punctuation', 'merge_spaces']
            },
            'ko': {  # 韩文配置
                'name': 'Korean',
                'splitter_separators': ["\n\n", "\n", ".", "!", "?", ";", ",", " ", ""],
                'chunk_size': 400,
                'chunk_overlap': 80,
                'special_handling': ['normalize_spaces', 'clean_control_chars']
            },
            'default': {  # 默认配置
                'name': 'Default',
                'splitter_separators': ["\n\n", "\n", ".", "!", "?", ";", ",", " ", ""],
                'chunk_size': 400,
                'chunk_overlap': 80,
                'special_handling': ['basic_cleaning']
            }
        }
    
    def get_config(self, language_code: str) -> Dict[str, Any]:
        """获取语言特定配置"""
        # 标准化语言代码
        if language_code.startswith('zh'):
            lang_key = 'zh'
        elif language_code == 'en':
            lang_key = 'en'
        elif language_code == 'ja':
            lang_key = 'ja'
        elif language_code == 'ko':
            lang_key = 'ko'
        else:
            lang_key = 'default'
        
        return self.language_configs.get(lang_key, self.language_configs['default'])
    
    def create_language_aware_splitter(self, language_code: str):
        """创建语言感知的分割器"""
        from langchain_text_splitters import RecursiveCharacterTextSplitter
        
        config = self.get_config(language_code)
        
        return RecursiveCharacterTextSplitter(
            chunk_size=config['chunk_size'],
            chunk_overlap=config['chunk_overlap'],
            separators=config['splitter_separators']
        )
    
    def apply_language_specific_processing(self, text: str, language_code: str) -> str:
        """应用语言特定处理"""
        config = self.get_config(language_code)
        processed_text = text
        
        # 应用处理规则
        for handler in config['special_handling']:
            processed_text = self._apply_handler(processed_text, handler)
        
        return processed_text
    
    def _apply_handler(self, text: str, handler: str) -> str:
        """应用特定处理程序"""
        import re
        
        if handler == 'remove_extra_spaces':
            # 移除多余空格（中文）
            text = re.sub(r'([，。！？；："']) +', r'\1', text)
            text = re.sub(r' +([，。！？；："'])', r'\1', text)
        elif handler == 'normalize_punctuation':
            # 标准化标点（中英文）
            text = text.replace('，', ',').replace('。', '.').replace('！', '!').replace('？', '?').replace('；', ';').replace('：', ':')
        elif handler == 'normalize_whitespace':
            # 标准化空白字符
            text = re.sub(r'\s+', ' ', text).strip()
        elif handler == 'merge_spaces':
            # 合并连续空格
            text = re.sub(r' +', ' ', text)
        elif handler == 'clean_control_chars':
            # 清理控制字符
            text = re.sub(r'[\x00-\x08\x0B\x0C\x0E-\x1F\x7F]', '', text)
        elif handler == 'basic_cleaning':
            # 基础清理
            text = re.sub(r'\s+', ' ', text).strip()
            text = text.replace('\r\n', '\n').replace('\r', '\n')
        
        return text

# 使用示例
config_manager = LanguageSpecificConfig()

# 测试不同语言的配置
test_languages = ['zh-CN', 'en', 'ja', 'ko', 'fr']

for lang in test_languages:
    config = config_manager.get_config(lang)
    print(f"\n语言: {lang} ({config['name']})")
    print(f"块大小: {config['chunk_size']}")
    print(f"重叠大小: {config['chunk_overlap']}")
    print(f"分隔符数量: {len(config['splitter_separators'])}")
    print(f"特殊处理: {config['special_handling']}")

# 创建语言感知分割器
chinese_splitter = config_manager.create_language_aware_splitter('zh-CN')
english_splitter = config_manager.create_language_aware_splitter('en')

print(f"\n中文分割器配置: 块大小={chinese_splitter._chunk_size}, 重叠大小={chinese_splitter._chunk_overlap}")
print(f"英文分割器配置: 块大小={english_splitter._chunk_size}, 重叠大小={english_splitter._chunk_overlap}")
```
