# 12.1.2 并行执行与同步控制

## 概念讲解

并行执行与同步控制是复杂业务工作流编排的核心能力，它允许系统同时处理多个任务，显著提升处理效率。LangGraph v1.2.22通过先进的图计算引擎提供了强大的并行执行能力，同时确保数据一致性和正确性。

### 并行执行的类型

LangGraph支持多种并行执行模式：

1. **数据并行**：同一任务处理不同数据分片
2. **任务并行**：不同任务同时执行
3. **流水线并行**：任务形成流水线，上游输出作为下游输入
4. **混合并行**：组合多种并行模式

### 同步控制机制

并行执行需要精细的同步控制：

1. **屏障同步**：所有并行任务完成后才继续
2. **条件同步**：基于特定条件决定何时继续
3. **数据依赖同步**：基于数据可用性控制执行
4. **资源同步**：控制共享资源访问

### LangGraph的并行实现

LangGraph v1.2.22通过以下机制实现并行执行：

1. **Send API**：支持动态创建并行节点实例
2. **状态归约器**：并行节点结果自动合并
3. **条件边**：实现复杂的同步逻辑
4. **检查点**：支持并行执行的状态持久化

## 核心要点

- **并行提升效率**：合理利用多核CPU和分布式资源
- **同步保障正确性**：防止数据竞争和状态不一致
- **LangGraph自动管理**：开发者声明并行结构，运行时自动优化
- **错误处理复杂化**：需要专门的并行错误处理机制
- **资源管理重要**：并行任务可能竞争有限资源

## 简单示例

### 基础并行执行示例

```python
# 示例：并行数据处理工作流
from typing import TypedDict, List
from typing_extensions import Annotated
from langgraph.graph import StateGraph, START, END
from langgraph.types import Send
import time
import operator

class ParallelProcessingState(TypedDict):
    """并行处理状态"""
    input_data: List[str]
    processed_results: Annotated[List[str], operator.add]
    completion_status: dict
    execution_time: float

def data_splitter(state: ParallelProcessingState):
    """数据分割节点"""
    data = state["input_data"]
    
    # 将数据分成3个批次并行处理
    batch_size = len(data) // 3
    batches = [
        data[i:i + batch_size] 
        for i in range(0, len(data), batch_size)
    ]
    
    # 创建并行处理任务
    sends = []
    for i, batch in enumerate(batches):
        sends.append(Send(
            "process_batch",
            {"batch_data": batch, "batch_id": i}
        ))
    
    return sends

def process_batch(state: dict):
    """批量处理节点（并行执行）"""
    batch_data = state["batch_data"]
    batch_id = state["batch_id"]
    
    start_time = time.time()
    
    # 模拟处理时间
    time.sleep(0.2)
    
    # 处理数据
    results = [item.upper() for item in batch_data]
    
    processing_time = time.time() - start_time
    
    return {
        "processed_results": results,
        "completion_status": {f"batch_{batch_id}": "completed"},
        "execution_time": processing_time
    }

def aggregate_results(state: ParallelProcessingState):
    """结果聚合节点（同步点）"""
    results = state["processed_results"]
    completion = state["completion_status"]
    
    print(f"所有批次完成状态: {completion}")
    print(f"总处理结果数: {len(results)}")
    print(f"前3个结果: {results[:3] if results else '无结果'}")
    
    total_time = sum(
        t for t in [state.get("execution_time", 0)] 
        if isinstance(t, (int, float))
    )
    
    return {"execution_time": total_time}

# 构建并行工作流
parallel_workflow = StateGraph(ParallelProcessingState)

parallel_workflow.add_node("splitter", data_splitter)
parallel_workflow.add_node("process_batch", process_batch)
parallel_workflow.add_node("aggregator", aggregate_results)

# 工作流结构：分割 → 并行处理 → 聚合
parallel_workflow.add_edge(START, "splitter")
parallel_workflow.add_conditional_edges(
    "splitter",
    lambda state: "process_batch",
    {"process_batch": "process_batch"}
)
parallel_workflow.add_edge("process_batch", "aggregator")
parallel_workflow.add_edge("aggregator", END)

# 测试并行执行
print("=== 并行执行测试 ===")
test_data = [f"item_{i}" for i in range(100)]
initial_state = {
    "input_data": test_data,
    "processed_results": [],
    "completion_status": {},
    "execution_time": 0.0
}

graph = parallel_workflow.compile()

# 测量执行时间
import time
start = time.time()
result = graph.invoke(initial_state)
total_time = time.time() - start

print(f"总执行时间: {total_time:.3f}秒")
print(f"工作流统计时间: {result['execution_time']:.3f}秒")
print(f"加速比: {result['execution_time']/total_time:.2f}x")
```

### 屏障同步示例

```python
# 示例：屏障同步控制
from typing import Dict, Any
from concurrent.futures import ThreadPoolExecutor

class BarrierSyncState(TypedDict):
    """屏障同步状态"""
    tasks: List[Dict[str, Any]]
    completed_tasks: Dict[str, Any]
    barrier_ready: bool
    final_result: str

def execute_parallel_tasks(state: BarrierSyncState):
    """执行并行任务节点"""
    tasks = state["tasks"]
    completed = {}
    
    # 使用线程池并行执行
    with ThreadPoolExecutor(max_workers=3) as executor:
        futures = {}
        for task in tasks:
            task_id = task["id"]
            future = executor.submit(process_single_task, task)
            futures[future] = task_id
        
        # 收集结果
        for future in futures:
            task_id = futures[future]
            try:
                result = future.result(timeout=5.0)
                completed[task_id] = result
            except Exception as e:
                completed[task_id] = {"error": str(e)}
    
    return {
        "completed_tasks": completed,
        "barrier_ready": len(completed) == len(tasks)
    }

def process_single_task(task: Dict[str, Any]) -> Dict[str, Any]:
    """处理单个任务"""
    task_id = task["id"]
    data = task["data"]
    
    # 模拟处理时间
    import time
    time.sleep(0.1 * (task_id % 3 + 1))
    
    # 处理逻辑
    result = {
        "task_id": task_id,
        "processed": data.upper() if isinstance(data, str) else data,
        "status": "success"
    }
    
    print(f"任务 {task_id} 完成")
    return result

def barrier_check(state: BarrierSyncState) -> str:
    """屏障检查条件函数"""
    if state["barrier_ready"]:
        return "process_after_barrier"
    else:
        return "wait_at_barrier"

def wait_at_barrier(state: BarrierSyncState):
    """屏障等待节点"""
    print(f"等待其他任务完成... 已完成 {len(state['completed_tasks'])}/{len(state['tasks'])}")
    return {}

def process_after_barrier(state: BarrierSyncState):
    """屏障后处理节点"""
    completed = state["completed_tasks"]
    
    # 检查所有任务是否成功
    all_success = all(
        task.get("status") == "success"
        for task in completed.values()
        if isinstance(task, dict)
    )
    
    if all_success:
        # 聚合结果
        results = [
            task.get("processed", "") 
            for task in completed.values() 
            if isinstance(task, dict)
        ]
        final_result = " | ".join(str(r) for r in results)
        
        return {"final_result": final_result}
    else:
        # 处理失败
        errors = [
            f"任务{task_id}: {task.get('error', '未知错误')}"
            for task_id, task in completed.items()
            if isinstance(task, dict) and "error" in task
        ]
        
        return {"final_result": f"错误: {', '.join(errors)}"}

# 构建屏障同步工作流
barrier_workflow = StateGraph(BarrierSyncState)

barrier_workflow.add_node("execute_tasks", execute_parallel_tasks)
barrier_workflow.add_node("wait", wait_at_barrier)
barrier_workflow.add_node("after_barrier", process_after_barrier)

barrier_workflow.add_edge(START, "execute_tasks")
barrier_workflow.add_conditional_edges(
    "execute_tasks",
    barrier_check,
    {
        "wait_at_barrier": "wait",
        "process_after_barrier": "after_barrier"
    }
)
barrier_workflow.add_edge("wait", "execute_tasks")  # 循环检查
barrier_workflow.add_edge("after_barrier", END)

print("\n=== 屏障同步测试 ===")
test_tasks = [
    {"id": 1, "data": "task_one"},
    {"id": 2, "data": "task_two"},
    {"id": 3, "data": "task_three"},
    {"id": 4, "data": "task_four"},
]

initial_barrier_state = {
    "tasks": test_tasks,
    "completed_tasks": {},
    "barrier_ready": False,
    "final_result": ""
}

result = barrier_workflow.compile().invoke(initial_barrier_state)
print(f"屏障同步完成，最终结果: {result['final_result']}")
```

### 动态并行度调整

```python
# 动态并行度调整
class AdaptiveParallelism:
    """自适应并行控制器"""
    
    def __init__(self, min_workers: int = 1, max_workers: int = 10):
        self.min_workers = min_workers
        self.max_workers = max_workers
        self.history = []
    
    def calculate_optimal_workers(self, task_count: int, 
                                  system_load: float) -> int:
        """计算最优工作线程数"""
        # 基于任务数量和系统负载的动态调整
        if system_load > 0.8:  # 高负载
            workers = max(self.min_workers, task_count // 4)
        elif system_load > 0.5:  # 中负载
            workers = max(self.min_workers, task_count // 2)
        else:  # 低负载
            workers = min(self.max_workers, task_count)
        
        # 平滑调整
        if self.history:
            last_workers = self.history[-1]
            # 避免剧烈变化
            if abs(workers - last_workers) > 2:
                workers = last_workers + (1 if workers > last_workers else -1)
        
        self.history.append(workers)
        return workers
    
    def adjust_batch_size(self, data_size: int, workers: int) -> List[int]:
        """调整批次大小"""
        base_size = data_size // workers
        remainder = data_size % workers
        
        # 分配批次大小
        batch_sizes = [base_size] * workers
        for i in range(remainder):
            batch_sizes[i] += 1
        
        return batch_sizes
```

### 复杂同步模式

```python
# 复杂同步模式：生产者-消费者
from queue import Queue
from threading import Thread, Event

class ProducerConsumerSync:
    """生产者-消费者同步模式"""
    
    def __init__(self, buffer_size: int = 10):
        self.queue = Queue(maxsize=buffer_size)
        self.producer_done = Event()
        self.consumer_done = Event()
    
    def producer_task(self, data_items: List[Any]):
        """生产者任务"""
        for item in data_items:
            # 等待队列有空间
            self.queue.put(item)
            print(f"生产者添加: {item}")
        
        self.producer_done.set()
    
    def consumer_task(self, process_func):
        """消费者任务"""
        processed = []
        
        while not self.producer_done.is_set() or not self.queue.empty():
            try:
                item = self.queue.get(timeout=0.1)
                result = process_func(item)
                processed.append(result)
                self.queue.task_done()
                print(f"消费者处理: {item} -> {result}")
            except:
                continue
        
        self.consumer_done.set()
        return processed
    
    def execute_pipeline(self, data: List[Any], process_func):
        """执行生产者-消费者管道"""
        import threading
        
        # 启动生产者
        producer = threading.Thread(
            target=self.producer_task,
            args=(data,)
        )
        
        # 启动消费者
        consumer = threading.Thread(
            target=self.consumer_task,
            args=(process_func,)
        )
        
        producer.start()
        consumer.start()
        
        # 等待完成
        producer.join()
        consumer.join()
        
        # 获取结果
        return self.consumer_task(process_func)
```

## 常见问题

### Q1: 并行执行会引入哪些问题？
**A**: 主要问题：1) 数据竞争；2) 死锁；3) 资源争用；4) 错误传播复杂化；5) 调试困难。需要设计良好的同步机制和错误处理策略。

### Q2: 如何选择并行度？
**A**: 考虑因素：1) CPU核心数；2) 任务特性（CPU密集型 vs I/O密集型）；3) 系统负载；4) 数据依赖关系。建议动态调整并行度。

### Q3: LangGraph如何处理并行错误？
**A**: LangGraph提供：1) 节点级错误处理；2) 并行任务隔离；3) 优雅降级；4) 检查点恢复。可以设计专门的错误处理节点收集和处理并行错误。

### Q4: 并行执行如何影响状态一致性？
**A**: LangGraph通过以下机制保障一致性：1) 状态归约器定义合并逻辑；2) 事务性状态更新；3) 检查点确保原子性；4) 条件同步控制执行顺序。

### Q5: 如何测试并行工作流？
**A**: 测试策略：1) 单元测试每个节点；2) 集成测试并行路径；3) 压力测试并发场景；4) 使用模拟数据测试边界条件；5) 可视化调试执行路径。
