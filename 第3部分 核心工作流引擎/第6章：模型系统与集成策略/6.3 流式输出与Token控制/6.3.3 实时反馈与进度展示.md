## 传统实时反馈系统的复杂性

在LangChain等框架出现之前，实现实时反馈和进度展示需要复杂的多线程、事件驱动架构和手动状态管理。

### 手动实现流式输出和进度反馈

```python
# 传统手动实现流式输出和进度反馈
import threading
import time
import json
from queue import Queue
from typing import Dict, Any, List, Optional, Callable
from datetime import datetime, timedelta
import asyncio
import sys

class ManualStreamingHandler:
    """手动流式处理器"""
    
    def __init__(self, callback: Optional[Callable] = None):
        self.callback = callback
        self.token_queue = Queue()
        self.is_streaming = False
        self.progress = 0.0  # 0-100
        self.start_time = None
        self.end_time = None
        self.total_tokens_expected = 0
        self.tokens_received = 0
        
        # 性能指标
        self.metrics = {
            "tokens_per_second": 0,
            "avg_token_latency": 0,
            "total_time": 0
        }
        
        # 状态历史
        self.history = []
        
    def start_streaming(self, total_tokens_expected: int = 0):
        """开始流式处理"""
        self.is_streaming = True
        self.start_time = datetime.now()
        self.total_tokens_expected = total_tokens_expected
        self.tokens_received = 0
        self.progress = 0.0
        
        print(f"[{self.start_time.strftime('%H:%M:%S')}] 开始流式生成...")
        
        # 启动监控线程
        self._start_monitoring()
    
    def _start_monitoring(self):
        """启动监控线程"""
        def monitor():
            last_update = datetime.now()
            last_tokens = 0
            
            while self.is_streaming:
                time.sleep(0.1)  # 100ms更新频率
                
                current_time = datetime.now()
                time_elapsed = (current_time - last_update).total_seconds()
                
                if time_elapsed >= 1.0:  # 每秒更新一次指标
                    if self.start_time:
                        total_time = (current_time - self.start_time).total_seconds()
                        
                        # 计算tokens/s
                        if total_time > 0:
                            current_tokens = self.tokens_received
                            new_tokens = current_tokens - last_tokens
                            self.metrics["tokens_per_second"] = new_tokens / time_elapsed if time_elapsed > 0 else 0
                            last_tokens = current_tokens
                        
                        # 更新进度
                        if self.total_tokens_expected > 0:
                            self.progress = (self.tokens_received / self.total_tokens_expected) * 100
                        else:
                            # 基于时间估计
                            if self.progress < 95:  # 防止过早完成
                                self.progress += 0.5
                    
                    last_update = current_time
                    
                    # 记录状态
                    self.history.append({
                        "timestamp": current_time.isoformat(),
                        "progress": self.progress,
                        "tokens_received": self.tokens_received,
                        "tokens_per_second": self.metrics["tokens_per_second"]
                    })
        
        thread = threading.Thread(target=monitor, daemon=True)
        thread.start()
    
    def on_token_received(self, token: str):
        """接收到token时调用"""
        if not self.is_streaming:
            return
        
        self.token_queue.put(token)
        self.tokens_received += 1
        
        # 调用回调
        if self.callback:
            self.callback(token, {
                "progress": self.progress,
                "tokens_received": self.tokens_received,
                "tokens_per_second": self.metrics["tokens_per_second"]
            })
        
        # 直接输出（打字机效果）
        print(token, end='', flush=True)
    
    def end_streaming(self):
        """结束流式处理"""
        if not self.is_streaming:
            return
        
        self.is_streaming = False
        self.end_time = datetime.now()
        
        if self.start_time:
            total_time = (self.end_time - self.start_time).total_seconds()
            self.metrics["total_time"] = total_time
            
            if total_time > 0:
                self.metrics["avg_token_latency"] = total_time / self.tokens_received if self.tokens_received > 0 else 0
        
        print(f"\n[{self.end_time.strftime('%H:%M:%S')}] 流式生成完成")
        print(f"总token数: {self.tokens_received}")
        print(f"总时间: {self.metrics['total_time']:.2f}秒")
        print(f"平均速度: {self.metrics['tokens_per_second']:.1f} tokens/秒")
        
        return self.metrics
    
    def get_progress_bar(self, width: int = 40) -> str:
        """获取进度条字符串"""
        if not self.is_streaming:
            return ""
        
        filled = int(width * self.progress / 100)
        empty = width - filled
        
        bar = "█" * filled + "░" * empty
        
        eta = "未知"
        if self.metrics["tokens_per_second"] > 0 and self.total_tokens_expected > 0:
            remaining_tokens = self.total_tokens_expected - self.tokens_received
            remaining_time = remaining_tokens / self.metrics["tokens_per_second"]
            eta = f"{remaining_time:.1f}秒"
        
        return f"[{bar}] {self.progress:.1f}% | {self.tokens_received}/{self.total_tokens_expected} tokens | {self.metrics['tokens_per_second']:.1f} tokens/s | ETA: {eta}"
    
    def get_detailed_metrics(self) -> Dict[str, Any]:
        """获取详细指标"""
        return {
            "streaming_status": {
                "is_streaming": self.is_streaming,
                "start_time": self.start_time.isoformat() if self.start_time else None,
                "end_time": self.end_time.isoformat() if self.end_time else None,
                "duration": self.metrics["total_time"]
            },
            "progress_metrics": {
                "progress_percentage": self.progress,
                "tokens_received": self.tokens_received,
                "total_tokens_expected": self.total_tokens_expected,
                "remaining_tokens": max(0, self.total_tokens_expected - self.tokens_received) if self.total_tokens_expected > 0 else None
            },
            "performance_metrics": {
                "tokens_per_second": self.metrics["tokens_per_second"],
                "avg_token_latency": self.metrics["avg_token_latency"],
                "total_time": self.metrics["total_time"]
            },
            "history_summary": {
                "total_updates": len(self.history),
                "first_update": self.history[0]["timestamp"] if self.history else None,
                "last_update": self.history[-1]["timestamp"] if self.history else None
            }
        }

class ManualProgressSystem:
    """手动进度展示系统"""
    
    def __init__(self):
        self.tasks = {}
        self.task_counter = 0
        self.update_callbacks = []
        
    def create_task(self, task_name: str, total_steps: int = 100) -> str:
        """创建任务"""
        task_id = f"task_{self.task_counter}"
        self.task_counter += 1
        
        self.tasks[task_id] = {
            "task_name": task_name,
            "total_steps": total_steps,
            "completed_steps": 0,
            "start_time": datetime.now(),
            "status": "running",  # running, paused, completed, failed
            "current_step_name": "初始化",
            "progress_percentage": 0.0,
            "estimated_completion_time": None,
            "metadata": {}
        }
        
        # 通知回调
        self._notify_update(task_id)
        
        return task_id
    
    def update_task_progress(self, 
                           task_id: str, 
                           completed_steps: int = None,
                           step_increment: int = 1,
                           step_name: str = None,
                           metadata: Dict[str, Any] = None):
        """更新任务进度"""
        if task_id not in self.tasks:
            raise ValueError(f"任务不存在: {task_id}")
        
        task = self.tasks[task_id]
        
        # 更新完成步骤数
        if completed_steps is not None:
            task["completed_steps"] = completed_steps
        else:
            task["completed_steps"] += step_increment
        
        # 限制范围
        task["completed_steps"] = min(task["completed_steps"], task["total_steps"])
        
        # 更新进度百分比
        if task["total_steps"] > 0:
            task["progress_percentage"] = (task["completed_steps"] / task["total_steps"]) * 100
        
        # 更新步骤名称
        if step_name:
            task["current_step_name"] = step_name
        
        # 更新元数据
        if metadata:
            task["metadata"].update(metadata)
        
        # 估算完成时间
        if task["completed_steps"] > 0:
            elapsed_time = (datetime.now() - task["start_time"]).total_seconds()
            time_per_step = elapsed_time / task["completed_steps"]
            remaining_steps = task["total_steps"] - task["completed_steps"]
            remaining_time = time_per_step * remaining_steps
            
            task["estimated_completion_time"] = (
                datetime.now() + timedelta(seconds=remaining_time)
            )
        
        # 检查是否完成
        if task["completed_steps"] >= task["total_steps"]:
            task["status"] = "completed"
            task["end_time"] = datetime.now()
        
        # 通知回调
        self._notify_update(task_id)
    
    def pause_task(self, task_id: str):
        """暂停任务"""
        if task_id in self.tasks:
            self.tasks[task_id]["status"] = "paused"
            self._notify_update(task_id)
    
    def resume_task(self, task_id: str):
        """恢复任务"""
        if task_id in self.tasks:
            self.tasks[task_id]["status"] = "running"
            self._notify_update(task_id)
    
    def fail_task(self, task_id: str, error_message: str):
        """标记任务失败"""
        if task_id in self.tasks:
            self.tasks[task_id]["status"] = "failed"
            self.tasks[task_id]["error_message"] = error_message
            self.tasks[task_id]["end_time"] = datetime.now()
            self._notify_update(task_id)
    
    def complete_task(self, task_id: str):
        """标记任务完成"""
        if task_id in self.tasks:
            self.tasks[task_id]["status"] = "completed"
            self.tasks[task_id]["completed_steps"] = self.tasks[task_id]["total_steps"]
            self.tasks[task_id]["progress_percentage"] = 100.0
            self.tasks[task_id]["end_time"] = datetime.now()
            self._notify_update(task_id)
    
    def _notify_update(self, task_id: str):
        """通知更新"""
        for callback in self.update_callbacks:
            try:
                callback(task_id, self.tasks[task_id])
            except Exception as e:
                print(f"回调通知失败: {e}")
    
    def register_update_callback(self, callback: Callable):
        """注册更新回调"""
        self.update_callbacks.append(callback)
    
    def get_task_progress_bar(self, task_id: str, width: int = 30) -> str:
        """获取任务进度条"""
        if task_id not in self.tasks:
            return f"任务 {task_id} 不存在"
        
        task = self.tasks[task_id]
        
        filled = int(width * task["progress_percentage"] / 100)
        empty = width - filled
        
        bar = "█" * filled + "░" * empty
        
        # 状态图标
        status_icons = {
            "running": "▶",
            "paused": "⏸",
            "completed": "✓",
            "failed": "✗"
        }
        
        icon = status_icons.get(task["status"], "?")
        
        # ETA显示
        eta_str = ""
        if task["estimated_completion_time"] and task["status"] == "running":
            remaining_seconds = (task["estimated_completion_time"] - datetime.now()).total_seconds()
            if remaining_seconds > 0:
                if remaining_seconds < 60:
                    eta_str = f" {remaining_seconds:.0f}s"
                else:
                    eta_str = f" {remaining_seconds/60:.1f}m"
        
        return f"{icon} [{bar}] {task['progress_percentage']:.1f}% | {task['completed_steps']}/{task['total_steps']}{eta_str}"
    
    def get_dashboard_view(self) -> str:
        """获取仪表板视图"""
        lines = []
        lines.append("=" * 80)
        lines.append("任务进度仪表板")
        lines.append("=" * 80)
        
        for task_id, task in self.tasks.items():
            progress_bar = self.get_task_progress_bar(task_id, 40)
            
            lines.append(f"{task_id}: {task['task_name']}")
            lines.append(f"  {progress_bar}")
            lines.append(f"  状态: {task['status']} | 当前步骤: {task['current_step_name']}")
            
            if "start_time" in task:
                elapsed = (datetime.now() - task["start_time"]).total_seconds()
                lines.append(f"  已运行: {elapsed:.1f}秒")
            
            if task["estimated_completion_time"]:
                eta = (task["estimated_completion_time"] - datetime.now()).total_seconds()
                if eta > 0:
                    lines.append(f"  预计剩余: {eta:.1f}秒")
            
            lines.append("")
        
        return "\n".join(lines)
```

### 手动实现用户中断和控制

对于交互式应用，支持用户中断和实时控制更加复杂。

```python
# 手动实现用户中断和控制
import signal
import threading
from typing import Optional, Callable, Dict, Any
import sys
import select

class UserInterruptHandler:
    """用户中断处理器（传统方式）"""
    
    def __init__(self):
        self.should_stop = False
        self.pause_event = threading.Event()
        self.pause_event.set()  # 初始为运行状态
        
        self.adjustment_queue = Queue()
        self.adjustment_lock = threading.Lock()
        
        self.current_parameters = {}
        
        # 注册信号处理
        self._setup_signal_handlers()
        
    def _setup_signal_handlers(self):
        """设置信号处理器"""
        original_sigint = signal.getsignal(signal.SIGINT)
        
        def sigint_handler(signum, frame):
            print("\n接收到中断信号 (Ctrl+C)")
            response = input("是否中断当前生成？(y/n): ").strip().lower()
            
            if response == 'y':
                print("正在中断生成...")
                self.should_stop = True
                return  # 不传递信号，我们自己处理
            else:
                print("继续生成...")
                return  # 继续
            
            # 调用原始处理器
            if original_sigint != signal.SIG_IGN:
                original_sigint(signum, frame)
        
        signal.signal(signal.SIGINT, sigint_handler)
    
    def check_interrupt(self) -> bool:
        """检查是否应该中断"""
        return self.should_stop
    
    def pause(self):
        """暂停生成"""
        self.pause_event.clear()
        print("生成已暂停，按回车继续...")
    
    def resume(self):
        """恢复生成"""
        self.pause_event.set()
        print("生成已恢复")
    
    def wait_if_paused(self):
        """如果暂停则等待"""
        self.pause_event.wait()
    
    def request_adjustment(self, parameter: str, value: Any):
        """请求参数调整"""
        with self.adjustment_lock:
            self.adjustment_queue.put((parameter, value))
    
    def get_adjustments(self) -> Dict[str, Any]:
        """获取所有待处理的调整"""
        adjustments = {}
        
        with self.adjustment_lock:
            while not self.adjustment_queue.empty():
                param, value = self.adjustment_queue.get()
                adjustments[param] = value
        
        return adjustments
    
    def apply_adjustments(self, current_parameters: Dict[str, Any]) -> Dict[str, Any]:
        """应用调整到当前参数"""
        adjustments = self.get_adjustments()
        
        if adjustments:
            print(f"应用调整: {adjustments}")
            current_parameters.update(adjustments)
        
        return current_parameters
    
    def interactive_control_loop(self, update_callback: Callable):
        """交互式控制循环"""
        def control_thread():
            while not self.should_stop:
                try:
                    # 非阻塞读取输入
                    if select.select([sys.stdin], [], [], 0.1)[0]:
                        user_input = sys.stdin.readline().strip()
                        
                        if user_input:
                            self._process_user_command(user_input, update_callback)
                except Exception as e:
                    print(f"控制循环错误: {e}")
                
                # 检查暂停状态
                if not self.pause_event.is_set():
                    # 在暂停状态下等待用户输入
                    print("\n控制选项: (r)恢复, (a)调整参数, (s)停止")
                    command = input("命令: ").strip().lower()
                    
                    if command == 'r':
                        self.resume()
                    elif command == 'a':
                        self._adjust_parameters_interactive()
                    elif command == 's':
                        self.should_stop = True
                        print("停止生成")
        
        thread = threading.Thread(target=control_thread, daemon=True)
        thread.start()
    
    def _process_user_command(self, command: str, update_callback: Callable):

        elif action == "status":

                "is_paused": not self.pause_event.is_set(),
                "adjustment_queue_size": self.adjustment_queue.qsize()
            }
            print(f"状态: {status_info}")
            if update_callback:
                update_callback("status_update", status_info)
        elif action == "help":

            print("  resume - 恢复生成")
```

```python
from langchain_core.callbacks import BaseCallbackHandler, CallbackManager
from langchain_core.callbacks.streaming_stdout import StreamingStdOutCallbackHandler
from langchain_core.callbacks.file import FileCallbackHandler
from langchain_openai import ChatOpenAI
import time

from typing import Dict, Any, List
from dataclasses import dataclass, field
from datetime import datetime
import threading

@dataclass
class ProgressState:
    """进度状态"""
    task_id: str
    task_name: str
    total_steps: int
    completed_steps: int = 0
    start_time: datetime = field(default_factory=datetime.now)
    end_time: datetime = None
    status: str = "pending"  # pending, running, paused, completed, failed
    current_step_name: str = "初始化"
    metadata: Dict[str, Any] = field(default_factory=dict)
    
    @property
    def progress_percentage(self) -> float:
        """进度百分比"""
        if self.total_steps == 0:
            return 0.0
        return (self.completed_steps / self.total_steps) * 100
    
    @property
    def elapsed_time(self) -> float:
        """已用时间（秒）"""
        if self.end_time:
            return (self.end_time - self.start_time).total_seconds()
        return (datetime.now() - self.start_time).total_seconds()
    
    @property
    def estimated_completion_time(self) -> datetime:
        """预计完成时间"""
        if self.completed_steps == 0 or self.status != "running":
            return None
        
        time_per_step = self.elapsed_time / self.completed_steps
        remaining_steps = self.total_steps - self.completed_steps
        remaining_time = time_per_step * remaining_steps
        
        return datetime.now() + timedelta(seconds=remaining_time)

class ProgressManager:
    """进度管理器"""
    
    def __init__(self):
        self.tasks: Dict[str, ProgressState] = {}
        self.subscribers = []
        self.lock = threading.Lock()
        
    def create_task(self, task_name: str, total_steps: int = 100, metadata: Dict = None) -> str:
        """创建任务"""
        task_id = f"task_{len(self.tasks)}_{int(time.time())}"
        
        task = ProgressState(
            task_id=task_id,
            task_name=task_name,
            total_steps=total_steps,
            metadata=metadata or {}
        )
        
        with self.lock:
            self.tasks[task_id] = task
        
        self._notify_subscribers("task_created", task)
        
        return task_id
    
    def update_task(self, task_id: str, **kwargs):
        """更新任务"""
        with self.lock:
            if task_id not in self.tasks:
                raise ValueError(f"任务不存在: {task_id}")
            
            task = self.tasks[task_id]
            
            # 更新字段
            for key, value in kwargs.items():
                if hasattr(task, key):
                    setattr(task, key, value)
            
            # 如果状态改变为完成或失败，设置结束时间
            if task.status in ["completed", "failed"] and task.end_time is None:
                task.end_time = datetime.now()
        
        self._notify_subscribers("task_updated", task)
    
    def increment_progress(self, task_id: str, steps: int = 1, step_name: str = None):
        """增加进度"""
        with self.lock:
            if task_id not in self.tasks:
                raise ValueError(f"任务不存在: {task_id}")
            
            task = self.tasks[task_id]
            task.completed_steps = min(task.completed_steps + steps, task.total_steps)
            
            if step_name:
                task.current_step_name = step_name
            
            # 如果完成所有步骤，标记为完成
            if task.completed_steps >= task.total_steps and task.status == "running":
                task.status = "completed"
                task.end_time = datetime.now()
        
        self._notify_subscribers("progress_updated", task)
    
    def subscribe(self, callback):
        """订阅更新"""
        self.subscribers.append(callback)
    
    def _notify_subscribers(self, event_type: str, task: ProgressState):
        """通知订阅者"""
        for subscriber in self.subscribers:
            try:
                subscriber(event_type, task)
            except Exception as e:
                print(f"订阅者通知失败: {e}")
    
    def get_task(self, task_id: str) -> ProgressState:
        """获取任务"""
        with self.lock:
            return self.tasks.get(task_id)
    
    def get_all_tasks(self) -> List[ProgressState]:
        """获取所有任务"""
        with self.lock:
            return list(self.tasks.values())
    
    def get_progress_bar(self, task_id: str, width: int = 40) -> str:
        """获取进度条"""
        task = self.get_task(task_id)
        if not task:
            return "任务不存在"
        
        filled = int(width * task.progress_percentage / 100)
        empty = width - filled
        
        bar = "█" * filled + "░" * empty
        
        # 状态图标
        icons = {
            "pending": "⏳",
            "running": "▶",
            "paused": "⏸",
            "completed": "✓",
            "failed": "✗"
        }
        
        icon = icons.get(task.status, "?")
        
        # ETA
        eta_str = ""
        if task.status == "running" and task.estimated_completion_time:
            remaining = (task.estimated_completion_time - datetime.now()).total_seconds()
            if remaining > 0:
                if remaining < 60:
                    eta_str = f" {remaining:.0f}s"
                elif remaining < 3600:
                    eta_str = f" {remaining/60:.1f}m"
                else:
                    eta_str = f" {remaining/3600:.1f}h"
        
        return (f"{icon} [{bar}] {task.progress_percentage:.1f}% | "
                f"{task.completed_steps}/{task.total_steps} | "
                f"{task.current_step_name}{eta_str}")

class ProgressAwareCallback(BaseCallbackHandler):
    """进度感知回调处理器"""
    
    def __init__(self, progress_manager: ProgressManager, task_id: str):
        super().__init__()
        self.progress_manager = progress_manager
        self.task_id = task_id
        self.token_count = 0
        
    def on_llm_start(self, serialized: Dict[str, Any], prompts: List[str], **kwargs):
        """LLM开始时"""
        # 估算总token数
        total_words = sum(len(prompt.split()) for prompt in prompts)
        estimated_tokens = int(total_words / 0.75)
        
        # 更新任务状态
        self.progress_manager.update_task(
            self.task_id,
            status="running",
            total_steps=estimated_tokens,
            current_step_name="开始LLM处理"
        )
        
        print(f"[{datetime.now().strftime('%H:%M:%S')}] 开始LLM处理，估算{estimated_tokens}个token")
    
    def on_llm_new_token(self, token: str, **kwargs):
        """新token时"""
        self.token_count += 1
        
        # 每10个token更新一次进度
        if self.token_count % 10 == 0:
            self.progress_manager.increment_progress(
                self.task_id,
                steps=10,
                step_name=f"生成token {self.token_count}"
            )
            
            # 显示进度条
            progress_bar = self.progress_manager.get_progress_bar(self.task_id)
            print(f"\r{progress_bar}", end='', flush=True)
    
    def on_llm_end(self, response, **kwargs):
        """LLM结束时"""
        # 完成剩余进度
        task = self.progress_manager.get_task(self.task_id)
        if task:
            remaining = task.total_steps - task.completed_steps
            if remaining > 0:
                self.progress_manager.increment_progress(
                    self.task_id,
                    steps=remaining,
                    step_name="完成生成"
                )
        
        # 标记为完成
        self.progress_manager.update_task(
            self.task_id,
            status="completed",
            current_step_name="生成完成"
        )
        
        print(f"\n[✓] LLM处理完成，共{self.token_count}个token")
        
        # 获取使用统计（如果可用）
        if hasattr(response, 'usage_metadata'):
            usage = response.usage_metadata
            print(f"  输入token: {usage.get('input_tokens', 'N/A')}")
            print(f"  输出token: {usage.get('output_tokens', 'N/A')}")
            print(f"  总token: {usage.get('total_tokens', 'N/A')}")
    
    def on_llm_error(self, error, **kwargs):
        """LLM错误时"""
        self.progress_manager.update_task(
            self.task_id,
            status="failed",
            current_step_name=f"错误: {str(error)[:50]}..."
        )
        
        print(f"[✗] LLM错误: {error}")

class InterruptibleStreamingCallback(BaseCallbackHandler):
    """可中断的流式回调"""
    
    def __init__(self):
        super().__init__()
        self.should_stop = False
        self.is_paused = False
        self.pause_event = threading.Event()
        self.pause_event.set()
        
        # 参数调整队列
        self.adjustment_queue = []
        self.adjustment_lock = threading.Lock()
        
        # 当前参数
        self.current_params = {}
        
    def check_interrupt(self) -> bool:
        """检查是否应该中断"""
        return self.should_stop
    
    def wait_if_paused(self):
        """如果暂停则等待"""
        self.pause_event.wait()
    
    def pause(self):
        """暂停"""
        self.is_paused = True
        self.pause_event.clear()
    
    def resume(self):
        """恢复"""
        self.is_paused = False
        self.pause_event.set()
    
    def stop(self):
        """停止"""
        self.should_stop = True
        self.resume()  # 确保线程可以退出
    
    def request_adjustment(self, param: str, value: Any):
        """请求参数调整"""
        with self.adjustment_lock:
            self.adjustment_queue.append((param, value))
    
    def get_adjustments(self) -> Dict[str, Any]:
        """获取所有调整"""
        with self.adjustment_lock:
            adjustments = {}
            for param, value in self.adjustment_queue:
                adjustments[param] = value
            self.adjustment_queue.clear()
            return adjustments
    
    def on_llm_new_token(self, token: str, **kwargs):
        """新token时"""
        # 检查中断
        if self.check_interrupt():
            raise KeyboardInterrupt("用户请求中断")
        
        # 检查暂停
        self.wait_if_paused()
        
        # 应用参数调整
        adjustments = self.get_adjustments()
        if adjustments:
            print(f"\n应用调整: {adjustments}")
            self.current_params.update(adjustments)
        
        # 正常处理token
        print(token, end='', flush=True)
```

```python
class EnterpriseTask:

    user_id: str
    project_id: str
    task_type: str

class EnterpriseProgressSystem:
    """企业级进度系统"""
    
    def __init__(self, redis_url: str = "redis://localhost:6379"):
        # 连接Redis用于分布式状态共享
        self.redis_client = redis.Redis.from_url(redis_url)
        
        # WebSocket服务器用于实时更新
        self.websocket_clients = set()
        
        # 任务存储
        self.tasks: Dict[str, EnterpriseTask] = {}
        
        # 指标收集器
        self.metrics_collector = MetricsCollector()

        self.alert_system = AlertSystem()
    
    async def start_websocket_server(self, host: str = "localhost", port: int = 8765):
        """启动WebSocket服务器"""
        
        async def handler(websocket, path):
            self.websocket_clients.add(websocket)
            try:
                async for message in websocket:
                    # 处理客户端消息
                    await self._handle_websocket_message(websocket, message)
            finally:
                self.websocket_clients.remove(websocket)
        
        server = await websockets.serve(handler, host, port)
        print(f"WebSocket服务器启动在 ws://{host}:{port}")
        
        return server
    
    async def _handle_websocket_message(self, websocket, message: str):
        """处理WebSocket消息"""
        try:
            data = json.loads(message)
            message_type = data.get("type")
            
            if message_type == "subscribe":
                # 客户端订阅任务更新
                task_id = data.get("task_id")
                await self._send_task_update(websocket, task_id)
                
            elif message_type == "control":
                # 控制命令
                task_id = data.get("task_id")
                command = data.get("command")
                await self._handle_control_command(task_id, command, data.get("params", {}))
                
        except json.JSONDecodeError:
            print(f"无效的JSON消息: {message}")
        except Exception as e:
            print(f"处理WebSocket消息错误: {e}")
    
    async def _send_task_update(self, websocket, task_id: str):
        """发送任务更新"""
        if task_id in self.tasks:
            task = self.tasks[task_id]
            update_message = {
                "type": "task_update",
                "task": task.to_dict(),
                "timestamp": datetime.now().isoformat()
            }
            
            await websocket.send(json.dumps(update_message))
    
    async def _handle_control_command(self, task_id: str, command: str, params: Dict[str, Any]):
        """处理控制命令"""
        if task_id not in self.tasks:
            return
        
        task = self.tasks[task_id]
        
        if command == "pause":
            task.status = "paused"
            self._broadcast_task_update(task_id)
            
        elif command == "resume":
            task.status = "running"
            self._broadcast_task_update(task_id)
            
        elif command == "cancel":
            task.status = "failed"
            task.metadata["cancelled_by_user"] = True
            self._broadcast_task_update(task_id)
            
        elif command == "adjust":
            # 调整参数
            if "parameters" in params:
                task.metadata["adjusted_parameters"] = params["parameters"]
                self._broadcast_task_update(task_id)
    
    def _broadcast_task_update(self, task_id: str):
        """广播任务更新到所有客户端"""
        if task_id not in self.tasks:
            return
        
        task = self.tasks[task_id]
        update_message = {
            "type": "task_update",
            "task": task.to_dict(),
            "timestamp": datetime.now().isoformat()
        }
        
        message_json = json.dumps(update_message)
        
        # 广播到所有客户端
        for client in list(self.websocket_clients):
            try:
                asyncio.create_task(client.send(message_json))
            except Exception as e:

    def create_task(self, user_id: str, project_id: str, task_type: str, 
                   metadata: Dict[str, Any] = None) -> str:
        """创建任务"""
        task_id = f"{user_id}_{project_id}_{task_type}_{int(time.time())}"
        
        task = EnterpriseTask(
            task_id=task_id,
            user_id=user_id,
            project_id=project_id,
            task_type=task_type,
            status="pending",
            progress=0.0,
            start_time=datetime.now(),
            metadata=metadata or {},
            metrics={}
        )
        
        # 存储到内存和Redis
        self.tasks[task_id] = task
        self._save_task_to_redis(task)
        
        # 广播创建事件
        self._broadcast_task_update(task_id)
        
        return task_id
    
    def update_task_progress(self, task_id: str, progress: float, 
                           status: str = None, metrics: Dict[str, Any] = None):
        """更新任务进度"""
        if task_id not in self.tasks:
            return
        
        task = self.tasks[task_id]
        task.progress = progress
        
        if status:
            task.status = status
        
        if metrics:
            if task.metrics is None:
                task.metrics = {}
            task.metrics.update(metrics)
        
        # 更新估计完成时间
        if progress > 0 and task.status == "running":
            elapsed = (datetime.now() - task.start_time).total_seconds()
            if elapsed > 0:
                total_time_estimated = elapsed / (progress / 100)
                remaining_time = total_time_estimated - elapsed
                task.estimated_completion = datetime.now() + timedelta(seconds=remaining_time)
        
        # 如果完成，设置结束时间
        if progress >= 100 or status in ["completed", "failed"]:
            task.end_time = datetime.now()
        
        # 保存到Redis
        self._save_task_to_redis(task)
        
        # 广播更新
        self._broadcast_task_update(task_id)
        
        # 收集指标
        self.metrics_collector.record_task_update(task)
        
        # 检查告警
        self.alert_system.check_alerts(task)
    
    def _save_task_to_redis(self, task: EnterpriseTask):
        """保存任务到Redis"""
        try:
            task_data = json.dumps(task.to_dict())
            self.redis_client.set(f"task:{task.task_id}", task_data)
            
            # 设置过期时间（24小时）
            self.redis_client.expire(f"task:{task.task_id}", 24 * 3600)
            
        except Exception as e:
            print(f"保存任务到Redis错误: {e}")
    
    def get_task(self, task_id: str) -> Optional[EnterpriseTask]:
        """获取任务"""
        # 首先尝试内存
        if task_id in self.tasks:
            return self.tasks[task_id]
        
        # 然后尝试Redis
        try:
            task_data = self.redis_client.get(f"task:{task_id}")

                task_dict = json.loads(task_data)
                
                # 转换datetime字符串

                        task_dict[key] = datetime.fromisoformat(task_dict[key])
                
                task = EnterpriseTask(**task_dict)
                self.tasks[task_id] = task
                return task
        except Exception as e:
            print(f"从Redis获取任务错误: {e}")
        
        return None
    
    def get_user_tasks(self, user_id: str) -> List[EnterpriseTask]:
        """获取用户的所有任务"""
        user_tasks = []
        
        for task in self.tasks.values():
            if task.user_id == user_id:

            if task.project_id == project_id:

class MetricsCollector:
    """指标收集器"""
    
    def __init__(self):
        self.metrics_history = []
        self.aggregated_metrics = {}
    
    def record_task_update(self, task: EnterpriseTask):
        """记录任务更新"""
        metric_point = {
            "timestamp": datetime.now().isoformat(),
            "task_id": task.task_id,
            "task_type": task.task_type,
            "progress": task.progress,
            "status": task.status,
            "metrics": task.metrics or {}
        }
        
        self.metrics_history.append(metric_point)
        
        # 更新聚合指标
        self._update_aggregated_metrics(task)
    
    def _update_aggregated_metrics(self, task: EnterpriseTask):
        """更新聚合指标"""
        task_type = task.task_type
        
        if task_type not in self.aggregated_metrics:
            self.aggregated_metrics[task_type] = {
                "total_tasks": 0,
                "completed_tasks": 0,
                "failed_tasks": 0,
                "avg_completion_time": 0,
                "avg_progress_speed": 0,
                "recent_tasks": []
            }
        
        agg = self.aggregated_metrics[task_type]
        agg["total_tasks"] += 1
        
        if task.status == "completed":
            agg["completed_tasks"] += 1
            
            # 计算完成时间
            if task.start_time and task.end_time:
                completion_time = (task.end_time - task.start_time).total_seconds()
                
                # 更新平均完成时间
                current_avg = agg["avg_completion_time"]
                completed_count = agg["completed_tasks"]
                
                agg["avg_completion_time"] = (
                    (current_avg * (completed_count - 1) + completion_time) / completed_count
                )
        
        elif task.status == "failed":
            agg["failed_tasks"] += 1
        
        # 更新平均进度速度
        if task.progress > 0 and task.start_time:
            elapsed = (datetime.now() - task.start_time).total_seconds()
            if elapsed > 0:
                progress_speed = task.progress / elapsed
                
                # 简单平均
                running_tasks = [t for t in self.metrics_history[-100:] 
                               if t["task_type"] == task_type and t["progress"] > 0]
                
                if running_tasks:
                    total_speed = sum(t["progress"] / (
                        (datetime.fromisoformat(t["timestamp"]) - 
                         datetime.fromisoformat(t.get("start_time", t["timestamp"]))).total_seconds()
                    ) for t in running_tasks if "start_time" in t)
                    
                    agg["avg_progress_speed"] = total_speed / len(running_tasks)
        
        # 添加最近任务
        agg["recent_tasks"].append({
            "task_id": task.task_id,
            "status": task.status,
            "progress": task.progress,
            "timestamp": datetime.now().isoformat()
        })
        
        # 保持最近100个任务
        agg["recent_tasks"] = agg["recent_tasks"][-100:]
    
    def get_metrics_report(self, task_type: str = None) -> Dict[str, Any]:
        """获取指标报告"""
        if task_type and task_type in self.aggregated_metrics:
            return self.aggregated_metrics[task_type]
        
        # 总体报告
        overall = {
            "total_tasks": sum(agg["total_tasks"] for agg in self.aggregated_metrics.values()),
            "completed_tasks": sum(agg["completed_tasks"] for agg in self.aggregated_metrics.values()),
            "failed_tasks": sum(agg["failed_tasks"] for agg in self.aggregated_metrics.values()),
            "task_types": {}
        }
        
        for ttype, agg in self.aggregated_metrics.items():
            overall["task_types"][ttype] = {
                "total": agg["total_tasks"],
                "completed": agg["completed_tasks"],
                "failed": agg["failed_tasks"],
                "success_rate": agg["completed_tasks"] / agg["total_tasks"] * 100 if agg["total_tasks"] > 0 else 0,
                "avg_completion_time": agg["avg_completion_time"],
                "avg_progress_speed": agg["avg_progress_speed"]
            }
        
        return overall

                if rule["condition"](task):
                    alert = {
                        "rule_name": rule["name"],
                        "task_id": task.task_id,
                        "message": rule["message"],
                        "severity": rule["severity"],
                        "timestamp": datetime.now().isoformat(),
                        "task_details": {

                            "status": task.status,
                            "user_id": task.user_id,
                            "project_id": task.project_id
                        }
                    }
                    
                    self.alerts.append(alert)
                    
                    # 触发告警动作
                    self._trigger_alert_action(alert)
                    
            except Exception as e:
                print(f"检查告警规则错误: {e}")

        """触发告警动作"""
        print(f"[{alert['severity'].upper()}] {alert['message']} - 任务: {alert['task_id']}")
        
        # 这里可以添加更多动作，如：

        # - 发送Slack消息
        # - 记录到监控系统

        if alert["severity"] == "critical":
            # 对于严重告警，可以采取紧急措施

class EnterpriseCallbackHandler(BaseCallbackHandler):
    """企业级回调处理器"""
    
    def __init__(self, progress_system: EnterpriseProgressSystem, task_id: str):
        super().__init__()
        self.progress_system = progress_system
        self.task_id = task_id
        self.start_time = None
        self.token_count = 0
        self.last_progress_update = None
        
    def on_llm_start(self, serialized: Dict[str, Any], prompts: List[str], **kwargs):
        self.start_time = datetime.now()
        
        # 估算总token数
        total_words = sum(len(prompt.split()) for prompt in prompts)
        estimated_tokens = int(total_words / 0.75)
        
        # 更新任务状态
        self.progress_system.update_task_progress(
            self.task_id,
            progress=0,
            status="running",
            metrics={
                "estimated_tokens": estimated_tokens,
                "model": serialized.get("kwargs", {}).get("model_name", "unknown"),
                "prompt_count": len(prompts)
            }
        )
        
        # 记录最后进度更新时间
        self.last_progress_update = self.start_time
    
    def on_llm_new_token(self, token: str, **kwargs):
        current_time = datetime.now()
        self.token_count += 1
        
        # 每10个token或每秒更新一次进度
        update_interval = max(1.0, (current_time - self.last_progress_update).total_seconds())
        
        if self.token_count % 10 == 0 or update_interval >= 1.0:
            # 估算总进度
            task = self.progress_system.get_task(self.task_id)
            estimated_tokens = task.metrics.get("estimated_tokens", 100)
            
            progress = min(99.9, (self.token_count / estimated_tokens) * 100)
            
            # 计算实时指标
            elapsed = (current_time - self.start_time).total_seconds()
            tokens_per_second = self.token_count / elapsed if elapsed > 0 else 0
            
            # 更新进度
            self.progress_system.update_task_progress(
                self.task_id,
                progress=progress,
                metrics={
                    "token_count": self.token_count,
                    "tokens_per_second": tokens_per_second,
                    "elapsed_seconds": elapsed,
                    "last_progress_update": current_time.isoformat()
                }
            )
            
            self.last_progress_update = current_time
    
    def on_llm_end(self, response, **kwargs):
        end_time = datetime.now()
        elapsed = (end_time - self.start_time).total_seconds()
        
        # 最终进度
        self.progress_system.update_task_progress(
            self.task_id,
            progress=100,
            status="completed",
            metrics={
                "final_token_count": self.token_count,
                "total_elapsed_seconds": elapsed,
                "final_tokens_per_second": self.token_count / elapsed if elapsed > 0 else 0,
                "completion_time": end_time.isoformat()
            }
        )
        
        # 如果有使用统计，添加到指标
        if hasattr(response, 'usage_metadata'):
            usage = response.usage_metadata
            self.progress_system.update_task_progress(
                self.task_id,
                metrics={
                    "input_tokens": usage.get("input_tokens"),
                    "output_tokens": usage.get("output_tokens"),
                    "total_tokens": usage.get("total_tokens")
                }
            )
    
    def on_llm_error(self, error, **kwargs):
        self.progress_system.update_task_progress(
            self.task_id,
            progress=self.token_count / 100 * 100,  # 基于已完成的token估算进度
            status="failed",
            metrics={
                "error": str(error),
                "failed_at": datetime.now().isoformat(),
                "completed_tokens": self.token_count
            }
        )
```
