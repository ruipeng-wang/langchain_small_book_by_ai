## 传统多模型管理的复杂性

在LangChain等框架出现之前，实现多模型智能路由需要大量的手动编码和复杂的逻辑。

### 手动实现多模型路由

```python
# 传统多模型路由实现
import time
from typing import Dict, Any, List, Optional
import random

class ManualMultiModelRouter:
    """传统手动多模型路由器"""
    
    def __init__(self):
        # 初始化不同模型
        self.models = {}
        self.model_costs = {}
        self.model_latencies = {}
        self.model_capabilities = {}
        
        # 初始化统计
        self.request_history = []
        self.model_usage = {}
        
        # 配置路由策略
        self.routing_strategies = {
            "cost_first": self._route_by_cost,
            "performance_first": self._route_by_performance,
            "load_balance": self._route_by_load,
            "intelligent": self._route_intelligently
        }
    
    def register_model(self, name: str, model_obj, cost_per_token: float, 
                      capabilities: List[str], latency_ms: float = 100):
        """注册模型"""
        self.models[name] = model_obj
        self.model_costs[name] = cost_per_token
        self.model_latencies[name] = latency_ms
        self.model_capabilities[name] = capabilities
        self.model_usage[name] = {"requests": 0, "tokens": 0, "errors": 0}
    
    def predict(self, input_text: str, strategy: str = "intelligent", 
               task_type: str = None, budget: float = None) -> str:
        """预测入口"""
        # 1. 选择模型
        selected_model = self._select_model(input_text, strategy, task_type, budget)
        
        if not selected_model:
            raise Exception("没有可用模型")
        
        # 2. 记录请求开始
        start_time = time.time()
        request_info = {
            "model": selected_model,
            "input": input_text,
            "start_time": start_time,
            "task_type": task_type,
            "strategy": strategy
        }
        
        try:
            # 3. 执行预测
            model_obj = self.models[selected_model]
            response = model_obj.generate(input_text)
            
            # 4. 计算成本和延迟
            end_time = time.time()
            latency = end_time - start_time
            
            # 估算token数（简单估算）
            input_tokens = len(input_text) // 4
            output_tokens = len(response) // 4
            total_tokens = input_tokens + output_tokens
            
            cost = total_tokens * self.model_costs[selected_model]
            
            # 5. 更新统计
            self.model_usage[selected_model]["requests"] += 1
            self.model_usage[selected_model]["tokens"] += total_tokens
            
            # 更新延迟统计
            current_latency = self.model_latencies[selected_model]
            # 指数移动平均
            new_latency = current_latency * 0.9 + latency * 1000 * 0.1
            self.model_latencies[selected_model] = new_latency
            
            # 记录历史
            request_info.update({
                "end_time": end_time,
                "latency": latency,
                "tokens": total_tokens,
                "cost": cost,
                "success": True,
                "response": response
            })
            
            self.request_history.append(request_info)
            
            return response
            
        except Exception as e:
            # 6. 处理失败
            end_time = time.time()
            
            self.model_usage[selected_model]["errors"] += 1
            
            request_info.update({
                "end_time": end_time,
                "error": str(e),
                "success": False
            })
            
            self.request_history.append(request_info)
            
            # 7. 故障转移
            print(f"模型 {selected_model} 失败: {e}")
            return self._handle_failure(input_text, selected_model, strategy, task_type, budget)
    
    def _select_model(self, input_text: str, strategy: str, task_type: str, 
                     budget: float) -> Optional[str]:
        """选择模型"""
        available_models = list(self.models.keys())
        
        if not available_models:
            return None
        
        # 根据策略选择
        router = self.routing_strategies.get(strategy, self._route_intelligently)
        return router(input_text, available_models, task_type, budget)
    
    def _route_by_cost(self, input_text: str, available_models: List[str], 
                      task_type: str, budget: float) -> str:
        """成本优先路由"""
        # 选择成本最低的模型
        sorted_models = sorted(
            available_models,
            key=lambda m: self.model_costs[m]
        )
        
        # 如果有预算限制，选择在预算内的模型
        if budget:
            for model in sorted_models:
                # 估算成本
                input_tokens = len(input_text) // 4
                estimated_cost = input_tokens * self.model_costs[model]
                if estimated_cost <= budget:
                    return model
        
        return sorted_models[0] if sorted_models else None
    
    def _route_by_performance(self, input_text: str, available_models: List[str],
                            task_type: str, budget: float) -> str:
        """性能优先路由"""
        # 选择延迟最低的模型
        sorted_models = sorted(
            available_models,
            key=lambda m: self.model_latencies[m]
        )
        return sorted_models[0] if sorted_models else None
    
    def _route_by_load(self, input_text: str, available_models: List[str],
                      task_type: str, budget: float) -> str:
        """负载均衡路由"""
        # 选择请求数最少的模型
        sorted_models = sorted(
            available_models,
            key=lambda m: self.model_usage[m]["requests"]
        )
        return sorted_models[0] if sorted_models else None
    
    def _route_intelligently(self, input_text: str, available_models: List[str],
                           task_type: str, budget: float) -> str:
        """智能路由"""
        # 复杂的智能路由逻辑
        
        # 1. 根据任务类型选择
        if task_type:
            # 查找最擅长该任务的模型
            suitable_models = []
            for model in available_models:
                if task_type in self.model_capabilities[model]:
                    suitable_models.append(model)
            
            if suitable_models:
                available_models = suitable_models
        
        # 2. 根据输入复杂度选择
        input_length = len(input_text)
        complexity = self._estimate_complexity(input_text)
        
        if complexity == "simple" and input_length < 100:
            # 简单任务用低成本模型
            return self._route_by_cost(input_text, available_models, task_type, budget)
        elif complexity == "complex" or input_length > 1000:
            # 复杂任务用高性能模型
            return self._route_by_performance(input_text, available_models, task_type, budget)
        else:
            # 中等复杂度，平衡成本和性能
            # 计算每个模型的分数
            model_scores = {}
            for model in available_models:
                # 成本分数（越低越好）
                cost_score = 1 / (self.model_costs[model] + 0.001)
                
                # 性能分数（延迟越低越好）
                perf_score = 1 / (self.model_latencies[model] + 1)
                
                # 负载分数（请求越少越好）
                load_score = 1 / (self.model_usage[model]["requests"] + 1)
                
                # 综合分数
                total_score = cost_score * 0.4 + perf_score * 0.4 + load_score * 0.2
                model_scores[model] = total_score
            
            # 选择分数最高的模型
            if model_scores:
                return max(model_scores.items(), key=lambda x: x[1])[0]
        
        # 回退到负载均衡
        return self._route_by_load(input_text, available_models, task_type, budget)
    
    def _estimate_complexity(self, text: str) -> str:
        """估算任务复杂度"""
        # 简单的启发式规则
        length = len(text)
        
        # 检查是否是复杂问题
        complexity_indicators = [
            "如何", "为什么", "解释", "分析", "比较", "评估",
            "解决方案", "设计", "实现", "优化"
        ]
        
        contains_complex = any(indicator in text for indicator in complexity_indicators)
        
        if length > 500 or contains_complex:
            return "complex"
        elif length > 100:
            return "medium"
        else:
            return "simple"
    
    def _handle_failure(self, input_text: str, failed_model: str, 
                       strategy: str, task_type: str, budget: float) -> str:
        """处理模型失败"""
        print(f"尝试故障转移，原始模型: {failed_model}")
        
        # 从可用模型中移除失败的模型
        available_models = [
            m for m in self.models.keys() 
            if m != failed_model
        ]
        
        if not available_models:
            raise Exception("所有模型都失败了")
        
        # 使用备选策略重新选择
        # 首先尝试相同策略
        try:
            router = self.routing_strategies.get(strategy, self._route_intelligently)
            new_model = router(input_text, available_models, task_type, budget)
            
            if new_model:
                print(f"故障转移到: {new_model}")
                return self._retry_with_model(input_text, new_model)
        except:
            pass
        
        # 如果相同策略失败，尝试成本优先
        try:
            new_model = self._route_by_cost(input_text, available_models, task_type, budget)
            if new_model:
                print(f"故障转移到（成本优先）: {new_model}")
                return self._retry_with_model(input_text, new_model)
        except:
            pass
        
        # 最后尝试任何可用模型
        for model in available_models:
            try:
                print(f"尝试模型: {model}")
                return self._retry_with_model(input_text, model)
            except:
                continue
        
        raise Exception("所有故障转移尝试都失败了")
    
    def _retry_with_model(self, input_text: str, model_name: str) -> str:
        """使用指定模型重试"""
        model_obj = self.models[model_name]
        return model_obj.generate(input_text)
    
    def get_statistics(self) -> Dict[str, Any]:
        """获取统计信息"""
        total_requests = sum(usage["requests"] for usage in self.model_usage.values())
        total_tokens = sum(usage["tokens"] for usage in self.model_usage.values())
        total_errors = sum(usage["errors"] for usage in self.model_usage.values())
        
        total_cost = 0
        for model, usage in self.model_usage.items():
            tokens = usage["tokens"]
            cost_per_token = self.model_costs[model]
            total_cost += tokens * cost_per_token
        
        # 计算成功率
        success_rate = ((total_requests - total_errors) / total_requests * 100 
                       if total_requests > 0 else 0)
        
        # 计算平均延迟
        recent_requests = [r for r in self.request_history if r.get("success")]
        avg_latency = (sum(r["latency"] for r in recent_requests) / len(recent_requests)
                      if recent_requests else 0)
        
        return {
            "total_requests": total_requests,
            "total_tokens": total_tokens,
            "total_cost": total_cost,
            "total_errors": total_errors,
            "success_rate": f"{success_rate:.1f}%",
            "avg_latency": f"{avg_latency:.3f}秒",
            "model_usage": self.model_usage,
            "model_costs": {k: f"${v:.6f}/token" for k, v in self.model_costs.items()},
            "model_latencies": {k: f"{v:.1f}ms" for k, v in self.model_latencies.items()}
        }
```

```python
# 手动负载均衡器实现
class ManualLoadBalancer:
    """手动负载均衡器"""
    
    def __init__(self, model_instances: List[Any]):
        self.instances = model_instances
        self.instance_weights = {}
        self.instance_health = {}
        self.instance_stats = {}
        
        # 初始化
        for i, instance in enumerate(model_instances):
            self.instance_weights[instance] = 1.0  # 默认权重
            self.instance_health[instance] = True  # 健康状态
            self.instance_stats[instance] = {
                "requests": 0,
                "successful": 0,
                "failed": 0,
                "total_latency": 0.0,
                "concurrent": 0
            }
        
        # 健康检查线程
        self._start_health_check()
    
    def _start_health_check(self):
        """启动健康检查"""
        import threading
        import time
        
        def health_check_worker():
            while True:
                self._perform_health_checks()
                time.sleep(30)  # 每30秒检查一次
        
        thread = threading.Thread(target=health_check_worker, daemon=True)
        thread.start()
    
    def _perform_health_checks(self):
        """执行健康检查"""
        for instance in self.instances:
            try:
                # 发送测试请求
                start_time = time.time()
                # 这里应该是实际的健康检查请求
                # 模拟检查
                time.sleep(0.1)
                
                # 计算响应时间
                response_time = time.time() - start_time
                
                # 更新健康状态
                if response_time < 2.0:  # 2秒阈值
                    self.instance_health[instance] = True
                    # 根据响应时间调整权重
                    if response_time < 0.5:
                        self.instance_weights[instance] = 1.5  # 高性能，提高权重
                    else:
                        self.instance_weights[instance] = 1.0
                else:
                    self.instance_health[instance] = False
                    self.instance_weights[instance] = 0.1  # 低性能，降低权重
                    
            except Exception as e:
                print(f"健康检查失败 {instance}: {e}")
                self.instance_health[instance] = False
                self.instance_weights[instance] = 0.0
    
    def select_instance(self, strategy: str = "weighted") -> Any:
        """选择实例"""
        # 过滤健康实例
        healthy_instances = [
            inst for inst in self.instances 
            if self.instance_health.get(inst, False)
        ]
        
        if not healthy_instances:
            raise Exception("没有健康实例可用")
        
        if strategy == "round_robin":
            # 轮询 - 需要维护状态
            if not hasattr(self, '_rr_index'):
                self._rr_index = 0
            
            instance = healthy_instances[self._rr_index % len(healthy_instances)]
            self._rr_index += 1
            return instance
        
        elif strategy == "random":
            import random
            return random.choice(healthy_instances)
        
        elif strategy == "weighted":
            # 加权随机
            import random
            
            # 获取权重
            weights = []
            instances = []
            for inst in healthy_instances:
                weight = self.instance_weights.get(inst, 1.0)
                if weight > 0:
                    weights.append(weight)
                    instances.append(inst)
            
            if not instances:
                raise Exception("没有有权重的健康实例")
            
            # 加权随机选择
            total_weight = sum(weights)
            rand_val = random.uniform(0, total_weight)
            
            cumulative = 0
            for i, weight in enumerate(weights):
                cumulative += weight
                if rand_val <= cumulative:
                    return instances[i]
            
            # 回退
            return instances[0]
        
        elif strategy == "least_connections":
            # 最少连接数
            return min(
                healthy_instances,
                key=lambda inst: self.instance_stats[inst].get("concurrent", 0)
            )
        
        else:
            # 默认轮询
            return healthy_instances[0]
    
    def record_request_start(self, instance: Any):
        """记录请求开始"""
        stats = self.instance_stats.get(instance)
        if stats:
            stats["concurrent"] = stats.get("concurrent", 0) + 1
            stats["requests"] = stats.get("requests", 0) + 1
    
    def record_request_end(self, instance: Any, success: bool, latency: float):
        """记录请求结束"""
        stats = self.instance_stats.get(instance)
        if stats:
            stats["concurrent"] = max(0, stats.get("concurrent", 0) - 1)
            
            if success:
                stats["successful"] = stats.get("successful", 0) + 1
                stats["total_latency"] = stats.get("total_latency", 0.0) + latency
            else:
                stats["failed"] = stats.get("failed", 0) + 1
            
            # 如果失败次数过多，标记为不健康
            total = stats["successful"] + stats["failed"]
            if total > 10 and stats["failed"] / total > 0.3:
                self.instance_health[instance] = False
    
    def get_stats(self) -> Dict[str, Any]:
        """获取统计"""
        stats_summary = {}
        
        for instance, data in self.instance_stats.items():
            total = data["successful"] + data["failed"]
            success_rate = (data["successful"] / total * 100 if total > 0 else 0)
            
            avg_latency = (data["total_latency"] / data["successful"] 
                          if data["successful"] > 0 else 0)
            
            stats_summary[str(instance)] = {
                "requests": total,
                "success_rate": f"{success_rate:.1f}%",
                "avg_latency": f"{avg_latency:.3f}s",
                "concurrent": data["concurrent"],
                "healthy": self.instance_health.get(instance, False),
                "weight": self.instance_weights.get(instance, 1.0)
            }
        
        return stats_summary
```

## LangChain v1.2.22解决方案

LangChain通过内置的中间件、路由器和统一抽象，大幅简化了多模型混合使用的复杂性。

### ModelFallbackMiddleware：自动故障转移

LangChain v1.2.22提供了`ModelFallbackMiddleware`，可以自动处理模型失败并切换到备用模型。

```python
# LangChain v1.2.22 自动故障转移
"""
安装所需包:
pip install langchain-openai langchain-anthropic
"""

from langchain_openai import ChatOpenAI
from langchain_anthropic import ChatAnthropic
from langchain_ollama import ChatOllama
from langchain.agents import create_agent
from langchain.agents.middleware import ModelFallbackMiddleware

# 1. 创建多个模型实例
gpt4_model = ChatOpenAI(
    model="gpt-4",
    temperature=0.7,
    max_tokens=1000
)

gpt35_model = ChatOpenAI(
    model="gpt-3.5-turbo",
    temperature=0.7,
    max_tokens=1000
)

claude_model = ChatAnthropic(
    model="claude-3-5-sonnet-20241022",
    temperature=0.7,
    max_tokens=1000
)

local_model = ChatOllama(
    model="llama3.1:8b",
    temperature=0.7,
    num_predict=512
)

# 2. 使用ModelFallbackMiddleware实现自动故障转移
agent_with_fallback = create_agent(
    model=gpt4_model,  # 主模型
    tools=[],  # 可以添加工具
    middleware=[
        ModelFallbackMiddleware(
            # 备用模型列表（按优先级）
            gpt35_model,
            claude_model,
            local_model
        )
    ]
)

# 使用示例
try:
    response = agent_with_fallback.invoke(
        "解释量子计算的基本原理"
    )
    print(f"响应: {response}")
except Exception as e:
    print(f"所有模型都失败了: {e}")

# ModelFallbackMiddleware的工作原理：
# 1. 首先尝试主模型（gpt4_model）
# 2. 如果失败，自动尝试第一个备用模型（gpt35_model）
# 3. 继续尝试直到成功或所有模型都失败
# 4. 提供详细的错误信息和重试统计
```

```python
        for keyword in keywords:

        if capabilities_needed:
            # 检查模型是否具备所需能力

                cap in config["capabilities"]
                for cap in capabilities_needed
            )
            if not has_capabilities:
                continue

        if context_length > config["max_context"] * 0.8:  # 留20%余量
            continue
        
        candidate_models.append((model_name, config))

        return "gpt-3.5-turbo"  # 默认

    if task_complexity == "simple":

        return min(candidate_models, 
                  key=lambda x: x[1]["cost_per_token"])[0]
    
    elif task_complexity == "medium":
        # 中等任务：平衡成本和能力
        # 计算每个模型的分数
        model_scores = []
        for model_name, config in candidate_models:
            # 成本分数（越低越好）
            cost_score = 1 / (config["cost_per_token"] + 1e-10)

            if "reasoning" in config["capabilities"]:
                capability_score += 1
            
            # 综合分数
            total_score = cost_score * 0.6 + capability_score * 0.4
            model_scores.append((model_name, total_score))

            return min(strong_models, 
                      key=lambda x: x[1]["cost_per_token"])[0]
        else:
            # 回退到能力最强的候选

                      key=lambda x: len(x[1]["capabilities"]))[0]
```

### 基于权重的负载均衡

对于需要在高性能模型和低成本模型之间平衡的场景，可以使用基于权重的负载均衡。

```python
# LangChain v1.2.22 权重负载均衡
import random
from typing import List, Tuple
from langchain_core.runnables import RunnableLambda

class WeightedLoadBalancer:
    """基于权重的负载均衡器"""
    
    def __init__(self, models_with_weights: List[Tuple[Any, float]]):
        """
        初始化负载均衡器
        
        Args:
            models_with_weights: [(model实例, 权重), ...]
        """
        self.models_with_weights = models_with_weights
        self.total_weight = sum(weight for _, weight in models_with_weights)
        
        # 验证权重
        if self.total_weight <= 0:
            raise ValueError("总权重必须大于0")
        
        # 归一化权重
        self.normalized_weights = [
            (model, weight / self.total_weight)
            for model, weight in models_with_weights
        ]
        
        # 创建累积分布
        self.cumulative_dist = []
        cumulative = 0
        for model, norm_weight in self.normalized_weights:
            cumulative += norm_weight
            self.cumulative_dist.append((model, cumulative))
    
    def select_model(self) -> Any:
        """根据权重选择模型"""
        rand_val = random.random()
        
        for model, cumulative in self.cumulative_dist:
            if rand_val <= cumulative:
                return model
        
        # 回退到最后一个模型
        return self.cumulative_dist[-1][0]
    
    def create_balanced_chain(self):
        """创建负载均衡链"""
        def balanced_invoke(input_data):
            # 选择模型
            selected_model = self.select_model()
            
            # 记录选择（用于监控）
            model_name = selected_model.__class__.__name__
            print(f"负载均衡: 选择 {model_name}")
            
            # 使用选中的模型执行
            return selected_model.invoke(input_data)
        
        return RunnableLambda(balanced_invoke)

# 高级功能：动态调整权重
class DynamicWeightedLoadBalancer(WeightedLoadBalancer):
    """动态权重负载均衡器"""
    
    def __init__(self, models_with_weights: List[Tuple[Any, float]]):
        super().__init__(models_with_weights)
        
        # 监控统计
        self.model_stats = {
            model: {"requests": 0, "success": 0, "total_latency": 0.0}
            for model, _ in models_with_weights
        }
        
        # 性能指标
        self.performance_metrics = {}
        
    def select_model(self) -> Any:
        """动态选择模型（基于性能调整权重）"""
        # 如果有性能数据，使用动态权重
        if self.performance_metrics:
            return self._select_by_performance()
        else:
            # 初始阶段使用配置权重
            return super().select_model()
    
    def _select_by_performance(self) -> Any:
        """基于性能选择模型"""
        # 计算每个模型的性能分数
        performance_scores = []
        
        for model, _ in self.models_with_weights:
            stats = self.model_stats.get(model, {})
            perf = self.performance_metrics.get(model, {})
            
            # 成功率和延迟作为性能指标
            success_rate = (stats.get("success", 0) / stats.get("requests", 1) 
                          if stats.get("requests", 0) > 0 else 0.5)
            
            avg_latency = (perf.get("avg_latency", 1.0) 
                          if perf.get("avg_latency", 0) > 0 else 1.0)
            
            # 性能分数 = 成功率 / 平均延迟
            # 成功率越高越好，延迟越低越好
            performance_score = success_rate / (avg_latency + 0.001)
            
            performance_scores.append((model, performance_score))
        
        # 使用softmax转换为概率分布
        import math
        
        # 计算softmax
        exp_scores = [math.exp(score) for _, score in performance_scores]
        sum_exp_scores = sum(exp_scores)
        probabilities = [exp / sum_exp_scores for exp in exp_scores]
        
        # 根据概率选择
        rand_val = random.random()
        cumulative = 0
        
        for i, (model, _) in enumerate(performance_scores):
            cumulative += probabilities[i]
            if rand_val <= cumulative:
                return model
        
        return performance_scores[0][0]
    
    def record_request(self, model: Any, success: bool, latency: float):
        """记录请求结果"""
        stats = self.model_stats.get(model)
        if stats:
            stats["requests"] = stats.get("requests", 0) + 1
            if success:
                stats["success"] = stats.get("success", 0) + 1
                stats["total_latency"] = stats.get("total_latency", 0.0) + latency
                
                # 更新性能指标
                self._update_performance_metrics(model, latency)
    
    def _update_performance_metrics(self, model: Any, latency: float):
        """更新性能指标"""
        if model not in self.performance_metrics:
            self.performance_metrics[model] = {
                "avg_latency": latency,
                "sample_count": 1
            }
        else:
            metrics = self.performance_metrics[model]
            current_avg = metrics["avg_latency"]
            count = metrics["sample_count"]
            
            # 指数移动平均
            alpha = 0.1  # 学习率
            new_avg = current_avg * (1 - alpha) + latency * alpha
            
            metrics["avg_latency"] = new_avg
            metrics["sample_count"] = count + 1
    
    def get_performance_report(self) -> Dict[str, Any]:
        """获取性能报告"""
        report = {}
        
        for model, stats in self.model_stats.items():
            model_name = model.__class__.__name__
            
            requests = stats.get("requests", 0)
            success = stats.get("success", 0)
            total_latency = stats.get("total_latency", 0.0)
            
            success_rate = (success / requests * 100 if requests > 0 else 0)
            avg_latency = (total_latency / success if success > 0 else 0)
            
            perf_metrics = self.performance_metrics.get(model, {})
            
            report[model_name] = {
                "requests": requests,
                "success_rate": f"{success_rate:.1f}%",
                "avg_latency": f"{avg_latency:.3f}s",
                "performance_score": perf_metrics.get("avg_latency", 0)
            }
        
        return report
```

```python
# 企业级智能路由系统
from dataclasses import dataclass, field
from typing import Dict, List, Any, Optional, Callable
import time
import json

@dataclass
class ModelEndpoint:

@dataclass

class EnterpriseModelRouter:
    """企业级模型路由器"""
    
    def __init__(self, config_path: Optional[str] = None):
        # 模型端点池
        self.endpoints: Dict[str, ModelEndpoint] = {}
        
        # 路由规则链
        self.routing_rules: List[RoutingRule] = []
        
        # 请求历史
        self.request_history: List[Dict[str, Any]] = []
        
        # 监控指标
        self.metrics = {
            "total_requests": 0,
            "successful_requests": 0,
            "failed_requests": 0,
            "total_cost": 0.0,
            "total_tokens": 0
        }
        
        # 加载配置
        if config_path:
            self.load_config(config_path)
        
        # 初始化默认规则
        self._init_default_rules()
        
        # 启动监控线程
        self._start_monitoring()
    
    def _init_default_rules(self):
        """初始化默认路由规则"""
        
        # 规则1：预算敏感
        def budget_sensitive_condition(context: Dict[str, Any]) -> bool:
            budget = context.get("budget", float('inf'))
            return budget < 0.01  # 预算小于$0.01
        
        def budget_sensitive_selector(endpoints: List[ModelEndpoint], 
                                     context: Dict[str, Any]) -> Optional[str]:
            # 选择成本最低的端点
            suitable = [ep for ep in endpoints 
                       if ep.current_concurrent < ep.max_concurrent]
            
            if not suitable:
                return None
            
            return min(suitable, key=lambda ep: ep.cost_per_token).name
        
        self.add_rule(RoutingRule(
            name="budget_sensitive",
            condition=budget_sensitive_condition,
            model_selector=budget_sensitive_selector,
            description="预算敏感任务"
        ))
        
        # 规则2：高性能要求
        def high_performance_condition(context: Dict[str, Any]) -> bool:
            latency_req = context.get("latency_requirement", float('inf'))
            return latency_req < 2.0  # 延迟要求小于2秒
        
        def high_performance_selector(endpoints: List[ModelEndpoint],
                                     context: Dict[str, Any]) -> Optional[str]:
            suitable = [ep for ep in endpoints 
                       if ep.current_concurrent < ep.max_concurrent]
            
            if not suitable:
                return None
            
            # 选择延迟最低的端点
            return min(suitable, key=lambda ep: ep.avg_latency).name
        
        self.add_rule(RoutingRule(
            name="high_performance",
            condition=high_performance_condition,
            model_selector=high_performance_selector,
            description="高性能要求任务"
        ))
        
        # 规则3：能力匹配
        def capability_match_condition(context: Dict[str, Any]) -> bool:
            required_caps = context.get("required_capabilities", [])
            return bool(required_caps)
        
        def capability_match_selector(endpoints: List[ModelEndpoint],
                                     context: Dict[str, Any]) -> Optional[str]:
            required_caps = context.get("required_capabilities", [])
            
            # 过滤具备所有能力的端点
            suitable = []
            for ep in endpoints:
                if ep.current_concurrent >= ep.max_concurrent:
                    continue
                
                # 检查是否具备所有所需能力
                has_all_caps = all(
                    cap in ep.capabilities
                    for cap in required_caps
                )
                
                if has_all_caps:
                    suitable.append(ep)
            
            if not suitable:
                return None
            
            # 选择优先级最高的
            suitable.sort(key=lambda ep: (-ep.priority, ep.avg_latency))
            return suitable[0].name
        
        self.add_rule(RoutingRule(
            name="capability_match",
            condition=capability_match_condition,
            model_selector=capability_match_selector,
            description="特定能力要求任务"
        ))
    
    def add_endpoint(self, endpoint: ModelEndpoint):
        """添加模型端点"""
        self.endpoints[endpoint.name] = endpoint
    
    def add_rule(self, rule: RoutingRule):
        """添加路由规则"""
        self.routing_rules.append(rule)
    
    def route_request(self, input_data: str, context: Optional[Dict[str, Any]] = None) -> str:
        """路由请求"""
        if context is None:
            context = {}
        
        # 记录请求开始
        request_id = f"req_{int(time.time()*1000)}_{self.metrics['total_requests']}"
        start_time = time.time()
        
        request_info = {
            "id": request_id,
            "input": input_data,
            "context": context,
            "start_time": start_time,
            "selected_endpoint": None,
            "status": "processing"
        }
        
        self.metrics["total_requests"] += 1
        
        try:
            # 1. 选择端点
            selected_endpoint = self._select_endpoint(context)
            
            if not selected_endpoint:
                raise Exception("没有可用端点")
            
            request_info["selected_endpoint"] = selected_endpoint.name
            
            # 2. 执行请求
            with selected_endpoint.lock:
                selected_endpoint.current_concurrent += 1
            
            try:
                # 执行模型调用
                execute_start = time.time()
                response = selected_endpoint.model_instance.invoke(input_data)
                execute_time = time.time() - execute_start
                
                # 3. 更新统计
                self._update_endpoint_stats(
                    selected_endpoint,
                    success=True,
                    latency=execute_time,
                    input_text=input_data,
                    output_text=response.content if hasattr(response, 'content') else str(response)
                )
                
                # 4. 记录成功
                request_info.update({
                    "end_time": time.time(),
                    "latency": execute_time,
                    "status": "success",
                    "response": response.content[:500] if hasattr(response, 'content') else str(response)[:500]
                })
                
                self.metrics["successful_requests"] += 1
                
                return response.content if hasattr(response, 'content') else str(response)
                
            except Exception as e:
                # 执行失败
                execute_time = time.time() - start_time
                
                self._update_endpoint_stats(
                    selected_endpoint,
                    success=False,
                    latency=execute_time
                )
                
                raise e
                
            finally:
                # 减少并发计数
                with selected_endpoint.lock:
                    selected_endpoint.current_concurrent -= 1
        
        except Exception as e:
            # 整体失败
            end_time = time.time()
            
            request_info.update({
                "end_time": end_time,
                "latency": end_time - start_time,
                "status": "failed",
                "error": str(e)
            })
            
            self.metrics["failed_requests"] += 1
            
            # 尝试故障转移
            if context.get("enable_fallback", True):
                return self._handle_fallback(input_data, context, request_info)
            else:
                raise
        
        finally:
            # 记录请求历史
            self.request_history.append(request_info)
            
            # 限制历史记录大小
            if len(self.request_history) > 10000:
                self.request_history = self.request_history[-5000:]
    
    def _select_endpoint(self, context: Dict[str, Any]) -> Optional[ModelEndpoint]:
        """选择端点"""
        # 获取可用端点
        available_endpoints = [
            ep for ep in self.endpoints.values()
            if ep.current_concurrent < ep.max_concurrent
        ]
        
        if not available_endpoints:
            return None
        
        # 应用路由规则
        for rule in self.routing_rules:
            if rule.condition(context):
                selected_name = rule.model_selector(available_endpoints, context)
                if selected_name:
                    return self.endpoints.get(selected_name)
        
        # 默认规则：选择优先级最高且并发最低的
        available_endpoints.sort(key=lambda ep: (-ep.priority, ep.current_concurrent))
        return available_endpoints[0]
    
    def _update_endpoint_stats(self, endpoint: ModelEndpoint, success: bool, 
                              latency: float, input_text: str = "", 
                              output_text: str = ""):
        """更新端点统计"""
        with endpoint.lock:
            endpoint.total_requests += 1
            
            if success:
                # 估算token数
                input_tokens = len(input_text) // 4
                output_tokens = len(output_text) // 4
                total_tokens = input_tokens + output_tokens
                
                # 计算成本
                cost = total_tokens * endpoint.cost_per_token
                
                endpoint.total_tokens += total_tokens
                endpoint.total_cost += cost
                
                # 更新平均延迟（指数移动平均）
                alpha = 0.1
                endpoint.avg_latency = (endpoint.avg_latency * (1 - alpha) + 
                                       latency * alpha)
                
                # 更新全局指标
                self.metrics["total_tokens"] += total_tokens
                self.metrics["total_cost"] += cost
            else:
                endpoint.error_count += 1
    
    def _handle_fallback(self, input_data: str, context: Dict[str, Any], 
                        original_request: Dict[str, Any]) -> str:
        """处理故障转移"""
        print(f"故障转移: 原始端点 {original_request['selected_endpoint']} 失败")
        
        # 排除失败的端点
        failed_endpoint_name = original_request.get("selected_endpoint")
        available_endpoints = [
            ep for ep in self.endpoints.values()
            if ep.name != failed_endpoint_name and
               ep.current_concurrent < ep.max_concurrent
        ]
        
        if not available_endpoints:
            raise Exception(f"所有端点都失败，原始错误: {original_request.get('error')}")
        
        # 尝试其他端点
        for endpoint in available_endpoints:
            try:
                print(f"尝试故障转移到: {endpoint.name}")
                
                with endpoint.lock:
                    endpoint.current_concurrent += 1
                
                response = endpoint.model_instance.invoke(input_data)
                
                # 记录故障转移成功
                fallback_request = original_request.copy()
                fallback_request.update({
                    "fallback_to": endpoint.name,
                    "fallback_success": True,
                    "final_status": "fallback_success"
                })
                
                self.request_history.append(fallback_request)
                
                return response.content if hasattr(response, 'content') else str(response)
                
            except Exception as e:
                print(f"故障转移失败 {endpoint.name}: {e}")
                continue
            finally:
                with endpoint.lock:
                    endpoint.current_concurrent -= 1
        
        raise Exception(f"所有故障转移尝试都失败，原始错误: {original_request.get('error')}")
    
    def _start_monitoring(self):
        """启动监控线程"""
        def monitor_worker():
            while True:
                time.sleep(60)  # 每分钟报告一次
                self._report_metrics()
        
        thread = threading.Thread(target=monitor_worker, daemon=True)
        thread.start()
    
    def _report_metrics(self):
        """报告指标"""
        now = datetime.now().strftime("%Y-%m-%d %H:%M:%S")
        
        success_rate = (self.metrics["successful_requests"] / 
                       self.metrics["total_requests"] * 100 
                       if self.metrics["total_requests"] > 0 else 0)
        
        print(f"\n=== 路由系统监控报告 ({now}) ===")
        print(f"总请求数: {self.metrics['total_requests']}")

        print(f"总成本: ${self.metrics['total_cost']:.4f}")

                  f"错误 {endpoint.error_count} "
                  f"延迟 {endpoint.avg_latency:.3f}s")
        
        print("=" * 50)
```

```python
class MultiModelOptimizer:

    @staticmethod
    def optimize_cost_performance_balance(endpoints: List[ModelEndpoint], 
                                         historical_data: List[Dict]) -> Dict[str, float]:
        """优化成本-性能平衡"""
        
        # 分析历史数据
        endpoint_performance = {}
        
        for endpoint in endpoints:
            # 收集该端点的历史请求
            endpoint_data = [
                d for d in historical_data
                if d.get("endpoint") == endpoint.name
            ]
            
            if not endpoint_data:
                continue
            
            # 计算关键指标
            total_requests = len(endpoint_data)
            successful = sum(1 for d in endpoint_data if d.get("success"))
            total_latency = sum(d.get("latency", 0) for d in endpoint_data)
            total_cost = sum(d.get("cost", 0) for d in endpoint_data)
            
            success_rate = successful / total_requests if total_requests > 0 else 0
            avg_latency = total_latency / successful if successful > 0 else 0
            avg_cost = total_cost / total_requests if total_requests > 0 else 0
            
            # 计算质量分数
            # 综合考虑成功率、延迟、成本
            quality_score = (
                success_rate * 0.4 +  # 成功率权重40%
                (1 / (avg_latency + 0.001)) * 0.3 +  # 延迟权重30%（延迟越低越好）
                (1 / (avg_cost + 0.001)) * 0.3  # 成本权重30%（成本越低越好）
            )
            
            endpoint_performance[endpoint.name] = {
                "success_rate": success_rate,
                "avg_latency": avg_latency,
                "avg_cost": avg_cost,
                "quality_score": quality_score,
                "total_requests": total_requests
            }
        
        # 生成优化建议
        recommendations = {}
        
        # 1. 识别性能瓶颈
        for name, perf in endpoint_performance.items():

                    "suggestion": "考虑使用更经济的模型或优化token使用"
                }
        
        # 2. 调整路由权重
        if endpoint_performance:
            # 基于质量分数计算新权重
            total_score = sum(p["quality_score"] for p in endpoint_performance.values())
            
            new_weights = {}
            for name, perf in endpoint_performance.items():

                "suggestion": f"根据性能调整权重: {new_weights}",
                "new_weights": new_weights
            }
        
        return recommendations
    
    @staticmethod
    def analyze_traffic_patterns(historical_data: List[Dict], 
                                time_window_hours: int = 24) -> Dict[str, Any]:
        """分析流量模式"""
        
        if not historical_data:
            return {"error": "没有历史数据"}
        
        # 转换为DataFrame便于分析
        import pandas as pd
        
        df_data = []
        for record in historical_data:
            if "timestamp" in record:
                df_data.append({
                    "timestamp": pd.to_datetime(record["timestamp"], unit='s'),
                    "endpoint": record.get("endpoint"),
                    "success": record.get("success", False),
                    "latency": record.get("latency", 0),
                    "cost": record.get("cost", 0)
                })
        
        if not df_data:
            return {"error": "数据格式不正确"}
        
        df = pd.DataFrame(df_data)
        
        # 按时间窗口分析
        df["hour"] = df["timestamp"].dt.hour
        
        patterns = {
            "hourly_traffic": {},
            "endpoint_popularity": {},
            "peak_hours": [],
            "quiet_hours": []
        }
        
        # 小时流量分布
        hourly_counts = df.groupby("hour").size()
        patterns["hourly_traffic"] = hourly_counts.to_dict()
        
        # 端点使用频率
        endpoint_counts = df["endpoint"].value_counts()
        patterns["endpoint_popularity"] = endpoint_counts.to_dict()
        
        # 识别高峰时段
        if not hourly_counts.empty:
            avg_traffic = hourly_counts.mean()
            peak_threshold = avg_traffic * 1.5
            
            peak_hours = hourly_counts[hourly_counts > peak_threshold].index.tolist()
            patterns["peak_hours"] = peak_hours
            
            # 识别低峰时段
            quiet_threshold = avg_traffic * 0.5
            quiet_hours = hourly_counts[hourly_counts < quiet_threshold].index.tolist()
            patterns["quiet_hours"] = quiet_hours
        
        return patterns

        """生成自动扩缩容建议"""

        quiet_hours = patterns.get("quiet_hours", [])

        current_hour = datetime.now().hour
        
        if current_hour in peak_hours:

                    "endpoint": endpoint,
                    "action": "scale_up",
                    "reason": f"当前时段 {current_hour}:00 是高峰时段",
                    "suggested_capacity": int(capacity * 1.5)  # 扩容50%
                })
        
        # 低峰时段缩容建议
        elif current_hour in quiet_hours:
            for endpoint, capacity in current_capacity.items():
                if capacity > 1:  # 至少保留1个实例
                    recommendations.append({
                        "endpoint": endpoint,
                        "action": "scale_down",
                        "reason": f"当前时段 {current_hour}:00 是低峰时段",
                        "suggested_capacity": max(1, int(capacity * 0.5))  # 缩容50%
                    })
        
        return recommendations
```

## 总结：多模型混合使用的核心价值

通过本章学习，我们深入了解了LangChain v1.2.22对多模型混合使用的支持。以下是关键要点：

### 1. 多模型策略的核心优势
- **成本优化**：将简单任务路由到低成本模型，复杂任务使用高性能模型
- **性能提升**：根据任务特性选择最擅长的模型
- **可靠性增强**：故障转移机制确保系统高可用性
- **能力扩展**：组合不同模型的优势能力

### 2. LangChain统一抽象的价值
| 功能 | 传统方法复杂度 | LangChain v1.2.22简化 |
|------|---------------|---------------------|
| 故障转移 | 需要手动实现重试逻辑 | ModelFallbackMiddleware自动处理 |
| 智能路由 | 复杂的条件判断和模型选择 | @wrap_model_call装饰器轻松实现 |
| 负载均衡 | 手动管理权重和并发 | WeightedLoadBalancer内置支持 |
| 配置管理 | 分散的配置文件 | 统一的配置化路由器 |

### 3. 企业级架构设计原则
1. **分层路由策略**：
