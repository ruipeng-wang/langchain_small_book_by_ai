## 传统Anthropic API集成挑战

在LangChain出现之前，直接使用Anthropic Python SDK需要处理复杂的API结构和错误处理逻辑。

### 原生Anthropic API调用示例

```python
# 原生Anthropic API调用（传统方式）
import os
import anthropic
from typing import Dict, Any, List, Optional
import time
import json

class NativeAnthropicIntegration:
    """原生Anthropic API集成示例"""
    
    def __init__(self, api_key: str = None):
        # 初始化Anthropic客户端
        self.client = anthropic.Anthropic(api_key=api_key or os.environ.get("ANTHROPIC_API_KEY"))
        if not self.client.api_key:
            raise ValueError("ANTHROPIC_API_KEY环境变量未设置")
    
    def chat_completion(
        self,
        messages: List[Dict[str, str]],
        model: str = "claude-3-5-sonnet-20241022",
        temperature: float = 0.7,
        max_tokens: int = 1000,
        system: str = None,
        stream: bool = False
    ) -> Dict[str, Any]:
        """原生聊天补全调用"""
        try:
            # Anthropic的API结构与OpenAI不同
            params = {
                "model": model,
                "messages": messages,
                "max_tokens": max_tokens,
                "temperature": temperature
            }
            
            # 系统提示处理
            if system:
                params["system"] = system
            
            # 流式和非流式处理
            if stream:
                return self._handle_stream_completion(params)
            else:
                response = self.client.messages.create(**params)
                return {
                    "content": response.content[0].text,
                    "role": response.role,
                    "stop_reason": response.stop_reason,
                    "usage": {
                        "input_tokens": response.usage.input_tokens,
                        "output_tokens": response.usage.output_tokens
                    }
                }
                
        except Exception as e:
            # Anthropic特定的错误处理
            error_message = self._handle_anthropic_error(e)
            raise Exception(f"Anthropic API调用失败: {error_message}")
    
    def _handle_stream_completion(self, params: Dict[str, Any]) -> Dict[str, Any]:
        """处理Anthropic流式响应"""
        full_content = ""
        
        # Anthropic流式API调用
        with self.client.messages.stream(**params) as stream:
            for event in stream:
                if event.type == "content_block_delta":
                    # 内容增量
                    if event.delta.text is not None:
                        content = event.delta.text
                        full_content += content
                        print(content, end="", flush=True)
                elif event.type == "message_delta":
                    # 消息元数据更新
                    print(f"\n使用量: 输入{event.usage.input_tokens}, 输出{event.usage.output_tokens}")
        
        return {"content": full_content, "streamed": True}
    
    def _handle_anthropic_error(self, error) -> str:
        """处理Anthropic特定错误"""
        error_type = type(error).__name__
        
        # Anthropic特定的错误类型
        if error_type == "APIConnectionError":
            return "连接错误，请检查网络"
        elif error_type == "RateLimitError":
            return "速率限制，请稍后重试"
        elif error_type == "AuthenticationError":
            return "认证失败，请检查API密钥"
        elif error_type == "BadRequestError":
            # Anthropic特有的错误信息
            if "max_tokens" in str(error).lower():
                return "max_tokens参数超出模型限制"
            elif "context_length" in str(error).lower():
                return "上下文长度超出限制"
            return "请求参数错误"
        elif error_type == "InternalServerError":
            return "Anthropic服务器内部错误"
        else:
            return str(error)
    
    def count_tokens(self, text: str, model: str = "claude-3-5-sonnet") -> int:
        """计算文本的token数（Anthropic特有功能）"""
        try:
            return self.client.count_tokens(text, model=model)
        except Exception as e:
            print(f"Token计数失败: {e}")
            return len(text) // 4  # 近似估计
    
    def batch_process(self, prompts: List[str], **kwargs) -> List[Dict[str, Any]]:
        """批量处理提示（需要手动实现并发）"""
        import concurrent.futures
        
        results = []
        
        def process_prompt(prompt: str) -> Dict[str, Any]:
            try:
                return self.chat_completion(
                    messages=[{"role": "user", "content": prompt}],
                    **kwargs
                )
            except Exception as e:
                return {"error": str(e)}
        
        # 手动实现线程池
        with concurrent.futures.ThreadPoolExecutor(max_workers=5) as executor:
            future_to_prompt = {
                executor.submit(process_prompt, prompt): prompt 
                for prompt in prompts
            }
            
            for future in concurrent.futures.as_completed(future_to_prompt):
                results.append(future.result())
        
        return results

# 传统实现的问题：
# 1. API结构与OpenAI完全不同，学习成本高
# 2. 流式响应处理复杂，事件类型多样
# 3. 需要手动实现并发控制
# 4. 错误处理需要针对Anthropic特定调整
# 5. 缺少统一的批量处理接口
```

### Anthropic特有功能挑战

```python
# Anthropic特有功能的传统实现
class NativeAnthropicFeatures:
    """Anthropic特有功能的手动实现"""
    
    def __init__(self, client):
        self.client = client
    
    def use_system_prompt(self, system_prompt: str, user_prompt: str) -> str:
        """使用系统提示（Anthropic特有）"""
        try:
            response = self.client.messages.create(
                model="claude-3-5-sonnet",
                system=system_prompt,  # Anthropic特有参数
                messages=[{"role": "user", "content": user_prompt}],
                max_tokens=1000
            )
            return response.content[0].text
        except Exception as e:
            return f"错误: {e}"
    
    def tool_use_with_anthropic(self, tools: List[Dict], prompt: str) -> Dict[str, Any]:
        """Anthropic工具调用（与OpenAI格式不同）"""
        # Anthropic的工具调用格式
        anthropic_tools = []
        for tool in tools:
            anthropic_tools.append({
                "name": tool["function"]["name"],
                "description": tool["function"].get("description", ""),
                "input_schema": tool["function"]["parameters"]
            })
        
        try:
            response = self.client.messages.create(
                model="claude-3-5-sonnet",
                messages=[{"role": "user", "content": prompt}],
                tools=anthropic_tools,
                max_tokens=1000
            )
            
            # 处理工具调用响应
            result = {"content": response.content[0].text}
            
            # 检查是否有工具调用
            if hasattr(response.content[0], 'tool_calls'):
                result["tool_calls"] = response.content[0].tool_calls
            
            return result
        except Exception as e:
            return {"error": str(e)}
    
    def streaming_with_events(self, prompt: str):
        """Anthropic特有的流式事件处理"""
        with self.client.messages.stream(
            model="claude-3-5-sonnet",
            messages=[{"role": "user", "content": prompt}],
            max_tokens=1000
        ) as stream:
            for event in stream:
                # 需要处理多种事件类型
                if event.type == "message_start":
                    print(f"消息开始: {event.message.id}")
                elif event.type == "content_block_start":
                    print(f"内容块开始: {event.content_block.type}")
                elif event.type == "content_block_delta":
                    if event.delta.text:
                        print(event.delta.text, end="", flush=True)
                elif event.type == "content_block_stop":
                    print(f"\n内容块结束")
                elif event.type == "message_delta":
                    print(f"\n使用统计: {event.usage}")
                elif event.type == "message_stop":
                    print(f"消息结束")
```

# 智能模型选择器
def select_claude_model_by_task(
    task_description: str, 
    constraints: Dict[str, Any]
) -> str:
    """根据任务需求选择合适的Claude模型"""
    
    # 分析任务需求
    task_needs = {
        "complexity": "low",  # low, medium, high
        "context_length": 0,
        "needs_reasoning": False,
        "needs_coding": False,
        "budget": "medium"  # low, medium, high
    }
    
    # 简单任务分析
    if "简单" in task_description or "日常" in task_description:
        task_needs["complexity"] = "low"
    elif "复杂" in task_description or "分析" in task_description:
        task_needs["complexity"] = "high"
        task_needs["needs_reasoning"] = True
    
    if "代码" in task_description or "编程" in task_description:
        task_needs["needs_coding"] = True
    
    # 根据约束选择模型
    if constraints.get("budget") == "low" or task_needs["complexity"] == "low":
        return "claude-3-haiku-20240307"
    
    if task_needs["needs_coding"]:
        # 编码任务优先使用Claude 3.5 Sonnet
        return "claude-3-5-sonnet-20241022"
    
    if task_needs["complexity"] == "high" and constraints.get("budget") == "high":
        return "claude-3-opus-20240229"
    
    # 默认选择
    return "claude-3-5-sonnet-20241022"

from langchain_anthropic import ChatAnthropic
from langchain.text_splitter import RecursiveCharacterTextSplitter
from langchain_core.documents import Document

class LongContextProcessor:
    """长上下文处理器（针对Anthropic优化）"""
    
    def __init__(self, model_name: str = "claude-3-5-sonnet"):
        self.model = ChatAnthropic(model=model_name)
        self.max_context = 200000  # Claude最大上下文
        
        # 文本分割器（针对Claude优化）
        self.text_splitter = RecursiveCharacterTextSplitter(
            chunk_size=50000,  # 每个块的大小
            chunk_overlap=5000,  # 块之间的重叠
            length_function=len,
            separators=["\n\n", "\n", "。", "，", " ", ""]
        )
    
    def process_long_document(self, document_text: str, query: str) -> str:
        """处理长文档"""
        # 1. 检查文档长度
        estimated_tokens = len(document_text) // 4  # 粗略估计
        
        if estimated_tokens <= self.max_context * 0.8:
            # 文档可以一次性处理
            return self._process_with_full_context(document_text, query)
        else:
            # 需要分块处理
            return self._process_with_chunking(document_text, query)
    
    def _process_with_full_context(self, document_text: str, query: str) -> str:
        """使用完整上下文处理"""
        prompt = f"""
文档内容：
{document_text}

问题：
{query}

请基于文档内容回答问题。
"""
        
        response = self.model.invoke(prompt)
        return response.content
    
    def _process_with_chunking(self, document_text: str, query: str) -> str:
        """使用分块处理"""
        # 分割文档
        documents = self.text_splitter.create_documents([document_text])
        
        # 处理每个块
        chunk_summaries = []
        for i, doc in enumerate(documents):
            chunk_prompt = f"""
文档块 {i+1}/{len(documents)} 内容：
{doc.page_content}

请总结这个文档块的核心内容（不超过200字）。
"""
            
            chunk_response = self.model.invoke(chunk_prompt)
            chunk_summaries.append(f"块{i+1}: {chunk_response.content}")
        
        # 基于总结回答原始问题
        summary_text = "\n".join(chunk_summaries)
        final_prompt = f"""
原始问题：
{query}

以下是长文档的分块总结：
{summary_text}

请基于这些总结回答问题。
"""
        
        final_response = self.model.invoke(final_prompt)
        return final_response.content

class AnthropicService:
    """Anthropic服务层"""
    
    def __init__(self, config: AnthropicConfig = None):
        self.config = config or AnthropicConfig.from_environment()
        self._models_cache = {}
        self._metrics = {
            "total_calls": 0,
            "successful_calls": 0,
            "failed_calls": 0,
            "total_input_tokens": 0,
            "total_output_tokens": 0
        }
    
    def get_model(self, model_type: str = "default", **kwargs) -> ChatAnthropic:
        """获取模型实例（带缓存）"""
        model_name = self.config.default_model if model_type == "default" else model_type
        
        cache_key = f"{model_name}_{hash(str(kwargs))}"
        if cache_key in self._models_cache:
            return self._models_cache[cache_key]
        
        # 创建新模型实例
        model = ChatAnthropic(
            model=model_name,
            api_key=self.config.api_key,
            temperature=kwargs.get("temperature", 0.7),
            max_tokens=kwargs.get("max_tokens", 1000),
            timeout=kwargs.get("timeout", 30),
            max_retries=kwargs.get("max_retries", 3),
            # Anthropic特有参数
            system=kwargs.get("system", None)
        )
        
        self._models_cache[cache_key] = model
        return model
    
    def invoke_with_monitoring(self, prompt: str, **kwargs) -> str:
        """带监控的模型调用"""
        start_time = time.time()
        
        try:
            model = self.get_model(kwargs.get("model_type", "default"), **kwargs)
            response = model.invoke(prompt)
            
            # 记录成功指标
            self._record_success(response, start_time)
            
            return response.content
            
        except Exception as e:
            # 记录失败指标
            self._record_failure(start_time)
            
            # 尝试备用模型
            if kwargs.get("enable_fallback", True):
                print(f"主模型调用失败，尝试备用模型: {e}")
                return self._try_fallback_model(prompt, **kwargs)
            else:
                raise e
    
    def _record_success(self, response, start_time: float):
        """记录成功调用指标"""
        end_time = time.time()
        latency = end_time - start_time
        
        self._metrics["total_calls"] += 1
        self._metrics["successful_calls"] += 1
        
        # 提取token使用量
        if hasattr(response, 'response_metadata'):
            metadata = response.response_metadata
            input_tokens = metadata.get('token_usage', {}).get('prompt_tokens', 0)
            output_tokens = metadata.get('token_usage', {}).get('completion_tokens', 0)
            
            self._metrics["total_input_tokens"] += input_tokens
            self._metrics["total_output_tokens"] += output_tokens
        
        # 记录到日志/监控系统
        self._log_metrics("success", latency)
    
    def _record_failure(self, start_time: float):
        """记录失败调用指标"""
        end_time = time.time()
        latency = end_time - start_time
        
        self._metrics["total_calls"] += 1
        self._metrics["failed_calls"] += 1
        
        self._log_metrics("failure", latency)
    
    def _try_fallback_model(self, prompt: str, **kwargs) -> str:
        """尝试备用模型"""
        fallback_kwargs = kwargs.copy()
        fallback_kwargs["model_type"] = self.config.fallback_model
        
        try:
            model = self.get_model(self.config.fallback_model, **fallback_kwargs)
            response = model.invoke(prompt)
            return response.content
        except Exception as e:
            raise Exception(f"备用模型也失败: {e}")
    
    def _log_metrics(self, status: str, latency: float):
        """记录指标到日志"""
        log_entry = {
            "timestamp": datetime.now().isoformat(),
            "environment": self.config.environment,
            "status": status,
            "latency": latency,
            "metrics": self._metrics.copy()
        }
        
        # 实际应用中：发送到ELK、Prometheus等

        success_rate = (
            self._metrics["successful_calls"] / self._metrics["total_calls"] * 100
            if self._metrics["total_calls"] > 0 else 0
        )
        
        avg_input_tokens = (
            self._metrics["total_input_tokens"] / self._metrics["successful_calls"]
            if self._metrics["successful_calls"] > 0 else 0
        )
        
        avg_output_tokens = (
            self._metrics["total_output_tokens"] / self._metrics["successful_calls"]
            if self._metrics["successful_calls"] > 0 else 0
        )
        
        return {
            "environment": self.config.environment,
            "total_calls": self._metrics["total_calls"],
            "success_rate": f"{success_rate:.1f}%",
            "failed_calls": self._metrics["failed_calls"],

            "avg_input_tokens_per_call": f"{avg_input_tokens:.0f}",

        }

# LangChain v1.2.22 Anthropic成本优化
from langchain_anthropic import ChatAnthropic
from typing import Dict, Any, Tuple
import re

class AnthropicCostOptimizer:
    """Anthropic成本优化器"""
    
    # Anthropic定价（美元/每1000个token）
    PRICING = {
        "claude-3-opus": {"input": 0.015, "output": 0.075},
        "claude-3-5-sonnet": {"input": 0.003, "output": 0.015},
        "claude-3-sonnet": {"input": 0.003, "output": 0.015},
        "claude-3-haiku": {"input": 0.00025, "output": 0.00125}
    }
    
    def __init__(self):
        self.token_estimator = TokenEstimator()
    
    def estimate_cost(self, model_name: str, input_text: str, output_text: str = "") -> Tuple[float, Dict[str, Any]]:
        """估算调用成本"""
        # 提取模型系列
        model_series = self._extract_model_series(model_name)
        pricing = self.PRICING.get(model_series, self.PRICING["claude-3-haiku"])
        
        # 估算token数
        input_tokens = self.token_estimator.estimate_tokens(input_text)
        output_tokens = self.token_estimator.estimate_tokens(output_text) if output_text else 1000  # 默认假设
        
        # 计算成本
        input_cost = (input_tokens * pricing["input"]) / 1000
        output_cost = (output_tokens * pricing["output"]) / 1000
        total_cost = input_cost + output_cost
        
        details = {
            "model_series": model_series,
            "input_tokens": input_tokens,
            "output_tokens": output_tokens,
            "input_cost": input_cost,
            "output_cost": output_cost,
            "pricing": pricing
        }
        
        return total_cost, details
    
    def optimize_model_selection(self, task_description: str, input_text: str, budget: float = 0.01) -> Dict[str, Any]:
        """优化模型选择以控制成本"""
        
        # 分析任务复杂度
        complexity = self._analyze_task_complexity(task_description, input_text)
        
        # 根据复杂度选择候选模型
        candidates = []
        
        if complexity["level"] == "low":
            candidates = ["claude-3-haiku", "claude-3-sonnet"]
        elif complexity["level"] == "medium":
            candidates = ["claude-3-sonnet", "claude-3-5-sonnet"]
        else:  # high
            candidates = ["claude-3-5-sonnet", "claude-3-opus"]
        
        # 评估每个候选模型的成本
        recommendations = []
        for model in candidates:
            cost, details = self.estimate_cost(model, input_text)
            
            if cost <= budget:
                recommendations.append({
                    "model": model,
                    "estimated_cost": cost,
                    "within_budget": True,
                    "details": details
                })
            else:
                recommendations.append({
                    "model": model,
                    "estimated_cost": cost,
                    "within_budget": False,
                    "details": details
                })
        
        # 选择最佳模型（在预算内选择能力最强的）
        best_model = None
        for rec in recommendations:
            if rec["within_budget"]:
                if best_model is None or rec["estimated_cost"] < best_model["estimated_cost"]:
                    best_model = rec
        
        return {
            "task_complexity": complexity,
            "budget": budget,
            "recommendations": recommendations,
            "best_model": best_model
        }
    
    def _extract_model_series(self, model_name: str) -> str:
        """从完整模型名称提取系列"""
        patterns = [
            (r"claude-3-opus", "claude-3-opus"),
            (r"claude-3-5-sonnet", "claude-3-5-sonnet"),
            (r"claude-3-sonnet", "claude-3-sonnet"),
            (r"claude-3-haiku", "claude-3-haiku")
        ]
        
        for pattern, series in patterns:
            if re.search(pattern, model_name):
                return series
        
        return "claude-3-haiku"  # 默认
    
    def _analyze_task_complexity(self, task_description: str, input_text: str) -> Dict[str, Any]:
        """分析任务复杂度"""
        complexity_keywords = {
            "low": ["简单", "日常", "问候", "基本信息"],
            "medium": ["分析", "解释", "总结", "描述"],
            "high": ["复杂", "推理", "论证", "对比", "批判性", "创造性"]
        }
        
        level = "low"
        for keyword_level, keywords in complexity_keywords.items():
            for keyword in keywords:
                if keyword in task_description:
                    level = keyword_level
                    break
        
        # 考虑输入长度
        input_length = len(input_text)
        if input_length > 5000 and level == "low":
            level = "medium"
        elif input_length > 10000:
            level = "high"
        
        return {
            "level": level,
            "input_length": input_length,
            "keywords_found": [k for k in complexity_keywords[level] if k in task_description]
        }

class TokenEstimator:
    """Token估算器"""
    
    def estimate_tokens(self, text: str) -> int:
        """估算文本的token数"""
        # 简单估算：对于英文，大约1个token=4个字符
        # 对于中文，大约1个token=2个字符
        if self._is_mostly_chinese(text):
            return len(text) // 2
        else:
            return len(text) // 4
    
    def _is_mostly_chinese(self, text: str) -> bool:
        """判断文本是否主要为中文"""
        chinese_chars = sum(1 for char in text if '\u4e00' <= char <= '\u9fff')
        return chinese_chars / len(text) > 0.3 if text else False

            "solution": "检查参数格式，特别是system参数和消息格式",
            "retryable": False
        },
        "service_unavailable": {
            "pattern": ["service unavailable", "503", "timeout"],
            "solution": "Anthropic服务暂时不可用，稍后重试",
            "retryable": True,
            "wait_seconds": 30
        }
    }
    
    @classmethod
    def handle_error(cls, error: Exception, model: Optional[ChatAnthropic] = None) -> Dict[str, Any]:
        """处理Anthropic错误"""
        error_msg = str(error).lower()
        error_type_name = type(error).__name__
        
        # 识别错误类型
        identified_type = "unknown_error"
        solution = "未知错误，请查看Anthropic文档"
        retryable = False
        wait_seconds = 0
        
        for error_type, info in cls.ERROR_MAPPINGS.items():
            for pattern in info["pattern"]:
                if pattern.lower() in error_msg or pattern.lower() in error_type_name.lower():
                    identified_type = error_type
                    solution = info["solution"]
                    retryable = info.get("retryable", False)
                    wait_seconds = info.get("wait_seconds", 0)
                    break
        
        # 添加模型特定建议
        if model and identified_type == "context_length_exceeded":
            model_name = getattr(model, 'model_name', 'unknown')
            if 'haiku' in model_name or 'sonnet' in model_name:
                solution += "。Claude 3系列所有模型都支持200K上下文"
        
        return {
            "error_type": identified_type,
            "error_message": error_msg,
            "solution": solution,
            "retryable": retryable,
            "wait_seconds": wait_seconds,
            "original_error": str(error)
        }
    
    @classmethod
    def create_resilient_anthropic_model(cls, base_model: ChatAnthropic, max_retries: int = 2):
        """创建具有弹性的Anthropic模型包装器"""
        # Anthropic的速率限制更严格，重试次数应较少
        
        class ResilientChatAnthropic:
            def __init__(self, model, max_retries):
                self.model = model
                self.max_retries = max_retries
            
            def invoke(self, input, **kwargs):
                for attempt in range(self.max_retries):
                    try:
                        return self.model.invoke(input, **kwargs)
                    except Exception as e:
                        error_info = cls.handle_error(e, self.model)
                        print(f"尝试 {attempt + 1}/{self.max_retries} 失败: {error_info['error_type']}")
                        
                        if not error_info["retryable"]:
                            raise e
                        
                        # Anthropic需要更长的退避时间
                        wait_time = error_info.get("wait_seconds", min(30 * (attempt + 1), 300))
                        print(f"等待 {wait_time} 秒后重试...")
                        time.sleep(wait_time)
                
                raise Exception(f"所有{self.max_retries}次重试都失败")
        
        return ResilientChatAnthropic(base_model, max_retries)
