# 9.2.4 自定义代理开发

## 概念讲解

自定义代理开发是LangChain框架最强大的特性之一，它允许开发者根据特定业务需求创建高度定制化的智能代理。在LangChain v1.2.22中，自定义代理开发经历了重要的范式转变，从传统的继承模式演进为基于中间件和状态机的现代架构。

### 自定义代理的设计哲学

自定义代理的核心思想是"可组合性"和"可扩展性"：

1. **模块化设计**：将代理功能分解为独立的中间件组件
2. **声明式配置**：通过配置而非代码定义代理行为
3. **运行时适应性**：根据上下文动态调整代理能力
4. **渐进式增强**：从简单代理开始，逐步增加复杂性

### LangChain v1.2.22中的架构演进

**传统架构（v0.x）：**
```python
from langchain.agents import BaseAgent
# 需要继承基类，重写大量方法
```

**现代架构（v1.2.22）：**
```python
from langchain.agents import create_agent
from langchain.agents.middleware import wrap_tool_call
# 通过中间件和配置实现自定义
```

关键改进：
- **中间件优先**：通过可插拔中间件扩展功能
- **配置驱动**：声明式代理定义，减少样板代码
- **组合式设计**：复用现有组件构建新代理
- **深度集成**：与LangGraph状态机无缝集成

## 核心要点

### 1. 自定义代理的核心机制

#### 中间件系统
中间件是现代自定义代理的核心扩展机制：

```python
from langchain.agents.middleware import (
    wrap_tool_call,      # 包装工具调用
    wrap_model_call,     # 包装模型调用
    AgentMiddleware      # 基础中间件类
)
```

#### 上下文架构
```python
from typing import TypedDict
from langchain.agents import create_agent

class AgentContext(TypedDict):
    """代理上下文定义"""
    user_role: str      # 用户角色
    permissions: list   # 权限列表
    environment: dict   # 环境变量

agent = create_agent(
    model=model,
    tools=tools,
    context_schema=AgentContext,  # 上下文架构
    middleware=[custom_middleware]
)
```

### 2. 自定义代理的三种模式

#### 模式1：中间件扩展
通过中间件修改现有代理行为：
```python
@wrap_tool_call
def validate_tool_inputs(request, handler):
    """验证工具输入中间件"""
    # 在工具调用前验证参数
    return handler(request)
```

#### 模式2：子代理组合
将现有代理包装为子代理：
```python
from deepagents import CompiledSubAgent

subagent = CompiledSubAgent(
    name="data-analyzer",
    description="专业数据分析代理",
    runnable=data_analysis_agent
)
```

#### 模式3：自定义状态机
从头构建LangGraph状态机：
```python
from langgraph.graph import StateGraph

workflow = StateGraph(CustomState)
# 自定义节点和边
```

### 3. 自定义代理的关键组件

#### 工具过滤中间件
```python
def context_based_tools(context: AgentContext):
    """基于上下文的工具过滤"""
    # 根据用户角色动态选择可用工具
    pass
```

#### 错误处理中间件
```python
@wrap_tool_call
def handle_tool_errors(request, handler):
    """工具错误处理中间件"""
    try:
        return handler(request)
    except Exception as e:
        # 返回友好的错误消息
        return ToolMessage(
            content=f"工具执行出错: {str(e)}",
            tool_call_id=request.tool_call["id"]
        )
```

## 简单示例

### 示例1：基础自定义代理（中间件扩展）

```python
from langchain.agents import create_agent
from langchain.agents.middleware import wrap_tool_call, wrap_model_call
from langchain.messages import ToolMessage
from langchain.tools import tool
from langchain.chat_models import init_chat_model
from typing import TypedDict, List

# 1. 定义上下文架构
class SecurityContext(TypedDict):
    """安全上下文"""
    user_id: str
    user_role: str  # admin, user, guest
    allowed_tools: List[str]
    session_ip: str

# 2. 定义工具
@tool
def read_data(query: str) -> str:
    """读取数据（公开工具）"""
    return f"读取数据: {query}"

@tool
def write_data(data: str) -> str:
    """写入数据（需要权限）"""
    return f"写入数据: {data}"

@tool
def delete_data(id: str) -> str:
    """删除数据（管理员专用）"""
    return f"删除数据ID: {id}"

# 3. 创建基于角色的工具过滤中间件
def role_based_tool_filter(context: SecurityContext):
    """基于角色的工具过滤中间件"""
    
    @wrap_tool_call
    def middleware(request, handler):
        # 获取当前用户角色
        user_role = context.get("user_role", "guest")
        
        # 根据角色过滤工具
        if user_role == "admin":
            # 管理员可以使用所有工具
            tools = request.tools
        elif user_role == "user":
            # 普通用户不能使用delete_data
            tools = [t for t in request.tools if t.name != "delete_data"]
        else:  # guest
            # 访客只能使用read_data
            tools = [t for t in request.tools if t.name == "read_data"]
        
        # 更新请求
        request = request.override(tools=tools)
        return handler(request)
    
    return middleware

# 4. 创建日志记录中间件
@wrap_model_call
def log_agent_decisions(request, handler):
    """记录代理决策中间件"""
    import logging
    logger = logging.getLogger(__name__)
    
    # 记录请求信息
    logger.info(f"代理收到请求: {request.messages[-1].content if request.messages else '无消息'}")
    
    # 处理请求
    response = handler(request)
    
    # 记录响应信息
    if hasattr(response, 'tool_calls') and response.tool_calls:
        for tool_call in response.tool_calls:
            logger.info(f"代理调用工具: {tool_call['name']} with args: {tool_call['args']}")
    
    return response

# 5. 初始化模型和工具
model = init_chat_model("gpt-4-turbo-preview")
tools = [read_data, write_data, delete_data]

# 6. 创建自定义代理
security_agent = create_agent(
    model=model,
    tools=tools,
    system_prompt="""你是一个安全敏感的助手，需要根据用户角色提供不同的功能。
    
    注意：某些工具可能需要特定权限才能使用。
    请确保只使用用户有权使用的工具。""",
    context_schema=SecurityContext,
    middleware=[
        role_based_tool_filter,  # 工具过滤中间件
        log_agent_decisions      # 日志记录中间件
    ]
)

# 7. 测试不同角色的代理行为
print("=== 测试管理员角色 ===")
admin_context = {
    "user_id": "admin_001",
    "user_role": "admin",
    "allowed_tools": ["read_data", "write_data", "delete_data"],
    "session_ip": "192.168.1.100"
}

admin_result = security_agent.invoke(
    {
        "messages": [{"role": "user", "content": "请删除ID为123的数据，然后读取所有数据"}],
        "context": admin_context
    }
)
print(f"管理员响应: {admin_result['messages'][-1].content}")

print("\n=== 测试普通用户角色 ===")
user_context = {
    "user_id": "user_001",
    "user_role": "user",
    "allowed_tools": ["read_data", "write_data"],
    "session_ip": "192.168.1.101"
}

user_result = security_agent.invoke(
    {
        "messages": [{"role": "user", "content": "请删除ID为123的数据"}],
        "context": user_context
    }
)
print(f"普通用户响应: {user_result['messages'][-1].content}")
```

```python
from deepagents import create_deep_agent, CompiledSubAgent
from langchain.agents import create_agent
from langchain.tools import tool

import json

# 1. 创建专业化子代理
# 数据分析子代理
@tool
def analyze_dataset(dataset_name: str) -> str:

    analysis_results = {
        "customer_data": "数据集包含1000条记录，5个特征，完整率98%",
        "sales_data": "月度销售数据，趋势向上，季节性明显",

def generate_report(analysis: str, format: str = "text") -> str:
    """生成分析报告"""
    return f"{format.upper()}格式报告: {analysis}"

data_analysis_agent = create_agent(
    model=init_chat_model("gpt-4-turbo-preview"),
    tools=[analyze_dataset, generate_report],
    system_prompt="你是数据分析专家，负责分析数据集和生成报告。"
)

# 2. 创建可视化子代理
@tool
def create_chart(data: str, chart_type: str) -> str:
    """创建图表"""
    return f"创建{chart_type}图表: {data[:50]}..."

@tool
def format_visualization(chart_data: str, style: str) -> str:
    """格式化可视化"""
    return f"{style}风格的可视化: {chart_data}"

visualization_agent = create_agent(
    model=init_chat_model("gpt-4-turbo-preview"),
    tools=[create_chart, format_visualization],
    system_prompt="你是数据可视化专家，负责创建美观的图表和可视化。"
)

# 3. 将子代理包装为CompiledSubAgent
data_subagent = CompiledSubAgent(
    name="data-analyzer",
    description="专业数据分析代理，擅长数据集分析和报告生成",
    runnable=data_analysis_agent
)

viz_subagent = CompiledSubAgent(
    name="visualization-expert",
    description="数据可视化专家，创建各种图表和可视化效果",
    runnable=visualization_agent
)

# 4. 创建主代理（协调多个子代理）
@tool

    """搜索互联网获取信息"""
    return f"搜索结果: {query}"

# 使用Deep Agents创建协调代理
coordinator_agent = create_deep_agent(
    model=init_chat_model("claude-sonnet-4-5-20250929"),
    tools=[search_internet],
    system_prompt="""你是数据分析项目的协调员，负责：
    1. 理解用户的数据分析需求
    2. 将任务分发给合适的专家子代理
    3. 整合各专家的工作成果
    4. 提供最终的综合报告
    
    可用的专家子代理：
    - data-analyzer: 数据分析专家
    - visualization-expert: 可视化专家
    
    根据任务类型选择合适的专家。""",
    subagents=[data_subagent, viz_subagent]
)

# 5. 测试多代理协作
print("=== 测试多代理协作 ===")
task = """
我需要分析客户数据集，了解用户行为模式。
请完成以下任务：
1. 分析customer_data数据集
2. 基于分析结果创建可视化图表
3. 生成综合报告
"""

result = coordinator_agent.invoke({
    "messages": [{"role": "user", "content": task}]
})

print("协调员响应:")
print(result['messages'][-1].content)

# 检查是否有子代理调用
for msg in result['messages']:
    if hasattr(msg, 'tool_calls') and msg.tool_calls:
        for tool_call in msg.tool_calls:
            if tool_call.get('name') in ['data-analyzer', 'visualization-expert']:
                print(f"\n调用了子代理: {tool_call['name']}")
                print(f"参数: {tool_call.get('args', {})}")
```

```python
                "required_capabilities": [

                    AgentCapability.REPORT_GENERATION
                ],
                "default_middleware": ["security", "logging", "caching"]
            },
            # 更多模板...
        }
    
    def create_custom_agent(
        self,
        profile: AgentProfile,
        custom_tools: Optional[List[Any]] = None,
        custom_middleware: Optional[List[str]] = None,
        context_schema: Optional[Any] = None
    ) -> Any:
        """创建自定义代理"""
        
        # 1. 选择基础模板
        template_name = self._select_template(profile.capabilities)
        template = self.agent_templates[template_name]
        
        # 2. 准备工具
        tools = self._prepare_tools(profile, custom_tools)
        
        # 3. 准备中间件
        middleware = self._prepare_middleware(
            template["default_middleware"],
            custom_middleware,
            profile
        )
        
        # 4. 创建代理
        agent = create_agent(
            model=self.model_registry[template["model"]],
            tools=tools,
            system_prompt=self._customize_prompt(template["system_prompt"], profile),
            middleware=middleware,
            context_schema=context_schema,
            checkpointer=self.checkpointer
        )
        
        # 5. 添加监控（如果启用）
        if self.enable_monitoring:
            agent = self._wrap_with_monitoring(agent, profile)
        
        self.logger.info(f"创建自定义代理: {profile.name}")
        return agent
    
    def _select_template(self, capabilities: List[AgentCapability]) -> str:
        """选择最匹配的模板"""
        # 简化的模板选择逻辑
        if AgentCapability.CODE_EXECUTION in capabilities:
            return "data_scientist"
        elif AgentCapability.VISUALIZATION in capabilities:
            return "business_analyst"
        else:
            return "data_scientist"  # 默认模板
    
    def _prepare_tools(
        self,
        profile: AgentProfile,
        custom_tools: Optional[List[Any]]
    ) -> List[Any]:
        """准备工具列表"""
        tools = []
        
        # 根据能力选择工具
        if AgentCapability.DATA_ANALYSIS in profile.capabilities:
            tools.append(self.tool_registry.get("data_analysis"))
        
        if AgentCapability.VISUALIZATION in profile.capabilities:
            tools.append(self.tool_registry.get("visualization"))
        
        if AgentCapability.REPORT_GENERATION in profile.capabilities:
            tools.append(self.tool_registry.get("report_generation"))
        
        # 添加自定义工具
        if custom_tools:
            tools.extend(custom_tools)
        
        return [tool for tool in tools if tool is not None]
    
    def _prepare_middleware(
        self,
        default_middleware: List[str],
        custom_middleware: Optional[List[str]],
        profile: AgentProfile
    ) -> List[AgentMiddleware]:
        """准备中间件列表"""
        middleware = []
        
        # 添加默认中间件
        for mw_name in default_middleware:
            if mw_name in self.middleware_registry:
                middleware.append(self.middleware_registry[mw_name])
        
        # 添加自定义中间件

            for mw_name in custom_middleware:
                if mw_name in self.middleware_registry:
                    middleware.append(self.middleware_registry[mw_name])

            middleware.append(rate_limit_mw)
        
        return middleware
    
    def _customize_prompt(self, base_prompt: str, profile: AgentProfile) -> str:
        """自定义提示词"""
        capabilities_text = "\n".join([
            f"- {cap.value}" for cap in profile.capabilities

        你的专长包括：
        {capabilities_text}

        代理名称：{profile.name}
        代理描述：{profile.description}
        
        请根据你的专长提供最专业的服务。"""
    
    def _wrap_with_monitoring(self, agent: Any, profile: AgentProfile) -> Any:

                )
                
                return result
                
            except Exception as e:

                    f"代理执行失败: {profile.name}, "
                    f"耗时: {execution_time:.2f}s, "
                    f"错误: {str(e)}"
                )
                raise
        
        return monitored_agent
```

### 动态代理生成系统

```python
class DynamicAgentGenerator:
    """动态代理生成器"""
    
    def __init__(self, llm, available_tools: List[Any]):
        self.llm = llm
        self.available_tools = available_tools
        self.generated_agents = {}  # 缓存生成的代理
    
    async def generate_agent_for_task(
        self,
        task_description: str,
        constraints: Dict[str, Any]
    ) -> Any:
        """为特定任务生成代理"""
        
        # 检查缓存
        task_hash = hash(f"{task_description}{str(constraints)}")
        if task_hash in self.generated_agents:
            return self.generated_agents[task_hash]
        
        # 1. 分析任务需求
        task_analysis = await self._analyze_task_requirements(
            task_description, constraints
        )
        
        # 2. 选择合适工具
        selected_tools = self._select_tools_for_task(
            task_analysis, self.available_tools
        )
        
        # 3. 生成定制提示词
        custom_prompt = await self._generate_custom_prompt(
            task_description, task_analysis, selected_tools
        )
        
        # 4. 选择中间件
        middleware = self._select_middleware(task_analysis)
        
        # 5. 创建代理
        agent = create_agent(
            model=self.llm,
            tools=selected_tools,
            system_prompt=custom_prompt,
            middleware=middleware
        )
        
        # 缓存代理
        self.generated_agents[task_hash] = agent
        
        return agent
    
    async def _analyze_task_requirements(
        self,
        task_description: str,
        constraints: Dict[str, Any]
    ) -> Dict[str, Any]:
        """分析任务需求"""
        analysis_prompt = f"""分析以下任务的需求：
        
        任务描述：{task_description}
        约束条件：{constraints}
        
        请分析：
        1. 任务类型（数据分析、内容生成、代码编写等）
        2. 需要的专业知识领域
        3. 必要的工具类型
        4. 可能的挑战和注意事项
        
        以JSON格式返回分析结果。"""
        
        response = await self.llm.ainvoke(analysis_prompt)
        return json.loads(response.content)
    
    def _select_tools_for_task(
        self,
        task_analysis: Dict[str, Any],
        available_tools: List[Any]
    ) -> List[Any]:
        """为任务选择工具"""
        required_tool_types = task_analysis.get("required_tool_types", [])
        
        selected_tools = []
        for tool in available_tools:
            tool_description = tool.description.lower()
            
            # 检查工具是否匹配需求
            for tool_type in required_tool_types:
                if tool_type.lower() in tool_description:
                    selected_tools.append(tool)
                    break
        
        return selected_tools
    
    async def _generate_custom_prompt(
        self,
        task_description: str,
        task_analysis: Dict[str, Any],
        selected_tools: List[Any]
    ) -> str:
        """生成定制提示词"""
        tools_description = "\n".join([
            f"- {tool.name}: {tool.description}" 
            for tool in selected_tools
        ])
        
        prompt_generation_request = f"""为以下任务创建代理提示词：
        
        任务：{task_description}
        分析结果：{json.dumps(task_analysis, indent=2, ensure_ascii=False)}
        可用工具：{tools_description}
        
        请生成一个专业的系统提示词，指导代理完成这个特定任务。
        提示词应包括：
        1. 代理的角色定位
        2. 任务的具体要求
        3. 可用的工具和使用建议
        4. 输出格式要求
        5. 注意事项和最佳实践
        
        返回纯文本提示词。"""
        
        response = await self.llm.ainvoke(prompt_generation_request)
        return response.content
    
    def _select_middleware(self, task_analysis: Dict[str, Any]) -> List[str]:
        """选择中间件"""
        middleware = ["logging"]  # 默认包含日志中间件
        
        # 根据任务分析添加中间件
        if task_analysis.get("requires_validation", False):
            middleware.append("validation")
        
        if task_analysis.get("requires_caching", False):
            middleware.append("caching")
        
        if task_analysis.get("sensitive_data", False):
            middleware.append("security")
        
        return middleware
```

### Q1: 何时应该创建自定义代理而不是使用内置代理？

**A:** 考虑以下场景创建自定义代理：

4. **集成需求**：与现有系统深度集成
5. **特殊错误处理**：需要领域特定的错误恢复策略

### Q3: 如何测试自定义代理？

**A:** 多层次测试策略：

```python
class AgentTestSuite:
    """代理测试套件"""
    
    def __init__(self, agent):
        self.agent = agent
    
    async def run_unit_tests(self):
        """单元测试：测试单个组件"""
        tests = [
            self._test_tool_selection,
            self._test_middleware_chain,
            self._test_error_handling,
            self._test_performance
        ]
        
        results = {}
        for test in tests:
            try:
                results[test.__name__] = await test()
            except Exception as e:
                results[test.__name__] = f"失败: {str(e)}"
        
        return results
    
    async def run_integration_tests(self, test_scenarios: List[Dict]):
        """集成测试：测试完整工作流"""
        results = []
        
        for scenario in test_scenarios:
            start_time = datetime.now()
            
            try:
                result = await self.agent.ainvoke(scenario["input"])
                
                # 验证结果
                is_valid = self._validate_result(
                    result, 
                    scenario.get("expected_output")
                )
                
                execution_time = (datetime.now() - start_time).total_seconds()
                
                results.append({
                    "scenario": scenario["name"],
                    "status": "通过" if is_valid else "失败",
                    "execution_time": execution_time,
                    "result": result
                })
                
            except Exception as e:
                results.append({
                    "scenario": scenario["name"],
                    "status": "异常",
                    "error": str(e),
                    "execution_time": (datetime.now() - start_time).total_seconds()
                })
        
        return results
    
    async def run_load_tests(self, concurrent_users: int, duration: int):
        """负载测试：测试并发性能"""
        import asyncio
        
        async def simulate_user(user_id: int):
            """模拟单个用户"""
            tasks_completed = 0
            start_time = datetime.now()
            
            while (datetime.now() - start_time).total_seconds() < duration:
                try:
                    await self.agent.ainvoke({
                        "messages": [{"role": "user", "content": f"测试任务 {user_id}_{tasks_completed}"}]
                    })
                    tasks_completed += 1
                except Exception:
                    pass
                
                await asyncio.sleep(0.1)  # 添加延迟
            
            return tasks_completed
        
        # 并发执行
        tasks = [simulate_user(i) for i in range(concurrent_users)]
        results = await asyncio.gather(*tasks)
        
        return {
            "total_tasks": sum(results),
            "tasks_per_second": sum(results) / duration,
            "concurrent_users": concurrent_users
        }
    
    def _validate_result(self, actual_result, expected_output):
        """验证结果"""
        if expected_output is None:
            return True  # 没有预期输出，总是通过
        
        # 简化的验证逻辑
        # 实际应用中可能需要更复杂的验证
        return str(actual_result).lower() == str(expected_output).lower()
```

```python
class AgentDeploymentManager:
    """代理部署管理器"""
    
    def __init__(self, registry_url: str):
        self.registry_url = registry_url
        self.version_manager = AgentVersionManager()
        self.rollback_manager = RollbackManager()
    
    async def deploy_agent(
        self,
        agent_config: Dict[str, Any],
        environment: str,
        version_tag: str
    ) -> Dict[str, Any]:
        """部署代理"""
        
        # 1. 版本控制
        version_info = await self.version_manager.create_version(
            agent_config, version_tag
        )
        
        # 2. 环境配置
        env_config = self._prepare_environment_config(
            agent_config, environment
        )
        
        # 3. 部署验证

        if not validation_result["success"]:
            raise Exception(f"部署验证失败: {validation_result['errors']}")
        
        # 4. 执行部署
        deployment_result = await self._execute_deployment(env_config)
        
        # 5. 健康检查
        health_status = await self._check_agent_health(
            deployment_result["endpoint"]
        )
        
        return {
            "version": version_info["version"],
            "environment": environment,
            "endpoint": deployment_result["endpoint"],
            "deployment_time": datetime.now().isoformat(),
            "health_status": health_status
        }
    
    async def rollback_agent(
        self,
        agent_name: str,
        environment: str,
        target_version: str
    ) -> Dict[str, Any]:
        """回滚代理版本"""
        
        # 1. 获取当前版本
        current_version = await self.version_manager.get_current_version(
            agent_name, environment
        )
        
        # 2. 执行回滚
        rollback_result = await self.rollback_manager.execute_rollback(
            agent_name, environment, target_version
        )
        
        # 3. 验证回滚
        validation_result = await self._validate_rollback(
            agent_name, environment, target_version
        )
        
        return {
            "agent_name": agent_name,
            "from_version": current_version,
            "to_version": target_version,
            "rollback_time": datetime.now().isoformat(),
            "success": validation_result["success"]
        }
```
