## 传统模板渲染的复杂性

在LangChain等框架出现之前，实现模板渲染需要手动处理字符串拼接、变量替换、转义和安全等问题。

### 手动模板引擎实现

```python
# 传统手动模板引擎实现
import re
import json
from typing import Dict, Any, List, Optional, Callable
from datetime import datetime
import html

class ManualTemplateEngine:
    """手动模板引擎"""
    
    def __init__(self):
        # 变量正则表达式
        self.var_pattern = re.compile(r'\{([^{}]+)\}')
        
        # 条件语句正则
        self.if_pattern = re.compile(r'\{if\s+([^{}]+)\}(.*?)\{else\}(.*?)\{/if\}', re.DOTALL)
        
        # 循环语句正则
        self.for_pattern = re.compile(r'\{for\s+(\w+)\s+in\s+(\w+)\}(.*?)\{/for\}', re.DOTALL)
        
        # 转义映射
        self.escape_map = {
            '<': '<',
            '>': '>',
            '&': '&',
            '"': '"',
            "'": '&#39;'
        }
        
        # 自定义过滤器
        self.filters = {
            'upper': str.upper,
            'lower': str.lower,
            'capitalize': str.capitalize,
            'trim': str.strip,
            'length': len,
            'json': json.dumps
        }
    
    def register_filter(self, name: str, filter_func: Callable):
        """注册自定义过滤器"""
        self.filters[name] = filter_func
    
    def _apply_filters(self, value: Any, filter_str: str) -> Any:
        """应用过滤器"""
        if not filter_str:
            return value
        
        filters = filter_str.split('|')
        result = value
        
        for filter_name in filters:
            filter_name = filter_name.strip()
            
            # 检查是否有参数
            if ':' in filter_name:
                filter_name, args_str = filter_name.split(':', 1)
                args = [arg.strip() for arg in args_str.split(',')]
            else:
                args = []
            
            # 查找过滤器
            if filter_name in self.filters:
                try:
                    if args:
                        result = self.filters[filter_name](result, *args)
                    else:
                        result = self.filters[filter_name](result)
                except Exception as e:
                    raise ValueError(f"过滤器 '{filter_name}' 执行失败: {e}")
            else:
                raise ValueError(f"未知过滤器: {filter_name}")
        
        return result
    
    def _escape_html(self, text: str) -> str:
        """HTML转义"""
        return ''.join(self.escape_map.get(c, c) for c in text)
    
    def render(self, template: str, context: Dict[str, Any], 
               autoescape: bool = True) -> str:
        """渲染模板"""
        
        # 1. 处理循环语句
        def replace_for(match):
            var_name = match.group(1)
            iterable_name = match.group(2)
            loop_body = match.group(3)
            
            if iterable_name not in context:
                return f"<!-- 未找到变量: {iterable_name} -->"
            
            iterable = context[iterable_name]
            if not hasattr(iterable, '__iter__'):
                return f"<!-- {iterable_name} 不是可迭代对象 -->"
            
            results = []
            for item in iterable:
                # 创建循环上下文
                loop_context = context.copy()
                loop_context[var_name] = item
                loop_context['loop'] = {
                    'index': len(results) + 1,
                    'index0': len(results),
                    'first': len(results) == 0,
                    'last': len(results) == len(list(iterable)) - 1,
                    'length': len(list(iterable))
                }
                
                # 渲染循环体
                loop_result = self._render_loop_body(loop_body, loop_context, autoescape)
                results.append(loop_result)
            
            return ''.join(results)
        
        # 处理所有循环
        while self.for_pattern.search(template):
            template = self.for_pattern.sub(replace_for, template)
        
        # 2. 处理条件语句
        def replace_if(match):
            condition_expr = match.group(1)
            true_block = match.group(2)
            false_block = match.group(3) if match.group(3) else ""
            
            # 简单的条件表达式求值
            # 实际实现需要更复杂的表达式解析
            try:
                # 检查变量是否存在且为真
                condition_parts = condition_expr.strip().split()
                if len(condition_parts) == 1:
                    # 简单变量检查
                    var_name = condition_parts[0]
                    condition = bool(context.get(var_name, False))
                else:
                    # 简单表达式（实际应使用eval或表达式引擎）
                    # 这里简化处理
                    condition = False
                    
                    # 支持简单的比较
                    if '==' in condition_expr:
                        left, right = condition_expr.split('==', 1)
                        left = left.strip()
                        right = right.strip().strip("'\"")
                        
                        left_value = context.get(left, left)
                        condition = str(left_value) == right
                    elif '!=' in condition_expr:
                        left, right = condition_expr.split('!=', 1)
                        left = left.strip()
                        right = right.strip().strip("'\"")
                        
                        left_value = context.get(left, left)
                        condition = str(left_value) != right
                    elif '>' in condition_expr:
                        left, right = condition_expr.split('>', 1)
                        left = left.strip()
                        right = right.strip()
                        
                        left_value = context.get(left, 0)
                        right_value = context.get(right, int(right) if right.isdigit() else 0)
                        condition = left_value > right_value
                
                return true_block if condition else false_block
            except Exception as e:
                return f"<!-- 条件表达式错误: {e} -->"
        
        # 处理所有条件语句
        while self.if_pattern.search(template):
            template = self.if_pattern.sub(replace_if, template)
        
        # 3. 处理变量插值（带过滤器）
        def replace_var(match):
            var_expr = match.group(1).strip()
            
            # 检查是否有过滤器
            if '|' in var_expr:
                var_name, filter_str = var_expr.split('|', 1)
                var_name = var_name.strip()
                filter_str = filter_str.strip()
            else:
                var_name = var_expr
                filter_str = ""
            
            # 获取变量值
            if var_name not in context:
                return f"{{{{{var_name}}}}}"  # 保持原样
            
            value = context[var_name]
            
            # 应用过滤器
            if filter_str:
                try:
                    value = self._apply_filters(value, filter_str)
                except Exception as e:
                    return f"<!-- 过滤器错误: {e} -->"
            
            # 转义处理
            if autoescape and isinstance(value, str):
                value = self._escape_html(value)
            
            return str(value)
        
        # 处理所有变量
        template = self.var_pattern.sub(replace_var, template)
        
        return template
    
    def _render_loop_body(self, body: str, context: Dict[str, Any], 
                         autoescape: bool) -> str:
        """渲染循环体（递归处理内部变量）"""
        # 简化实现：直接调用render，但需要避免无限递归
        # 这里只处理变量插值
        def replace_loop_var(match):
            var_expr = match.group(1).strip()
            
            if '|' in var_expr:
                var_name, filter_str = var_expr.split('|', 1)
                var_name = var_name.strip()
                filter_str = filter_str.strip()
            else:
                var_name = var_expr
                filter_str = ""
            
            if var_name not in context:
                return f"{{{{{var_name}}}}}"  # 保持原样
            
            value = context[var_name]
            
            if filter_str:
                try:
                    value = self._apply_filters(value, filter_str)
                except Exception:
                    return f"{{{{{var_expr}}}}}"  # 保持原样
            
            if autoescape and isinstance(value, str):
                value = self._escape_html(value)
            
            return str(value)
        
        return self.var_pattern.sub(replace_loop_var, body)
    
    def validate_template(self, template: str) -> List[str]:
        """验证模板语法"""
        errors = []
        
        # 检查未闭合的标签
        if template.count('{if') != template.count('{/if}'):
            errors.append("条件语句未闭合")
        
        if template.count('{for') != template.count('{/for}'):
            errors.append("循环语句未闭合")
        
        # 检查嵌套深度
        stack = []
        for i, char in enumerate(template):
            if template[i:].startswith('{if'):
                stack.append(('if', i))
            elif template[i:].startswith('{for'):
                stack.append(('for', i))
            elif template[i:].startswith('{/if}'):
                if not stack or stack[-1][0] != 'if':
                    errors.append(f"位置{i}: 未匹配的{{/if}}")
                else:
                    stack.pop()
            elif template[i:].startswith('{/for}'):
                if not stack or stack[-1][0] != 'for':
                    errors.append(f"位置{i}: 未匹配的{{/for}}")
                else:
                    stack.pop()
        
        if stack:
            for tag_type, pos in stack:
                errors.append(f"位置{pos}: 未闭合的{{{tag_type}}}")
        
        # 提取所有变量
        variables = set(self.var_pattern.findall(template))
        
        # 清理变量名（移除过滤器）
        clean_vars = set()
        for var in variables:
            if '|' in var:
                var_name = var.split('|')[0].strip()
                clean_vars.add(var_name)
            else:
                clean_vars.add(var.strip())
        
        return errors, list(clean_vars)

# 自定义过滤器示例
def format_date(value, format_str='%Y-%m-%d'):
    """日期格式化过滤器"""
    if isinstance(value, datetime):
        return value.strftime(format_str)
    elif isinstance(value, str):
        try:
            dt = datetime.fromisoformat(value)
            return dt.strftime(format_str)
        except ValueError:
            return value
    return value

def truncate(value, length=50, suffix='...'):
    """截断文本过滤器"""
    if not isinstance(value, str):
        value = str(value)
    
    if len(value) <= length:
        return value
    
    return value[:length] + suffix
```

### 复杂模板场景的手动处理

对于需要支持继承、包含、宏等高级功能的模板系统，实现更加复杂。

```python
# 传统模板继承系统
import os
from typing import Dict, Any, List, Optional
import re

class ManualTemplateInheritanceSystem:
    """手动模板继承系统"""
    
    def __init__(self, template_dirs: List[str]):
        self.template_dirs = template_dirs
        self.template_cache = {}
        self.blocks = {}
        
        # 模板继承语法
        self.extends_pattern = re.compile(r'\{%\s*extends\s+[\'"](.+?)[\'"]\s*%\}')
        self.block_pattern = re.compile(r'\{%\s*block\s+(\w+)\s*%\}(.*?)\{%\s*endblock\s*%\}', re.DOTALL)
        self.include_pattern = re.compile(r'\{%\s*include\s+[\'"](.+?)[\'"]\s*%\}')
        self.macro_pattern = re.compile(r'\{%\s*macro\s+(\w+)\((.*?)\)\s*%\}(.*?)\{%\s*endmacro\s*%\}', re.DOTALL)
        self.call_pattern = re.compile(r'\{\{\s*(\w+)\((.*?)\)\s*\}\}')
    
    def _load_template(self, template_name: str) -> str:
        """加载模板文件"""
        if template_name in self.template_cache:
            return self.template_cache[template_name]
        
        for template_dir in self.template_dirs:
            template_path = os.path.join(template_dir, template_name)
            
            if os.path.exists(template_path):
                try:
                    with open(template_path, 'r', encoding='utf-8') as f:
                        content = f.read()
                        self.template_cache[template_name] = content
                        return content
                except Exception as e:
                    raise IOError(f"无法读取模板 {template_name}: {e}")
        
        raise FileNotFoundError(f"模板未找到: {template_name}")
    
    def _parse_macros(self, template_content: str, template_name: str) -> Dict[str, Any]:
        """解析宏定义"""
        macros = {}
        
        for match in self.macro_pattern.finditer(template_content):
            macro_name = match.group(1)
            params_str = match.group(2)
            macro_body = match.group(3)
            
            # 解析参数
            params = [p.strip() for p in params_str.split(',')] if params_str else []
            
            macros[macro_name] = {
                'params': params,
                'body': macro_body,
                'template': template_name
            }
        
        return macros
    
    def _render_macro(self, macro: Dict[str, Any], args: List[str], 
                     context: Dict[str, Any]) -> str:
        """渲染宏"""
        # 创建宏的局部上下文
        macro_context = context.copy()
        
        # 绑定参数
        params = macro['params']
        for i, param in enumerate(params):
            if i < len(args):
                # 解析参数表达式（简化处理）
                arg_expr = args[i].strip()
                
                # 简单变量或字面量
                if arg_expr.startswith('"') and arg_expr.endswith('"'):
                    macro_context[param] = arg_expr[1:-1]  # 移除引号
                elif arg_expr.startswith("'") and arg_expr.endswith("'"):
                    macro_context[param] = arg_expr[1:-1]  # 移除引号
                elif arg_expr.isdigit():
                    macro_context[param] = int(arg_expr)
                elif arg_expr in context:
                    macro_context[param] = context[arg_expr]
                else:
                    macro_context[param] = arg_expr
        
        # 渲染宏体（简化：只处理变量插值）
        body = macro['body']
        
        def replace_var(match):
            var_name = match.group(1).strip()
            return str(macro_context.get(var_name, f"{{{{{var_name}}}}}"))
        
        # 简单变量插值
        var_pattern = re.compile(r'\{\{\s*(\w+)\s*\}\}')
        result = var_pattern.sub(replace_var, body)
        
        return result
    
    def _resolve_inheritance(self, template_name: str, 
                           context: Dict[str, Any]) -> str:
        """解析模板继承链"""
        # 加载模板
        template_content = self._load_template(template_name)
        
        # 检查是否继承其他模板
        extends_match = self.extends_pattern.search(template_content)
        
        if extends_match:
            parent_template = extends_match.group(1)
            
            # 解析父模板
            parent_content = self._resolve_inheritance(parent_template, context)
            
            # 提取当前模板的块定义
            current_blocks = {}
            for match in self.block_pattern.finditer(template_content):
                block_name = match.group(1)
                block_content = match.group(2)
                current_blocks[block_name] = block_content
            
            # 提取父模板的块定义
            parent_blocks = {}
            for match in self.block_pattern.finditer(parent_content):
                block_name = match.group(1)
                block_content = match.group(2)
                parent_blocks[block_name] = block_content
            
            # 合并块：当前模板覆盖父模板
            merged_blocks = parent_blocks.copy()
            merged_blocks.update(current_blocks)
            
            # 渲染父模板，用当前模板的块替换
            result = parent_content
            
            for block_name, block_content in merged_blocks.items():
                block_tag = f'{{% block {block_name} %}}'
                end_tag = '{% endblock %}'
                
                # 查找并替换块
                start_pos = result.find(block_tag)
                if start_pos != -1:
                    end_pos = result.find(end_tag, start_pos)
                    if end_pos != -1:
                        before = result[:start_pos]
                        after = result[end_pos + len(end_tag):]
                        result = before + block_content + after
            
            return result
        else:
            # 没有继承，直接返回
            return template_content
    
    def render(self, template_name: str, context: Dict[str, Any]) -> str:
        """渲染模板"""
        # 解析继承链
        resolved_content = self._resolve_inheritance(template_name, context)
        
        # 处理包含指令
        def replace_include(match):
            include_template = match.group(1)
            
            try:
                included_content = self._resolve_inheritance(include_template, context)
                return included_content
            except Exception as e:
                return f"<!-- 包含错误: {e} -->"
        
        while self.include_pattern.search(resolved_content):
            resolved_content = self.include_pattern.sub(replace_include, resolved_content)
        
        # 处理宏调用
        # 首先收集所有宏定义
        all_macros = {}
        
        # 从所有模板中收集宏
        for cached_name, cached_content in self.template_cache.items():
            template_macros = self._parse_macros(cached_content, cached_name)
            all_macros.update(template_macros)
        
        def replace_macro_call(match):
            macro_name = match.group(1)
            args_str = match.group(2)
            
            if macro_name not in all_macros:
                return f"{{{{ {macro_name}({args_str}) }}}}"  # 保持原样
            
            # 解析参数
            args = []
            if args_str:
                # 简单分割，忽略嵌套括号
                current_arg = ""
                paren_depth = 0
                
                for char in args_str:
                    if char == '(':
                        paren_depth += 1
                        current_arg += char
                    elif char == ')':
                        paren_depth -= 1
                        current_arg += char
                    elif char == ',' and paren_depth == 0:
                        args.append(current_arg.strip())
                        current_arg = ""
                    else:
                        current_arg += char
                
                if current_arg:
                    args.append(current_arg.strip())
            
            # 渲染宏
            macro = all_macros[macro_name]
            return self._render_macro(macro, args, context)
        
        while self.call_pattern.search(resolved_content):
            resolved_content = self.call_pattern.sub(replace_macro_call, resolved_content)
        
        # 简单变量插值（最终处理）
        def replace_final_var(match):
            var_name = match.group(1).strip()
            return str(context.get(var_name, f"{{{{{var_name}}}}}"))
        
        var_pattern = re.compile(r'\{\{\s*(\w+)\s*\}\}')
        resolved_content = var_pattern.sub(replace_final_var, resolved_content)
        
        return resolved_content

            }
```

```python
def enterprise_template_system():
    """企业级模板管理系统"""
    
    import hashlib
    from datetime import datetime
    from typing import Dict, Any
    
    class EnterpriseTemplateManager:
        """企业级模板管理器"""
        
        def __init__(self):
            self.templates: Dict[str, Dict] = {}
            self.categories = {}
            self.version_history = {}
            
            # 模板验证规则
            self.validation_rules = {
                "max_length": 10000,
                "allowed_variables": 20,
                "required_variables": [],
                "disallowed_patterns": [r"\{\{.*?\}\}"]  # 不允许双重花括号
            }
        
        def create_template(self, 
                          name: str, 
                          template: str, 
                          category: str = "general",
                          metadata: Dict[str, Any] = None) -> str:
            """创建模板"""
            
            # 验证模板
            validation_result = self._validate_template(template)
            if not validation_result["valid"]:
                raise ValueError(f"模板验证失败: {validation_result['errors']}")
            
            # 提取变量
            variables = self._extract_variables(template)
            
            # 生成模板ID
            template_id = hashlib.md5(f"{name}_{template}".encode()).hexdigest()[:8]
            
            # 创建PromptTemplate
            prompt_template = PromptTemplate.from_template(template)
            
            # 存储模板
            self.templates[template_id] = {
                "id": template_id,
                "name": name,
                "template": prompt_template,
                "original_template": template,
                "variables": variables,
                "category": category,
                "metadata": metadata or {},
                "created_at": datetime.now(),
                "updated_at": datetime.now(),
                "version": 1,
                "usage_count": 0
            }
            
            # 更新分类
            if category not in self.categories:
                self.categories[category] = []
            self.categories[category].append(template_id)
            
            # 记录版本历史
            self.version_history[template_id] = [{
                "version": 1,
                "template": template,
                "timestamp": datetime.now(),
                "changes": "初始版本"
            }]
            
            return template_id
        
        def _validate_template(self, template: str) -> Dict[str, Any]:
            """验证模板"""
            errors = []
            warnings = []
            
            # 检查长度
            if len(template) > self.validation_rules["max_length"]:
                errors.append(f"模板过长 ({len(template)} > {self.validation_rules['max_length']})")
            
            # 提取变量
            variables = self._extract_variables(template)
            
            # 检查变量数量
            if len(variables) > self.validation_rules["allowed_variables"]:
                warnings.append(f"变量数量较多 ({len(variables)} 个)")
            
            # 检查必填变量
            for required_var in self.validation_rules["required_variables"]:
                if required_var not in variables:
                    errors.append(f"缺少必填变量: {required_var}")
            
            # 检查禁止的模式
            import re
            for pattern in self.validation_rules["disallowed_patterns"]:
                if re.search(pattern, template):
                    errors.append(f"包含禁止的模式: {pattern}")
            
            return {
                "valid": len(errors) == 0,
                "errors": errors,
                "warnings": warnings,
                "variables": variables,
                "length": len(template)
            }
        
        def _extract_variables(self, template: str) -> List[str]:
            """提取变量"""
            import re
            variables = re.findall(r'\{(\w+)\}', template)
            return list(set(variables))  # 去重
        
        def get_template(self, template_id: str) -> PromptTemplate:
            """获取模板"""
            if template_id not in self.templates:
                raise ValueError(f"模板未找到: {template_id}")
            
            # 更新使用计数
            self.templates[template_id]["usage_count"] += 1
            self.templates[template_id]["updated_at"] = datetime.now()
            
            return self.templates[template_id]["template"]
        
        def render_template(self, template_id: str, **kwargs) -> str:
            """渲染模板"""
            template = self.get_template(template_id)
            return template.format(**kwargs)
        
        def search_templates(self, 
                           category: str = None,
                           keyword: str = None,
                           min_variables: int = None,
                           max_variables: int = None) -> List[Dict]:
            """搜索模板"""
            results = []
            
            for template_id, template_data in self.templates.items():
                # 分类过滤
                if category and template_data["category"] != category:
                    continue
                
                # 关键词过滤
                if keyword:
                    keyword_lower = keyword.lower()
                    name_match = keyword_lower in template_data["name"].lower()
                    template_match = keyword_lower in template_data["original_template"].lower()
                    
                    if not (name_match or template_match):
                        continue
                
                # 变量数量过滤
                var_count = len(template_data["variables"])
                if min_variables is not None and var_count < min_variables:
                    continue
                if max_variables is not None and var_count > max_variables:
                    continue
                
                results.append({
                    "id": template_id,
                    "name": template_data["name"],
                    "category": template_data["category"],
                    "variables": template_data["variables"],
                    "usage_count": template_data["usage_count"],
                    "created_at": template_data["created_at"],
                    "updated_at": template_data["updated_at"]
                })
            
            return results
        
        def export_templates(self, format: str = "json") -> str:
            """导出模板"""
            export_data = {
                "templates": self.templates,
                "categories": self.categories,
                "export_time": datetime.now().isoformat(),
                "count": len(self.templates)
            }
            
            if format == "json":
                # 转换datetime为字符串
                import json
                from datetime import datetime
                
                class DateTimeEncoder(json.JSONEncoder):
                    def default(self, obj):
                        if isinstance(obj, datetime):
                            return obj.isoformat()
                        return super().default(obj)
                
                return json.dumps(export_data, cls=DateTimeEncoder, indent=2, ensure_ascii=False)
            else:
                raise ValueError(f"不支持的格式: {format}")
        
        def import_templates(self, data: str, format: str = "json"):
            """导入模板"""
            if format == "json":
                import json
                import_data = json.loads(data)
                
                for template_id, template_data in import_data.get("templates", {}).items():
                    # 检查是否已存在
                    if template_id in self.templates:
                        print(f"警告: 模板 {template_id} 已存在，跳过")
                        continue
                    
                    # 重新创建PromptTemplate
                    template_data["template"] = PromptTemplate.from_template(
                        template_data["original_template"]
                    )
                    
                    # 转换字符串为datetime
                    for date_field in ["created_at", "updated_at"]:
                        if date_field in template_data and isinstance(template_data[date_field], str):
                            template_data[date_field] = datetime.fromisoformat(template_data[date_field])
                    
                    self.templates[template_id] = template_data
                
                print(f"导入完成，新增 {len(import_data.get('templates', {}))} 个模板")
            else:
                raise ValueError(f"不支持的格式: {format}")
    
    # 使用企业级模板管理器
    print("企业级模板管理系统示例:")
    print("="*60)
    
    manager = EnterpriseTemplateManager()
    
    # 创建模板
    customer_service_id = manager.create_template(
        name="客户服务响应",

        category="customer_service",
        metadata={
            "author": "客服部",
            "description": "标准客户服务响应模板",
```
