# 7.1.2 条件模板与动态内容

## 条件模板的核心价值：场景适应与个性化体验

在真实的企业AI应用中，单一静态的提示模板往往难以满足复杂多变的业务需求。条件模板（Conditional Templates）通过根据运行时上下文动态调整提示内容，实现了场景自适应与个性化体验。其核心价值体现在三个方面：

1. **场景适应性**：根据用户输入、系统状态、业务规则等条件生成不同风格的提示
2. **个性化体验**：针对不同用户群体、使用场景或设备类型提供定制化交互
3. **效率与成本优化**：通过条件逻辑避免不必要的模型调用，优化Token使用

## 传统条件提示的复杂性：手动拼接与逻辑分散

在LangChain出现之前，开发人员需要手动处理条件逻辑，导致代码重复且难以维护。让我们通过一个电商客服场景来展示传统实现的复杂性：

### 场景需求
电商客服系统需要根据用户问题类型提供不同帮助：
- 订单查询：需要订单号信息
- 商品咨询：需要商品ID和具体问题
- 售后支持：需要问题描述和联系方式
- 紧急投诉：需要立即转人工客服

### 传统手动实现

```python
# 传统条件提示实现 - 繁琐且难以维护
def generate_customer_service_prompt(user_query: str, user_type: str = "普通用户", urgency: str = "一般"):
    """
    根据用户查询和上下文生成客服提示
    问题：逻辑分散、模板硬编码、难以扩展
    """
    # 基础模板
    base_template = """你是一个专业的电商客服助手。请根据用户的问题提供帮助。"""
    
    # 条件逻辑分散在代码各处
    if "订单" in user_query or "物流" in user_query:
        # 订单相关模板
        prompt = f"""{base_template}
用户问题：{user_query}

请按以下步骤处理：
1. 首先询问用户的订单号
2. 根据订单号查询物流状态
3. 提供预计送达时间
"""
        if urgency == "紧急":
            prompt += "\n注意：这是一个紧急查询，需要优先处理！"
            
    elif "商品" in user_query or "产品" in user_query:
        # 商品咨询模板
        prompt = f"""{base_template}
用户问题：{user_query}

请按以下步骤处理：
1. 询问用户具体的商品ID或名称
2. 提供商品详细信息
3. 回答用户的具体问题
"""
        if user_type == "VIP用户":
            prompt += "\n注意：这是VIP用户，请提供更详细的服务！"
            
    elif "售后" in user_query or "退货" in user_query:
        # 售后支持模板
        prompt = f"""{base_template}
用户问题：{user_query}

请按以下步骤处理：
1. 了解具体售后问题
2. 提供解决方案选项
3. 收集用户联系方式以便跟进
"""
    elif "投诉" in user_query or "不满意" in user_query:
        # 投诉处理模板
        prompt = f"""{base_template}
用户问题：{user_query}

注意：这是一个投诉问题，请立即转接人工客服。
请回复：
"您好，我理解您的问题很重要。为了给您更好的服务，我将立即为您转接专业的人工客服。"
"""
    else:
        # 默认通用模板
        prompt = f"""{base_template}
用户问题：{user_query}

请提供专业的帮助。
"""
    
    return prompt

# 测试不同场景
test_queries = [
    ("我的订单什么时候到？", "普通用户", "一般"),
    ("这个商品有货吗？", "VIP用户", "一般"),
    ("我想退货", "普通用户", "紧急"),
    ("我要投诉！", "普通用户", "紧急")
]

for query, user_type, urgency in test_queries:
    print(f"查询：{query}")
    print(f"生成的提示：\n{generate_customer_service_prompt(query, user_type, urgency)}\n{'='*50}")
```

### 传统实现的问题分析

1. **逻辑与模板耦合**：条件判断逻辑与模板内容混合在一起，违反关注点分离原则
2. **代码重复**：每个条件分支中都需要重复基础模板和格式化逻辑
3. **难以维护**：新增场景需要修改复杂的if-else链，容易引入错误
4. **缺乏复用性**：不同条件分支中的相似模板片段无法复用
5. **测试困难**：每个条件分支都需要单独测试，组合条件更是难以覆盖

## LangChain v1.2.22条件模板解决方案：声明式条件逻辑

LangChain通过声明式条件模板系统，将条件逻辑从代码实现中解耦，提供清晰、可维护的条件提示管理方案。让我们探索LangChain v1.2.22中的多种条件模板实现方式。

### 方案二：ChatPromptTemplate + 条件消息选择

对于聊天模型，LangChain提供了更灵活的条件消息管理方案：

```python
from langchain_core.prompts import ChatPromptTemplate, HumanMessagePromptTemplate, SystemMessagePromptTemplate
from langchain_core.messages import AIMessage, HumanMessage, SystemMessage

# 定义条件消息选择函数
def select_messages_based_on_context(context: dict) -> list:
    """根据上下文选择不同的消息序列"""
    messages = []
    
    # 系统消息（始终包含）
    system_content = "你是一个专业的电商客服助手。"
    
    # 根据用户类型添加不同的系统提示
    if context.get("user_type") == "VIP用户":
        system_content += " 当前服务的是VIP用户，请提供最高级别的服务。"
    elif context.get("user_type") == "新用户":
        system_content += " 当前服务的是新用户，请耐心引导并提供详细说明。"
    
    messages.append(SystemMessage(content=system_content))
    
    # 根据紧急程度添加优先级提示
    if context.get("urgency") == "紧急":
        messages.append(SystemMessage(content="注意：这是一个紧急查询，需要优先处理！"))
    
    # 历史对话上下文（条件性包含）
    if context.get("chat_history"):
        messages.extend(context["chat_history"][-3:])  # 只包含最近3条历史
    
    # 当前用户消息
    messages.append(HumanMessage(content=context["user_query"]))
    
    return messages

# 创建动态聊天提示
def create_dynamic_chat_prompt(context: dict):
    """创建基于上下文的动态聊天提示"""
    messages = select_messages_based_on_context(context)
    return ChatPromptTemplate.from_messages(messages)

# 测试动态聊天提示

    },

        "user_query": "我想投诉！",
        "user_type": "普通用户",
        "urgency": "紧急",

            HumanMessage(content="我之前买了个商品"),

```

### 方案三：RunnableBranch - 工作流级别的条件路由

LangChain的RunnableBranch提供了更强大的条件路由能力，可以在整个工作流中实现条件分支：

```python
from langchain_core.runnables import RunnableBranch, RunnableLambda, RunnableParallel
from langchain_openai import ChatOpenAI
from langchain_core.output_parsers import StrOutputParser

# 定义不同的处理分支
def classify_query(user_query: str) -> str:
    """查询分类"""
    user_query_lower = user_query.lower()
    
    if any(keyword in user_query_lower for keyword in ["订单", "物流"]):
        return "order"
    elif any(keyword in user_query_lower for keyword in ["商品", "产品"]):
        return "product"
    elif any(keyword in user_query_lower for keyword in ["售后", "退货"]):
        return "after_sales"
    elif any(keyword in user_query_lower for keyword in ["投诉", "不满意"]):
        return "complaint"
    else:
        return "general"

# 定义不同分支的提示模板
order_prompt = ChatPromptTemplate.from_messages([
    ("system", "你是订单查询专家。"),
    ("human", "用户查询订单：{query}\n请提供专业帮助。")
])

product_prompt = ChatPromptTemplate.from_messages([
    ("system", "你是商品咨询专家。"),
    ("human", "用户查询商品：{query}\n请提供商品详细信息。")
])

after_sales_prompt = ChatPromptTemplate.from_messages([
    ("system", "你是售后支持专家。"),
    ("human", "用户需要售后支持：{query}\n请提供解决方案。")
])

complaint_prompt = ChatPromptTemplate.from_messages([
    ("system", "你是投诉处理专家。"),
    ("human", "用户投诉：{query}\n请妥善处理。")
])

general_prompt = ChatPromptTemplate.from_messages([
    ("system", "你是通用客服助手。"),
    ("human", "用户查询：{query}\n请提供帮助。")
])

# 创建条件分支路由
branch = RunnableBranch(
    (lambda x: classify_query(x["query"]) == "order", order_prompt),
    (lambda x: classify_query(x["query"]) == "product", product_prompt),
    (lambda x: classify_query(x["query"]) == "after_sales", after_sales_prompt),
    (lambda x: classify_query(x["query"]) == "complaint", complaint_prompt),
    general_prompt
)

# 创建完整的工作流链
chain = (
    RunnableParallel({
        "query": lambda x: x["query"],
        "context": lambda x: {"user_type": x.get("user_type", "普通用户")}
    })
    | branch
    | ChatOpenAI(model="gpt-3.5-turbo", temperature=0.7)
    | StrOutputParser()
)

# 测试条件路由工作流

print("=" * 60)

for i, input_data in enumerate(test_inputs, 1):

    # 注意：实际调用需要API密钥
```

### 模式二：动态变量注入

```python
from langchain.prompts import FewShotPromptTemplate

# 动态示例选择
def select_examples_based_on_query(user_query: str, examples: list) -> list:
    """根据查询内容动态选择相关示例"""
    query_lower = user_query.lower()
    
    selected = []
    for example in examples:
        # 基于相似度或关键词匹配选择示例
        if any(keyword in query_lower for keyword in example.get("keywords", [])):
            selected.append(example)
    
    # 确保至少返回一个示例
    return selected[:3] if selected else examples[:1]

# 动态Few-Shot模板
def create_dynamic_fewshot_prompt(user_query: str, context: dict):
    """创建动态Few-Shot提示"""
    
    # 示例库
    example_pool = [
        {
            "input": "我的订单什么时候发货？",
            "output": "请问您的订单号是多少？我来为您查询发货状态。",
            "keywords": ["订单", "发货", "物流"]
        },
        {
            "input": "商品有折扣吗？",
            "output": "目前这款商品正在促销中，可以享受8折优惠。",
            "keywords": ["商品", "折扣", "价格"]
        },
        {
            "input": "怎么退货？",
            "output": "请提供订单号和退货原因，我将为您处理退货申请。",
            "keywords": ["退货", "售后", "退款"]
        }
    ]
    
    # 动态选择示例
    selected_examples = select_examples_based_on_query(user_query, example_pool)
    
    # 创建动态模板
    example_prompt = PromptTemplate(
        input_variables=["input", "output"],
        template="用户：{input}\n助手：{output}"
    )
    
    dynamic_prompt = FewShotPromptTemplate(
        examples=selected_examples,
        example_prompt=example_prompt,
        prefix="你是一个客服助手。请参考以下类似问题的回答方式：",
        suffix="用户：{query}\n助手：",
        input_variables=["query"],
        example_separator="\n\n"
    )
    
    return dynamic_prompt.format(query=user_query)
```

## 企业级条件模板管理系统

### 架构设计

```python
from dataclasses import dataclass
from typing import Dict, List, Optional, Callable
from enum import Enum
import json

class TemplateType(Enum):
    """模板类型枚举"""
    ORDER = "order"
    PRODUCT = "product"
    AFTER_SALES = "after_sales"
    COMPLAINT = "complaint"
    GENERAL = "general"

@dataclass
class TemplateCondition:
    """模板条件定义"""
    condition_id: str
    condition_func: Callable[[Dict], bool]
    template_id: str
    priority: int = 0

@dataclass
class BusinessTemplate:
    """业务模板定义"""
    template_id: str
    template_type: TemplateType
    content: str
    variables: List[str]
    metadata: Dict
    version: str = "1.0"

class EnterpriseTemplateManager:
    """企业级模板管理器"""
    
    def __init__(self):
        self.templates: Dict[str, BusinessTemplate] = {}
        self.conditions: List[TemplateCondition] = []
        self.template_history: Dict[str, List[BusinessTemplate]] = {}
    
    def register_template(self, template: BusinessTemplate):
        """注册模板"""
        self.templates[template.template_id] = template
        
        # 记录版本历史
        if template.template_id not in self.template_history:
            self.template_history[template.template_id] = []
        self.template_history[template.template_id].append(template)
    
    def add_condition(self, condition: TemplateCondition):
        """添加条件"""
        self.conditions.append(condition)
        # 按优先级排序
        self.conditions.sort(key=lambda x: x.priority, reverse=True)
    
    def select_template(self, context: Dict) -> Optional[BusinessTemplate]:
        """根据上下文选择模板"""
        # 按优先级检查所有条件
        for condition in self.conditions:
            try:
                if condition.condition_func(context):
                    return self.templates.get(condition.template_id)
            except Exception as e:
                print(f"条件检查失败: {condition.condition_id}, 错误: {e}")
                continue
        
        # 默认模板
        return self.templates.get("default")
    
    def render_template(self, template_id: str, variables: Dict) -> str:
        """渲染模板"""
        template = self.templates.get(template_id)
        if not template:
            raise ValueError(f"模板不存在: {template_id}")
        
        # 检查必需变量
        missing_vars = [v for v in template.variables if v not in variables]
        if missing_vars:
            raise ValueError(f"缺少必需变量: {missing_vars}")
        
        # 简单渲染（实际应使用安全模板引擎）
        rendered = template.content
        for key, value in variables.items():
            placeholder = "{" + key + "}"
            rendered = rendered.replace(placeholder, str(value))
        
        return rendered
```

### 性能优化与缓存策略

```python
import time
from functools import lru_cache
from typing import Tuple

class OptimizedTemplateManager:
    """优化性能的模板管理器"""
    
    def __init__(self):
        self.template_cache = {}
        self.condition_cache = {}
        self.render_cache = {}
    
    @lru_cache(maxsize=128)
    def _get_template_hash(self, context_tuple: Tuple) -> str:
        """生成上下文哈希用于缓存"""
        # 将字典转换为可哈希的元组
        sorted_items = tuple(sorted(context_tuple.items()))
        return hash(sorted_items)
    
    def select_template_with_cache(self, context: Dict) -> Optional[BusinessTemplate]:
        """带缓存的模板选择"""
        # 生成缓存键
        cache_key = self._get_template_hash(tuple(sorted(context.items())))
        
        # 检查缓存
        if cache_key in self.template_cache:
            return self.template_cache[cache_key]
        
        # 实际选择逻辑
        start_time = time.time()
        template = self.select_template(context)  # 调用基类方法
        elapsed = time.time() - start_time
        
        # 缓存结果（仅当选择时间超过阈值）
        if elapsed > 0.001:  # 1毫秒阈值
            self.template_cache[cache_key] = template
        
        return template
    
    @lru_cache(maxsize=1024)
    def render_with_cache(self, template_id: str, variables_tuple: Tuple) -> str:
        """带缓存的模板渲染"""
        # 将变量字典转换为元组
        variables = dict(variables_tuple)
        return self.render_template(template_id, variables)
```
