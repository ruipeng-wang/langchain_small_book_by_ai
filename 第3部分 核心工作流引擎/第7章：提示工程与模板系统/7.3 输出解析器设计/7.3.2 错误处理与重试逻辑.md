# 7.3.2 错误处理与重试逻辑

## 错误处理与重试逻辑的核心价值：系统稳定性与用户体验

在大语言模型应用开发中，错误处理与重试逻辑（Error Handling and Retry Logic）是确保系统稳定性、提升用户体验和保障数据质量的关键技术。其核心价值体现在：

1. **系统稳定性**：处理模型API调用失败、网络异常、速率限制等临时性问题
2. **数据完整性**：确保即使在部分失败情况下也能获取尽可能完整的数据
3. **用户体验**：通过优雅的降级处理和自动重试减少用户感知的失败
4. **成本优化**：避免因偶发性错误导致的无效API调用和资源浪费
5. **可靠性工程**：建立可观测的错误分类、监控和告警机制
6. **容错能力**：支持故障转移、备用模型和降级策略
7. **业务连续性**：确保核心功能在异常情况下仍能提供服务

## 传统错误处理实现的复杂性：手动重试与降级处理

### 场景需求
智能客服系统需要实时分析客户消息情感并提取关键信息，要求：
1. **情感分析**：识别客户情绪（正面、负面、中性）
2. **问题分类**：自动分类客户问题类型（产品、售后、技术等）
3. **紧急程度评估**：判断问题紧急程度（低、中、高）
4. **关键信息提取**：提取客户姓名、订单号、问题描述等关键信息
5. **可靠性要求**：系统可用性99.9%，平均响应时间<2秒

```python
# 传统错误处理与重试实现 - 繁琐的手动重试和降级逻辑
import time
import random
import json
from typing import Dict, Any, Optional, Tuple, List, Callable
from enum import Enum
from dataclasses import dataclass
from datetime import datetime, timedelta
import logging
import requests
from functools import wraps

# 配置日志
logging.basicConfig(level=logging.INFO)
logger = logging.getLogger(__name__)

class Sentiment(Enum):
    """情感分类"""
    POSITIVE = "positive"
    NEGATIVE = "negative"
    NEUTRAL = "neutral"

class ProblemCategory(Enum):
    """问题分类"""
    PRODUCT = "product"
    AFTER_SALES = "after_sales"
    TECHNICAL = "technical"
    BILLING = "billing"
    GENERAL = "general"

class UrgencyLevel(Enum):
    """紧急程度"""
    LOW = "low"
    MEDIUM = "medium"
    HIGH = "high"

@dataclass
class CustomerMessageAnalysis:
    """客户消息分析结果"""
    sentiment: Sentiment
    category: ProblemCategory
    urgency: UrgencyLevel
    key_info: Dict[str, Any]
    confidence: float
    processing_time_ms: int
    errors: List[str]
    fallback_used: bool

class ModelAPIError(Exception):
    """模型API错误"""
    def __init__(self, message: str, status_code: Optional[int] = None, error_type: str = "api_error"):
        super().__init__(message)
        self.status_code = status_code
        self.error_type = error_type

class RateLimitError(ModelAPIError):
    """速率限制错误"""
    def __init__(self, message: str, retry_after: Optional[int] = None):
        super().__init__(message, status_code=429, error_type="rate_limit")
        self.retry_after = retry_after

class TimeoutError(ModelAPIError):
    """超时错误"""
    def __init__(self, message: str):
        super().__init__(message, error_type="timeout")

class NetworkError(ModelAPIError):
    """网络错误"""
    def __init__(self, message: str):
        super().__init__(message, error_type="network")

class TraditionalErrorHandler:
    """传统错误处理器 - 手动实现重试和降级逻辑"""
    
    def __init__(self, config: Dict[str, Any]):
        self.config = config
        self.stats = self._initialize_stats()
        self.circuit_breaker = self._initialize_circuit_breaker()
    
    def _initialize_stats(self) -> Dict[str, Any]:
        """初始化统计信息"""
        return {
            "total_calls": 0,
            "successful_calls": 0,
            "failed_calls": 0,
            "retry_attempts": 0,
            "circuit_breaker_trips": 0,
            "fallback_used": 0,
            "average_response_time_ms": 0,
            "error_distribution": {}
        }
    
    def _initialize_circuit_breaker(self) -> Dict[str, Any]:
        """初始化熔断器"""
        return {
            "state": "CLOSED",  # CLOSED, OPEN, HALF_OPEN
            "failure_count": 0,
            "last_failure_time": None,
            "open_until": None,
            "success_threshold": 5,
            "failure_threshold": 10,
            "open_timeout_seconds": 60
        }
    
    def _update_circuit_breaker(self, success: bool):
        """更新熔断器状态"""
        current_time = datetime.now()
        
        if not success:
            self.circuit_breaker["failure_count"] += 1
            self.circuit_breaker["last_failure_time"] = current_time
            
            # 检查是否需要打开熔断器
            if (self.circuit_breaker["failure_count"] >= self.circuit_breaker["failure_threshold"] and
                self.circuit_breaker["state"] == "CLOSED"):
                self.circuit_breaker["state"] = "OPEN"
                self.circuit_breaker["open_until"] = current_time + timedelta(
                    seconds=self.circuit_breaker["open_timeout_seconds"]
                )
                self.stats["circuit_breaker_trips"] += 1
                logger.warning(f"熔断器打开，将在 {self.circuit_breaker['open_until']} 后尝试恢复")
        else:
            if self.circuit_breaker["state"] == "HALF_OPEN":
                # 半开状态下成功，关闭熔断器
                self.circuit_breaker["success_count"] = self.circuit_breaker.get("success_count", 0) + 1
                if self.circuit_breaker["success_count"] >= self.circuit_breaker["success_threshold"]:
                    self.circuit_breaker["state"] = "CLOSED"
                    self.circuit_breaker["failure_count"] = 0
                    self.circuit_breaker["success_count"] = 0
                    logger.info("熔断器关闭")
            elif self.circuit_breaker["state"] == "CLOSED":
                # 正常状态下，重置失败计数
                self.circuit_breaker["failure_count"] = max(0, self.circuit_breaker["failure_count"] - 1)
    
    def _check_circuit_breaker(self) -> bool:
        """检查熔断器状态"""
        current_time = datetime.now()
        
        if self.circuit_breaker["state"] == "OPEN":
            if self.circuit_breaker["open_until"] and current_time >= self.circuit_breaker["open_until"]:
                # 熔断器超时，转为半开状态
                self.circuit_breaker["state"] = "HALF_OPEN"
                self.circuit_breaker["success_count"] = 0
                logger.info("熔断器进入半开状态")
                return True
            else:
                # 熔断器仍处于打开状态
                logger.warning("熔断器阻止请求")
                return False
        elif self.circuit_breaker["state"] == "HALF_OPEN":
            # 半开状态允许有限尝试
            return random.random() < 0.5  # 50%概率允许请求
        
        return True  # CLOSED状态允许所有请求
    
    def _exponential_backoff(self, attempt: int) -> float:
        """指数退避延迟计算"""
        base_delay = self.config.get("base_delay_seconds", 1.0)
        max_delay = self.config.get("max_delay_seconds", 60.0)
        backoff_factor = self.config.get("backoff_factor", 2.0)
        
        delay = base_delay * (backoff_factor ** attempt)
        jitter = random.uniform(0, delay * 0.1)  # 添加10%抖动
        
        return min(delay + jitter, max_delay)
    
    def _should_retry(self, error: Exception) -> bool:
        """判断是否应该重试"""
        
        # 可重试的错误类型
        retryable_errors = [
            TimeoutError,
            NetworkError,
            RateLimitError,
            requests.exceptions.Timeout,
            requests.exceptions.ConnectionError,
            requests.exceptions.HTTPError
        ]
        
        # 检查错误类型
        for error_type in retryable_errors:
            if isinstance(error, error_type):
                return True
        
        # 检查HTTP状态码
        if hasattr(error, 'status_code'):
            status_code = getattr(error, 'status_code')
            if status_code in [408, 429, 500, 502, 503, 504]:
                return True
        
        return False
    
    def _classify_error(self, error: Exception) -> str:
        """错误分类"""
        
        error_mapping = {
            TimeoutError: "timeout",
            NetworkError: "network",
            RateLimitError: "rate_limit",
            requests.exceptions.Timeout: "timeout",
            requests.exceptions.ConnectionError: "network",
            requests.exceptions.HTTPError: "http",
            json.JSONDecodeError: "json_decode",
            ValueError: "validation",
            KeyError: "data_missing"
        }
        
        for error_type, error_category in error_mapping.items():
            if isinstance(error, error_type):
                return error_category
        
        return "unknown"
    
    def _fallback_sentiment_analysis(self, message: str) -> CustomerMessageAnalysis:
        """降级情感分析 - 基于关键词的简单实现"""
        
        positive_keywords = ["满意", "好", "不错", "感谢", "谢谢", "棒", "优秀"]
        negative_keywords = ["不满意", "差", "糟糕", "问题", "投诉", "退货", "退款", "生气"]
        
        category_keywords = {
            ProblemCategory.PRODUCT: ["产品", "商品", "质量", "功能", "性能"],
            ProblemCategory.AFTER_SALES: ["售后", "客服", "服务", "支持", "咨询"],
            ProblemCategory.TECHNICAL: ["技术", "故障", "错误", "bug", "问题", "异常"],
            ProblemCategory.BILLING: ["付款", "账单", "发票", "退款", "价格", "费用"]
        }
        
        urgency_keywords = {
            UrgencyLevel.HIGH: ["紧急", "立刻", "马上", "尽快", "重要", "立即"],
            UrgencyLevel.MEDIUM: ["需要", "希望", "尽快", "适当"],
            UrgencyLevel.LOW: ["建议", "如果", "考虑", "以后"]
        }
        
        # 简单情感分析
        sentiment_score = 0
        for keyword in positive_keywords:
            if keyword in message:
                sentiment_score += 1
        
        for keyword in negative_keywords:
            if keyword in message:
                sentiment_score -= 1
        
        if sentiment_score > 0:
            sentiment = Sentiment.POSITIVE
            confidence = min(0.7, 0.5 + sentiment_score * 0.1)
        elif sentiment_score < 0:
            sentiment = Sentiment.NEGATIVE
            confidence = min(0.7, 0.5 + abs(sentiment_score) * 0.1)
        else:
            sentiment = Sentiment.NEUTRAL
            confidence = 0.5
        
        # 问题分类
        category_scores = {}
        for category, keywords in category_keywords.items():
            score = sum(1 for keyword in keywords if keyword in message)
            category_scores[category] = score
        
        category = max(category_scores.items(), key=lambda x: x[1])[0]
        if max(category_scores.values()) == 0:
            category = ProblemCategory.GENERAL
        
        # 紧急程度
        urgency_scores = {}
        for urgency, keywords in urgency_keywords.items():
            score = sum(1 for keyword in keywords if keyword in message)
            urgency_scores[urgency] = score
        
        urgency = max(urgency_scores.items(), key=lambda x: x[1])[0]
        if max(urgency_scores.values()) == 0:
            urgency = UrgencyLevel.MEDIUM
        
        # 提取关键信息（简化版）
        key_info = {
            "has_product_mention": any(keyword in message for keyword in ["产品", "商品"]),
            "has_order_mention": any(keyword in message for keyword in ["订单", "单号"]),
            "has_contact_info": any(keyword in message for keyword in ["电话", "邮箱", "微信"]),
            "message_length": len(message)
        }
        
        return CustomerMessageAnalysis(
            sentiment=sentiment,
            category=category,
            urgency=urgency,
            key_info=key_info,
            confidence=confidence,
            processing_time_ms=10,  # 简单实现，处理速度快
            errors=["使用降级分析"],
            fallback_used=True
        )
    
    def call_with_retry(self, func: Callable, *args, max_retries: int = 3, **kwargs) -> Any:
        """带重试的函数调用"""
        
        self.stats["total_calls"] += 1
        
        # 检查熔断器
        if not self._check_circuit_breaker():
            logger.warning("熔断器阻止调用，使用降级处理")
            self.stats["fallback_used"] += 1
            
            # 尝试从参数中提取消息进行降级处理
            if len(args) > 0 and isinstance(args[0], str):
                return self._fallback_sentiment_analysis(args[0])
            else:
                raise ModelAPIError("熔断器打开且无法降级处理")
        
        last_error = None
        
        for attempt in range(max_retries + 1):  # 包括初始尝试
            try:
                start_time = datetime.now()
                
                # 如果不是第一次尝试，等待退避时间
                if attempt > 0:
                    delay = self._exponential_backoff(attempt - 1)
                    logger.info(f"重试尝试 {attempt}/{max_retries}，等待 {delay:.2f}秒")
                    time.sleep(delay)
                
                # 执行函数
                result = func(*args, **kwargs)
                
                # 更新成功统计
                end_time = datetime.now()
                processing_time = (end_time - start_time).total_seconds() * 1000
                
                self.stats["successful_calls"] += 1
                self.stats["retry_attempts"] += attempt
                
                # 更新平均响应时间
                current_avg = self.stats["average_response_time_ms"]
                total_successful = self.stats["successful_calls"]
                self.stats["average_response_time_ms"] = (
                    current_avg * (total_successful - 1) + processing_time
                ) / total_successful
                
                # 更新熔断器
                self._update_circuit_breaker(True)
                
                return result
                
            except Exception as e:
                last_error = e
                
                # 更新失败统计
                error_category = self._classify_error(e)
                self.stats["error_distribution"][error_category] = self.stats["error_distribution"].get(error_category, 0) + 1
                
                logger.error(f"调用失败 (尝试 {attempt + 1}/{max_retries + 1}): {type(e).__name__}: {e}")
                
                # 检查是否应该重试
                should_retry = self._should_retry(e) and attempt < max_retries
                
                if not should_retry:
                    self.stats["failed_calls"] += 1
                    self._update_circuit_breaker(False)
                    break
        
        # 所有重试都失败
        if isinstance(last_error, RateLimitError) and hasattr(last_error, 'retry_after'):
            # 如果是速率限制，等待指定时间
            wait_time = getattr(last_error, 'retry_after', 60)
            logger.warning(f"速率限制，等待 {wait_time}秒后使用降级处理")
            time.sleep(wait_time)
        
        # 使用降级处理
        logger.warning("所有重试失败，使用降级处理")
        self.stats["failed_calls"] += 1
        self.stats["fallback_used"] += 1
        
        if len(args) > 0 and isinstance(args[0], str):
            return self._fallback_sentiment_analysis(args[0])
        else:
            raise ModelAPIError(f"所有重试失败且无法降级: {last_error}")
    
    def get_stats(self) -> Dict[str, Any]:
        """获取统计信息"""
        return {
            **self.stats,
            "circuit_breaker_state": self.circuit_breaker["state"],
            "failure_count": self.circuit_breaker["failure_count"],
            "success_rate": (
                self.stats["successful_calls"] / self.stats["total_calls"] * 100
                if self.stats["total_calls"] > 0 else 100
            ),
            "fallback_rate": (
                self.stats["fallback_used"] / self.stats["total_calls"] * 100
                if self.stats["total_calls"] > 0 else 0
            )
        }
```

### 传统实现的问题分析

1. **代码复杂度爆炸**：错误处理逻辑往往比业务逻辑更复杂，维护困难
2. **模式重复**：每个服务调用都需要相似的错误处理代码，导致重复劳动
3. **配置分散**：重试策略、超时设置、熔断器参数分散在各个地方
4. **缺乏标准化**：不同开发人员实现不同的错误处理策略
5. **监控缺失**：难以统一收集和分析错误处理相关的指标
6. **测试覆盖率低**：错误处理逻辑难以全面测试
7. **降级策略单一**：通常只有简单的降级逻辑，缺乏智能降级
8. **错误恢复复杂**：从错误状态恢复需要复杂的逻辑
9. **依赖管理困难**：多个依赖服务的错误处理需要协调
10. **技术债务累积**：随着时间的推移，错误处理代码变得越来越难以维护

### 核心错误处理模式分析

通过分析LangChain v1.2.22文档，我们发现了以下核心错误处理模式：

```python
class EnterpriseErrorHandlingSystem:
    """企业级错误处理系统"""
    
    def __init__(self):
        self.error_handlers = {}
        self.metrics_collector = MetricsCollector()
        self.alert_system = AlertSystem()
        self.circuit_breakers = {}
    
    def register_error_handler(self, service_name: str, config: Dict[str, Any]):
        """注册错误处理器"""
        
        # 创建熔断器
        circuit_breaker = CircuitBreaker(
            failure_threshold=config.get("failure_threshold", 10),
            recovery_timeout=config.get("recovery_timeout", 60),
            half_open_max_attempts=config.get("half_open_max_attempts", 3)
        )
        
        # 创建重试策略
        retry_strategy = RetryStrategy(
            max_retries=config.get("max_retries", 3),
            backoff_factor=config.get("backoff_factor", 2.0),
            initial_delay=config.get("initial_delay", 1.0),
            retry_on=config.get("retry_on", None),
            on_failure=config.get("on_failure", "continue")
        )
        
        # 创建降级策略
        fallback_strategy = FallbackStrategy(
            primary_function=config.get("primary_function"),
            fallback_function=config.get("fallback_function"),
            fallback_conditions=config.get("fallback_conditions", ["timeout", "unavailable"])
        )
        
        self.error_handlers[service_name] = {
            "circuit_breaker": circuit_breaker,
            "retry_strategy": retry_strategy,
            "fallback_strategy": fallback_strategy,
            "config": config
        }
        
        self.circuit_breakers[service_name] = circuit_breaker
    
    def execute_with_error_handling(self, service_name: str, func: callable, *args, **kwargs):
        """执行带错误处理的函数"""
        
        if service_name not in self.error_handlers:
            raise ValueError(f"未注册的服务: {service_name}")
        
        handler = self.error_handlers[service_name]
        circuit_breaker = handler["circuit_breaker"]
        retry_strategy = handler["retry_strategy"]
        fallback_strategy = handler["fallback_strategy"]
        
        start_time = time.time()
        
        try:
            # 检查熔断器
            if not circuit_breaker.allow_request():
                self.metrics_collector.record_circuit_open(service_name)
                self.alert_system.notify_circuit_open(service_name)
                
                # 尝试降级
                if fallback_strategy.has_fallback():
                    return fallback_strategy.execute_fallback(*args, **kwargs)
                else:
                    raise CircuitOpenError(f"服务 {service_name} 熔断器打开")
            
            # 执行带重试的函数
            result = retry_strategy.execute_with_retry(func, *args, **kwargs)
            
            # 记录成功
            execution_time = time.time() - start_time
            self.metrics_collector.record_success(service_name, execution_time)
            circuit_breaker.record_success()
            
            return result
            
        except Exception as e:
            # 记录失败
            execution_time = time.time() - start_time
            error_type = type(e).__name__
            
            self.metrics_collector.record_failure(service_name, error_type, execution_time)
            circuit_breaker.record_failure()
            
            # 检查是否需要告警
            if self._should_alert(service_name, error_type):
                self.alert_system.notify_failure(service_name, error_type, str(e))
            
            # 尝试降级
            if fallback_strategy.should_fallback(e):
                try:
                    fallback_result = fallback_strategy.execute_fallback(*args, **kwargs)
                    self.metrics_collector.record_fallback(service_name)
                    return fallback_result
                except Exception as fallback_error:
                    # 降级也失败
                    raise AggregateError([
                        f"主服务失败: {e}",
                        f"降级服务失败: {fallback_error}"
                    ])
            else:
                raise
    
    def _should_alert(self, service_name: str, error_type: str) -> bool:
        """判断是否需要告警"""
        # 实现告警逻辑
        return error_type in ["TimeoutError", "ConnectionError", "RateLimitError"]
    
    def get_service_health(self, service_name: str) -> Dict[str, Any]:
        """获取服务健康状态"""
        if service_name not in self.error_handlers:
            return {"status": "unknown"}
        
        circuit_breaker = self.circuit_breakers[service_name]
        metrics = self.metrics_collector.get_service_metrics(service_name)
        
        return {
            "status": circuit_breaker.state,
            "failure_rate": metrics.get("failure_rate", 0),
            "success_rate": metrics.get("success_rate", 0),
            "average_response_time": metrics.get("average_response_time", 0),
            "circuit_breaker_trips": metrics.get("circuit_breaker_trips", 0),
            "fallback_usage": metrics.get("fallback_usage", 0)
        }

# 辅助类（简化实现）
class CircuitBreaker:
    """熔断器实现"""
    
    def __init__(self, failure_threshold=10, recovery_timeout=60, half_open_max_attempts=3):
        self.failure_threshold = failure_threshold
        self.recovery_timeout = recovery_timeout
        self.half_open_max_attempts = half_open_max_attempts
        
        self.state = "CLOSED"  # CLOSED, OPEN, HALF_OPEN
        self.failure_count = 0
        self.half_open_attempts = 0
        self.last_failure_time = None
        self.open_until = None
    
    def allow_request(self) -> bool:
        """检查是否允许请求"""
        current_time = time.time()
        
        if self.state == "OPEN":
            if self.open_until and current_time >= self.open_until:
                # 超时，转为半开状态
                self.state = "HALF_OPEN"
                self.half_open_attempts = 0
                return True
            else:
                return False
        elif self.state == "HALF_OPEN":
            # 半开状态允许有限尝试
            if self.half_open_attempts < self.half_open_max_attempts:
                self.half_open_attempts += 1
                return True
            else:
                return False
        
        return True  # CLOSED状态
    
    def record_success(self):
        """记录成功"""
        if self.state == "HALF_OPEN":
            # 半开状态成功，关闭熔断器
            self.state = "CLOSED"
            self.failure_count = 0
            self.half_open_attempts = 0
        elif self.state == "CLOSED":
            # 正常状态，减少失败计数
            self.failure_count = max(0, self.failure_count - 1)
    
    def record_failure(self):
        """记录失败"""
        self.failure_count += 1
        self.last_failure_time = time.time()
        
        if self.failure_count >= self.failure_threshold and self.state == "CLOSED":
            # 达到阈值，打开熔断器
            self.state = "OPEN"
            self.open_until = time.time() + self.recovery_timeout

class RetryStrategy:
    """重试策略"""
    
    def __init__(self, max_retries=3, backoff_factor=2.0, initial_delay=1.0, 
                 retry_on=None, on_failure="continue"):
        self.max_retries = max_retries
        self.backoff_factor = backoff_factor
        self.initial_delay = initial_delay
        self.retry_on = retry_on
        self.on_failure = on_failure
    
    def execute_with_retry(self, func: callable, *args, **kwargs):
        """执行带重试的函数"""
        last_error = None
        
        for attempt in range(self.max_retries + 1):  # 包括初始尝试
            try:
                if attempt > 0:  # 重试
                    delay = self.initial_delay * (self.backoff_factor ** (attempt - 1))
                    time.sleep(delay)
                
                return func(*args, **kwargs)
                
            except Exception as e:
                last_error = e
                
                # 检查是否应该重试
                should_retry = self._should_retry(e) and attempt < self.max_retries
                
                if not should_retry:
                    break
        
        # 所有重试都失败
        if self.on_failure == "continue":
            # 继续执行，返回错误消息
            if callable(self.on_failure):
                return self.on_failure(last_error)
            else:
                return f"操作失败: {last_error}"
        else:
            # 抛出异常
            raise last_error
    
    def _should_retry(self, error: Exception) -> bool:
        """判断是否应该重试"""
        if self.retry_on is None:
            # 默认重试所有异常
            return True
        
        if callable(self.retry_on):
            # 自定义判断函数
            return self.retry_on(error)
        
        # 检查异常类型
        if isinstance(self.retry_on, list):
            for error_type in self.retry_on:
                if isinstance(error, error_type):
                    return True
        
        return False

class FallbackStrategy:
    """降级策略"""
    
    def __init__(self, primary_function=None, fallback_function=None, 
                 fallback_conditions=None):
        self.primary_function = primary_function
        self.fallback_function = fallback_function
        self.fallback_conditions = fallback_conditions or []
    
    def has_fallback(self) -> bool:
        """是否有降级函数"""
        return self.fallback_function is not None
    
    def should_fallback(self, error: Exception) -> bool:
        """是否应该降级"""
        if not self.has_fallback():
            return False
        
        error_message = str(error).lower()
        error_type = type(error).__name__
        
        # 检查降级条件
        for condition in self.fallback_conditions:
            if condition in error_message.lower() or condition == error_type:
                return True
        
        return False
    
    def execute_fallback(self, *args, **kwargs):
        """执行降级函数"""
        if not self.has_fallback():
            raise ValueError("没有可用的降级函数")
        
        return self.fallback_function(*args, **kwargs)

class MetricsCollector:
    """指标收集器"""
    
    def __init__(self):
        self.metrics = {}
    
    def record_success(self, service_name: str, execution_time: float):
        """记录成功"""
        if service_name not in self.metrics:
            self.metrics[service_name] = self._initialize_service_metrics()
        
        metrics = self.metrics[service_name]
        metrics["total_calls"] += 1
        metrics["successful_calls"] += 1
        metrics["total_response_time"] += execution_time
    
    def record_failure(self, service_name: str, error_type: str, execution_time: float):
        """记录失败"""
        if service_name not in self.metrics:
            self.metrics[service_name] = self._initialize_service_metrics()
        
        metrics = self.metrics[service_name]
        metrics["total_calls"] += 1
        metrics["failed_calls"] += 1
        metrics["total_response_time"] += execution_time
        
        # 记录错误分布
        if "error_distribution" not in metrics:
            metrics["error_distribution"] = {}
        metrics["error_distribution"][error_type] = metrics["error_distribution"].get(error_type, 0) + 1
    
    def record_circuit_open(self, service_name: str):
        """记录熔断器打开"""
        if service_name not in self.metrics:
            self.metrics[service_name] = self._initialize_service_metrics()
        
        self.metrics[service_name]["circuit_breaker_trips"] += 1
    
    def record_fallback(self, service_name: str):
        """记录降级使用"""
        if service_name not in self.metrics:
            self.metrics[service_name] = self._initialize_service_metrics()
        
        self.metrics[service_name]["fallback_usage"] += 1
    
    def get_service_metrics(self, service_name: str) -> Dict[str, Any]:
        """获取服务指标"""
        if service_name not in self.metrics:
            return self._initialize_service_metrics()
        
        metrics = self.metrics[service_name].copy()
        
        # 计算派生指标
        total_calls = metrics["total_calls"]
        if total_calls > 0:
            metrics["success_rate"] = (metrics["successful_calls"] / total_calls) * 100
            metrics["failure_rate"] = (metrics["failed_calls"] / total_calls) * 100
            
            if metrics["successful_calls"] > 0:
                metrics["average_response_time"] = metrics["total_response_time"] / metrics["successful_calls"]
            else:
                metrics["average_response_time"] = 0
        
        return metrics
    
    def _initialize_service_metrics(self) -> Dict[str, Any]:
        """初始化服务指标"""
        return {
            "total_calls": 0,
            "successful_calls": 0,
            "failed_calls": 0,
            "total_response_time": 0,
            "circuit_breaker_trips": 0,
            "fallback_usage": 0
        }
```

```python
class IntelligentModelSelector:
    """智能模型选择器"""
    
    def __init__(self):
        self.models = []
        self.model_stats = {}
        self.selection_strategy = "cost_performance"  # cost, performance, reliability, balanced
    
    def add_model(self, name: str, cost_per_token: float, avg_response_time: float, 
                  success_rate: float, capabilities: list):
        """添加模型"""
        
        model = {
            "name": name,
            "cost_per_token": cost_per_token,
            "avg_response_time": avg_response_time,
            "success_rate": success_rate,
            "capabilities": capabilities,
            "call_count": 0,
            "success_count": 0,
            "total_cost": 0.0,
            "total_response_time": 0.0
        }
        
        self.models.append(model)
        self.model_stats[name] = {
            "call_count": 0,
            "success_count": 0,
            "total_cost": 0.0,
            "total_response_time": 0.0
        }
    
    def select_model(self, requirements: dict) -> dict:
        """根据需求选择模型"""
        
        # 过滤满足能力要求的模型
        capable_models = []
        for model in self.models:
            if self._meets_capability_requirements(model, requirements):
                capable_models.append(model)
        
        if not capable_models:
            raise Exception("没有模型满足能力要求")
        
        # 根据策略选择模型
        if self.selection_strategy == "cost":
            # 选择成本最低的模型
            selected = min(capable_models, key=lambda x: x["cost_per_token"])
        
        elif self.selection_strategy == "performance":
            # 选择响应时间最短的模型
            selected = min(capable_models, key=lambda x: x["avg_response_time"])
        
        elif self.selection_strategy == "reliability":
            # 选择成功率最高的模型
            selected = max(capable_models, key=lambda x: x["success_rate"])
        
        elif self.selection_strategy == "cost_performance":
            # 成本性能平衡：成本/响应时间的比值
            capable_models_with_score = []
            for model in capable_models:
                # 归一化成本（0-1）
                costs = [m["cost_per_token"] for m in capable_models]
                min_cost, max_cost = min(costs), max(costs)
                if max_cost > min_cost:
                    normalized_cost = (model["cost_per_token"] - min_cost) / (max_cost - min_cost)
                else:
                    normalized_cost = 0.5
                
                # 归一化响应时间（0-1）
                response_times = [m["avg_response_time"] for m in capable_models]
                min_time, max_time = min(response_times), max(response_times)
                if max_time > min_time:
                    normalized_time = (model["avg_response_time"] - min_time) / (max_time - min_time)
                else:
                    normalized_time = 0.5
                
                # 成本性能分数（越低越好）
                # 权重：成本60%，性能40%
                score = 0.6 * normalized_cost + 0.4 * normalized_time
                
                capable_models_with_score.append((model, score))
            
            selected = min(capable_models_with_score, key=lambda x: x[1])[0]
        
        elif self.selection_strategy == "balanced":
            # 平衡策略：考虑成本、性能和可靠性
            capable_models_with_score = []
            for model in capable_models:
                # 归一化成本
                costs = [m["cost_per_token"] for m in capable_models]
                min_cost, max_cost = min(costs), max(costs)
                if max_cost > min_cost:
                    normalized_cost = (model["cost_per_token"] - min_cost) / (max_cost - min_cost)
                else:
                    normalized_cost = 0.5
                
                # 归一化响应时间
                response_times = [m["avg_response_time"] for m in capable_models]
                min_time, max_time = min(response_times), max(response_times)
                if max_time > min_time:
                    normalized_time = (model["avg_response_time"] - min_time) / (max_time - min_time)
                else:
                    normalized_time = 0.5
                
                # 归一化成功率（越高越好，所以用1-）
                success_rates = [m["success_rate"] for m in capable_models]
                min_success, max_success = min(success_rates), max(success_rates)
                if max_success > min_success:
                    normalized_success = 1 - (model["success_rate"] - min_success) / (max_success - min_success)
                else:
                    normalized_success = 0.5
                
                # 综合分数（越低越好）
                # 权重：成本40%，性能30%，可靠性30%
                score = (0.4 * normalized_cost + 
                        0.3 * normalized_time + 
                        0.3 * normalized_success)
                
                capable_models_with_score.append((model, score))
            
            selected = min(capable_models_with_score, key=lambda x: x[1])[0]
        
        else:
            # 默认选择第一个模型
            selected = capable_models[0]
        
        return selected
    
    def _meets_capability_requirements(self, model: dict, requirements: dict) -> bool:
        """检查模型是否满足能力要求"""
        
        required_capabilities = requirements.get("capabilities", [])
        if not required_capabilities:
            return True
        
        model_capabilities = set(model["capabilities"])
        required_capabilities_set = set(required_capabilities)
        
        return required_capabilities_set.issubset(model_capabilities)
    
    def record_model_performance(self, model_name: str, success: bool, 
                                 cost: float, response_time: float):
        """记录模型性能"""
        
        if model_name not in self.model_stats:
            return
        
        stats = self.model_stats[model_name]
        stats["call_count"] += 1
        stats["total_response_time"] += response_time
        
        if success:
            stats["success_count"] += 1
            stats["total_cost"] += cost
        
        # 更新模型成功率
        for model in self.models:
            if model["name"] == model_name:
                model_stats = self.model_stats[model_name]
                if model_stats["call_count"] > 0:
                    model["success_rate"] = model_stats["success_count"] / model_stats["call_count"]
                if model_stats["success_count"] > 0:
                    model["avg_response_time"] = model_stats["total_response_time"] / model_stats["success_count"]
                break
    
    def get_model_stats(self) -> dict:
        """获取模型统计信息"""
        return self.model_stats

                }
```

```python
    def _calculate_error_rate(self, service_name: str) -> float:
        """计算错误率"""
        
        if service_name not in self.sliding_window["request_counts"]:
            return 0.0
        
        requests = self.sliding_window["request_counts"][service_name]
        if not requests:
            return 0.0
        
        total_requests = len(requests)
        failed_requests = sum(1 for req in requests if not req["success"])
        
        return failed_requests / total_requests if total_requests > 0 else 0.0
    
    def _count_consecutive_failures(self, service_name: str) -> int:
        """计算连续失败次数"""
        
        if service_name not in self.sliding_window["request_counts"]:
            return 0
        
        requests = self.sliding_window["request_counts"][service_name]
        if not requests:
            return 0
        
        # 从最新请求开始向前计数
        consecutive_failures = 0
        for req in reversed(requests):
            if req["success"]:
                break
            consecutive_failures += 1
        
        return consecutive_failures
    
    def _calculate_p95_response_time(self, service_name: str) -> float:
        """计算P95响应时间"""
        
        if service_name not in self.sliding_window["response_times"]:
            return 0.0
        
        response_times = self.sliding_window["response_times"][service_name]
        if not response_times:
            return 0.0
        
        # 提取响应时间
        durations = [item["duration_ms"] for item in response_times]
        durations.sort()
        
        # 计算P95
        index = int(len(durations) * 0.95)
        return durations[index] if index < len(durations) else durations[-1]
    
    def get_service_health(self, service_name: str) -> Dict[str, Any]:
        """获取服务健康状态"""
        
        error_rate = self._calculate_error_rate(service_name)
        consecutive_failures = self._count_consecutive_failures(service_name)
        p95_response_time = self._calculate_p95_response_time(service_name)
        
        # 确定健康状态
        if error_rate > 0.2 or consecutive_failures > 10:
            health_status = "unhealthy"
        elif error_rate > 0.05 or p95_response_time > 10000:
            health_status = "degraded"
        else:
            health_status = "healthy"
        
        return {
            "service_name": service_name,
            "health_status": health_status,
            "error_rate": error_rate,
            "consecutive_failures": consecutive_failures,
            "p95_response_time": p95_response_time,
            "total_requests": len(self.sliding_window["request_counts"].get(service_name, [])),
            "total_errors": len(self.sliding_window["error_counts"].get(service_name, [])),
            "window_size_minutes": self.sliding_window["window_size"].total_seconds() / 60
        }
    
    def get_recent_alerts(self, limit: int = 10) -> List[Dict[str, Any]]:
        """获取最近的告警"""
        return self.alerts[-limit:] if self.alerts else []
    
    def export_metrics(self, format: str = "json") -> str:
        """导出指标数据"""
        
        data = {
            "timestamp": datetime.now().isoformat(),
            "error_events": [event.to_dict() for event in self.error_events[-100:]],
            "performance_metrics": [metric.to_dict() for metric in self.performance_metrics[-100:]],
            "recent_alerts": self.get_recent_alerts(10),
            "service_health": {}
        }
        
        # 添加所有服务的健康状态
        all_services = set()
        all_services.update(self.sliding_window["error_counts"].keys())
        all_services.update(self.sliding_window["request_counts"].keys())
        
        for service_name in all_services:
            data["service_health"][service_name] = self.get_service_health(service_name)
        
        if format == "json":
            return json.dumps(data, indent=2, ensure_ascii=False)
        else:
            # 可以添加其他格式支持
            return str(data)

    def execute_with_comprehensive_handling(self, service_name: str, func: callable, 
                                           *args, strategy: dict = None, **kwargs):
        """执行带综合错误处理的函数"""
        
        # 使用服务特定策略或默认策略
        strategy = strategy or self.default_strategy
        
        # 1. 检查熔断器

                        error_type="CircuitBreakerOpen",
                        error_message="熔断器打开，阻止请求",
                        severity=ErrorSeverity.WARNING,
                        category=ErrorCategory.BUSINESS_ERROR,
                        context={"strategy": strategy}
                    )
                )
                
                # 尝试降级
                if strategy.get("fallback_enabled", True):
                    fallback_result = self._execute_fallback(service_name, func, *args, **kwargs)
                    if fallback_result is not None:
                        return fallback_result
                
                raise Exception(f"服务 {service_name} 熔断器打开且无降级策略")
        
        # 2. 执行带重试的函数
        last_error = None
        max_retries = strategy.get("max_retries", 3)
        
        for attempt in range(max_retries + 1):
            try:
                start_time = datetime.now()
                
                # 如果不是第一次尝试，等待退避时间
                if attempt > 0:
                    delay = self._calculate_backoff(attempt, strategy)
                    print(f"重试尝试 {attempt}/{max_retries}，等待 {delay:.2f}秒")
                    time.sleep(delay)
                
                # 执行函数
                result = func(*args, **kwargs)
                
                # 记录成功

                self.monitoring_system.record_performance(

                        service_name=service_name,
                        operation_name=func.__name__,
                        duration_ms=duration_ms,
                        success=True,
                        input_size=len(str(args)) + len(str(kwargs))

            except Exception as e:

                )

                        ErrorEvent(
                            timestamp=datetime.now(),
                            service_name=service_name,
                            error_type="FallbackUsed",
                            error_message="使用降级策略",
                            severity=ErrorSeverity.WARNING,
                            category=ErrorCategory.BUSINESS_ERROR,
                            context={
                                "original_error": str(last_error),
                                "fallback_success": True
                            }
                        )
                    )
                    
                    return fallback_result
            except Exception as fallback_error:
                # 降级也失败
                self.monitoring_system.record_error(
                    ErrorEvent(
                        timestamp=datetime.now(),
                        service_name=service_name,
                        error_type="FallbackFailed",
                        error_message=f"降级策略失败: {fallback_error}",
                        severity=ErrorSeverity.ERROR,
                        category=ErrorCategory.BUSINESS_ERROR,
                        context={
                            "original_error": str(last_error),
                            "fallback_error": str(fallback_error)
                        }
                    )
                )
        
        # 4. 完全失败
        raise Exception(f"服务 {service_name} 完全失败: {last_error}")

    def _calculate_backoff(self, attempt: int, strategy: dict) -> float:
        """计算退避延迟"""
        
        base_delay = strategy.get("initial_delay", 1.0)
        backoff_factor = strategy.get("backoff_factor", 2.0)
        max_delay = strategy.get("max_delay", 60.0)
        
        delay = base_delay * (backoff_factor ** (attempt - 1))
        
        # 添加抖动
        jitter = random.uniform(0, delay * 0.1)
        
        return min(delay + jitter, max_delay)

        """判断是否应该重试"""

            return False
        
        # 检查错误类型是否应该重试

        retry_on = strategy.get("retry_on", [])

            # 检查错误消息是否包含重试关键词

            retry_keywords = ["timeout", "rate limit", "busy", "retry", "temporarily"]
            
            if not any(keyword in error_msg for keyword in retry_keywords):
                return False
        
        return True
```
