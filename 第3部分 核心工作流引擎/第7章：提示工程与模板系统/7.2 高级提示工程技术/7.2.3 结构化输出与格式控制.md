# 7.2.3 结构化输出与格式控制

## 结构化输出与格式控制的核心价值：从自由文本到可编程数据

在AI应用开发中，结构化输出（Structured Output）和格式控制（Format Control）是连接大语言模型自由文本生成与企业系统可编程接口的关键桥梁。其核心价值体现在：

1. **数据一致性保证**：确保模型输出符合预定义的数据结构，便于下游系统处理
2. **自动化处理支持**：结构化数据可以直接被代码解析和处理，无需人工干预
3. **系统集成简化**：标准化的数据结构简化了与其他系统的集成
4. **数据质量提升**：通过格式验证和类型检查，提高输出数据的准确性和完整性
5. **开发效率提高**：减少手动解析和验证的工作量，提高开发效率
6. **错误处理优化**：结构化输出更容易进行错误检测和异常处理

## 传统结构化输出实现的复杂性：手动解析与格式验证

在LangChain出现之前，开发人员需要手动处理模型输出的解析、验证和转换，导致代码复杂且容易出错。让我们通过一个企业产品信息提取系统来展示传统实现的复杂性：

### 场景需求
企业需要从产品描述文本中提取结构化信息：
1. **产品基本信息**：名称、品牌、型号、类别
2. **技术规格**：尺寸、重量、颜色、材质
3. **价格信息**：原价、折扣价、促销信息
4. **库存状态**：可用性、预计到货时间
5. **相关产品**：配件、兼容产品、升级选项

# 传统结构化输出实现 - 繁琐的解析和验证逻辑
import re
import json
from typing import Dict, Any, Optional, List
from datetime import datetime
from dataclasses import dataclass
from enum import Enum

class ProductCategory(Enum):
    """产品类别枚举"""
    ELECTRONICS = "electronics"
    CLOTHING = "clothing"
    HOME_APPLIANCES = "home_appliances"
    BOOKS = "books"
    OTHER = "other"

@dataclass
class ProductSpecification:
    """产品规格"""
    dimensions: Optional[str] = None
    weight: Optional[str] = None
    color: Optional[str] = None
    material: Optional[str] = None
    warranty: Optional[str] = None

@dataclass
class PriceInfo:
    """价格信息"""
    original_price: float
    discount_price: Optional[float] = None
    currency: str = "CNY"
    promotion_text: Optional[str] = None
    valid_until: Optional[datetime] = None

@dataclass
class InventoryStatus:
    """库存状态"""
    available: bool
    quantity: Optional[int] = None
    estimated_delivery: Optional[str] = None
    backorder_allowed: bool = False

class TraditionalStructuredOutputSystem:
    """传统结构化输出系统 - 手动解析实现"""
    
    def __init__(self):
        self.extraction_patterns = self._initialize_extraction_patterns()
        self.validation_rules = self._initialize_validation_rules()
    
    def _initialize_extraction_patterns(self) -> Dict[str, re.Pattern]:
        """初始化提取模式 - 复杂的正则表达式"""
        return {
            "product_name": re.compile(r"产品[：:]\s*([^\n\r]+?)(?=\s*品牌|\s*型号|\s*规格|$)"),
            "brand": re.compile(r"品牌[：:]\s*([^\n\r]+?)(?=\s*型号|\s*规格|$)"),
            "model": re.compile(r"型号[：:]\s*([^\n\r]+?)(?=\s*规格|$)"),
            "category": re.compile(r"类别[：:]\s*([^\n\r]+?)(?=\s*规格|$)"),
            "dimensions": re.compile(r"尺寸[：:]\s*([^\n\r]+?)(?=\s*重量|$)"),
            "weight": re.compile(r"重量[：:]\s*([^\n\r]+?)(?=\s*颜色|$)"),
            "color": re.compile(r"颜色[：:]\s*([^\n\r]+?)(?=\s*材质|$)"),
            "material": re.compile(r"材质[：:]\s*([^\n\r]+?)(?=\s*价格|$)"),
            "original_price": re.compile(r"原价[：:]\s*([0-9.]+)\s*([A-Z]{3})?"),
            "discount_price": re.compile(r"折扣价[：:]\s*([0-9.]+)\s*([A-Z]{3})?"),
            "availability": re.compile(r"库存[：:]\s*(有货|缺货|预定)"),
            "quantity": re.compile(r"数量[：:]\s*([0-9]+)"),
        }
    
    def _initialize_validation_rules(self) -> Dict[str, callable]:
        """初始化验证规则 - 复杂的验证逻辑"""
        return {
            "product_name": lambda x: len(x.strip()) > 0 and len(x.strip()) <= 100,
            "brand": lambda x: len(x.strip()) > 0 and len(x.strip()) <= 50,
            "model": lambda x: len(x.strip()) > 0 and len(x.strip()) <= 50,
            "category": lambda x: x.strip() in ["电子产品", "服装", "家电", "图书", "其他"],
            "original_price": lambda x: float(x) > 0,
            "discount_price": lambda x: float(x) > 0,
            "availability": lambda x: x in ["有货", "缺货", "预定"],
            "quantity": lambda x: int(x) >= 0,
        }
    
    def extract_product_info(self, product_description: str) -> Dict[str, Any]:
        """提取产品信息 - 复杂的解析逻辑"""
        
        extracted_data = {}
        
        # 1. 使用正则表达式提取信息
        for field, pattern in self.extraction_patterns.items():
            match = pattern.search(product_description)
            if match:
                if field in ["original_price", "discount_price"]:
                    extracted_data[field] = float(match.group(1))
                    if match.group(2):
                        extracted_data["currency"] = match.group(2)
                elif field in ["quantity"]:
                    extracted_data[field] = int(match.group(1))
                else:
                    extracted_data[field] = match.group(1).strip()
        
        # 2. 后处理：分类映射
        if "category" in extracted_data:
            category_map = {
                "电子产品": ProductCategory.ELECTRONICS,
                "服装": ProductCategory.CLOTHING,
                "家电": ProductCategory.HOME_APPLIANCES,
                "图书": ProductCategory.BOOKS,
                "其他": ProductCategory.OTHER
            }
            extracted_data["category"] = category_map.get(
                extracted_data["category"], 
                ProductCategory.OTHER
            )
        
        # 3. 数据验证
        validation_results = {}
        errors = []
        
        for field, value in extracted_data.items():
            if field in self.validation_rules:
                try:
                    is_valid = self.validation_rules[field](value)
                    validation_results[field] = {
                        "valid": is_valid,
                        "value": value
                    }
                    if not is_valid:
                        errors.append(f"字段 {field} 的值 {value} 无效")
                except Exception as e:
                    validation_results[field] = {
                        "valid": False,
                        "value": value,
                        "error": str(e)
                    }
                    errors.append(f"字段 {field} 验证失败: {e}")
        
        # 4. 构建数据结构
        try:
            # 产品规格
            specification = ProductSpecification(
                dimensions=extracted_data.get("dimensions"),
                weight=extracted_data.get("weight"),
                color=extracted_data.get("color"),
                material=extracted_data.get("material")
            )
            
            # 价格信息
            price_info = PriceInfo(
                original_price=extracted_data.get("original_price", 0.0),
                discount_price=extracted_data.get("discount_price"),
                currency=extracted_data.get("currency", "CNY")
            )
            
            # 库存状态
            inventory_status = InventoryStatus(
                available=extracted_data.get("availability") == "有货",
                quantity=extracted_data.get("quantity"),
                estimated_delivery=None  # 需要额外提取
            )
            
            result = {
                "basic_info": {
                    "product_name": extracted_data.get("product_name"),
                    "brand": extracted_data.get("brand"),
                    "model": extracted_data.get("model"),
                    "category": extracted_data.get("category"),
                },
                "specification": specification,
                "price_info": price_info,
                "inventory_status": inventory_status,
                "extracted_fields": list(extracted_data.keys()),
                "validation_results": validation_results,
                "errors": errors,
                "success": len(errors) == 0
            }
            
            return result
            
        except Exception as e:
            return {
                "basic_info": {},
                "specification": None,
                "price_info": None,
                "inventory_status": None,
                "extracted_fields": list(extracted_data.keys()),
                "validation_results": validation_results,
                "errors": errors + [f"数据结构构建失败: {e}"],
                "success": False
            }
    
    def generate_extraction_prompt(self, product_description: str) -> str:
        """生成信息提取提示 - 复杂的提示工程"""
        
        prompt = f"""请从以下产品描述中提取结构化信息：

产品描述：
{product_description}

请提取以下信息，并以JSON格式返回：
1. 产品基本信息：
   - 产品名称
   - 品牌
   - 型号
   - 类别（电子产品、服装、家电、图书、其他）

2. 技术规格：
   - 尺寸（如果有）
   - 重量（如果有）
   - 颜色（如果有）
   - 材质（如果有）

3. 价格信息：
   - 原价（单位：元）
   - 折扣价（如果有）
   - 促销信息（如果有）

4. 库存状态：
   - 是否有货（有货/缺货/预定）
   - 数量（如果有）
   - 预计到货时间（如果缺货）

5. 相关产品：
   - 配件（如果有）
   - 兼容产品（如果有）
   - 升级选项（如果有）

要求：
1. 如果信息不存在，字段值为null
2. 价格以数字格式表示，单位元
3. 所有文本字段去除前后空格
4. 严格按照以下JSON格式返回：

  "product_info": {{
    "name": "string",
    "brand": "string",
    "model": "string",
    "category": "string"
  }},
  "specification": {{
    "dimensions": "string | null",
    "weight": "string | null",

    "material": "string | null"
  }},

    "original_price": "number",

    "promotion_text": "string | null"

    "available": "boolean",
    "quantity": "number | null",

  }},

    "compatible_products": ["string"] | null,
    "upgrade_options": ["string"] | null

        return prompt
    
    def parse_model_response(self, response: str) -> Dict[str, Any]:
        """解析模型响应 - 复杂的JSON解析和验证"""
        
        # 1. 尝试提取JSON
        json_pattern = re.compile(r'```json\s*(.*?)\s*```', re.DOTALL)
        match = json_pattern.search(response)
        
        if match:
            json_text = match.group(1)
        else:
            # 尝试直接解析整个响应
            json_text = response
        
        try:
            # 2. 解析JSON
            parsed_data = json.loads(json_text)
            
            # 3. 验证必需字段
            required_fields = ["product_info"]
            missing_fields = []
            
            for field in required_fields:
                if field not in parsed_data:
                    missing_fields.append(field)
            
            if missing_fields:
                return {
                    "success": False,
                    "parsed_data": parsed_data,
                    "errors": [f"缺少必需字段: {', '.join(missing_fields)}"],
                    "validation_passed": False
                }
            
            # 4. 类型检查
            type_errors = []
            
            # 检查产品信息
            if "product_info" in parsed_data:
                product_info = parsed_data["product_info"]
                if not isinstance(product_info, dict):
                    type_errors.append("product_info 必须是对象")
                else:
                    if "name" in product_info and not isinstance(product_info["name"], str):
                        type_errors.append("product_info.name 必须是字符串")
                    if "brand" in product_info and not isinstance(product_info["brand"], str):
                        type_errors.append("product_info.brand 必须是字符串")
            
            # 检查价格信息
            if "price_info" in parsed_data:
                price_info = parsed_data["price_info"]
                if not isinstance(price_info, dict):
                    type_errors.append("price_info 必须是对象")
                else:
                    if "original_price" in price_info and not isinstance(price_info["original_price"], (int, float)):
                        type_errors.append("price_info.original_price 必须是数字")
            
            if type_errors:
                return {
                    "success": False,
                    "parsed_data": parsed_data,
                    "errors": type_errors,
                    "validation_passed": False
                }
            
            return {
                "success": True,
                "parsed_data": parsed_data,
                "errors": [],
                "validation_passed": True
            }
            
        except json.JSONDecodeError as e:
            return {
                "success": False,
                "parsed_data": None,
                "errors": [f"JSON解析失败: {e}"],
                "validation_passed": False
            }
        except Exception as e:
            return {
                "success": False,
                "parsed_data": None,
                "errors": [f"解析过程出错: {e}"],
                "validation_passed": False
            }

### 传统实现的问题分析

1. **正则表达式局限性**：无法处理复杂的自然语言变化和上下文
2. **验证逻辑分散**：类型检查、格式验证、业务规则分散在多个函数中
3. **JSON解析脆弱**：依赖模型严格遵循JSON格式，容易解析失败
4. **提示工程复杂**：需要详细描述JSON结构，容易出错
5. **错误处理冗余**：每个解析步骤都需要独立的错误处理
6. **缺乏标准化**：每个项目需要重新实现解析和验证逻辑

### 方案一：Pydantic模型与with_structured_output方法

```python
from langchain_core.prompts import ChatPromptTemplate
from langchain_openai import ChatOpenAI
from pydantic import BaseModel, Field, validator
from typing import List, Optional, Dict, Any
from datetime import datetime
import os

# 设置API密钥
os.environ["OPENAI_API_KEY"] = "your-api-key-here"

# 1. 使用Pydantic定义数据结构
class ProductSpecification(BaseModel):
    """产品规格Pydantic模型"""
    dimensions: Optional[str] = Field(
        default=None,
        description="产品尺寸，格式如'159.9 x 76.7 x 8.25 mm'"
    )
    weight: Optional[str] = Field(
        default=None, 
        description="产品重量，格式如'221g'或'2.5kg'"
    )
    color: Optional[str] = Field(
        default=None,
        description="产品颜色"
    )
    material: Optional[str] = Field(
        default=None,
        description="产品材质"
    )
    warranty: Optional[str] = Field(
        default=None,
        description="保修信息，如'1年保修'"
    )
    
    @validator('weight')
    def validate_weight_format(cls, v):
        """验证重量格式"""
        if v is not None:
            if not any(unit in v.lower() for unit in ['g', 'kg', '克', '千克']):
                raise ValueError("重量必须包含单位（g/kg/克/千克）")
        return v

class PriceInfo(BaseModel):
    """价格信息Pydantic模型"""
    original_price: float = Field(
        description="原价，单位元"
    )
    discount_price: Optional[float] = Field(
        default=None,
        description="折扣价，单位元"
    )
    currency: str = Field(
        default="CNY",
        description="货币代码，默认CNY"
    )
    promotion_text: Optional[str] = Field(
        default=None,
        description="促销信息文本"
    )
    valid_until: Optional[str] = Field(
        default=None,
        description="有效期，格式'YYYY-MM-DD'"
    )
    
    @validator('original_price', 'discount_price')
    def validate_price_positive(cls, v):
        """验证价格为正数"""
        if v is not None and v <= 0:
            raise ValueError("价格必须大于0")
        return v
    
    @validator('discount_price')
    def validate_discount_lower(cls, v, values):
        """验证折扣价低于原价"""
        if v is not None and 'original_price' in values:
            if v >= values['original_price']:
                raise ValueError("折扣价必须低于原价")
        return v

class InventoryStatus(BaseModel):
    """库存状态Pydantic模型"""
    available: bool = Field(
        description="是否有货"
    )
    quantity: Optional[int] = Field(
        default=None,
        description="库存数量"
    )
    estimated_delivery: Optional[str] = Field(
        default=None,
        description="预计到货时间，如'3-5天'"
    )
    backorder_allowed: bool = Field(
        default=False,
        description="是否允许预订"
    )
    
    @validator('quantity')
    def validate_quantity_nonnegative(cls, v):
        """验证数量非负"""
        if v is not None and v < 0:
            raise ValueError("库存数量不能为负数")
        return v

class RelatedProducts(BaseModel):
    """相关产品Pydantic模型"""
    accessories: Optional[List[str]] = Field(
        default=None,
        description="配件列表"
    )
    compatible_products: Optional[List[str]] = Field(
        default=None,
        description="兼容产品列表"
    )
    upgrade_options: Optional[List[str]] = Field(
        default=None,
        description="升级选项列表"

class StructuredOutputWithRetry:
    """带重试的结构化输出"""
    
    def __init__(self, max_retries: int = 3):
        self.max_retries = max_retries
        self.llm = ChatOpenAI(model="gpt-3.5-turbo", temperature=0.0)
    
    def extract_with_retry(self, schema: Any, text: str) -> Any:
        """带重试的提取"""
        
        structured_llm = self.llm.with_structured_output(schema)
        
        prompt = ChatPromptTemplate.from_messages([
            ("system", "请从文本中提取结构化信息。如果信息不完整，尽可能提取可用信息。"),
            ("human", "文本：{text}")
        ])
        
        chain = prompt | structured_llm
        
        # 重试逻辑
        for attempt in range(self.max_retries):
            try:
                result = chain.invoke({"text": text})
                return {
                    "success": True,
                    "result": result,
                    "attempts": attempt + 1
                }
            except Exception as e:
                print(f"尝试 {attempt + 1} 失败: {e}")
                
                # 最后一次尝试仍然失败
                if attempt == self.max_retries - 1:
                    return {
                        "success": False,
                        "error": str(e),
                        "attempts": attempt + 1
                    }
        
        return {
            "success": False,
            "error": "达到最大重试次数",
            "attempts": self.max_retries
        }
```

from langchain_core.output_parsers import BaseOutputParser
from typing import Any

class CustomStructuredParser(BaseOutputParser):
    """自定义结构化解析器"""
    
    def __init__(self, expected_fields: List[str]):
        super().__init__()
        self.expected_fields = expected_fields
    
    def parse(self, text: str) -> Dict[str, Any]:
        """解析文本为结构化数据"""
        
        # 1. 清理文本
        cleaned_text = text.strip()
        
        # 2. 尝试JSON解析
        try:
            # 尝试提取JSON块
            import re
            json_pattern = re.compile(r'```(?:json)?\s*(.*?)\s*```', re.DOTALL)
            match = json_pattern.search(cleaned_text)
            
            if match:
                json_text = match.group(1)
            else:
                # 尝试直接解析
                json_text = cleaned_text
            
            data = json.loads(json_text)
            
            # 3. 验证必需字段
            missing_fields = []
            for field in self.expected_fields:
                if field not in data:
                    missing_fields.append(field)
            
            if missing_fields:
                raise ValueError(f"缺少必需字段: {missing_fields}")
            
            return data
            
        except json.JSONDecodeError as e:
            # 4. 回退到正则表达式提取
            extracted_data = {}
            
            for field in self.expected_fields:
                # 简单模式匹配
                pattern = re.compile(rf'{field}[：:]\s*([^\n\r]+)', re.IGNORECASE)
                match = pattern.search(cleaned_text)
                if match:
                    extracted_data[field] = match.group(1).strip()
            
            if len(extracted_data) >= len(self.expected_fields) // 2:
                return extracted_data
            else:
                raise ValueError(f"无法解析文本: {e}")
    
    @property
    def _type(self) -> str:
        """解析器类型"""
        return "custom_structured_parser"

```python
            input_variables=["text"],
            partial_variables={"structure": structure_text}
        )

        """提取文本为XML格式"""
        
        prompt = self.create_xml_prompt(structure)

  <currency>CNY</currency>
  <in_stock>true</in_stock>
</product>"""
    
    def parse_xml_response(self, xml_text: str) -> Dict:
        """解析XML响应"""
        
        try:
            # 解析XML
            root = ET.fromstring(xml_text)
            
            # 转换为字典
            result = {}
            for child in root:
                # 处理文本值
                if child.text:
                    # 尝试类型转换
                    text = child.text.strip()
                    if text.lower() in ['true', 'false']:
                        result[child.tag] = text.lower() == 'true'
                    elif text.isdigit():
                        result[child.tag] = int(text)
                    elif self._is_float(text):
                        result[child.tag] = float(text)
                    else:
                        result[child.tag] = text
                else:
                    result[child.tag] = None
            
            return {
                "success": True,
                "data": result,
                "root_tag": root.tag
            }
            
        except ET.ParseError as e:
            return {
                "success": False,
                "error": f"XML解析错误: {e}",
                "data": None
            }
    
    def _is_float(self, text: str) -> bool:
        """检查是否为浮点数"""
        try:
            float(text)
            return True
        except ValueError:
            return False
```

```python
from typing import Type, Any, Dict, Optional, List
from pydantic import BaseModel, Field, create_model
from enum import Enum
import json
import yaml

class OutputFormat(str, Enum):
    """输出格式枚举"""
    JSON = "json"
    XML = "xml"
    YAML = "yaml"
    CSV = "csv"
    PYDANTIC = "pydantic"

class FieldDefinition(BaseModel):
    """字段定义"""
    name: str = Field(description="字段名称")
    type: str = Field(description="字段类型：string, number, boolean, array, object")
    description: str = Field(description="字段描述")
    required: bool = Field(default=True, description="是否必需")
    default: Optional[Any] = Field(default=None, description="默认值")
    constraints: Optional[Dict[str, Any]] = Field(default=None, description="约束条件")
    
    class Config:
        json_schema_extra = {
            "example": {
                "name": "product_name",
                "type": "string",
                "description": "产品名称",
                "required": True,
                "default": None,
                "constraints": {"max_length": 100}
            }
        }

class SchemaDefinition(BaseModel):
    """模式定义"""
    name: str = Field(description="模式名称")
    description: str = Field(description="模式描述")
    fields: List[FieldDefinition] = Field(description="字段列表")
    output_format: OutputFormat = Field(default=OutputFormat.JSON, description="输出格式")
    
    def to_pydantic_model(self) -> Type[BaseModel]:
        """转换为Pydantic模型"""
        
        field_definitions = {}
        
        for field in self.fields:
            # 确定Python类型
            python_type = self._map_type_to_python(field.type, field.required)
            
            # 创建Field配置
            field_config = {
                "description": field.description,
                "default": ... if field.required else field.default
            }
            
            # 添加约束
            if field.constraints:
                field_config.update(field.constraints)
            
            field_definitions[field.name] = (python_type, Field(**field_config))
        
        # 动态创建Pydantic模型
        return create_model(
            self.name,
            __config__=type('Config', (), {
                'schema_extra': {
                    'example': self._generate_example()
                }
            }),
            **field_definitions
        )
    
    def _map_type_to_python(self, field_type: str, required: bool) -> Any:
        """映射字段类型到Python类型"""
        from typing import Optional, List
        
        type_map = {
            "string": str,
            "number": float,
            "integer": int,
            "boolean": bool,
            "array": List[str],  # 简化处理
            "object": Dict[str, Any]
        }
        
        python_type = type_map.get(field_type, str)
        
        if not required:
            return Optional[python_type]
        
        return python_type
    
    def _generate_example(self) -> Dict[str, Any]:
        """生成示例数据"""
        example = {}
        
        for field in self.fields:
            if field.type == "string":
                example[field.name] = f"示例{field.name}"
            elif field.type == "number":
                example[field.name] = 100.0
            elif field.type == "integer":
                example[field.name] = 100
            elif field.type == "boolean":
                example[field.name] = True
            elif field.type == "array":
                example[field.name] = ["示例1", "示例2"]
            elif field.type == "object":
                example[field.name] = {"key": "value"}
        
        return example
    
    def to_json_schema(self) -> Dict[str, Any]:
        """转换为JSON Schema"""
        
        properties = {}
        required_fields = []
        
        for field in self.fields:
            # 构建属性定义
            property_def = {
                "type": field.type,
                "description": field.description
            }
            
            # 添加约束
            if field.constraints:
                property_def.update(field.constraints)
            
            properties[field.name] = property_def
            
            if field.required:
                required_fields.append(field.name)
        
        return {
            "title": self.name,
            "description": self.description,
            "type": "object",
            "properties": properties,
            "required": required_fields
        }
    
    def to_xml_schema(self) -> str:
        """转换为XML Schema（简化）"""
        
        elements = []
        
        for field in self.fields:
            element = f"""  <element name="{field.name}" type="{field.type}">
    <annotation>
      <documentation>{field.description}</documentation>
    </annotation>
  </element>"""
            elements.append(element)
        
        return f"""<?xml version="1.0" encoding="UTF-8"?>
<schema xmlns="http://www.w3.org/2001/XMLSchema">
  <element name="{self.name}">
    <complexType>
      <sequence>
{chr(10).join(elements)}
      </sequence>
    </complexType>
  </element>
</schema>"""

class StructuredOutputManager:
    """结构化输出管理器"""
    
    def __init__(self):
        self.schemas: Dict[str, SchemaDefinition] = {}
        self.llm = ChatOpenAI(model="gpt-3.5-turbo", temperature=0.0)
    
    def register_schema(self, schema: SchemaDefinition):
        """注册模式"""
        self.schemas[schema.name] = schema
    
    def get_extraction_chain(self, schema_name: str):
        """获取提取链"""
        
        if schema_name not in self.schemas:
            raise ValueError(f"模式不存在: {schema_name}")
        
        schema = self.schemas[schema_name]
        
        # 根据输出格式选择不同的链
        if schema.output_format == OutputFormat.PYDANTIC:
            # 使用Pydantic模型
            pydantic_model = schema.to_pydantic_model()
            structured_llm = self.llm.with_structured_output(pydantic_model)
            
            prompt = ChatPromptTemplate.from_messages([
                ("system", f"请从文本中提取信息，遵循{schema.name}模式。"),
                ("human", "文本：{text}")
            ])
            
            return prompt | structured_llm
            
        elif schema.output_format == OutputFormat.JSON:
            # 使用JSON Schema
            json_schema = schema.to_json_schema()
            structured_llm = self.llm.with_structured_output(
                json_schema,
                method="json_schema"
            )
            
            prompt = ChatPromptTemplate.from_messages([
                ("system", f"请从文本中提取信息，输出JSON格式。"),
                ("human", "文本：{text}")
            ])
            
            return prompt | structured_llm
            
        elif schema.output_format == OutputFormat.XML:
            # 使用XML输出
            from langchain_core.output_parsers import XMLOutputParser
            
            xml_parser = XMLOutputParser()
            
            prompt_template = """请从以下文本中提取信息，输出为XML格式。

文本：{text}

XML结构要求：
{xml_structure}

请确保XML格式正确。"""

            prompt = PromptTemplate(
                template=prompt_template,
                input_variables=["text"],
                partial_variables={
                    "xml_structure": self._format_xml_structure(schema)
                }
            )
            
            return prompt | self.llm | xml_parser
            
        else:
            raise ValueError(f"不支持的输出格式: {schema.output_format}")
    
    def _format_xml_structure(self, schema: SchemaDefinition) -> str:
        """格式化XML结构描述"""
        structure_lines = []
        for field in schema.fields:
            structure_lines.append(f"<{field.name}>: {field.description}")
        return "\n".join(structure_lines)
    
    def validate_output(self, schema_name: str, output: Any) -> Dict[str, Any]:
        """验证输出"""
        
        if schema_name not in self.schemas:
            raise ValueError(f"模式不存在: {schema_name}")
        
        schema = self.schemas[schema_name]
        
        validation_result = {
            "schema": schema_name,
            "valid": True,
            "errors": [],
            "warnings": []
        }
        
        try:
            if schema.output_format == OutputFormat.PYDANTIC:
                # Pydantic验证
                pydantic_model = schema.to_pydantic_model()
                if isinstance(output, dict):
                    validated = pydantic_model(**output)
                    validation_result["validated_data"] = validated.dict()
                else:
                    validation_result["valid"] = False
                    validation_result["errors"].append("输出必须是字典类型")
            
            elif schema.output_format == OutputFormat.JSON:
                # JSON Schema验证（简化）
                if isinstance(output, dict):
                    # 检查必需字段
                    for field in schema.fields:
                        if field.required and field.name not in output:
                            validation_result["valid"] = False
                            validation_result["errors"].append(f"缺少必需字段: {field.name}")
                else:
                    validation_result["valid"] = False
                    validation_result["errors"].append("输出必须是字典类型")
            
            elif schema.output_format == OutputFormat.XML:
                # XML验证（简化）
                if not isinstance(output, str) or "<" not in output or ">" not in output:
                    validation_result["valid"] = False
                    validation_result["errors"].append("输出必须是XML字符串")
            
        except Exception as e:
            validation_result["valid"] = False
            validation_result["errors"].append(f"验证失败: {e}")
        
        return validation_result
```
