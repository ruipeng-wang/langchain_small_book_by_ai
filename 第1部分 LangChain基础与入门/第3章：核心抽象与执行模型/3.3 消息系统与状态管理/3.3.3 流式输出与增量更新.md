```python
def traditional_streaming_handler():
    """传统流式处理器：手动管理所有异步逻辑"""
    
    import asyncio
    import json
    from typing import AsyncGenerator, Dict, Any
    
    class ManualStreamProcessor:
        """手动流式处理器"""
        
        def __init__(self):
            self.buffer = ""
            self.callbacks = []
            self.is_streaming = False
            self.chunk_count = 0
        
        def add_callback(self, callback):
            """添加回调函数"""
            self.callbacks.append(callback)
        
        def _notify_callbacks(self, chunk: str, is_complete: bool = False):
            """通知所有回调函数"""
            for callback in self.callbacks:
                try:
                    callback(chunk, is_complete)
                except Exception as e:
                    print(f"回调函数错误: {e}")
        
        async def simulate_streaming_response(self, full_response: str, chunk_size: int = 10):
            """模拟流式响应"""
            self.is_streaming = True
            self.chunk_count = 0
            
            # 分块处理
            total_chunks = (len(full_response) + chunk_size - 1) // chunk_size
            
            for i in range(total_chunks):
                start = i * chunk_size
                end = min(start + chunk_size, len(full_response))
                chunk = full_response[start:end]
                
                # 更新缓冲区
                self.buffer += chunk
                self.chunk_count += 1
                
                # 通知回调
                is_complete = (i == total_chunks - 1)
                self._notify_callbacks(chunk, is_complete)
                
                # 模拟延迟
                await asyncio.sleep(0.05)
            
            self.is_streaming = False
            return self.buffer
        
        def get_stream_statistics(self) -> Dict[str, Any]:
            """获取流式统计"""
            return {
                "total_chunks": self.chunk_count,
                "buffer_length": len(self.buffer),
                "is_streaming": self.is_streaming,
                "completion_percentage": self.chunk_count * 10  # 简化计算
            }
    
    # 使用示例
    processor = ManualStreamProcessor()
    
    # 定义回调函数
    def print_callback(chunk: str, is_complete: bool):
        """打印回调"""
        print(f"收到块: '{chunk}' {'[完成]' if is_complete else ''}")

        """累积回调"""
        if not hasattr(accumulate_callback, 'accumulated'):
            accumulate_callback.accumulated = ""
        accumulate_callback.accumulated += chunk
        
        if is_complete:
            print(f"累积完成，总长度: {len(accumulate_callback.accumulated)}")

    processor.add_callback(accumulate_callback)

        print(f"\n流式完成，最终结果: {result[:50]}...")
        
        stats = processor.get_stream_statistics()
        print(f"流式统计: {stats}")
    
    # 运行流式处理

    # 3. 状态管理：需要手动跟踪流式状态
```

```python
from dataclasses import dataclass
from enum import Enum

class UpdateType(Enum):
    """更新类型"""
    TOKEN = "token"          # 新token
    TOOL_CALL = "tool_call"  # 工具调用
    TOOL_RESULT = "tool_result"  # 工具结果
    COMPLETION = "completion"    # 完成

@dataclass
class StreamUpdate:
    """流式更新"""
    type: UpdateType
    content: Any
    timestamp: str
    metadata: Dict[str, Any] = None
    
    def __post_init__(self):
        if self.metadata is None:
            self.metadata = {}

class IncrementalStateManager:
    """增量状态管理器"""
    
    def __init__(self):
        self.state = {
            "tokens": [],
            "tool_calls": [],
            "tool_results": [],
            "full_response": "",
            "is_complete": False
        }
        self.update_history: List[StreamUpdate] = []
        self.subscribers = []
    
    def process_update(self, update: StreamUpdate):
        """处理更新"""
        self.update_history.append(update)
        
        # 更新状态
        if update.type == UpdateType.TOKEN:
            token = update.content
            self.state["tokens"].append(token)
            self.state["full_response"] += token
            
        elif update.type == UpdateType.TOOL_CALL:
            tool_call = update.content
            self.state["tool_calls"].append(tool_call)
            
        elif update.type == UpdateType.TOOL_RESULT:
            tool_result = update.content
            self.state["tool_results"].append(tool_result)
            
        elif update.type == UpdateType.COMPLETION:
            self.state["is_complete"] = True
        
        # 通知订阅者
        self._notify_subscribers(update)
    
    def _notify_subscribers(self, update: StreamUpdate):
        """通知订阅者"""
        for subscriber in self.subscribers:
            try:
                subscriber(update, self.state.copy())
            except Exception as e:
                print(f"订阅者通知错误: {e}")
    
    def subscribe(self, callback):
        """订阅更新"""
        self.subscribers.append(callback)
    
    def get_state_summary(self) -> Dict[str, Any]:
        """获取状态摘要"""
        return {
            "total_tokens": len(self.state["tokens"]),
            "total_tool_calls": len(self.state["tool_calls"]),
            "total_tool_results": len(self.state["tool_results"]),
            "response_length": len(self.state["full_response"]),
            "is_complete": self.state["is_complete"],
            "update_count": len(self.update_history)
        }
    
    def get_recent_updates(self, n: int = 5) -> List[StreamUpdate]:
        """获取最近更新"""
        return self.update_history[-n:] if self.update_history else []
```

```python
# 高级流式模式
from typing import List, Dict, Any
from enum import Enum

class StreamMode(Enum):
    """流式模式"""
    VALUES = "values"        # 完整状态值
    UPDATES = "updates"      # 状态更新
    MESSAGES = "messages"    # 消息流
    DEBUG = "debug"          # 调试信息
    CUSTOM = "custom"        # 自定义数据

class AdvancedStreamManager:
    """高级流式管理器"""
    
    def __init__(self, supported_modes: List[StreamMode] = None):
        self.supported_modes = supported_modes or [
            StreamMode.VALUES,
            StreamMode.UPDATES,
            StreamMode.MESSAGES
        ]
        self.mode_handlers = self._initialize_handlers()
        self.active_modes = []
    
    def _initialize_handlers(self) -> Dict[StreamMode, Any]:
        """初始化模式处理器"""
        return {
            StreamMode.VALUES: self._handle_values_mode,
            StreamMode.UPDATES: self._handle_updates_mode,
            StreamMode.MESSAGES: self._handle_messages_mode,
            StreamMode.DEBUG: self._handle_debug_mode,
            StreamMode.CUSTOM: self._handle_custom_mode
        }
    
    def _handle_values_mode(self, data: Any, metadata: Dict[str, Any]):
        """处理VALUES模式"""
        # 返回完整状态值
        return {
            "type": "values",
            "data": data,
            "timestamp": metadata.get("timestamp", ""),
            "sequence": metadata.get("sequence", 0)
        }
    
    def _handle_updates_mode(self, data: Any, metadata: Dict[str, Any]):
        """处理UPDATES模式"""
        # 返回状态更新
        return {
            "type": "updates",
            "data": {
                "delta": data,  # 状态变化量
                "full_state": metadata.get("full_state", {})
            },
            "timestamp": metadata.get("timestamp", "")
        }
    
    def _handle_messages_mode(self, data: Any, metadata: Dict[str, Any]):
        """处理MESSAGES模式"""
        # 返回消息流（token + 元数据）
        return {
            "type": "messages",
            "data": (data, metadata),  # (token, 元数据)二元组
            "timestamp": metadata.get("timestamp", "")
        }
    
    def _handle_debug_mode(self, data: Any, metadata: Dict[str, Any]):
        """处理DEBUG模式"""
        # 返回调试信息
        return {
            "type": "debug",
            "data": {
                "raw": data,
                "metadata": metadata,
                "processing_time_ms": metadata.get("processing_time_ms", 0),
                "memory_usage_mb": metadata.get("memory_usage_mb", 0)
            },
            "timestamp": metadata.get("timestamp", "")
        }
    
    def _handle_custom_mode(self, data: Any, metadata: Dict[str, Any]):
        """处理CUSTOM模式"""
        # 返回自定义数据
        custom_config = metadata.get("custom_config", {})
        return {
            "type": "custom",
            "data": {
                "original": data,
                "processed": self._process_custom_data(data, custom_config),
                "config": custom_config
            },
            "timestamp": metadata.get("timestamp", "")
        }
    
    def _process_custom_data(self, data: Any, config: Dict[str, Any]) -> Any:
        """处理自定义数据"""
        # 根据配置处理数据
        processing_type = config.get("processing", "raw")
        
        if processing_type == "summary":
            # 生成摘要
            if isinstance(data, str):
                return data[:100] + "..." if len(data) > 100 else data
            else:
                return str(data)[:100] + "..."
        
        elif processing_type == "metrics":
            # 计算指标
            if isinstance(data, str):
                return {
                    "length": len(data),
                    "word_count": len(data.split()),
                    "contains_chinese": any('\u4e00' <= c <= '\u9fff' for c in data)
                }
            else:
                return {"type": type(data).__name__, "size": len(str(data))}
        
        else:
            # 原始数据
            return data
    
    def set_active_modes(self, modes: List[StreamMode]):
        """设置活动模式"""
        valid_modes = [m for m in modes if m in self.supported_modes]
        self.active_modes = valid_modes
        return valid_modes
    
    def process_stream_data(self, data: Any, metadata: Dict[str, Any] = None) -> List[Dict[str, Any]]:
        """处理流式数据"""
        if metadata is None:
            metadata = {}
        
        results = []
        
        for mode in self.active_modes:
            handler = self.mode_handlers.get(mode)
            if handler:
                try:
                    result = handler(data, metadata)
                    results.append(result)
                except Exception as e:
                    print(f"模式 {mode} 处理错误: {e}")
                    # 添加错误结果
                    results.append({
                        "type": "error",
                        "mode": mode.value,
                        "error": str(e),
                        "timestamp": metadata.get("timestamp", "")
                    })
        
        return results
    
    def create_multi_mode_stream(self, data_source, modes: List[StreamMode] = None):
        """创建多模式流"""
        if modes:
            self.set_active_modes(modes)
        
        import time
        
        def stream_generator():
            """流式生成器"""
            sequence = 0
            
            for data_item in data_source:
                sequence += 1
                
                # 准备元数据
                metadata = {
                    "timestamp": time.time(),
                    "sequence": sequence,
                    "source": "stream_generator",
                    "item_count": sequence
                }
                
                # 处理数据
                processed_results = self.process_stream_data(data_item, metadata)
                
                # 返回结果
                for result in processed_results:
                    yield result
                
                # 模拟延迟
                time.sleep(0.1)
        
        return stream_generator()
```

```python
                # 添加到缓冲区
                self.token_buffer.append({
                    "token": token,
                    "timestamp": current_time,
                    "index": len(self.token_buffer)
                })
                
                # 实时计算指标
                if len(self.token_buffer) >= 2:
                    time_diff = (current_time - self.start_time).total_seconds()
                    tokens_per_second = len(self.token_buffer) / time_diff if time_diff > 0 else 0
                    self.stream_metrics.tokens_per_second = tokens_per_second
                
                # 流式发送给学习者
                self.system._stream_to_learner(
                    self.session_id,
                    {
                        "type": "explanation_chunk",
                        "content": token,
                        "chunk_index": len(self.token_buffer),
                        "metrics": {
                            "tokens_per_second": self.stream_metrics.tokens_per_second
                        }
                    }
                )
            
            def on_llm_end(self, response, **kwargs):
                """LLM结束"""
                end_time = datetime.now()
                duration = (end_time - self.start_time).total_seconds()
                
                # 计算最终指标
                full_explanation = "".join([t["token"] for t in self.token_buffer])
                
                # 简单的内容质量评估
                explanation_length = len(full_explanation)
                word_count = len(full_explanation.split())
                unique_words = len(set(full_explanation.split()))
                
                diversity_score = unique_words / word_count if word_count > 0 else 0
                detail_score = min(explanation_length / 500, 1.0)  # 500字符为理想长度
                
                self.stream_metrics.response_quality = (diversity_score + detail_score) / 2
                
                # 通知学习完成
                self.system._notify_learning_event(
                    self.session_id,
                    "explanation_completed",
                    {
                        "full_explanation": full_explanation,
                        "duration_seconds": duration,
                        "total_chunks": len(self.token_buffer),
                        "metrics": {
                            "tokens_per_second": self.stream_metrics.tokens_per_second,
                            "response_quality": self.stream_metrics.response_quality,
                            "explanation_length": explanation_length,
                            "word_count": word_count
                        }
                    }
                )
        
        self.callback_class = EducationalCallback

        session_id = f"learn_{student_id}_{datetime.now().strftime('%Y%m%d_%H%M%S')}"

        callback = self.callback_class(session_id, self)
        
        session = {
            "session_id": session_id,
            "student_id": student_id,
            "learning_topic": learning_topic,
            "created_at": datetime.now().isoformat(),
            "questions_asked": 0,
            "explanations_given": 0,
            "total_learning_time": 0,
            "callback": callback,
            "learner_handlers": [],
            "learning_history": []
        }
        
        self.sessions[session_id] = session
        self.learning_metrics[session_id] = LearningStreamMetrics()
        
        print(f"创建学习会话: {session_id}")
        print(f"学习主题: {learning_topic}")
        
        return session

            return True
        return False

        if session_id not in self.sessions:
            return
        
        session = self.sessions[session_id]
        
        # 调用所有学习者处理器
        for handler in session["learner_handlers"]:
            try:
                handler(data)
            except Exception as e:
                print(f"学习者处理器错误: {e}")
        
        # 添加到学习历史
        if data["type"] == "explanation_chunk":
            session["learning_history"].append({
                "type": "chunk",
                "timestamp": datetime.now().isoformat(),
                "data": data
            })

        self._stream_to_learner(session_id, {
            "type": "learning_event",
            "event": event_type,
            "data": data
        })
        
        # 更新会话数据
        if session_id in self.sessions:
            session = self.sessions[session_id]

                session["explanations_given"] += 1

                    self.learning_metrics[session_id].tokens_per_second = metrics.get("tokens_per_second", 0)
```

```python
class StreamUXOptimizer:

        """应用优化"""
        if optimization_config:
            self.optimizations.update(optimization_config)

        for i, chunk in enumerate(stream_data):
            optimized_chunk = self._optimize_chunk(chunk, i)
            optimized_stream.append(optimized_chunk)

        if self.optimizations["progressive_display"] and index == 0:
            optimized["metadata"] = optimized.get("metadata", {})
            optimized["metadata"]["first_chunk"] = True
            optimized["metadata"]["display_delay_ms"] = 0  # 立即显示
        
        elif self.optimizations["progressive_display"]:
            # 后续chunk延迟显示，创造渐进效果
            delay_ms = min(50, index * 10)  # 最大50ms延迟
            optimized["metadata"] = optimized.get("metadata", {})
            optimized["metadata"]["display_delay_ms"] = delay_ms
        
        # 进度指示
        if self.optimizations["progress_bar"] and index % 10 == 0:
            progress = min((index + 1) / 100, 1.0)  # 假设最多100个chunk
            optimized["metadata"]["progress"] = progress
        
        # 预计时间
        if self.optimizations["estimated_time"] and index == 5:
            # 基于前5个chunk的速度估算
            if hasattr(self, '_first_five_times'):

        if index == 0:

            # 初始化时间记录
            if not hasattr(self, '_chunk_times'):
                self._chunk_times = []
                self._first_five_times = []
            
            self._chunk_times.append(current_time)
            
            if len(self._chunk_times) >= 2:
                interval = self._chunk_times[-1] - self._chunk_times[-2]
                self._first_five_times.append(interval)
                
                if len(self._first_five_times) <= 5:
```

```python
# 流式错误处理与监控
from typing import Dict, List, Any
from datetime import datetime
import time

class StreamErrorMonitor:
    """流式错误监控器"""
    
    def __init__(self):
        self.error_log: List[Dict[str, Any]] = []
        self.recovery_attempts = {}
        self.metrics = {
            "total_streams": 0,
            "successful_streams": 0,
            "failed_streams": 0,
            "total_errors": 0,
            "recovery_success_rate": 0.0
        }
        
        # 错误阈值配置
        self.error_thresholds = {
            "consecutive_errors": 3,
            "error_rate_per_minute": 10,  # 每分钟最多10个错误
            "recovery_timeout_seconds": 30
        }
    
    def record_error(self, error_type: str, error_msg: str, stream_id: str = None, context: Dict[str, Any] = None):
        """记录错误"""
        error_entry = {
            "timestamp": datetime.now().isoformat(),
            "error_type": error_type,
            "error_message": error_msg,
            "stream_id": stream_id,
            "context": context or {},
            "recovered": False
        }
        
        self.error_log.append(error_entry)
        self.metrics["total_errors"] += 1
        
        # 更新流式统计
        if stream_id:
            if error_type == "stream_failed":
                self.metrics["failed_streams"] += 1
            elif error_type == "stream_completed":
                self.metrics["successful_streams"] += 1
        
        print(f"📛 记录错误: {error_type} - {error_msg[:50]}...")
        
        # 检查是否超过阈值
        self._check_error_thresholds()
        
        return error_entry
    
    def record_recovery(self, error_entry: Dict[str, Any], recovery_method: str):
        """记录恢复"""
        error_entry["recovered"] = True
        error_entry["recovery_method"] = recovery_method
        error_entry["recovery_timestamp"] = datetime.now().isoformat()
        
        # 更新恢复统计
        stream_id = error_entry.get("stream_id")
        if stream_id:
            if stream_id not in self.recovery_attempts:
                self.recovery_attempts[stream_id] = {"attempts": 0, "successes": 0}
            
            self.recovery_attempts[stream_id]["attempts"] += 1
            self.recovery_attempts[stream_id]["successes"] += 1
        
        # 更新总体恢复率
        total_recoveries = sum(1 for e in self.error_log if e.get("recovered"))
        total_errors = len(self.error_log)
        
        if total_errors > 0:
            self.metrics["recovery_success_rate"] = total_recoveries / total_errors
        
        print(f"✅ 恢复成功: {recovery_method}")
    
    def _check_error_thresholds(self):
        """检查错误阈值"""
        # 检查连续错误
        recent_errors = self.error_log[-self.error_thresholds["consecutive_errors"]:]
        if len(recent_errors) >= self.error_thresholds["consecutive_errors"]:
            all_recent_failed = all(not e.get("recovered", False) for e in recent_errors)
            if all_recent_failed:
                print(f"⚠️  警告: 连续{self.error_thresholds['consecutive_errors']}个错误未恢复")
        
        # 检查错误频率（最近1分钟）
        one_minute_ago = datetime.now().timestamp() - 60
        recent_errors_count = sum(
            1 for e in self.error_log 
            if datetime.fromisoformat(e["timestamp"]).timestamp() > one_minute_ago
        )
        
        if recent_errors_count > self.error_thresholds["error_rate_per_minute"]:
            print(f"⚠️  警告: 错误频率过高 - 最近1分钟{recent_errors_count}个错误")
    
    def get_error_report(self, time_window_minutes: int = 5) -> Dict[str, Any]:
        """获取错误报告"""
        cutoff_time = datetime.now().timestamp() - (time_window_minutes * 60)
        
        recent_errors = [
            e for e in self.error_log
            if datetime.fromisoformat(e["timestamp"]).timestamp() > cutoff_time
        ]
        
        # 错误分类
        error_types = {}
        for error in recent_errors:
            error_type = error["error_type"]
            error_types[error_type] = error_types.get(error_type, 0) + 1
        
        # 恢复统计
        recovered_count = sum(1 for e in recent_errors if e.get("recovered", False))
        recovery_rate = recovered_count / len(recent_errors) if recent_errors else 1.0
        
        return {
            "time_window_minutes": time_window_minutes,
            "total_errors": len(recent_errors),
            "error_types": error_types,
            "recovery_rate": recovery_rate,
            "recent_errors": recent_errors[-5:],  # 最近5个错误
            "system_metrics": self.metrics.copy(),
            "recommendations": self._generate_error_recommendations(recent_errors)
        }
    
    def _generate_error_recommendations(self, recent_errors: List[Dict[str, Any]]) -> List[str]:
        """生成错误处理建议"""
        recommendations = []
        
        if not recent_errors:
            return ["✅ 系统运行正常，无错误"]
        
        # 分析错误类型
        network_errors = sum(1 for e in recent_errors if "network" in e["error_type"].lower())
        timeout_errors = sum(1 for e in recent_errors if "timeout" in e["error_type"].lower())
        model_errors = sum(1 for e in recent_errors if "model" in e["error_type"].lower())
        
        if network_errors > len(recent_errors) * 0.5:  # 超过50%是网络错误
            recommendations.append("网络错误较多，建议检查网络连接和API端点")
        
        if timeout_errors > 3:
            recommendations.append("超时错误频繁，建议增加超时时间或优化模型配置")
        
        if model_errors > 2:
            recommendations.append("模型错误出现，建议检查模型配置和凭证")
        
        # 恢复率建议
        recovery_rate = sum(1 for e in recent_errors if e.get("recovered", False)) / len(recent_errors)
        if recovery_rate < 0.7:  # 恢复率低于70%
            recommendations.append(f"错误恢复率较低({recovery_rate:.0%})，建议增强恢复机制")
        
        return recommendations
    
    def create_error_handling_stream(self, data_stream, stream_id: str = None):
        """创建带错误处理的流"""
        self.metrics["total_streams"] += 1
        
        def error_handling_generator():
            """带错误处理的生成器"""
            consecutive_errors = 0
            
            for data in data_stream:
                try:
                    # 处理数据
                    yield data
                    consecutive_errors = 0  # 重置连续错误计数
                    
                except Exception as e:
                    # 记录错误
                    error_type = type(e).__name__
                    self.record_error(
                        error_type,
                        str(e),
                        stream_id,
                        {"data_item": str(data)[:100]}
                    )
                    
                    consecutive_errors += 1
                    
                    # 尝试恢复
                    if consecutive_errors <= self.error_thresholds["consecutive_errors"]:
                        print(f"尝试恢复，连续错误: {consecutive_errors}")
                        # 可以在这里实现具体的恢复逻辑
                        # 例如：重试、跳过、使用备用数据等
                        
                        # 模拟恢复成功
                        recovery_entry = self.error_log[-1]
                        self.record_recovery(recovery_entry, "skip_and_continue")
                        
                    else:
                        # 超过阈值，停止流
                        print(f"❌ 连续错误超过阈值，停止流")
                        self.record_error("stream_failed", "连续错误超过阈值", stream_id)
                        break
            
            # 流完成
            self.record_error("stream_completed", "流式处理完成", stream_id)
        
        return error_handling_generator()
```
