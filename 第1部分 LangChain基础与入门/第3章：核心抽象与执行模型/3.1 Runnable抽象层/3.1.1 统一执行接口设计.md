# 3.1.1 统一执行接口设计

## 从传统函数调用到统一执行接口的演进

### 传统AI应用开发的接口碎片化问题

在LangChain v1.2.22之前，AI应用开发面临严重的接口碎片化问题。不同的组件、模型和服务提供商都有各自独特的调用方式，导致代码难以维护、复用和测试。

```python
# 传统AI应用开发中的接口碎片化示例
class FragmentedAISystem:
    def __init__(self):
        # OpenAI GPT调用方式
        self.openai_client = OpenAI(api_key="sk-...")
        
        # Hugging Face模型调用方式
        self.hf_pipeline = pipeline("text-generation", model="gpt2")
        
        # 自定义规则引擎
        self.rule_engine = RuleEngine()
        
        # 外部API服务
        self.external_api = ExternalAPIClient()
    
    def process_query(self, query: str) -> str:
        """处理用户查询（接口碎片化严重）"""
        result = ""
        
        # 1. 调用OpenAI GPT（REST API方式）
        try:
            openai_response = self.openai_client.chat.completions.create(
                model="gpt-4",
                messages=[
                    {"role": "system", "content": "You are a helpful assistant."},
                    {"role": "user", "content": query}
                ]
            )
            gpt_result = openai_response.choices[0].message.content
        except Exception as e:
            gpt_result = f"OpenAI调用失败: {e}"
        
        # 2. 调用Hugging Face模型（Pipeline方式）
        try:
            hf_inputs = self.hf_tokenizer(query, return_tensors="pt")
            hf_outputs = self.hf_model.generate(**hf_inputs)
            hf_result = self.hf_tokenizer.decode(hf_outputs[0], skip_special_tokens=True)
        except Exception as e:
            hf_result = f"Hugging Face调用失败: {e}"
        
        # 3. 调用规则引擎（自定义接口）
        try:
            rule_result = self.rule_engine.apply_rules(query)
        except Exception as e:
            rule_result = f"规则引擎失败: {e}"
        
        # 4. 调用外部API（HTTP请求方式）
        try:
            api_response = self.external_api.post("/process", json={"text": query})
            api_result = api_response.json().get("result", "")
        except Exception as e:
            api_result = f"API调用失败: {e}"
        
        # 5. 结果合并（手动处理）
        all_results = {
            "gpt": gpt_result,
            "huggingface": hf_result,
            "rules": rule_result,
            "api": api_result
        }
        
        # 6. 最终决策（硬编码逻辑）
        if len(query) < 10:
            result = all_results["rules"]
        elif "technical" in query.lower():
            result = all_results["gpt"]
        elif "creative" in query.lower():
            result = all_results["huggingface"]
        else:
            result = all_results["api"]
        
        return result
    
    def batch_process(self, queries: List[str]) -> List[str]:
        """批量处理查询（手动实现）"""
        results = []
        for query in queries:
            try:
                result = self.process_query(query)
                results.append(result)
            except Exception as e:
                results.append(f"处理失败: {e}")
        return results
    
    def stream_process(self, query: str):
        """流式处理查询（复杂的手动实现）"""
        # 模拟流式处理（实际实现更复杂）
        yield f"开始处理: {query}\n"
        import time
        
        for i in range(5):
            time.sleep(0.5)
            yield f"处理进度: {20 * (i + 1)}%\n"
        
        result = self.process_query(query)
        yield f"最终结果: {result}"
    
    async def async_process(self, query: str) -> str:
        """异步处理查询（需要手动管理）"""
        import asyncio
        
        # 模拟异步调用
        async def call_openai():
            await asyncio.sleep(0.5)
            return self.process_query(query)
        
        return await call_openai()

# 传统方式的问题总结：
# 1. 接口不统一：每个服务有不同的调用方式
# 2. 错误处理分散：每个调用需要单独的错误处理
# 3. 难以组合：无法轻松将多个服务组合成管道
# 4. 缺少抽象：无法统一处理同步、异步、批量、流式
# 5. 测试困难：每个组件需要单独的mock和测试
```

### LangChain v1.2.22的解决方案：Runnable统一接口

LangChain v1.2.22通过引入**Runnable接口**解决了接口碎片化问题。Runnable是LangChain的核心抽象层，为所有组件提供了统一的执行接口。

```python
# LangChain Runnable接口的简化定义
from typing import Any, Dict, List, Optional, AsyncIterator, Iterator
from abc import ABC, abstractmethod
from pydantic import BaseModel

class RunnableConfig(BaseModel):
    """可运行对象的配置"""
    callbacks: Optional[List[Any]] = None
    tags: Optional[List[str]] = None
    metadata: Optional[Dict[str, Any]] = None
    run_name: Optional[str] = None
    max_concurrency: Optional[int] = None

class Runnable(ABC):
    """Runnable抽象基类 - 所有LangChain组件的统一接口"""
    
    @abstractmethod
    def invoke(self, input: Any, config: Optional[RunnableConfig] = None) -> Any:
        """同步执行单个输入"""
        pass
    
    @abstractmethod
    async def ainvoke(self, input: Any, config: Optional[RunnableConfig] = None) -> Any:
        """异步执行单个输入"""
        pass
    
    @abstractmethod
    def batch(self, inputs: List[Any], config: Optional[RunnableConfig] = None) -> List[Any]:
        """批量执行多个输入"""
        pass
    
    @abstractmethod
    async def abatch(self, inputs: List[Any], config: Optional[RunnableConfig] = None) -> List[Any]:
        """异步批量执行多个输入"""
        pass
    
    @abstractmethod
    def stream(self, input: Any, config: Optional[RunnableConfig] = None) -> Iterator[Any]:
        """流式执行（同步）"""
        pass
    
    @abstractmethod
    async def astream(self, input: Any, config: Optional[RunnableConfig] = None) -> AsyncIterator[Any]:
        """流式执行（异步）"""
        pass
    
    def __or__(self, other: "Runnable") -> "RunnableSequence":
        """管道操作符重载，支持链式组合"""
        from langchain_core.runnables import RunnableSequence
        return RunnableSequence(self, other)
    
    def __ror__(self, other: Any) -> "Runnable":
        """反向管道操作符重载"""
        return self
```

### Runnable接口的核心设计哲学

**1. 统一接口设计模式**
- **单一职责**：每个Runnable只做一件事
- **开放封闭**：对扩展开放，对修改封闭
- **里氏替换**：任何Runnable都可以替换另一个Runnable
- **接口隔离**：最小化接口依赖

**2. 执行模式标准化**
- **同步执行**：`invoke()` - 阻塞调用，返回完整结果
- **异步执行**：`ainvoke()` - 非阻塞调用，返回Future
- **批量执行**：`batch()` - 高效处理多个输入
- **流式执行**：`stream()`, `astream()` - 实时处理数据流

**3. 配置管理统一化**
- **回调系统**：统一的事件回调机制
- **标签系统**：统一的组件标记和追踪
- **元数据**：统一的上下文传递机制
- **并发控制**：统一的并行度管理

```python
from langchain_huggingface import HuggingFacePipeline
from langchain_core.runnables import RunnableLambda
```

```python
# 传统批量处理 vs Runnable批量处理对比
import time

# 传统批量处理方式
class TraditionalBatchProcessing:
    def __init__(self):
        self.counter = 0
    
    def process_item(self, item: str) -> str:
        """处理单个项目（模拟耗时操作）"""
        time.sleep(0.1)  # 模拟处理时间
        self.counter += 1
        return f"Processed_{self.counter}: {item}"
    
    def batch_process(self, items: List[str]) -> List[str]:
        """传统批量处理（顺序执行）"""
        results = []
        for item in items:
            results.append(self.process_item(item))
        return results
    
    def parallel_batch_process(self, items: List[str]) -> List[str]:
        """传统并行批量处理（手动线程管理）"""
        from concurrent.futures import ThreadPoolExecutor
        
        with ThreadPoolExecutor(max_workers=5) as executor:
            futures = [executor.submit(self.process_item, item) for item in items]
            results = [future.result() for future in futures]
        
        return results

# Runnable批量处理方式
from langchain_core.runnables import RunnableLambda

class RunnableBatchProcessing:
    def __init__(self):
        self.processor = RunnableLambda(self._process_item)
        self.counter = 0
    
    def _process_item(self, item: str) -> str:
        """处理单个项目"""
        time.sleep(0.1)  # 模拟处理时间
        self.counter += 1
        return f"Processed_{self.counter}: {item}"
    
    def batch_process(self, items: List[str]) -> List[str]:
        """使用Runnable批量处理"""
        # 自动优化执行策略
        return self.processor.batch(items)
```

```python
# 传统工作流组合
class TraditionalWorkflow:
    def __init__(self):
        self.preprocessor = TextPreprocessor()
        self.classifier = IntentClassifier()
        self.generator = ResponseGenerator()
        self.formatter = OutputFormatter()
    
    def execute(self, input_text: str) -> str:
        """传统工作流执行 - 硬编码顺序"""
        # 步骤1: 预处理
        processed = self.preprocessor.process(input_text)
        
        # 步骤2: 意图分类
        intent = self.classifier.classify(processed)
        
        # 步骤3: 生成响应
        response = self.generator.generate(processed, intent)
        
        # 步骤4: 格式化输出
        formatted = self.formatter.format(response)
        
        return formatted
    
    def execute_async(self, input_text: str) -> str:
        """传统异步工作流 - 更复杂"""
        # 需要手动管理异步调用链
        pass

# Runnable工作流组合
from langchain_core.runnables import RunnableSequence

class RunnableWorkflow:
    def __init__(self):
        # 声明式工作流组合
        self.workflow = (
            RunnableLambda(lambda x: self._preprocess(x))
            | RunnableLambda(lambda x: self._classify(x))
            | RunnableLambda(lambda x: self._generate(x))
            | RunnableLambda(lambda x: self._format(x))
        )
    
    def _preprocess(self, text: str) -> Dict:
        return {"text": text.lower(), "original": text}
    
    def _classify(self, data: Dict) -> Dict:
        text = data["text"]
        intent = "general" if len(text) < 50 else "detailed"
        return {**data, "intent": intent}
    
    def _generate(self, data: Dict) -> Dict:
        response = f"Response to '{data['original']}' (intent: {data['intent']})"
        return {**data, "response": response}
    
    def _format(self, data: Dict) -> str:
        return f"【{data['intent'].upper()}】 {data['response']}"
    
    def execute(self, input_text: str) -> str:
        """使用Runnable工作流执行"""
        return self.workflow.invoke(input_text)
    
    async def execute_async(self, input_text: str) -> str:
        """异步执行工作流"""
        return await self.workflow.ainvoke(input_text)
    
    def stream_execute(self, input_text: str):
        """流式执行工作流"""
        return self.workflow.stream(input_text)
```

```python
# Runnable中间件系统
from langchain_core.runnables import RunnableLambda

class RunnableMiddlewareSystem:
    def __init__(self):
        self.base_processor = RunnableLambda(lambda x: f"Processed: {x}")
    
    def add_logging_middleware(self, runnable):
        """添加日志中间件"""
        def logging_wrapper(input_data):
            print(f"🚀 开始处理: {input_data}")
            result = runnable.invoke(input_data)
            print(f"✅ 处理完成: {result}")
            return result
        
        return RunnableLambda(logging_wrapper)
    
    def add_error_handling_middleware(self, runnable):
        """添加错误处理中间件"""
        def error_wrapper(input_data):
            try:
                return runnable.invoke(input_data)
            except Exception as e:
                print(f"❌ 处理错误: {e}")
                return f"Fallback for: {input_data}"
        
        return RunnableLambda(error_wrapper)
    
    def add_caching_middleware(self, runnable):
        """添加缓存中间件"""
        cache = {}
        
        def caching_wrapper(input_data):
            if input_data in cache:
                print(f"📦 缓存命中: {input_data}")
                return cache[input_data]
            
            result = runnable.invoke(input_data)
            cache[input_data] = result
            print(f"💾 缓存保存: {input_data}")
            return result
        
        return RunnableLambda(caching_wrapper)
    
    def demonstrate_middleware(self):
        """演示中间件组合"""
        # 构建带中间件的处理器
        processor = self.base_processor
        processor = self.add_logging_middleware(processor)
        processor = self.add_error_handling_middleware(processor)
        processor = self.add_caching_middleware(processor)
        
        print("第一次调用（会处理并缓存）:")
        result1 = processor.invoke("test input")
        
        print("\n第二次调用（从缓存读取）:")
        result2 = processor.invoke("test input")
        
        print("\n错误输入测试:")
        error_processor = RunnableLambda(lambda x: 1/0)  # 会出错的处理器
        error_processor = self.add_error_handling_middleware(error_processor)
        result3 = error_processor.invoke("error test")
        
        return result1, result2, result3
```

#### 步骤2：包装为Runnable

```python
# 包装传统组件为Runnable
from langchain_core.runnables import RunnableLambda

class RunnableMigration:
    """Runnable迁移工具类"""
    
    @staticmethod
    def wrap_legacy_function(func):
        """包装传统函数为Runnable"""
        return RunnableLambda(func)
    
    @staticmethod
    def migrate_legacy_system(legacy_system):
        """迁移整个传统系统"""
        # 包装每个组件
        cleaner_runnable = RunnableLambda(legacy_system.text_cleaner)
        analyzer_runnable = RunnableLambda(legacy_system.sentiment_analyzer)
        formatter_runnable = RunnableLambda(
            lambda x: legacy_system.response_formatter(x["sentiment"], x["text"])
        )
        
        # 构建Runnable管道
        runnable_pipeline = (
            cleaner_runnable
            | RunnableLambda(lambda text: {"text": text, "sentiment": None})
            | RunnableLambda(lambda data: {
                **data,
                "sentiment": analyzer_runnable.invoke(data["text"])
            })
            | formatter_runnable
        )
        
        return runnable_pipeline
    
    def demonstrate_migration(self):
        """演示迁移过程"""
        legacy = LegacyComponents()
        
        print("传统管道执行:")
        legacy_result = legacy.legacy_pipeline("This is a GOOD product!")
        print(f"结果: {legacy_result}")
        
        print("\n迁移到Runnable管道:")
        runnable_pipeline = self.migrate_legacy_system(legacy)
        
        # 使用统一的Runnable接口
        print("1. 同步执行:")
        sync_result = runnable_pipeline.invoke("This is a GOOD product!")
        print(f"  结果: {sync_result}")
        
        print("\n2. 批量执行:")
        batch_result = runnable_pipeline.batch([
            "This is GOOD",
            "This is bad",
            "Excellent work!"
        ])
        print(f"  结果: {batch_result}")
        
        print("\n3. 流式执行:")
        for chunk in runnable_pipeline.stream("This is a GOOD product!"):
            print(f"  流式块: {chunk}")
        
        return legacy_result, sync_result, batch_result
```

```python
# Runnable统一错误处理
class RunnableErrorHandling:
    def __init__(self):
        self.llm = ChatOpenAI(model="gpt-4")
    
    def demonstrate_error_handling(self):
        """演示Runnable错误处理"""
        
        # 1. 创建可能失败的Runnable
        unstable_runnable = RunnableLambda(
            lambda x: 1/0 if "error" in x else f"Success: {x}"
        )
        
        # 2. 添加错误处理的装饰器
        from functools import wraps
        
        def with_error_handling(runnable):
            @wraps(runnable.invoke)
            def safe_invoke(input_data, config=None):
                try:
                    return runnable.invoke(input_data, config)
                except Exception as e:
                    return f"Error handled: {e}"
            
            # 创建新的Runnable包装器
            return RunnableLambda(safe_invoke)
        
        # 3. 使用带错误处理的Runnable
        safe_runnable = with_error_handling(unstable_runnable)
        
        print("正常输入:")
        result1 = safe_runnable.invoke("normal input")
        print(f"结果: {result1}")
        
        print("\n错误输入:")
        result2 = safe_runnable.invoke("trigger error")
```

```python
import time
from contextlib import contextmanager

class RunnablePerformanceMonitoring:
    def __init__(self):
        self.llm = ChatOpenAI(model="gpt-4")
    
    @contextmanager
    def measure_performance(self, operation_name: str):
        """性能测量上下文管理器"""
        start_time = time.time()
        start_memory = self._get_memory_usage()
        
        try:
            yield
        finally:
            end_time = time.time()
            end_memory = self._get_memory_usage()
            
            print(f"\n📊 性能报告 - {operation_name}:")
            print(f"   执行时间: {end_time - start_time:.3f}秒")
            print(f"   内存变化: {end_memory - start_memory:.2f} MB")
    
    def _get_memory_usage(self) -> float:
        """获取内存使用量"""
        import psutil
        return psutil.Process().memory_info().rss / 1024 / 1024
    
    def demonstrate_performance_monitoring(self):
        """演示性能监控"""
        
        print("性能监控演示:")
        
        # 1. 单个调用性能
        with self.measure_performance("单个调用"):

            print(f"结果: {result.content[:50]}...")
```
