# 3.1.3 配置管理与上下文传递

## 从硬编码配置到动态上下文管理的演进

### 传统配置管理的局限性

在传统的AI应用开发中，配置管理通常采用硬编码、全局变量或配置文件的方式，导致系统缺乏灵活性和可维护性。上下文传递更是通过手动参数传递实现，增加了代码复杂性和耦合度。

```python
# 传统配置管理和上下文传递的问题
class TraditionalConfigSystem:
    def __init__(self):
        # 硬编码配置
        self.model_config = {
            "model_name": "gpt-4",
            "temperature": 0.7,
            "max_tokens": 1000
        }
        
        # 全局状态
        self.global_context = {
            "user_id": None,
            "session_id": None,
            "request_id": None
        }
        
        # 手动管理的回调列表
        self.callbacks = []
    
    def register_callback(self, callback_func):
        """手动注册回调函数"""
        self.callbacks.append(callback_func)
    
    def execute_with_context(self, query: str, user_context: Dict[str, Any]) -> str:
        """传统上下文传递方式"""
        
        # 1. 保存当前全局状态
        previous_context = self.global_context.copy()
        
        # 2. 更新全局状态（破坏性操作）
        self.global_context.update(user_context)
        
        # 3. 执行处理（依赖全局状态）
        try:
            # 触发回调开始
            for callback in self.callbacks:
                callback("start", {
                    "query": query,
                    "config": self.model_config,
                    "context": self.global_context
                })
            
            # 模拟LLM调用
            import time
            time.sleep(0.1)
            
            result = f"Response to '{query}' for user {self.global_context['user_id']}"
            
            # 触发回调结束
            for callback in self.callbacks:
                callback("end", {
                    "result": result,
                    "context": self.global_context
                })
            
            return result
            
        finally:
            # 4. 恢复全局状态（容易遗漏）
            self.global_context = previous_context
    
    def execute_batch_with_contexts(self, 
                                   queries: List[str], 
                                   contexts: List[Dict[str, Any]]) -> List[str]:
        """批量处理多个上下文（传统方式）"""
        
        results = []
        
        for i, (query, context) in enumerate(zip(queries, contexts)):
            # 为每个查询设置不同的请求ID
            context_with_id = {**context, "request_id": f"req_{i}"}
            result = self.execute_with_context(query, context_with_id)
            results.append(result)
        
        return results
    
    def demonstrate_problems(self):
        """演示传统配置管理的问题"""
        
        # 问题1：竞态条件
        import threading
        
        def worker(worker_id: int):
            context = {"user_id": f"user_{worker_id}", "session_id": f"session_{worker_id}"}
            result = self.execute_with_context(f"Query from worker {worker_id}", context)
            print(f"Worker {worker_id}: {result}")
        
        print("问题1：多线程环境下的竞态条件")
        threads = []
        for i in range(3):
            t = threading.Thread(target=worker, args=(i,))
            threads.append(t)
            t.start()
        
        for t in threads:
            t.join()
        
        # 问题2：上下文污染
        print("\n问题2：上下文污染")
        
        # 第一次调用
        result1 = self.execute_with_context("First query", {"user_id": "user1"})
        print(f"第一次调用: {result1}")
        
        # 忘记清除上下文导致的污染
        result2 = self.execute_with_context("Second query", {"session_id": "session2"})
        print(f"第二次调用: {result2}")
        # 注意：这里仍然包含user_id=user1
        
        # 问题3：回调管理复杂
        print("\n问题3：回调管理复杂")
        
        def logging_callback(event_type: str, data: Dict[str, Any]):
            print(f"  Callback: {event_type} - {data}")
        
        self.register_callback(logging_callback)
        result3 = self.execute_with_context("Logged query", {"user_id": "test_user"})
        print(f"带回调的调用: {result3}")
        
        # 问题4：配置覆盖困难
        print("\n问题4：配置覆盖困难")
        
        # 临时修改配置（影响后续所有调用）
        original_temp = self.model_config["temperature"]
        self.model_config["temperature"] = 0.9
        
        result4 = self.execute_with_context("High temp query", {"user_id": "temp_test"})
        print(f"高温调用: {result4}")
        
        # 需要手动恢复
        self.model_config["temperature"] = original_temp
```

### LangChain v1.2.22的解决方案：RunnableConfig系统

LangChain通过`RunnableConfig`提供了一套完整的配置管理和上下文传递系统，解决了传统方式的诸多问题。

```python
# LangChain RunnableConfig系统
from langchain_core.runnables import RunnableConfig
from langchain_core.callbacks import BaseCallbackHandler, StdOutCallbackHandler
from langchain_openai import ChatOpenAI
from typing import Dict, Any, List, Optional
import time

class LangChainConfigSystem:

    def demonstrate_basic_config(self):
        """演示基础配置使用"""
        
        print("1. 基础RunnableConfig使用")
        
        # 创建基本配置
        config = RunnableConfig(
            run_name="demo_execution",      # 运行名称（用于追踪）
            tags=["demo", "test"],          # 标签（用于分类）
            metadata={                      # 元数据（任意键值对）
                "user_id": "12345",
                "environment": "test",
                "version": "1.2.22"
            },
            callbacks=[StdOutCallbackHandler()],  # 回调处理器
            max_concurrency=5               # 最大并发数
        )
        
        # 使用配置执行
        query = "What is LangChain configuration?"
        result = self.llm.invoke(query, config=config)
        
        print(f"查询: {query}")
        print(f"结果: {result.content[:50]}...")
        print(f"使用的配置:")
        print(f"  run_name: {config.run_name}")
        print(f"  tags: {config.tags}")
        print(f"  metadata: {config.metadata}")
    
    def demonstrate_config_inheritance(self):
        """演示配置继承"""
        
        print("\n2. 配置继承和覆盖")
        
        # 基础配置
        base_config = RunnableConfig(
            run_name="base_execution",
            tags=["base", "production"],
            metadata={"environment": "prod", "region": "us-east-1"},
            max_concurrency=10
        )
        
        # 继承并覆盖
        child_config = base_config.copy(update={
            "run_name": "child_execution",
            "tags": ["base", "production", "debug"],  # 添加新标签
            "metadata": {**base_config.metadata, "debug_mode": True},  # 合并元数据
            "max_concurrency": 3  # 覆盖并发数
        })
        
        print(f"基础配置:")
        print(f"  run_name: {base_config.run_name}")
        print(f"  tags: {base_config.tags}")
        print(f"  metadata: {base_config.metadata}")
        print(f"  max_concurrency: {base_config.max_concurrency}")
        
        print(f"\n子配置（继承自基础配置）:")
        print(f"  run_name: {child_config.run_name}")
        print(f"  tags: {child_config.tags}")
        print(f"  metadata: {child_config.metadata}")
        print(f"  max_concurrency: {child_config.max_concurrency}")
        
        # 执行对比
        query = "Explain configuration inheritance"
        
        print(f"\n使用基础配置执行:")
        result1 = self.llm.invoke(query, config=base_config)
        
        print(f"\n使用子配置执行:")
        result2 = self.llm.invoke(query, config=child_config)
    
    def demonstrate_context_passing(self):
        """演示上下文传递"""
        
        print("\n3. 上下文传递机制")
        
        from langchain_core.runnables import RunnableLambda
        
        # 创建支持上下文的管道
        context_aware_chain = (
            RunnableLambda(lambda x, config: self._extract_context(x, config))
            | RunnableLambda(lambda data: self._process_with_context(data))
            | self.llm
            | RunnableLambda(lambda x: {"response": x.content})
        )
        
        # 不同上下文配置
        contexts = [
            RunnableConfig(
                metadata={
                    "user_type": "premium",
                    "language": "en",
                    "formality": "high"
                }
            ),
            RunnableConfig(
                metadata={
                    "user_type": "free",
                    "language": "zh",
                    "formality": "low"
                }
            )
        ]
        
        queries = [
            "How does LangChain handle configuration?",
            "What is RunnableConfig?"
        ]
        
        for i, (query, config) in enumerate(zip(queries, contexts)):
            print(f"\n查询 {i+1}: {query}")
            print(f"上下文: {config.metadata}")
            
            result = context_aware_chain.invoke(query, config=config)
            print(f"响应: {result['response'][:60]}...")
    
    def _extract_context(self, query: str, config: RunnableConfig) -> Dict[str, Any]:
        """从配置中提取上下文"""
        metadata = config.metadata or {}
        return {
            "query": query,
            "user_type": metadata.get("user_type", "standard"),
            "language": metadata.get("language", "en"),
            "formality": metadata.get("formality", "medium")
        }
    
    def _process_with_context(self, data: Dict[str, Any]) -> str:
        """根据上下文处理查询"""
        user_type = data["user_type"]
        language = data["language"]
        formality = data["formality"]
        query = data["query"]
        
        # 根据上下文调整提示
        if user_type == "premium":
            prefix = "Please provide a detailed and comprehensive answer: "
        elif user_type == "free":
            prefix = "Please provide a concise answer: "
        else:
            prefix = "Please answer: "
        
        if language == "zh":
            prefix = "请用中文回答: " + prefix
        
        if formality == "high":
            prefix = "尊敬的用户，" + prefix
        elif formality == "low":
            prefix = "嘿，" + prefix
        
        return prefix + query
    
    def demonstrate_callback_system(self):
        """演示回调系统"""
        
        print("\n4. 回调系统")
        
        # 自定义回调处理器
        class CustomCallbackHandler(BaseCallbackHandler):
            def on_chain_start(self, serialized: Dict[str, Any], inputs: Dict[str, Any], **kwargs):
                run_id = kwargs.get("run_id", "unknown")
                print(f"🔗 链开始 [Run ID: {run_id}]")
                print(f"   输入: {inputs}")
            
            def on_chain_end(self, outputs: Dict[str, Any], **kwargs):
                run_id = kwargs.get("run_id", "unknown")
                print(f"✅ 链结束 [Run ID: {run_id}]")
                print(f"   输出: {outputs}")
            
            def on_llm_start(self, serialized: Dict[str, Any], prompts: List[str], **kwargs):
                print(f"🤖 LLM开始处理 {len(prompts)} 个提示")
            
            def on_llm_end(self, response: Any, **kwargs):
                print(f"🎯 LLM处理完成")
        
        # 创建带回调的配置
        config = RunnableConfig(
            run_name="callback_demo",
            callbacks=[CustomCallbackHandler(), StdOutCallbackHandler()],
            metadata={"demo": "callback_system"}
        )
        
        # 创建复杂管道
        from langchain_core.runnables import RunnableSequence
        
        complex_chain = RunnableSequence(
            RunnableLambda(lambda x: {"query": x, "timestamp": time.time()}),
            RunnableLambda(lambda x: {**x, "processed": True}),
            self.llm,
            RunnableLambda(lambda x: {"response": x.content, "completed_at": time.time()})
        )
        
        print("执行带回调的复杂管道:")
        query = "Explain callback system in LangChain"
        result = complex_chain.invoke(query, config=config)
        
        print(f"\n最终结果:")
        print(f"  响应: {result['response'][:50]}...")
        print(f"  处理时间: {result['completed_at'] - result['timestamp']:.3f}秒")
    
    def demonstrate_advanced_features(self):
        """演示高级特性"""
        
        print("\n5. 高级特性")
        
        # 动态配置生成
        def create_dynamic_config(context: Dict[str, Any]) -> RunnableConfig:
            """根据上下文动态生成配置"""
            user_type = context.get("user_type", "standard")
            
            if user_type == "premium":
                return RunnableConfig(
                    run_name="premium_execution",
                    tags=["premium", "high_priority"],
                    metadata={**context, "quality_level": "high"},
                    max_concurrency=1  # 独占资源
                )
            elif user_type == "free":
                return RunnableConfig(
                    run_name="free_execution",
                    tags=["free", "low_priority"],
                    metadata={**context, "quality_level": "basic"},
                    max_concurrency=10  # 高并发
                )
            else:
                return RunnableConfig(
                    run_name="standard_execution",
                    tags=["standard"],
                    metadata=context
                )
        
        # 配置作用域
        from contextlib import contextmanager
        from langchain_core.runnables import RunnableConfig
        
        class ConfigScopeManager:
            """配置作用域管理器"""
            
            def __init__(self):
                self._config_stack = []
            
            @contextmanager
            def scope(self, config: RunnableConfig):
                """进入配置作用域"""
                self._config_stack.append(config)
                try:
                    yield
                finally:
                    self._config_stack.pop()
            
            def get_current_config(self) -> Optional[RunnableConfig]:
                """获取当前作用域配置"""
                if self._config_stack:
                    return self._config_stack[-1]
                return None
        
        # 演示动态配置
        print("动态配置生成:")
        
        contexts = [
            {"user_type": "premium", "user_id": "user_001"},
            {"user_type": "free", "user_id": "user_002"},
            {"user_type": "standard", "user_id": "user_003"}
        ]
        
        for ctx in contexts:
            dynamic_config = create_dynamic_config(ctx)
            print(f"  上下文: {ctx}")
            print(f"  生成配置:")
            print(f"    run_name: {dynamic_config.run_name}")
            print(f"    tags: {dynamic_config.tags}")
            print(f"    quality_level: {dynamic_config.metadata.get('quality_level')}")
            print()
        
        # 演示配置作用域
        print("配置作用域管理:")
        
        scope_manager = ConfigScopeManager()
        
        # 外层作用域
        outer_config = RunnableConfig(
            run_name="outer_scope",
            metadata={"level": "outer"}
        )
        
        # 内层作用域
        inner_config = RunnableConfig(
            run_name="inner_scope",
            metadata={"level": "inner"}
        )
        
        with scope_manager.scope(outer_config):
            print(f"  进入外层作用域")
            print(f"  当前配置: {scope_manager.get_current_config().run_name}")
            
            with scope_manager.scope(inner_config):
                print(f"  进入内层作用域")
                print(f"  当前配置: {scope_manager.get_current_config().run_name}")
            
            print(f"  返回外层作用域")
            print(f"  当前配置: {scope_manager.get_current_config().run_name}")
```

```python
                         traffic_allocation: float = 0.5) -> str:
        """创建A/B测试实验"""
        
        self.experiments[experiment_id] = {
            "variants": dict(variants),
            "traffic_allocation": traffic_allocation,
            "created_at": datetime.now(),
            "status": "active"
        }
        self.results[experiment_id] = []
        
        print(f"创建实验: {experiment_id}")

        for variant_name, config in variants:

            print(f"    tags: {config.tags}")

        return experiment_id
    
    def assign_variant(self, experiment_id: str, user_id: str) -> Tuple[str, RunnableConfig]:
        """分配变体给用户"""
        
        experiment = self.experiments.get(experiment_id)
        if not experiment or experiment["status"] != "active":
            # 返回默认配置
            default_config = RunnableConfig(
                run_name="default_config",
                tags=["default", "fallback"]
            )
            return "default", default_config
        
        # 根据流量分配决定是否参与实验
        if random.random() > experiment["traffic_allocation"]:
            default_config = RunnableConfig(
                run_name=f"control_{experiment_id}",
                tags=["control", experiment_id]
            )
            return "control", default_config
        
        # 随机分配变体
        variants = experiment["variants"]
        variant_name = random.choice(list(variants.keys()))
        variant_config = variants[variant_name]
        
        # 为用户添加实验标记
        updated_config = variant_config.copy(update={
            "metadata": {
                **variant_config.metadata,
                "ab_testing": {
                    "experiment_id": experiment_id,
                    "variant": variant_name,
                    "user_id": user_id,
                    "assigned_at": datetime.now()
                }
            }
        })
        
        return variant_name, updated_config
    
    def execute_with_experiment(self, 
                               experiment_id: str, 
                               user_id: str, 
                               query: str) -> Dict[str, Any]:
        """使用实验配置执行"""
        
        # 分配变体
        variant_name, config = self.assign_variant(experiment_id, user_id)

                "variant": variant_name,

                "execution_time": execution_time,
                "success": True,
                "timestamp": datetime.now(),
                "response_length": len(result.content)

            return {
                "variant": variant_name,
                "response": result.content,
                "execution_time": execution_time,

                    "sample_config": variant_data[0]["config"] if variant_data else None
```

```python
class ConfigValidationAndSecurity:
    def validate_config(self, config: RunnableConfig) -> List[str]:
        """验证配置安全性"""
        
        issues = []
        
        # 检查run_name
        if not config.run_name or len(config.run_name) > 100:
            issues.append("run_name必须存在且长度不超过100字符")
        
        # 检查tags
        if config.tags:
            for tag in config.tags:
                if not isinstance(tag, str) or len(tag) > 50:
                    issues.append(f"标签 '{tag}' 无效")
        
        # 检查metadata
        if config.metadata:
            # 防止敏感信息泄露
            sensitive_keys = ["password", "secret", "key", "token"]
            for key in config.metadata:
                if any(sensitive in key.lower() for sensitive in sensitive_keys):
                    issues.append(f"元数据包含可能敏感的键: {key}")
            
            # 检查嵌套深度
            if self._get_dict_depth(config.metadata) > 3:
                issues.append("元数据嵌套过深")
        
        # 检查并发限制
        if config.max_concurrency and config.max_concurrency > 100:
            issues.append("并发数过高，可能影响系统稳定性")
        
        return issues
    
    def _get_dict_depth(self, d: Dict[str, Any], current_depth: int = 0) -> int:
        """获取字典嵌套深度"""
        if not isinstance(d, dict):
            return current_depth
        
        max_depth = current_depth
        for value in d.values():
            if isinstance(value, dict):
                depth = self._get_dict_depth(value, current_depth + 1)
                max_depth = max(max_depth, depth)
        
        return max_depth
```

```python
    """上下文隔离管理器"""
    
    def __init__(self):
        self._context_stack = []
    
    @contextmanager
    def isolated_context(self, config: RunnableConfig):
        """进入隔离的上下文"""
        # 保存当前上下文
        self._context_stack.append(config)
        
        try:
            # 执行代码
            yield config
        finally:
            # 清理上下文
            self._context_stack.pop()
    
    def get_current_context(self) -> Optional[RunnableConfig]:
        """获取当前上下文"""
        if self._context_stack:
            return self._context_stack[-1]
        return None
    
    def clear_all_contexts(self):
        """清理所有上下文"""
        self._context_stack.clear()
```

```python
from functools import lru_cache
from typing import Dict, Any

class ConfigCacheStrategy:
    """配置缓存策略"""
    
    def __init__(self):
        self._cache = {}
        self._hits = 0
        self._misses = 0
    
    @lru_cache(maxsize=256)
    def get_cached_config(self, config_hash: str) -> Optional[RunnableConfig]:
        """获取缓存的配置"""
        return self._cache.get(config_hash)
    
    def create_config_with_cache(self, 
                                run_name: str,
                                tags: List[str],
                                metadata: Dict[str, Any],
                                **kwargs) -> RunnableConfig:
        """创建配置（带缓存）"""
        
        # 生成缓存键
        cache_key = self._generate_cache_key(run_name, tags, metadata, kwargs)
        
        # 检查缓存
        cached = self.get_cached_config(cache_key)
        if cached:
            self._hits += 1
            return cached
        
        # 创建新配置
        self._misses += 1
        config = RunnableConfig(
            run_name=run_name,
            tags=tags,
            metadata=metadata,
            **kwargs
        )
        
        # 缓存配置
        self._cache[cache_key] = config
        return config
    
    def _generate_cache_key(self, 
                           run_name: str,
                           tags: List[str],
                           metadata: Dict[str, Any],
                           kwargs: Dict[str, Any]) -> str:
        """生成缓存键"""
        sorted_tags = ",".join(sorted(tags))
        sorted_metadata = str(sorted(metadata.items()))
        sorted_kwargs = str(sorted(kwargs.items()))
        
        return f"{run_name}:{sorted_tags}:{sorted_metadata}:{sorted_kwargs}"
    
    def get_cache_stats(self) -> Dict[str, Any]:
        """获取缓存统计"""
        total = self._hits + self._misses
        hit_rate = (self._hits / total * 100) if total > 0 else 0
        
        return {
            "cache_size": len(self._cache),
            "hits": self._hits,
            "misses": self._misses,
            "hit_rate": f"{hit_rate:.1f}%",
            "memory_usage": self._estimate_memory_usage()
        }
    
    def _estimate_memory_usage(self) -> str:
        """估计内存使用"""
        # 简化实现
        approx_bytes = len(self._cache) * 1000  # 每个配置约1KB
        if approx_bytes < 1024:
            return f"{approx_bytes} B"
        elif approx_bytes < 1024 * 1024:
            return f"{approx_bytes/1024:.1f} KB"
        else:
            return f"{approx_bytes/(1024*1024):.1f} MB"
```

```python
# 智能并发控制
class IntelligentConcurrencyControl:
    """智能并发控制"""
    
    def __init__(self, 
                 min_concurrency: int = 1,
                 max_concurrency: int = 20,
                 adjustment_interval: int = 60):
        self.min_concurrency = min_concurrency
        self.max_concurrency = max_concurrency
        self.adjustment_interval = adjustment_interval
        
        self.current_concurrency = min_concurrency
        self.performance_history = []
        self.last_adjustment = time.time()
    
    def adjust_concurrency_based_on_performance(self, 
                                               execution_times: List[float],
                                               success_rate: float) -> int:
        """基于性能调整并发数"""
        
        current_time = time.time()
        
        # 检查是否需要调整
        if current_time - self.last_adjustment < self.adjustment_interval:
            return self.current_concurrency
        
        self.last_adjustment = current_time
        
        if not execution_times:
            return self.current_concurrency
        
        # 计算性能指标
        avg_execution_time = sum(execution_times) / len(execution_times)
        
        # 保存性能历史
        self.performance_history.append({
            "timestamp": current_time,
            "concurrency": self.current_concurrency,
            "avg_execution_time": avg_execution_time,
            "success_rate": success_rate
        })
        
        # 保持历史大小
        if len(self.performance_history) > 100:
            self.performance_history.pop(0)
        
        # 调整逻辑
        if success_rate < 0.9:
            # 成功率低，降低并发
            new_concurrency = max(self.min_concurrency, self.current_concurrency - 1)
        elif avg_execution_time > 2.0:
            # 执行时间长，降低并发
            new_concurrency = max(self.min_concurrency, self.current_concurrency - 1)
        elif avg_execution_time < 0.5 and self.current_concurrency < self.max_concurrency:
            # 执行时间短，可以增加并发
            new_concurrency = min(self.max_concurrency, self.current_concurrency + 1)
        else:
            # 保持当前并发
            new_concurrency = self.current_concurrency
        
        # 应用调整
        old_concurrency = self.current_concurrency
        self.current_concurrency = new_concurrency
        
        if old_concurrency != new_concurrency:
            print(f"并发数调整: {old_concurrency} -> {new_concurrency}")
            print(f"  原因: 平均执行时间={avg_execution_time:.2f}s, 成功率={success_rate:.1%}")
        
        return self.current_concurrency
```

```python
        else:

                         old_config: RunnableConfig,
                         new_config: RunnableConfig,
                         reason: str):
        """记录配置变更日志"""
        
        changes = self._detect_config_changes(old_config, new_config)
        
        log_data = {
            "event": "config_change",
            "timestamp": datetime.now().isoformat(),
            "old_config": old_config.dict(),
            "new_config": new_config.dict(),
            "changes": changes,
            "reason": reason
        }
        
        self.logger.info(json.dumps(log_data, default=str))
    
    def _detect_config_changes(self, 
                              old_config: RunnableConfig,
                              new_config: RunnableConfig) -> Dict[str, Any]:
        """检测配置变更"""
        
        changes = {}
        
        # 检查基本字段
        for field in ["run_name", "max_concurrency"]:
            old_value = getattr(old_config, field)
            new_value = getattr(new_config, field)
            
            if old_value != new_value:
                changes[field] = {"old": old_value, "new": new_value}
        
        # 检查标签
        if old_config.tags != new_config.tags:
            added = set(new_config.tags or []) - set(old_config.tags or [])
            removed = set(old_config.tags or []) - set(new_config.tags or [])
            
            if added or removed:
                changes["tags"] = {
                    "added": list(added),
                    "removed": list(removed)
                }
        
        # 检查元数据
        if old_config.metadata != new_config.metadata:
            changes["metadata"] = "modified"
        
        return changes
```

```python
# 配置健康检查
class ConfigHealthChecker:
    """配置健康检查器"""
    
    def __init__(self, warning_thresholds: Dict[str, float] = None):
        self.warning_thresholds = warning_thresholds or {
            "execution_time": 2.0,     # 秒
            "error_rate": 0.05,        # 5%
            "concurrency_usage": 0.8,  # 80%
            "cache_miss_rate": 0.3     # 30%
        }
        
        self.health_metrics = {}
    
    def check_config_health(self, config_id: str, metrics: Dict[str, Any]) -> Dict[str, Any]:
        """检查配置健康状态"""
        
        warnings = []
        recommendations = []
        
        # 检查执行时间
        exec_time = metrics.get("avg_execution_time", 0)
        if exec_time > self.warning_thresholds["execution_time"]:
            warnings.append(f"执行时间过长: {exec_time:.2f}秒")
            recommendations.append("考虑优化提示或降低温度参数")
        
        # 检查错误率
        error_rate = 1 - metrics.get("success_rate", 1.0)
        if error_rate > self.warning_thresholds["error_rate"]:
            warnings.append(f"错误率过高: {error_rate:.1%}")
            recommendations.append("检查API配额和网络连接")
        
        # 检查并发使用率
        concurrency_usage = metrics.get("concurrency_usage", 0)
        if concurrency_usage > self.warning_thresholds["concurrency_usage"]:
            warnings.append(f"并发使用率过高: {concurrency_usage:.1%}")
            recommendations.append("考虑增加最大并发数或优化资源分配")
        
        # 检查缓存命中率
        cache_hit_rate = metrics.get("cache_hit_rate", 1.0)
        cache_miss_rate = 1 - cache_hit_rate
        if cache_miss_rate > self.warning_thresholds["cache_miss_rate"]:
            warnings.append(f"缓存未命中率过高: {cache_miss_rate:.1%}")
            recommendations.append("考虑调整缓存策略或增加缓存容量")
        
        # 评估整体健康状态
        if warnings:
            health_status = "warning" if len(warnings) < 3 else "critical"
        else:
            health_status = "healthy"
        
        # 更新健康指标
        self.health_metrics[config_id] = {
            "status": health_status,
            "last_check": datetime.now(),
            "warnings": warnings,
            "metrics": metrics
        }
        
        return {
            "config_id": config_id,
            "status": health_status,
            "warnings": warnings,
            "recommendations": recommendations,
            "metrics": metrics
        }
    
    def get_health_summary(self) -> Dict[str, Any]:
        """获取健康摘要"""
        
        config_statuses = {}
        for config_id, health_data in self.health_metrics.items():
            config_statuses[config_id] = health_data["status"]
        
        healthy_count = sum(1 for status in config_statuses.values() if status == "healthy")
        warning_count = sum(1 for status in config_statuses.values() if status == "warning")
        critical_count = sum(1 for status in config_statuses.values() if status == "critical")
        
        total_configs = len(config_statuses)
        
        return {
            "total_configs": total_configs,
            "healthy_configs": healthy_count,
            "warning_configs": warning_count,
            "critical_configs": critical_count,
            "health_percentage": (healthy_count / total_configs * 100) if total_configs > 0 else 0,
            "config_statuses": config_statuses
        }
```
