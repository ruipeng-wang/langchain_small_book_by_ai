# 2.2.1 任务分解与顺序执行

## 从硬编码流程到声明式链式执行的演进

### 传统硬编码任务分解的问题

在LangChain出现之前，AI应用通常采用**硬编码的任务分解逻辑**，将复杂的业务流程直接写在代码中：

```python
# 传统硬编码任务分解示例：内容生成系统
class HardcodedContentGenerator:
    def __init__(self):
        # 直接依赖具体实现
        self.research_client = ResearchClient()
        self.outline_generator = OutlineGenerator()
        self.content_writer = ContentWriter()
        self.editor = Editor()
        self.formatter = Formatter()
    
    def generate_article(self, topic: str) -> Dict[str, Any]:
        """硬编码的文章生成流程"""
        try:
            # 1. 研究阶段（硬编码顺序）
            print("开始研究...")
            research_data = self.research_client.research(topic)
            
            # 2. 大纲生成（硬编码参数传递）
            print("生成大纲...")
            outline = self.outline_generator.generate(
                topic=topic,
                research_data=research_data,
                style="professional"
            )
            
            # 3. 内容写作（硬编码错误处理）
            print("撰写内容...")
            try:
                content = self.content_writer.write(
                    outline=outline,
                    research_data=research_data
                )
            except ContentWritingError as e:
                print(f"内容写作失败: {e}")
                # 硬编码的降级策略
                content = self._generate_fallback_content(topic)
            
            # 4. 编辑校对（硬编码条件判断）
            print("编辑校对...")
            if len(content) > 1000:
                edited_content = self.editor.edit(content, mode="thorough")
            else:
                edited_content = self.editor.edit(content, mode="light")
            
            # 5. 格式整理（硬编码后处理）
            print("格式化...")
            final_content = self.formatter.format(
                content=edited_content,
                format_type="markdown",
                include_metadata=True
            )
            
            return {
                "status": "success",
                "content": final_content,
                "steps_completed": 5,
                "metadata": {
                    "topic": topic,
                    "length": len(final_content)
                }
            }
            
        except Exception as e:
            # 硬编码的全局错误处理
            print(f"文章生成失败: {e}")
            return {
                "status": "error",
                "error": str(e),
                "content": None
            }
    
    def _generate_fallback_content(self, topic: str) -> str:
        """硬编码的降级方案"""
        return f"Fallback content about {topic}"
    
    # 添加新步骤需要修改多个地方
    def add_proofreading_step(self):
        """添加校对步骤（需要修改generate_article方法）"""
        # 需要在generate_article中：
        # 1. 添加新的调用
        # 2. 更新步骤计数
        # 3. 调整错误处理
        # 4. 可能影响其他步骤
        pass
```

**传统硬编码任务分解的问题：**

1. **紧耦合**：任务顺序和逻辑硬编码在方法中，难以修改
2. **缺乏灵活性**：无法动态调整任务顺序或跳过某些步骤
3. **错误处理复杂**：每个步骤需要单独的错误处理逻辑
4. **测试困难**：整个流程需要集成测试，难以单元测试单个步骤
5. **扩展性差**：添加新步骤需要修改现有代码，风险高
6. **代码冗余**：相似的任务分解逻辑在不同方法中重复
7. **可维护性低**：业务逻辑与执行逻辑混杂，难以理解和维护

### LangChain的链式执行解决方案：LCEL

LangChain v1.2.22通过**LangChain Expression Language (LCEL)** 提供了声明式的链式执行方案：

```python
# LangChain LCEL链式执行示例：内容生成系统
from langchain_core.runnables import RunnableSequence, RunnableLambda, RunnableParallel
from langchain_core.output_parsers import StrOutputParser
from langchain_core.prompts import ChatPromptTemplate
from langchain_openai import ChatOpenAI
from typing import Dict, Any, List
import asyncio

class LCELContentGenerator:
    def __init__(self):
        # 初始化LLM
        self.llm = ChatOpenAI(model="gpt-4", temperature=0.7)
        
        # 1. 研究组件（独立、可测试）
        self.research_component = RunnableLambda(
            lambda topic: self._conduct_research(topic)
        )
        
        # 2. 大纲生成组件
        self.outline_prompt = ChatPromptTemplate.from_messages([
            ("system", "You are a professional content strategist."),
            ("human", "Generate an outline for an article about {topic} based on this research: {research_data}")
        ])
        self.outline_generator = self.outline_prompt | self.llm | StrOutputParser()
        
        # 3. 内容写作组件
        self.writing_prompt = ChatPromptTemplate.from_messages([
            ("system", "You are a skilled writer. Write content based on the outline and research."),
            ("human", "Outline: {outline}\n\nResearch: {research_data}\n\nWrite the full article.")
        ])
        self.content_writer = self.writing_prompt | self.llm | StrOutputParser()
        
        # 4. 编辑组件
        self.editing_prompt = ChatPromptTemplate.from_messages([
            ("system", "You are an editor. Improve the content for clarity and professionalism."),
            ("human", "Please edit this content:\n\n{content}")
        ])
        self.editor = self.editing_prompt | self.llm | StrOutputParser()
        
        # 5. 格式化组件
        self.formatter = RunnableLambda(
            lambda content: self._format_content(content)
        )
        
        # 使用LCEL构建完整的工作流
        self.article_workflow = (
            # 步骤1: 研究
            RunnableLambda(lambda x: {"topic": x})
            | RunnableParallel({
                "research_data": self.research_component,
                "original_topic": RunnableLambda(lambda x: x["topic"])
            })
            # 步骤2: 生成大纲
            | RunnableLambda(lambda x: {
                "topic": x["original_topic"],
                "research_data": x["research_data"],
                "outline_input": {
                    "topic": x["original_topic"],
                    "research_data": x["research_data"]
                }
            })
            | RunnableParallel({
                "outline": RunnableLambda(lambda x: self.outline_generator.invoke(x["outline_input"])),
                "research_data": RunnableLambda(lambda x: x["research_data"]),
                "topic": RunnableLambda(lambda x: x["topic"])
            })
            # 步骤3: 写作
            | RunnableLambda(lambda x: {
                "outline": x["outline"],
                "research_data": x["research_data"],
                "writing_input": {
                    "outline": x["outline"],
                    "research_data": x["research_data"]
                }
            })
            | RunnableParallel({
                "content": RunnableLambda(lambda x: self.content_writer.invoke(x["writing_input"])),
                "outline": RunnableLambda(lambda x: x["outline"]),
                "research_data": RunnableLambda(lambda x: x["research_data"])
            })
            # 步骤4: 编辑
            | RunnableLambda(lambda x: {
                "content": x["content"],
                "editing_input": {"content": x["content"]}
            })
            | RunnableParallel({
                "edited_content": RunnableLambda(lambda x: self.editor.invoke(x["editing_input"])),
                "original_content": RunnableLambda(lambda x: x["content"])
            })
            # 步骤5: 格式化
            | RunnableLambda(lambda x: {
                "content": x["edited_content"],
                "formatting_input": x["edited_content"]
            })
            | RunnableParallel({
                "formatted_content": RunnableLambda(lambda x: self.formatter.invoke(x["formatting_input"])),
                "edited_content": RunnableLambda(lambda x: x["content"])
            })
            # 最终结果组装
            | RunnableLambda(lambda x: {
                "status": "success",
                "content": x["formatted_content"],
                "steps_completed": 5,
                "metadata": {
                    "content_length": len(x["formatted_content"]),
                    "original_content_length": len(x["edited_content"])
                }
            })
        )
    
    def _conduct_research(self, topic: str) -> str:
        """模拟研究过程"""
        # 实际中可能调用搜索API、数据库等
        return f"Research data about {topic}: key points, statistics, and trends."
    
    def _format_content(self, content: str) -> str:
        """格式化内容"""
        return f"# Article\n\n{content}\n\n---\n*Generated by AI*"
    
    def generate_article(self, topic: str) -> Dict[str, Any]:
        """使用LCEL工作流生成文章"""
        return self.article_workflow.invoke(topic)
    
    async def generate_article_async(self, topic: str) -> Dict[str, Any]:
        """异步生成文章"""
        return await self.article_workflow.ainvoke(topic)
    
    def stream_article_generation(self, topic: str):
        """流式生成文章"""
        # LCEL天然支持流式处理
        for chunk in self.article_workflow.stream(topic):
            yield chunk

# 使用示例
generator = LCELContentGenerator()

# 同步执行
result = generator.generate_article("Artificial Intelligence in Healthcare")
print(f"生成结果: {result['status']}")
print(f"内容长度: {len(result['content'])} 字符")
print(f"完成步骤: {result['steps_completed']}")

# 异步执行

```

**LCEL链式执行的核心优势：**

1. **声明式语法**：使用 `|` 操作符直观地组合组件
2. **松耦合**：组件之间通过标准接口交互，易于替换和测试
3. **灵活性**：可以动态调整、重新排列或跳过步骤
4. **错误处理统一**：通过统一的错误处理机制
5. **可测试性**：每个组件可以独立测试
6. **可扩展性**：添加新步骤不影响现有代码
7. **支持多种执行模式**：同步、异步、批量、流式

## 任务分解的核心模式

### 模式1：线性管道（顺序执行）

```python
# 传统线性管道：硬编码顺序
class TraditionalLinearPipeline:
    def process_data(self, input_data: str) -> str:
        """硬编码的线性处理"""
        # 步骤1: 清洗
        cleaned = self._clean_input(input_data)
        
        # 步骤2: 分析
        analyzed = self._analyze_content(cleaned)
        
        # 步骤3: 转换
        transformed = self._transform_format(analyzed)
        
        # 步骤4: 输出
        output = self._generate_output(transformed)
        
        return output
    
    def _clean_input(self, data: str) -> str:
        return data.strip().lower()
    
    def _analyze_content(self, data: str) -> Dict:
        return {"words": len(data.split()), "chars": len(data)}
    
    def _transform_format(self, analysis: Dict) -> str:
        return f"Analysis: {analysis}"
    
    def _generate_output(self, transformed: str) -> str:
        return f"Final output: {transformed}"

# LangChain线性管道：声明式顺序
from langchain_core.runnables import RunnableSequence

class LCELinearPipeline:
    def __init__(self):
        # 定义独立组件
        self.cleaner = RunnableLambda(lambda x: x.strip().lower())
        self.analyzer = RunnableLambda(lambda x: {"words": len(x.split()), "chars": len(x)})
        self.transformer = RunnableLambda(lambda x: f"Analysis: {x}")
        self.output_generator = RunnableLambda(lambda x: f"Final output: {x}")
        
        # 使用 | 操作符构建线性管道
        self.pipeline = (
            self.cleaner
            | self.analyzer
            | self.transformer
            | self.output_generator
        )
    
    def process_data(self, input_data: str) -> str:
        """声明式的线性处理"""
        return self.pipeline.invoke(input_data)

# 对比使用
traditional = TraditionalLinearPipeline()
lce = LCELinearPipeline()

input_data = "  Hello World! This is a test.  "

# 传统方式

print(f"传统结果: {traditional_result}")

# LCEL方式
lce_result = lce.process_data(input_data)
print(f"LCEL结果: {lce_result}")
```

### 模式2：带有中间结果的管道

```python
# 传统方式：手动传递中间结果
class TraditionalPipelineWithIntermediates:
    def process_with_intermediates(self, input_data: str) -> Dict[str, Any]:
        """手动管理中间结果"""
        intermediates = {}
        
        # 步骤1
        step1_result = self._step1(input_data)
        intermediates["step1"] = step1_result
        
        # 步骤2（需要步骤1的结果）
        step2_result = self._step2(step1_result)
        intermediates["step2"] = step2_result
        
        # 步骤3（需要步骤1和步骤2的结果）
        step3_result = self._step3(step1_result, step2_result)
        intermediates["step3"] = step3_result
        
        # 最终结果
        final_result = self._final_step(step1_result, step2_result, step3_result)
        intermediates["final"] = final_result
        
        return intermediates
    
    def _step1(self, data: str) -> str:
        return f"Step1({data})"
    
    def _step2(self, data: str) -> str:
        return f"Step2({data})"
    
    def _step3(self, data1: str, data2: str) -> str:
        return f"Step3({data1}, {data2})"
    
    def _final_step(self, s1: str, s2: str, s3: str) -> str:
        return f"Final({s1}, {s2}, {s3})"

# LangChain方式：自动管理中间结果
from langchain_core.runnables import RunnableParallel

class LCELPipelineWithIntermediates:
    def __init__(self):
        # 定义步骤组件
        self.step1 = RunnableLambda(lambda x: f"Step1({x})")
        self.step2 = RunnableLambda(lambda x: f"Step2({x})")
        self.step3 = RunnableLambda(lambda x: f"Step3({x['step1']}, {x['step2']})")
        self.final_step = RunnableLambda(
            lambda x: f"Final({x['step1']}, {x['step2']}, {x['step3']})"
        )
        
        # 构建管道，自动传递中间结果
        self.pipeline = (
            # 并行执行步骤1和步骤2（步骤2需要步骤1的结果）
            RunnableLambda(lambda x: {"input": x})
            | RunnableParallel({
                "step1": RunnableLambda(lambda x: x["input"]) | self.step1,
                "step2": RunnableLambda(lambda x: x["input"]) | self.step1 | self.step2,
            })
            # 执行步骤3（需要步骤1和步骤2的结果）
            | RunnableParallel({
                "step1": RunnableLambda(lambda x: x["step1"]),
                "step2": RunnableLambda(lambda x: x["step2"]),
                "step3": RunnableLambda(lambda x: {"step1": x["step1"], "step2": x["step2"]}) | self.step3,
            })
            # 最终步骤
            | self.final_step
        )
    
    def process_with_intermediates(self, input_data: str) -> Dict[str, Any]:
        """自动管理中间结果"""
        # LCEL自动处理数据流，无需手动管理
        result = self.pipeline.invoke(input_data)
        
        # 在实际应用中，可以使用RunnableConfig记录中间结果
        from langchain_core.runnables import RunnableConfig
        
        config = RunnableConfig(
            callbacks=[self._logging_callback],
            metadata={"track_intermediates": True}
        )
        
        return {"result": result, "config": config}
    
    def _logging_callback(self, info):
        """回调函数记录中间结果"""
        print(f"Step completed: {info}")

# 使用对比
traditional = TraditionalPipelineWithIntermediates()
lce = LCELPipelineWithIntermediates()

input_data = "test"

# 传统方式：需要手动管理中间结果
traditional_result = traditional.process_with_intermediates(input_data)
print(f"传统中间结果: {traditional_result}")

# LCEL方式：自动管理中间结果
lce_result = lce.process_with_intermediates(input_data)
print(f"LCEL结果: {lce_result}")
```

```python
# 传统方式：复杂的条件逻辑
class TraditionalConditionalPipeline:
    def process_with_conditions(self, input_data: str, user_type: str) -> str:
        """硬编码的条件逻辑"""
        result = input_data
        
        # 条件1：基于用户类型
        if user_type == "expert":
            # 专家用户流程
            result = self._expert_processing(result)
            result = self._advanced_analysis(result)
        elif user_type == "beginner":
            # 初学者流程
            result = self._simplified_processing(result)
            result = self._basic_analysis(result)
        else:
            # 默认流程
            result = self._default_processing(result)
        
        # 条件2：基于内容长度
        if len(result) > 1000:
            result = self._summarize_content(result)
        
        # 条件3：基于内容类型
        if "error" in result.lower():
            result = self._error_recovery(result)
        
        return result
    
    def _expert_processing(self, data: str) -> str:
        return f"Expert: {data}"
    
    def _advanced_analysis(self, data: str) -> str:
        return f"Advanced({data})"
    
    def _simplified_processing(self, data: str) -> str:
        return f"Simple: {data}"
    
    def _basic_analysis(self, data: str) -> str:
        return f"Basic({data})"
    
    def _default_processing(self, data: str) -> str:
        return f"Default: {data}"
    
    def _summarize_content(self, data: str) -> str:
        return data[:500] + "...[summarized]"
    
    def _error_recovery(self, data: str) -> str:
        return data.replace("error", "issue")

# LangChain方式：声明式条件路由
from langchain_core.runnables import RunnableBranch

class LCELConditionalPipeline:
    def __init__(self):
        # 定义处理组件
        self.expert_processor = RunnableLambda(lambda x: f"Expert: {x}")
        self.advanced_analyzer = RunnableLambda(lambda x: f"Advanced({x})")
        self.simple_processor = RunnableLambda(lambda x: f"Simple: {x}")
        self.basic_analyzer = RunnableLambda(lambda x: f"Basic({x})")
        self.default_processor = RunnableLambda(lambda x: f"Default: {x}")
        self.summarizer = RunnableLambda(lambda x: x[:500] + "...[summarized]" if len(x) > 1000 else x)
        self.error_recoverer = RunnableLambda(lambda x: x.replace("error", "issue") if "error" in x.lower() else x)
        
        # 构建条件分支
        self.user_type_branch = RunnableBranch(
            # 条件1: 专家用户
            (lambda x: x.get("user_type") == "expert", 
             RunnableSequence(
                 RunnableLambda(lambda x: x["input_data"]),
                 self.expert_processor,
                 self.advanced_analyzer
             )),
            # 条件2: 初学者
            (lambda x: x.get("user_type") == "beginner",
             RunnableSequence(
                 RunnableLambda(lambda x: x["input_data"]),
                 self.simple_processor,
                 self.basic_analyzer
             )),
            # 默认情况
            RunnableSequence(
                RunnableLambda(lambda x: x["input_data"]),
                self.default_processor
            )
        )
        
        # 构建完整管道
        self.pipeline = (
            # 用户类型分支
            self.user_type_branch
            # 内容长度条件
            | RunnableLambda(lambda x: self.summarizer.invoke(x) if len(x) > 1000 else x)
            # 错误恢复条件
            | RunnableLambda(lambda x: self.error_recoverer.invoke(x) if "error" in x.lower() else x)
        )
    
    def process_with_conditions(self, input_data: str, user_type: str) -> str:
        """声明式的条件处理"""
        return self.pipeline.invoke({"input_data": input_data, "user_type": user_type})

# 使用对比
traditional = TraditionalConditionalPipeline()
lce = LCELConditionalPipeline()

test_cases = [
    ("This is a test input with error message", "expert"),
    ("Short input", "beginner"),
    ("A" * 1500, "unknown"),  # 长内容
]

for input_data, user_type in test_cases:
    # 传统方式
    traditional_result = traditional.process_with_conditions(input_data, user_type)
    
    # LCEL方式
    lce_result = lce.process_with_conditions(input_data, user_type)
    
    print(f"\n输入: {input_data[:50]}..., 用户类型: {user_type}")
    print(f"传统结果: {traditional_result[:100]}...")
    print(f"LCEL结果: {lce_result[:100]}...")
```

```python
# 不好的实践：任务粒度过大
class CoarseGrainedTasks:
    def process_document(self, document: str) -> Dict[str, Any]:
        """粒度过大的任务：做了太多事情"""
        # 这个任务包含了：
        # 1. 文本清洗
        # 2. 实体识别
        # 3. 情感分析
        # 4. 摘要生成
        # 5. 分类
        # 难以测试和维护
        
        cleaned = document.strip().lower()
        entities = self._extract_entities(cleaned)
        sentiment = self._analyze_sentiment(cleaned)
        summary = self._generate_summary(cleaned)
        category = self._classify_document(cleaned)
        
        return {
            "cleaned": cleaned,
            "entities": entities,
            "sentiment": sentiment,
            "summary": summary,
            "category": category
        }
    
    def _extract_entities(self, text: str) -> List[str]:
        return ["entity1", "entity2"]
    
    def _analyze_sentiment(self, text: str) -> str:
        return "positive"
    
    def _generate_summary(self, text: str) -> str:
        return text[:100]
    
    def _classify_document(self, text: str) -> str:
        return "general"

# 好的实践：适当的任务粒度
class FineGrainedTasks:
    def __init__(self):
        # 每个组件只做一件事
        self.cleaner = RunnableLambda(lambda x: x.strip().lower())
        self.entity_extractor = RunnableLambda(
            lambda x: ["entity1", "entity2"] if "entity" in x else []
        )
        self.sentiment_analyzer = RunnableLambda(
            lambda x: "positive" if "good" in x else "negative"
        )
        self.summarizer = RunnableLambda(lambda x: x[:100] + "..." if len(x) > 100 else x)
        self.classifier = RunnableLambda(
            lambda x: "technical" if "code" in x else "general"
        )
        
        # 组合成工作流
        self.document_processor = (
            RunnableLambda(lambda x: {"original": x, "cleaned": self.cleaner.invoke(x)})
            | RunnableParallel({
                "entities": RunnableLambda(lambda x: self.entity_extractor.invoke(x["cleaned"])),
                "sentiment": RunnableLambda(lambda x: self.sentiment_analyzer.invoke(x["cleaned"])),
                "summary": RunnableLambda(lambda x: self.summarizer.invoke(x["cleaned"])),
                "category": RunnableLambda(lambda x: self.classifier.invoke(x["cleaned"])),
                "cleaned": RunnableLambda(lambda x: x["cleaned"]),

    def process_document(self, document: str) -> Dict[str, Any]:
```

```python
# LangChain可配置的任务分解
from typing import List, Optional, Callable
from pydantic import BaseModel, Field
from enum import Enum

class ProcessingStep(str, Enum):
    CLEANING = "cleaning"
    EXTRACTION = "extraction"
    ANALYSIS = "analysis"
    SUMMARIZATION = "summarization"
    CLASSIFICATION = "classification"

class ProcessingConfig(BaseModel):
    """处理配置"""
    enabled_steps: List[ProcessingStep] = Field(
        default_factory=lambda: [ProcessingStep.CLEANING, ProcessingStep.ANALYSIS]
    )
    skip_steps: List[ProcessingStep] = Field(default_factory=list)
    custom_processors: Dict[ProcessingStep, Callable] = Field(default_factory=dict)
    max_length: Optional[int] = None
    language: str = "en"

class ConfigurableTaskDecomposition:
    """可配置的任务分解"""
    
    def __init__(self):
        # 定义所有可用的处理步骤
        self.step_registry = {
            ProcessingStep.CLEANING: self._create_cleaner(),
            ProcessingStep.EXTRACTION: self._create_extractor(),
            ProcessingStep.ANALYSIS: self._create_analyzer(),
            ProcessingStep.SUMMARIZATION: self._create_summarizer(),
            ProcessingStep.CLASSIFICATION: self._create_classifier(),
        }
    
    def _create_cleaner(self) -> Runnable:
        return RunnableLambda(lambda x: x.strip().lower())
    
    def _create_extractor(self) -> Runnable:
        return RunnableLambda(lambda x: f"Extracted: {x[:50]}")
    
    def _create_analyzer(self) -> Runnable:
        return RunnableLambda(lambda x: {"length": len(x), "words": len(x.split())})
    
    def _create_summarizer(self) -> Runnable:
        return RunnableLambda(lambda x: x[:100] + "..." if len(x) > 100 else x)
    
    def _create_classifier(self) -> Runnable:
        return RunnableLambda(lambda x: "technical" if "ai" in x else "general")
    
    def build_pipeline(self, config: ProcessingConfig) -> Runnable:
        """根据配置构建管道"""
        steps = []
        
        # 添加启用的步骤
        for step in config.enabled_steps:
            if step in config.skip_steps:
                continue
                
            # 使用自定义处理器或默认处理器
            if step in config.custom_processors:
                processor = RunnableLambda(config.custom_processors[step])
            else:
                processor = self.step_registry[step]
            
            steps.append(processor)
        
        # 如果有长度限制，添加长度检查
        if config.max_length:
            steps.append(
                RunnableLambda(lambda x: x[:config.max_length] if len(x) > config.max_length else x)
            )
        
        # 构建管道
        if not steps:
            return RunnableLambda(lambda x: x)
        
        pipeline = steps[0]
        for step in steps[1:]:
            pipeline = pipeline | step
        
        return pipeline
    
    def process(self, text: str, config: ProcessingConfig) -> Any:
        """根据配置处理文本"""
        pipeline = self.build_pipeline(config)
        return pipeline.invoke(text)
```

```python
# 任务分解的性能优化
import time
from concurrent.futures import ThreadPoolExecutor, as_completed
from typing import List, Any, Dict

class PerformanceOptimizedDecomposition:
    """性能优化的任务分解"""
    
    def __init__(self):
        # 定义处理步骤
        self.steps = {
            "clean": self._clean_text,
            "tokenize": self._tokenize_text,
            "analyze": self._analyze_tokens,
            "extract": self._extract_entities,
            "summarize": self._summarize_content,
        }
        
        # 步骤依赖关系
        self.dependencies = {
            "tokenize": ["clean"],
            "analyze": ["tokenize"],
            "extract": ["clean"],
            "summarize": ["analyze", "extract"],
        }
    
    def _clean_text(self, text: str) -> str:
        time.sleep(0.1)  # 模拟耗时操作
        return text.strip().lower()
    
    def _tokenize_text(self, text: str) -> List[str]:
        time.sleep(0.2)
        return text.split()
    
    def _analyze_tokens(self, tokens: List[str]) -> Dict[str, Any]:
        time.sleep(0.3)
        return {
            "count": len(tokens),
            "unique": len(set(tokens)),
            "avg_length": sum(len(t) for t in tokens) / len(tokens) if tokens else 0
        }
    
    def _extract_entities(self, text: str) -> List[str]:
        time.sleep(0.15)
        return [word for word in text.split() if len(word) > 5]
    
    def _summarize_content(self, data: Dict[str, Any]) -> str:
        time.sleep(0.25)
        analysis = data["analysis"]
        entities = data["entities"]
        return f"Summary: {analysis['count']} tokens, {len(entities)} entities"
    
    def process_sequential(self, text: str) -> Dict[str, Any]:
        """顺序执行（性能较差）"""
        results = {}
        
        # 必须按顺序执行
        results["clean"] = self.steps["clean"](text)
        results["tokenize"] = self.steps["tokenize"](results["clean"])
        results["analyze"] = self.steps["analyze"](results["tokenize"])
        results["extract"] = self.steps["extract"](results["clean"])
        results["summarize"] = self.steps["summarize"]({
            "analysis": results["analyze"],
            "entities": results["extract"]
        })
        
        return results
    
    def process_parallel(self, text: str) -> Dict[str, Any]:
        """并行执行（性能优化）"""
        results = {}
        
        # 第一步：清理（无依赖）
        results["clean"] = self.steps["clean"](text)
        
        # 并行执行可以并行的步骤
        with ThreadPoolExecutor(max_workers=3) as executor:
            # 提交并行任务
            future_tokenize = executor.submit(self.steps["tokenize"], results["clean"])
            future_extract = executor.submit(self.steps["extract"], results["clean"])
            
            # 等待并行任务完成
            results["tokenize"] = future_tokenize.result()
            results["extract"] = future_extract.result()
        
        # 依赖步骤：分析（需要tokenize的结果）
        results["analyze"] = self.steps["analyze"](results["tokenize"])
        
        # 最后一步：总结（需要analyze和extract的结果）
        results["summarize"] = self.steps["summarize"]({
            "analysis": results["analyze"],
            "entities": results["extract"]
        })
        
        return results
    
    def process_with_lcel(self, text: str) -> Dict[str, Any]:
        """使用LCEL的并行执行"""
        from langchain_core.runnables import RunnableParallel
        
        # 定义LCEL组件
        cleaner = RunnableLambda(self._clean_text)
        tokenizer = RunnableLambda(self._tokenize_text)
        analyzer = RunnableLambda(self._analyze_tokens)
        extractor = RunnableLambda(self._extract_entities)
        summarizer = RunnableLambda(self._summarize_content)
        
        # 构建并行执行的管道
        pipeline = (
            cleaner
            | RunnableParallel({
                "clean": RunnableLambda(lambda x: x),
                # 并行执行tokenize和extract
                "tokenize": tokenizer,
                "extract": extractor,
            })
            | RunnableParallel({
                "clean": RunnableLambda(lambda x: x["clean"]),
                "tokenize": RunnableLambda(lambda x: x["tokenize"]),
                "extract": RunnableLambda(lambda x: x["extract"]),
                "analyze": RunnableLambda(lambda x: analyzer.invoke(x["tokenize"])),
            })
            | RunnableLambda(lambda x: {
                "clean": x["clean"],
                "tokenize": x["tokenize"],
                "extract": x["extract"],
                "analyze": x["analyze"],
                "summarize": summarizer.invoke({
                    "analysis": x["analyze"],
                    "entities": x["extract"]
                })
            })
        )
        
        return pipeline.invoke(text)
```

## 顺序执行的错误处理模式

```python
# 顺序执行的错误处理
from typing import Optional, Callable
from langchain_core.runnables import RunnableConfig

class SequentialExecutionWithErrorHandling:
    """带有错误处理的顺序执行"""
    
    def __init__(self):
        # 定义处理步骤
        self.steps = [
            self._step1_validate,
            self._step2_process,
            self._step3_transform,
            self._step4_finalize,
        ]
        
        # 错误处理策略
        self.error_handlers = {
            "validation_error": self._handle_validation_error,
            "processing_error": self._handle_processing_error,
            "transformation_error": self._handle_transformation_error,
        }
    
    def _step1_validate(self, input_data: str) -> str:
        """验证步骤"""
        if not input_data or len(input_data) < 3:
            raise ValueError("validation_error: Input too short")
        return input_data
    
    def _step2_process(self, data: str) -> str:
        """处理步骤"""
        if "error" in data.lower():
            raise RuntimeError("processing_error: Contains error keyword")
        return f"Processed: {data}"
    
    def _step3_transform(self, data: str) -> str:
        """转换步骤"""
        if len(data) > 100:
            raise ValueError("transformation_error: Data too long")
        return data.upper()
    
    def _step4_finalize(self, data: str) -> Dict[str, Any]:
        """最终步骤"""
        return {"result": data, "status": "success", "timestamp": time.time()}
    
    def _handle_validation_error(self, error: Exception, input_data: Any) -> Any:
        """验证错误处理"""
        return {"result": "default_value", "status": "validation_failed", "error": str(error)}
    
    def _handle_processing_error(self, error: Exception, input_data: Any) -> Any:
        """处理错误处理"""
        # 尝试降级处理
        return f"Fallback processing: {input_data}"
    
    def _handle_transformation_error(self, error: Exception, input_data: Any) -> Any:
        """转换错误处理"""
        # 截断并继续
        return input_data[:100].upper() if len(input_data) > 100 else input_data.upper()
    
    def execute_with_error_handling(self, input_data: str) -> Dict[str, Any]:
        """带有错误处理的顺序执行"""
        current_data = input_data
        executed_steps = []
        errors = []
        
        for i, step in enumerate(self.steps):
            step_name = f"step_{i+1}"
            executed_steps.append(step_name)
            
            try:
                current_data = step(current_data)
            except Exception as e:
                errors.append({
                    "step": step_name,
                    "error": str(e),
                    "error_type": type(e).__name__
                })
                
                # 根据错误类型选择处理策略
                error_str = str(e)
                for error_key, handler in self.error_handlers.items():
                    if error_key in error_str:
                        try:
                            current_data = handler(e, current_data)
                            break
                        except Exception as handler_error:
                            # 错误处理器也失败了
                            errors.append({
                                "step": f"{step_name}_handler",
                                "error": str(handler_error),
                                "error_type": "handler_error"
                            })
                            # 无法恢复，返回错误
                            return {
                                "status": "critical_failure",
                                "errors": errors,
                                "executed_steps": executed_steps,
                                "partial_result": current_data
                            }
                else:
                    # 没有匹配的错误处理器
                    return {
                        "status": "unhandled_error",
                        "errors": errors,
                        "executed_steps": executed_steps,
                        "partial_result": current_data
                    }
        
        return {
            "status": "success",
            "result": current_data,
            "executed_steps": executed_steps,
            "errors": errors if errors else None
        }

# 使用带有错误处理的顺序执行
executor = SequentialExecutionWithErrorHandling()

test_cases = [
    "valid input",  # 正常情况
    "er",           # 验证错误（太短）
    "This input contains error message",  # 处理错误
    "A" * 150,      # 转换错误（太长）
    "normal input",  # 正常情况
]

for test_input in test_cases:
    result = executor.execute_with_error_handling(test_input)
    print(f"\n输入: {test_input}")
    print(f"结果状态: {result['status']}")
    print(f"执行步骤: {result['executed_steps']}")
    if result.get('errors'):
        print(f"错误: {result['errors']}")
    else:
        print(f"最终结果: {result.get('result', 'N/A')}")
```

## 总结：任务分解与顺序执行的核心价值

### 技术价值对比

| 维度 | 传统硬编码方式 | LangChain LCEL方式 |
|------|---------------|-------------------|
| **代码复杂度** | 高（嵌套条件、手动错误处理） | 低（声明式、自动错误传播） |

| **扩展性** | 差（添加步骤影响现有代码） | 好（可插拔组件） |
