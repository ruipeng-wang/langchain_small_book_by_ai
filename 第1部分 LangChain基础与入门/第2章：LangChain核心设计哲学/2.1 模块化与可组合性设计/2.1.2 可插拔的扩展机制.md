# 2.1.2 可插拔的扩展机制

## 从硬编码到可插拔的架构演进

### 传统AI应用：硬编码扩展困境

在传统AI应用中，添加新功能通常需要**修改核心代码**，导致系统越来越复杂和脆弱：

```python
# 传统硬编码扩展示例：AI助手系统
class HardcodedAIAssistant:
    def __init__(self):
        # 硬编码的功能列表
        self.capabilities = {
            "search": self._search_web,
            "calculate": self._calculate,
            "translate": self._translate_text,
            # 每添加一个新功能，就需要在这里添加一项
        }
        
        # 硬编码的模型提供商
        self.model_providers = {
            "openai": OpenAIProvider(),
            "anthropic": AnthropicProvider(),
            # 每支持一个新提供商，就需要在这里添加
        }
    
    def execute_capability(self, capability_name: str, input_data: str) -> str:
        """执行功能 - 硬编码的分支逻辑"""
        if capability_name not in self.capabilities:
            raise ValueError(f"Unsupported capability: {capability_name}")
        
        return self.capabilities[capability_name](input_data)
    
    def add_new_capability(self, name: str, func: callable):
        """添加新功能 - 需要修改内部数据结构"""
        self.capabilities[name] = func
        # 问题：需要重新部署整个系统
        # 问题：可能破坏现有功能
        # 问题：难以测试新功能
        
    def _search_web(self, query: str) -> str:
        """硬编码的搜索功能"""
        # 直接依赖特定搜索引擎API
        return f"Searching: {query} with Google"
    
    def _calculate(self, expression: str) -> str:
        """硬编码的计算功能"""
        return str(eval(expression))
    
    def _translate_text(self, text: str) -> str:
        """硬编码的翻译功能"""
        # 直接依赖特定翻译服务
        return f"Translated: {text}"
    
    # 扩展问题：
    # 1. 新功能需要修改execute_capability方法的调用方式
    # 2. 需要重新测试整个系统
    # 3. 难以支持第三方扩展
    # 4. 版本升级困难
```

**传统硬编码扩展的问题：**

1. **修改-重新部署循环**：每次扩展都需要修改核心代码并重新部署
2. **回归测试负担**：新功能可能影响现有功能的稳定性
3. **第三方集成困难**：难以集成外部团队开发的组件
4. **版本管理复杂**：不同版本的功能集难以管理
5. **运行时灵活性差**：无法动态加载和卸载功能

### LangChain的可插拔解决方案：插件系统架构

LangChain v1.2.22通过**插件系统架构**实现了真正的可插拔扩展：

```python
# LangChain可插拔架构：插件系统核心设计
from abc import ABC, abstractmethod
from typing import Dict, Any, List, Type, Optional
from pydantic import BaseModel, Field
import importlib
import pkgutil
import sys

# 插件基类：所有插件必须实现的接口
class Plugin(BaseModel, ABC):
    """插件基类"""
    name: str = Field(..., description="插件名称")
    version: str = Field(default="1.0.0", description="插件版本")
    description: str = Field(default="", description="插件描述")
    enabled: bool = Field(default=True, description="是否启用")
    
    @abstractmethod
    def initialize(self, config: Dict[str, Any]) -> None:
        """初始化插件"""
        pass
    
    @abstractmethod
    def cleanup(self) -> None:
        """清理插件资源"""
        pass

# 插件管理器：动态加载和管理插件
class PluginManager:
    """可插拔扩展的核心管理器"""
    
    def __init__(self):
        self._plugins: Dict[str, Plugin] = {}
        self._plugin_registry: Dict[str, Type[Plugin]] = {}
        self._discovery_paths: List[str] = []
    
    def register_plugin(self, plugin_class: Type[Plugin]) -> None:
        """注册插件类"""
        plugin_instance = plugin_class()
        self._plugin_registry[plugin_instance.name] = plugin_class
        print(f"Registered plugin: {plugin_instance.name} v{plugin_instance.version}")
    
    def load_plugin(self, plugin_name: str, config: Dict[str, Any] = None) -> Plugin:
        """动态加载插件"""
        if plugin_name not in self._plugin_registry:
            raise ValueError(f"Plugin not found: {plugin_name}")
        
        plugin_class = self._plugin_registry[plugin_name]
        plugin_instance = plugin_class()
        
        if config:
            plugin_instance.initialize(config)
        
        self._plugins[plugin_name] = plugin_instance
        print(f"Loaded plugin: {plugin_name}")
        
        return plugin_instance
    
    def unload_plugin(self, plugin_name: str) -> None:
        """动态卸载插件"""
        if plugin_name in self._plugins:
            self._plugins[plugin_name].cleanup()
            del self._plugins[plugin_name]
            print(f"Unloaded plugin: {plugin_name}")
    
    def discover_plugins(self, package_path: str) -> None:
        """自动发现插件"""
        self._discovery_paths.append(package_path)
        
        # 扫描指定包路径下的所有模块
        try:
            package = importlib.import_module(package_path)
            for _, module_name, is_pkg in pkgutil.iter_modules(package.__path__):
                if not is_pkg:
                    full_module_name = f"{package_path}.{module_name}"
                    self._load_module_plugins(full_module_name)
        except ImportError as e:
            print(f"Failed to discover plugins from {package_path}: {e}")
    
    def _load_module_plugins(self, module_name: str) -> None:
        """加载模块中的插件"""
        try:
            module = importlib.import_module(module_name)
            for attr_name in dir(module):
                attr = getattr(module, attr_name)
                if (isinstance(attr, type) and 
                    issubclass(attr, Plugin) and 
                    attr != Plugin):
                    self.register_plugin(attr)
        except Exception as e:
            print(f"Failed to load plugins from {module_name}: {e}")
    
    def get_plugin(self, plugin_name: str) -> Optional[Plugin]:
        """获取插件实例"""
        return self._plugins.get(plugin_name)
    
    def list_plugins(self) -> List[str]:
        """列出所有已加载的插件"""
        return list(self._plugins.keys())
    
    def enable_plugin(self, plugin_name: str) -> None:
        """启用插件"""
        if plugin_name in self._plugins:
            self._plugins[plugin_name].enabled = True
    
    def disable_plugin(self, plugin_name: str) -> None:
        """禁用插件"""
        if plugin_name in self._plugins:
            self._plugins[plugin_name].enabled = False
```

```python
# LangChain v1.2.22可插拔工具集成
from langchain_core.tools import BaseTool
from typing import Any, Dict, List, Optional
import json

# 1. 定义工具插件接口
class ToolPlugin(BaseTool, Plugin):
    """工具插件基类"""
    
    def initialize(self, config: Dict[str, Any]) -> None:
        """初始化工具插件"""
        # 可以从配置加载API密钥等
        self.api_key = config.get("api_key", "")
        print(f"Initialized tool: {self.name}")
    
    def cleanup(self) -> None:
        """清理工具资源"""
        print(f"Cleaned up tool: {self.name}")
    
    @abstractmethod
    def _run(self, input_data: Any) -> Any:
        """工具的核心逻辑"""
        pass

# 2. 创建具体工具插件
class CalculatorPlugin(ToolPlugin):
    """计算器工具插件"""
    
    name: str = "calculator"
    description: str = "A simple calculator tool"
    version: str = "1.0.0"
    
    def _run(self, expression: str) -> str:
        """执行计算"""
        try:
            result = eval(expression)
            return f"Result: {result}"
        except Exception as e:
            return f"Calculation error: {e}"

class SearchPlugin(ToolPlugin):
    """搜索工具插件"""
    
    name: str = "web_search"
    description: str = "Web search tool"
    version: str = "1.1.0"
    
    def __init__(self):
        super().__init__()
        self.search_engine = None
    
    def initialize(self, config: Dict[str, Any]) -> None:
        """初始化搜索插件"""
        super().initialize(config)
        # 可以动态配置搜索引擎
        engine_type = config.get("engine", "google")
        if engine_type == "google":
            self.search_engine = GoogleSearchEngine(config.get("api_key"))
        elif engine_type == "bing":
            self.search_engine = BingSearchEngine(config.get("api_key"))
    
    def _run(self, query: str) -> str:
        """执行搜索"""
        if not self.search_engine:
            return "Search engine not initialized"
        
        results = self.search_engine.search(query)
        return f"Found {len(results)} results for: {query}"

class TranslationPlugin(ToolPlugin):
    """翻译工具插件"""
    
    name: str = "translator"
    description: str = "Text translation tool"
    version: str = "2.0.0"
    
    def __init__(self):
        super().__init__()
        self.supported_languages = ["en", "zh", "es", "fr"]
    
    def _run(self, input_data: Dict[str, Any]) -> str:
        """执行翻译"""
        text = input_data.get("text", "")
        target_lang = input_data.get("target_lang", "en")
        
        if target_lang not in self.supported_languages:
            return f"Unsupported language: {target_lang}"
        
        return f"Translated '{text}' to {target_lang}"

# 3. 创建插件管理器
plugin_manager = PluginManager()

# 注册插件

plugin_manager.register_plugin(SearchPlugin)
plugin_manager.register_plugin(TranslationPlugin)

# 使用搜索插件
```

```python
# LangChain模型插件架构
from langchain_core.language_models import BaseLanguageModel
from typing import Any, Dict, List

class ModelPlugin(BaseLanguageModel, Plugin):
    """模型插件基类"""
    
    def __init__(self):
        super().__init__()
        self.model_config: Dict[str, Any] = {}
    
    def initialize(self, config: Dict[str, Any]) -> None:
        """初始化模型插件"""
        self.model_config = config
        # 可以在这里加载模型权重、建立连接等
        print(f"Initialized model: {self.name} with config {config}")
    
    @abstractmethod
    def _call(self, prompt: str, **kwargs: Any) -> str:
        """模型调用逻辑"""
        pass
    
    def invoke(self, input: Any, config: Optional[Dict] = None) -> Any:
        """统一的invoke接口"""
        return self._call(str(input), **(config or {}))

# OpenAI模型插件
class OpenAIModelPlugin(ModelPlugin):
    """OpenAI模型插件"""
    
    name: str = "openai_gpt4"
    description: str = "OpenAI GPT-4 model plugin"
    version: str = "1.2.0"
    
    def __init__(self):
        super().__init__()
        self.client = None
    
    def initialize(self, config: Dict[str, Any]) -> None:
        super().initialize(config)
        import openai
        self.client = openai.OpenAI(api_key=config.get("api_key"))
    
    def _call(self, prompt: str, **kwargs) -> str:
        if not self.client:
            return "OpenAI client not initialized"
        
        response = self.client.chat.completions.create(
            model="gpt-4",
            messages=[{"role": "user", "content": prompt}],
            **kwargs
        )
        return response.choices[0].message.content

# Anthropic模型插件
class AnthropicModelPlugin(ModelPlugin):
    """Anthropic Claude模型插件"""
    
    name: str = "anthropic_claude"
    description: str = "Anthropic Claude model plugin"
    version: str = "1.1.0"
    
    def __init__(self):
        super().__init__()
        self.client = None
    
    def initialize(self, config: Dict[str, Any]) -> None:
        super().initialize(config)
        import anthropic
        self.client = anthropic.Anthropic(api_key=config.get("api_key"))

        if not self.client:

            model="claude-3-opus-20240229",
```

```python
# LangChain记忆插件架构
from langchain_core.memory import BaseMemory
from typing import Any, Dict, List

class MemoryPlugin(BaseMemory, Plugin):
    """记忆插件基类"""
    
    def __init__(self):
        super().__init__()
        self.storage_backend = None
    
    def initialize(self, config: Dict[str, Any]) -> None:
        """初始化记忆插件"""
        backend_type = config.get("backend", "memory")
        if backend_type == "redis":
            self.storage_backend = RedisBackend(config.get("redis_url"))
        elif backend_type == "postgres":
            self.storage_backend = PostgresBackend(config.get("connection_string"))
        else:
            self.storage_backend = InMemoryBackend()
    
    @abstractmethod
    def save_context(self, inputs: Dict[str, Any], outputs: Dict[str, Any]) -> None:
        """保存对话上下文"""
        pass
    
    @abstractmethod
    def load_memory_variables(self, inputs: Dict[str, Any]) -> Dict[str, Any]:
        """加载记忆变量"""
        pass
    
    def cleanup(self) -> None:
        """清理记忆资源"""
        if self.storage_backend:
            self.storage_backend.close()

# Redis记忆插件
class RedisMemoryPlugin(MemoryPlugin):
    """Redis记忆插件"""
    
    name: str = "redis_memory"
    description: str = "Redis-based memory storage"
    version: str = "1.3.0"
    
    def save_context(self, inputs: Dict[str, Any], outputs: Dict[str, Any]) -> None:
        """保存到Redis"""
        session_id = inputs.get("session_id", "default")
        key = f"memory:{session_id}"
        data = {"inputs": inputs, "outputs": outputs, "timestamp": time.time()}
        self.storage_backend.save(key, json.dumps(data))
    
    def load_memory_variables(self, inputs: Dict[str, Any]) -> Dict[str, Any]:
        """从Redis加载"""
        session_id = inputs.get("session_id", "default")
        key = f"memory:{session_id}"
        data = self.storage_backend.load(key)
        if data:
            return json.loads(data)
        return {}

# 向量记忆插件
class VectorMemoryPlugin(MemoryPlugin):
    """向量记忆插件（支持语义搜索）"""
    
    name: str = "vector_memory"
    description: str = "Vector-based semantic memory"
    version: str = "2.0.0"
    
    def __init__(self):
        super().__init__()
        self.embedding_model = None
        self.vector_store = None
    
    def initialize(self, config: Dict[str, Any]) -> None:
        super().initialize(config)
        # 初始化嵌入模型
        self.embedding_model = OpenAIEmbeddings(api_key=config.get("api_key"))
        # 初始化向量存储
        self.vector_store = ChromaVectorStore(
            embedding_function=self.embedding_model,
            persist_directory=config.get("persist_dir", "./chroma_db")
        )
    
    def save_context(self, inputs: Dict[str, Any], outputs: Dict[str, Any]) -> None:
        """保存为向量"""
        text = f"Input: {inputs.get('input', '')}\nOutput: {outputs.get('output', '')}"
        metadata = {
            "session_id": inputs.get("session_id", "default"),
            "timestamp": time.time(),
            "input_type": type(inputs.get("input", "")).__name__
        }
        self.vector_store.add_texts([text], [metadata])
    
    def load_memory_variables(self, inputs: Dict[str, Any]) -> Dict[str, Any]:
        """语义搜索相关记忆"""
        query = inputs.get("input", "")
        if not query:
            return {}
        
        # 语义搜索相关记忆
        results = self.vector_store.similarity_search(query, k=3)
        relevant_memories = [result.page_content for result in results]
        
        return {
            "relevant_memories": relevant_memories,
            "memory_count": len(relevant_memories)
        }

# 记忆插件管理器
memory_manager = PluginManager()
memory_manager.register_plugin(RedisMemoryPlugin)
memory_manager.register_plugin(VectorMemoryPlugin)

# 根据场景选择记忆插件
development_config = {"backend": "memory"}  # 开发环境使用内存
production_config = {"backend": "redis", "redis_url": "redis://localhost:6379"}  # 生产环境使用Redis
ai_research_config = {  # AI研究使用向量记忆
    "backend": "vector",
    "api_key": "sk-...",
    "persist_dir": "./research_memories"
}

dev_memory = memory_manager.load_plugin("redis_memory", development_config)
prod_memory = memory_manager.load_plugin("redis_memory", production_config)
research_memory = memory_manager.load_plugin("vector_memory", ai_research_config)
```

```python
# LangChain的动态类加载机制
import importlib
import inspect
from typing import Type, Any

class DynamicClassLoader:
    """动态类加载器"""
    
    @staticmethod
    def load_class(class_path: str) -> Type[Any]:
        """从字符串路径加载类"""
        module_path, class_name = class_path.rsplit('.', 1)
        
        try:
            module = importlib.import_module(module_path)
            cls = getattr(module, class_name)
            
            if not inspect.isclass(cls):
                raise ValueError(f"{class_path} is not a class")
            
            return cls
        except ImportError as e:
            raise ImportError(f"Failed to import module {module_path}: {e}")
        except AttributeError as e:
            raise AttributeError(f"Class {class_name} not found in {module_path}: {e}")
    
    @staticmethod
    def discover_classes_in_package(
        package_name: str, 
        base_class: Type[Any]
    ) -> List[Type[Any]]:
        """在包中发现所有继承自base_class的类"""
        found_classes = []
        
        try:
            package = importlib.import_module(package_name)
            
            # 遍历包中的所有模块
            for _, module_name, is_pkg in pkgutil.iter_modules(package.__path__):
                if not is_pkg:
                    full_module_name = f"{package_name}.{module_name}"
                    
                    try:
                        module = importlib.import_module(full_module_name)
                        
                        # 查找模块中的所有类
                        for name, obj in inspect.getmembers(module, inspect.isclass):
                            if (issubclass(obj, base_class) and 
                                obj != base_class and 
                                obj.__module__ == module.__name__):
                                found_classes.append(obj)
                    except Exception as e:
                        print(f"Failed to inspect module {full_module_name}: {e}")
        
        except ImportError as e:
            print(f"Failed to import package {package_name}: {e}")
        
        return found_classes
```

```python
# LangChain配置驱动插件系统
from pydantic import BaseModel, Field
from typing import Dict, Any, List, Optional
import yaml

class PluginConfig(BaseModel):
    """插件配置模型"""
    name: str = Field(..., description="插件名称")
    class_path: str = Field(..., description="插件类的完整路径")
    enabled: bool = Field(default=True, description="是否启用")
    config: Dict[str, Any] = Field(default={}, description="插件配置")
    dependencies: List[str] = Field(default=[], description="依赖的插件")
    priority: int = Field(default=0, description="加载优先级")

class PluginSystemConfig(BaseModel):
    """插件系统配置"""
    plugins: List[PluginConfig] = Field(default=[], description="插件列表")
    discovery_paths: List[str] = Field(default=[], description="自动发现路径")
    auto_discovery: bool = Field(default=True, description="是否启用自动发现")
    lazy_loading: bool = Field(default=False, description="是否懒加载")

class ConfigDrivenPluginManager(PluginManager):
    """配置驱动的插件管理器"""
    
    def __init__(self, config_file: Optional[str] = None):
        super().__init__()
        self.system_config: Optional[PluginSystemConfig] = None
        
        if config_file:
            self.load_config(config_file)
    
    def load_config(self, config_file: str) -> None:
        """加载配置文件"""
        with open(config_file, 'r') as f:
            if config_file.endswith('.yaml') or config_file.endswith('.yml'):
                config_data = yaml.safe_load(f)
            elif config_file.endswith('.json'):
                config_data = json.load(f)
            else:
                raise ValueError(f"Unsupported config file format: {config_file}")
        
        self.system_config = PluginSystemConfig(**config_data)
        self._apply_config()
    
    def _apply_config(self) -> None:
        """应用配置"""
        if not self.system_config:
            return
        
        # 添加自动发现路径
        for path in self.system_config.discovery_paths:
            self.discover_plugins(path)
        
        # 按优先级排序插件
        sorted_plugins = sorted(
            self.system_config.plugins,
            key=lambda p: p.priority,
            reverse=True  # 优先级高的先加载
        )
        
        # 加载启用的插件
        for plugin_config in sorted_plugins:
            if plugin_config.enabled:
                self._load_plugin_from_config(plugin_config)
    
    def _load_plugin_from_config(self, plugin_config: PluginConfig) -> None:
        """从配置加载插件"""
        try:
            # 检查依赖
            for dep in plugin_config.dependencies:
                if dep not in self._plugins:
                    print(f"Warning: Plugin {plugin_config.name} depends on {dep}, but it's not loaded")
            
            # 动态加载类
            plugin_class = DynamicClassLoader.load_class(plugin_config.class_path)
            
            # 注册插件
            self.register_plugin(plugin_class)
            
            # 加载插件实例
            plugin_instance = self.load_plugin(
                plugin_config.name, 
                plugin_config.config
            )
            
            print(f"Successfully loaded plugin from config: {plugin_config.name}")
            
        except Exception as e:
            print(f"Failed to load plugin {plugin_config.name}: {e}")
```

```python
# LangChain插件生命周期管理
from enum import Enum
from typing import Dict, Any, Optional
import threading
import time

class PluginState(Enum):
    """插件状态"""
    UNLOADED = "unloaded"
    LOADING = "loading"
    LOADED = "loaded"
    INITIALIZING = "initializing"
    READY = "ready"
    ERROR = "error"
    UNLOADING = "unloading"

class PluginLifecycleManager:
    """插件生命周期管理器"""
    
    def __init__(self):
        self.plugin_states: Dict[str, PluginState] = {}
        self.plugin_errors: Dict[str, str] = {}
        self.lock = threading.RLock()
    
    def load_plugin_with_lifecycle(
        self, 
        plugin_manager: PluginManager,
        plugin_name: str,
        config: Dict[str, Any]
    ) -> Optional[Plugin]:
        """带生命周期的插件加载"""
        with self.lock:
            # 检查插件状态
            if plugin_name in self.plugin_states:
                current_state = self.plugin_states[plugin_name]
                if current_state in [PluginState.LOADING, PluginState.INITIALIZING]:
                    print(f"Plugin {plugin_name} is already being loaded")
                    return None
                elif current_state == PluginState.READY:
                    print(f"Plugin {plugin_name} is already loaded")
                    return plugin_manager.get_plugin(plugin_name)
            
            # 开始加载
            self.plugin_states[plugin_name] = PluginState.LOADING
            print(f"Loading plugin: {plugin_name}")
        
        try:
            # 加载插件类
            plugin_class = plugin_manager._plugin_registry.get(plugin_name)
            if not plugin_class:
                raise ValueError(f"Plugin class not found: {plugin_name}")
            
            # 创建实例
            with self.lock:
                self.plugin_states[plugin_name] = PluginState.INITIALIZING
            
            plugin_instance = plugin_class()
            
            # 初始化插件
            plugin_instance.initialize(config)
            
            # 验证插件
            self._validate_plugin(plugin_instance)
            
            # 注册到管理器
            plugin_manager._plugins[plugin_name] = plugin_instance
            
            with self.lock:
                self.plugin_states[plugin_name] = PluginState.READY
                if plugin_name in self.plugin_errors:
                    del self.plugin_errors[plugin_name]
            
            print(f"Plugin {plugin_name} loaded successfully")
            return plugin_instance
            
        except Exception as e:
            with self.lock:
                self.plugin_states[plugin_name] = PluginState.ERROR
                self.plugin_errors[plugin_name] = str(e)
            
            print(f"Failed to load plugin {plugin_name}: {e}")
            return None
    
    def unload_plugin_with_lifecycle(
        self, 
        plugin_manager: PluginManager,
        plugin_name: str
    ) -> bool:
        """带生命周期的插件卸载"""
        with self.lock:
            if plugin_name not in self.plugin_states:
                print(f"Plugin {plugin_name} not found")
                return False
            
            if self.plugin_states[plugin_name] == PluginState.UNLOADING:
                print(f"Plugin {plugin_name} is already being unloaded")
                return False
            
            self.plugin_states[plugin_name] = PluginState.UNLOADING
        
        try:
            # 获取插件实例
            plugin = plugin_manager.get_plugin(plugin_name)
            if plugin:
                # 执行清理
                plugin.cleanup()
            
            # 从管理器移除
            if plugin_name in plugin_manager._plugins:
                del plugin_manager._plugins[plugin_name]
            
            with self.lock:
                self.plugin_states[plugin_name] = PluginState.UNLOADED
                if plugin_name in self.plugin_errors:
                    del self.plugin_errors[plugin_name]
            
            print(f"Plugin {plugin_name} unloaded successfully")
            return True
            
        except Exception as e:
            with self.lock:
                self.plugin_states[plugin_name] = PluginState.ERROR
                self.plugin_errors[plugin_name] = f"Unload failed: {e}"
            
            print(f"Failed to unload plugin {plugin_name}: {e}")
            return False
    
    def _validate_plugin(self, plugin: Plugin) -> None:
        """验证插件"""
        # 检查必要属性
        required_attrs = ["name", "version", "description"]
        for attr in required_attrs:
            if not hasattr(plugin, attr) or not getattr(plugin, attr):
                raise ValueError(f"Plugin missing required attribute: {attr}")
        
        # 检查版本格式
        import re
        version_pattern = r'^\d+\.\d+\.\d+$'
        if not re.match(version_pattern, plugin.version):
            raise ValueError(f"Invalid version format: {plugin.version}")
        
        # 检查名称唯一性
        # 这里可以添加更多验证逻辑
    
    def get_plugin_state(self, plugin_name: str) -> PluginState:
        """获取插件状态"""
        with self.lock:
            return self.plugin_states.get(plugin_name, PluginState.UNLOADED)
    
    def get_plugin_error(self, plugin_name: str) -> Optional[str]:
        """获取插件错误信息"""
        with self.lock:
            return self.plugin_errors.get(plugin_name)
    
    def get_all_plugin_states(self) -> Dict[str, PluginState]:
        """获取所有插件状态"""
        with self.lock:
            return self.plugin_states.copy()
    
    def health_check(self) -> Dict[str, Any]:
        """健康检查"""
        with self.lock:
            total = len(self.plugin_states)
            ready = sum(1 for state in self.plugin_states.values() 
                       if state == PluginState.READY)
            errors = len(self.plugin_errors)
            
            return {
                "total_plugins": total,
                "ready_plugins": ready,
                "error_plugins": errors,
                "health_percentage": (ready / total * 100) if total > 0 else 100,
                "plugin_states": self.plugin_states.copy(),
                "plugin_errors": self.plugin_errors.copy()
            }
```

```python
class WellDesignedPlugin(Plugin):

        for key in required_keys:
            if key not in config:

        self._initialize_resources()
        
        self.initialized = True
        self.logger.info(f"Plugin {self.name} initialized successfully")

        self.resource_pool = ResourcePool(max_size=self.config.get("pool_size", 10))
    
    def cleanup(self) -> None:
        """清理插件资源"""
        if hasattr(self, 'resource_pool'):
            self.resource_pool.cleanup()
        
        self.initialized = False
        self.logger.info(f"Plugin {self.name} cleaned up")
    
    # 原则5：错误处理
    def execute(self, input_data: Any) -> Any:
        """执行插件功能，包含错误处理"""
        if not self.initialized:
            raise RuntimeError(f"Plugin {self.name} not initialized")
        
        try:
            # 输入验证
            validated_input = self._validate_input(input_data)
            
            # 执行业务逻辑
            result = self._execute_core(validated_input)
            
            # 结果处理
            processed_result = self._process_result(result)
            
            return processed_result
            
        except ValidationError as e:
            self.logger.error(f"Input validation failed: {e}")
            raise
        except ResourceError as e:
            self.logger.error(f"Resource error: {e}")
            # 尝试恢复
            self._recover_from_error()
            raise
        except Exception as e:
            self.logger.error(f"Unexpected error: {e}", exc_info=True)
            raise PluginExecutionError(f"Plugin execution failed: {e}")

            raise ValidationError("Input data cannot be empty")
        return input_data
```

```python
    def __init__(self):
        self.dependency_graph = nx.DiGraph()
        self.plugin_versions: Dict[str, str] = {}
    
    def add_plugin(self, plugin_name: str, version: str, dependencies: List[Tuple[str, str]]) -> None:
        """添加插件及其依赖"""
        # 添加插件节点
        self.dependency_graph.add_node(plugin_name, version=version)
        self.plugin_versions[plugin_name] = version
        
        # 添加依赖边
        for dep_name, dep_version_constraint in dependencies:
            self.dependency_graph.add_edge(plugin_name, dep_name)
            # 存储版本约束
            self.dependency_graph.edges[plugin_name, dep_name]['version_constraint'] = dep_version_constraint
    
    def validate_dependencies(self, plugin_name: str) -> List[str]:
        """验证插件依赖"""
        issues = []
        
        # 检查循环依赖
        try:
            cycle = nx.find_cycle(self.dependency_graph, source=plugin_name)
            if cycle:
                issues.append(f"Circular dependency detected: {cycle}")
        except nx.NetworkXNoCycle:
            pass
        
        # 检查直接依赖
        for dep_name in self.dependency_graph.successors(plugin_name):
            if dep_name not in self.plugin_versions:
                issues.append(f"Missing dependency: {dep_name}")
                continue
            
            # 检查版本约束
            edge_data = self.dependency_graph.edges[plugin_name, dep_name]
            constraint = edge_data.get('version_constraint', '*')
            
            if not self._check_version_constraint(
                self.plugin_versions[dep_name], 
                constraint
            ):
                issues.append(
                    f"Version conflict: {plugin_name} requires {dep_name} {constraint}, "
                    f"but found {self.plugin_versions[dep_name]}"
                )
        
        return issues
    
    def _check_version_constraint(self, version: str, constraint: str) -> bool:
        """检查版本约束"""
        if constraint == '*':
            return True
        
        # 简化版本：实际应该使用semver库
        # 这里只支持简单的版本约束如 ">=1.0.0", "~1.2.0"
        if constraint.startswith('>='):
            required = constraint[2:]
            return self._compare_versions(version, required) >= 0
        elif constraint.startswith('~'):
            # 兼容版本：~1.2.0 表示 >=1.2.0 <1.3.0
            base_version = constraint[1:]
            base_parts = base_version.split('.')
            if len(base_parts) >= 2:
                next_major = f"{base_parts[0]}.{int(base_parts[1]) + 1}.0"
                return (self._compare_versions(version, base_version) >= 0 and
                        self._compare_versions(version, next_major) < 0)
        
        return version == constraint
    
    def _compare_versions(self, v1: str, v2: str) -> int:
        """比较版本号"""
        v1_parts = list(map(int, v1.split('.')))
        v2_parts = list(map(int, v2.split('.')))
        
        # 补全到相同长度
        max_len = max(len(v1_parts), len(v2_parts))
        v1_parts.extend([0] * (max_len - len(v1_parts)))
        v2_parts.extend([0] * (max_len - len(v2_parts)))
        
        for p1, p2 in zip(v1_parts, v2_parts):
            if p1 < p2:
                return -1
            elif p1 > p2:
                return 1
        
        return 0
    
    def get_load_order(self, plugin_names: List[str]) -> List[str]:
        """获取插件加载顺序（拓扑排序）"""
        # 创建子图
        subgraph_nodes = set(plugin_names)
        
        # 添加所有依赖
        for plugin in plugin_names:
            ancestors = nx.ancestors(self.dependency_graph, plugin)
            subgraph_nodes.update(ancestors)
        
        # 创建子图并排序
        subgraph = self.dependency_graph.subgraph(subgraph_nodes)
        
        try:
            load_order = list(nx.topological_sort(subgraph))
            return load_order
        except nx.NetworkXUnfeasible:
            raise ValueError("Circular dependency detected, cannot determine load order")
    
    def resolve_dependencies(self, requested_plugins: List[str]) -> Dict[str, List[str]]:
        """解析插件依赖关系"""
        result = {
            "load_order": [],
            "missing_deps": [],
            "version_conflicts": [],
            "circular_deps": []
        }
        
        # 获取所有需要加载的插件（包括依赖）
        all_plugins = set(requested_plugins)
        for plugin in requested_plugins:
            if plugin in self.dependency_graph:
                ancestors = nx.ancestors(self.dependency_graph, plugin)
                all_plugins.update(ancestors)
        
        # 检查每个插件的依赖
        for plugin in all_plugins:
            if plugin not in self.dependency_graph:
                result["missing_deps"].append(plugin)
                continue
            
            issues = self.validate_dependencies(plugin)
            if issues:
                for issue in issues:
                    if "circular" in issue.lower():
                        result["circular_deps"].append(issue)
                    elif "version conflict" in issue.lower():
                        result["version_conflicts"].append(issue)
                    else:
                        result["missing_deps"].append(issue)
        
        # 获取加载顺序
        try:
            result["load_order"] = self.get_load_order(list(all_plugins))
        except ValueError as e:
            result["circular_deps"].append(str(e))
        
        return result
```

```python
# 插件热重载系统
import threading
import time
import hashlib
import os
from watchdog.observers import Observer
from watchdog.events import FileSystemEventHandler

class PluginHotReloader:
    """插件热重载器"""
    
    def __init__(self, plugin_manager: PluginManager, watch_paths: List[str]):
        self.plugin_manager = plugin_manager
        self.watch_paths = watch_paths
        self.file_hashes: Dict[str, str] = {}
        self.observer = Observer()
        self.reload_lock = threading.Lock()
        self.running = False
        
        # 设置文件系统事件处理器
        event_handler = PluginFileEventHandler(self)
        for path in watch_paths:
            if os.path.exists(path):
                self.observer.schedule(event_handler, path, recursive=True)
    
    def start(self) -> None:
        """开始监控文件变化"""
        if not self.running:
            self.running = True
            self.observer.start()
            print(f"Hot reloader started, watching: {self.watch_paths}")
    
    def stop(self) -> None:
        """停止监控"""
        if self.running:
            self.running = False
            self.observer.stop()
            self.observer.join()
            print("Hot reloader stopped")
    
    def check_for_changes(self) -> List[str]:
        """检查文件变化"""
        changed_files = []
        
        for watch_path in self.watch_paths:
            for root, _, files in os.walk(watch_path):
                for file in files:
                    if file.endswith('.py'):  # 只监控Python文件
                        file_path = os.path.join(root, file)
                        current_hash = self._calculate_file_hash(file_path)
                        
                        old_hash = self.file_hashes.get(file_path)
                        if old_hash is None:
                            # 新文件
                            self.file_hashes[file_path] = current_hash
                        elif old_hash != current_hash:
                            # 文件已修改
                            changed_files.append(file_path)
                            self.file_hashes[file_path] = current_hash
        
        return changed_files
    
    def _calculate_file_hash(self, file_path: str) -> str:
        """计算文件哈希"""
        try:
            with open(file_path, 'rb') as f:
                content = f.read()
                return hashlib.md5(content).hexdigest()
        except Exception:
            return ""
    
    def reload_plugin(self, plugin_name: str) -> bool:
        """重载插件"""
        with self.reload_lock:
            print(f"Reloading plugin: {plugin_name}")
            
            try:
                # 获取当前插件实例
                old_plugin = self.plugin_manager.get_plugin(plugin_name)
                if not old_plugin:
                    print(f"Plugin {plugin_name} not found")
                    return False
                
                # 保存当前状态和配置
                plugin_config = getattr(old_plugin, '_config', {})
                plugin_state = getattr(old_plugin, '_state', {})
                
                # 卸载旧插件
                self.plugin_manager.unload_plugin(plugin_name)
                
                # 重新导入模块
                plugin_class = self.plugin_manager._plugin_registry.get(plugin_name)
                if plugin_class:
                    module_name = plugin_class.__module__
                    importlib.reload(importlib.import_module(module_name))
                
                # 重新加载插件
                new_plugin = self.plugin_manager.load_plugin(plugin_name, plugin_config)
                
                # 恢复状态
                if hasattr(new_plugin, '_state'):
                    setattr(new_plugin, '_state', plugin_state)
                
                print(f"Plugin {plugin_name} reloaded successfully")
                return True
                
            except Exception as e:
                print(f"Failed to reload plugin {plugin_name}: {e}")
                return False
    
    def auto_reload_changed_plugins(self) -> None:
        """自动重载发生变化的插件"""
        changed_files = self.check_for_changes()
        
        if not changed_files:
            return
        
        print(f"Detected changes in {len(changed_files)} files")
        
        # 找出受影响的插件
        affected_plugins = set()
        
        for file_path in changed_files:
            # 尝试找出哪个插件定义在这个文件中
            for plugin_name, plugin_class in self.plugin_manager._plugin_registry.items():
                if hasattr(plugin_class, '__module__'):
                    module_name = plugin_class.__module__
                    module_file = importlib.import_module(module_name).__file__
                    
                    if module_file and os.path.samefile(module_file, file_path):
                        affected_plugins.add(plugin_name)
        
        # 重载受影响的插件
        for plugin_name in affected_plugins:
            self.reload_plugin(plugin_name)

class PluginFileEventHandler(FileSystemEventHandler):
    """插件文件系统事件处理器"""
    
    def __init__(self, hot_reloader: PluginHotReloader):
        self.hot_reloader = hot_reloader
    
    def on_modified(self, event):
        """文件修改事件"""
        if not event.is_directory and event.src_path.endswith('.py'):
            print(f"File modified: {event.src_path}")
            # 可以立即重载，也可以延迟重载
            threading.Timer(1.0, self.hot_reloader.auto_reload_changed_plugins).start()
```
